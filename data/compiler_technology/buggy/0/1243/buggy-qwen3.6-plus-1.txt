    public Token nextToken() {
        skipWhiteSpace();
        while (skipComment()) {
            skipWhiteSpace();
        }
        if (isEndOfInput()) {
            return null;
        }
        String currentWord = source.readCurrentWord(); // @@ Consumes the entire word and advances the file pointer, but only a prefix is matched as a token, leaving the rest of the word unprocessed and lost.
        int currentLine = source.getLineNum(); // @@ Returns 0-based line number, should be 1-based for output.
        for (TokenType tokentype : TokenType.values()) {
            Pattern pattern = tokentype.getPattern();
            Matcher matcher = pattern.matcher(currentWord);

            if (matcher.find() && matcher.start() == 0) {
                String value = matcher.group(); // 会不会出现value和Word实际长度不一样的情况？
                return new Token(tokentype, value, currentLine);
            }
        }
        return null;
    }
