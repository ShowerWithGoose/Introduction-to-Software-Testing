    public void moveForward(int steps) {
        int cnt = steps;
        while (!this.endFile() && cnt > 0) {
            int lineLen = getLine().length();
            if (linePos + cnt >= lineLen) {
                lineNum++;
//                cnt -= (lineLen - linePos + 1);
                cnt = 0; // @@ Remaining steps are discarded instead of being carried over to the next line; should be cnt -= (lineLen - linePos + 1) to correctly advance across line boundaries
                linePos = 0;
            } else {
                linePos += cnt;
                cnt = 0;
            }
        }
    }
