    private void parseCmt() {
        column++;
        char ch = curLine.charAt(column);
        if (ch == '/') {
            lineCmt = true;
            column++;
        } else if (ch == '*') {
            blockCmt = true;
            column++;
        } else {
            System.err.println("Undefined suffix of /");
        }
    }
