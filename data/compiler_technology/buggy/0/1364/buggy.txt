#include <stdio.h>
#include <math.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>
#include <iostream>
#include <bits/stdc++.h>
using namespace std;

struct Token{
	string type;
	string name;
};

struct Error{
	int line;
	string type;
};

	
int main(){
	
	freopen("testfile.txt", "r", stdin);
	
	vector<Token> keyWords;
	{
		keyWords.push_back({"MAINTK","main"});
		keyWords.push_back({"CONSTTK","const"});
		keyWords.push_back({"INTTK","int"});
		keyWords.push_back({"CHARTK","char"});
		keyWords.push_back({"BREAKTK","break"});
		keyWords.push_back({"CONTINUETK","continue"});
		keyWords.push_back({"IFTK","if"});
		keyWords.push_back({"ELSETK","else"});
		keyWords.push_back({"FORTK","for"});
		keyWords.push_back({"GETINTTK","getint"});
		keyWords.push_back({"GETCHARTK","getchar"});
		keyWords.push_back({"PRINTFTK","printf"});
		keyWords.push_back({"RETURNTK","return"});
		keyWords.push_back({"VOIDTK","void"});
	}
	vector<Token> tokens;
	vector<Error> errors;
	bool isWrong = false, isComment = false;
	int lineNum = 1, flag = 0;
	int num;
	string word, input;
	char *ch;
	tokens.clear();
	errors.clear();
	word.clear();
	while(getline(cin, input)){
		/*flag++;
		if(flag == 1){
			freopen("lexer.txt", "w", stdout);
			cout << flag << endl;
			fclose(stdout);
			break;
		}*/
		char line[1000] = {0};		
		strcpy(line, input.c_str());
		line[input.length()] = '\n';
		ch = line;
		for(; *ch != 0; ch++){
			num = 0;
			word.clear();
			if (isComment){
				if (*ch == '*'){
					ch++;
					if (*ch == '/'){
						isComment = false;
						continue;
					}
					ch--;
				}
				if (*ch == '\n'){
					lineNum++;
				}
				continue;
			}
			if (*ch == '/'){
				ch++;
				if (*ch == '/'){
					while (*ch != '\n')ch++;
					ch--;
					continue;
				}
				else if (*ch == '*'){
					isComment = true;
					continue;
				}
				else {
					tokens.push_back({"DIV", "/"});
					ch--;
					continue;
				}
			}
	
			if (*ch == '\n'){
				lineNum++;
				continue;
			}
			if (*ch == ' ' || *ch == '\t'){
				continue;
			}
			if (*ch >= '0' && *ch <= '9'){
				do{
					num = num * 10 + *ch - '0';
					ch++;
				}
				while (*ch >= '0' && *ch <= '9');
				ch--;
				tokens.push_back({"INTCON",to_string(num)});			
			}
			else if ((*ch >= 'a' && *ch <= 'z') || (*ch >= 'A' && *ch <= 'Z')){
				do{
					word += *ch;
					ch++;
				}			
				while ((*ch >= 'a' && *ch <= 'z') || (*ch >= 'A' && *ch <= 'Z') || (*ch >= '0' && *ch <= '9') || *ch == '_');
				ch--;
				bool isFind = false;
				for (vector<Token>::iterator it = keyWords.begin(); it != keyWords.end(); it++){
					if (it->name == word){
						isFind = true;
						tokens.push_back(*it);
						break;
					}
				}
				if (!isFind){
					tokens.push_back({"IDENFR", word});
				}
			}
			else if (*ch == '"'){
				do {
					word += *ch;
					ch++;				
				} while (*ch != '"');
				word += *ch;
				tokens.push_back({"STRCON", word});
			}
			else if (*ch == '\''){
				do {
					word += *ch;	
					ch++;			
				} while (*ch != '\'');
				word += *ch;
				tokens.push_back({"CHRCON", word});
			}
			else if (*ch == '!'){
				ch++;
				if (*ch == '='){
					tokens.push_back({"NEQ", "!="});
				}
				else {
					ch--;
					tokens.push_back({"NOT", "!"});
				}			
			}
			else if (*ch == '&'){
				ch++;
				if (*ch != '&'){
					isWrong = true;
					ch--;
					errors.push_back({lineNum, "a"});
					continue;
				}
				else {
					tokens.push_back({"AND", "&&"});
				}
			}
			else if (*ch == '|'){
				ch++;
				if (*ch != '|'){
					isWrong = true;
					ch--;
					errors.push_back({lineNum, "a"});
					continue;
				}
				else {
					tokens.push_back({"OR", "||"});
				}
			}
			else if (*ch == '+'){
				tokens.push_back({"PLUS", "+"});
			}
			else if (*ch == '-'){
				tokens.push_back({"MINU", "-"});
			}
			else if (*ch == '*'){
				tokens.push_back({"MULT", "*"});
			}
			else if (*ch == '%'){
				tokens.push_back({"MOD", "%"});
			}
			else if (*ch == '<'){
				ch++;
				if (*ch == '='){
					tokens.push_back({"LEQ", "<="});
				}
				else {
					ch--;
					tokens.push_back({"LSS", "<"});
				}
			}
			else if (*ch == '>'){
				ch++;
				if (*ch == '='){
					tokens.push_back({"GEQ", ">="});
				}
				else {
					ch--;
					tokens.push_back({"GRE", ">"});
				}
			}
			else if (*ch == '='){
				ch++;
				if (*ch == '='){
					tokens.push_back({"EQL", "=="});
				}
				else {
					ch--;
					tokens.push_back({"ASSIGN", "="});
				}
			}
			else if (*ch == ';'){
				tokens.push_back({"SEMICN", ";"});
			}
			else if (*ch == ','){
				tokens.push_back({"COMMA", ","});
			}
			else if (*ch == '('){
				tokens.push_back({"LPARENT", "("});
			}
			else if (*ch == ')'){
				tokens.push_back({"RPARENT", ")"});
			}
			else if (*ch == '['){
				tokens.push_back({"LBRACK", "["});
			}
			else if (*ch == ']'){
				tokens.push_back({"RBRACK", "]"});
			}
			else if (*ch == '{'){
				tokens.push_back({"LBRACE", "{"});
			}
			else if (*ch == '}'){
				tokens.push_back({"RBRACE", "}"});
			}		
		}
		
		/*if(flag == 1){
				freopen("lexer.txt", "w", stdout);
				cout << flag << endl;
				fclose(stdout);
		}*/
		//if (ch == -1)isEnd = true;
	}

	if (!isWrong){
		freopen("lexer.txt", "w", stdout);
		for (vector<Token>::iterator it = tokens.begin(); it != tokens.end(); it++){
			cout << it->type << ' ' << it->name << endl;
		}
		//cout << ch << endl;
		fclose(stdout);
	}
	else {
		freopen("error.txt", "w", stdout);
		for (vector<Error>::iterator it = errors.begin(); it != errors.end(); it++){
			cout << it->line << ' ' << it->type << endl;
		}
		fclose(stdout);
	}
	

	


	
	fclose(stdin);
	
	return 0;
}
