    // 处理其他符号（如运算符，分隔符等）
    private void analyzeSymbol() {
        if (currentChar == '&' && peekChar() == '&') {
            tokens.add(new Token("AND", "&&", lineNumber));
            nextChar();  // 跳过第二个 &
            currentChar = nextChar();
        } else if (currentChar == '|' && peekChar() == '|') {
            tokens.add(new Token("OR", "||", lineNumber));
            nextChar();  // 跳过第二个 |
            currentChar = nextChar();
        } else {
            String tokenType = Keywords.getTokenType(Character.toString(currentChar));
            if (!tokenType.equals("IDENFR")) {
                tokens.add(new Token(tokenType, Character.toString(currentChar), lineNumber));
            } else {
                errors.add(lineNumber + " a");  // 非法字符 @@ [This incorrectly treats standalone '&' or '|' as error 'a', but also misclassifies any unknown symbol (e.g., '@') as error 'a'. More critically, it fails to detect the specific lexical error where '&' or '|' appear alone — it should check explicitly for single '&' or '|' and report error 'a' only in those cases, not for all unrecognized symbols.]
            }
            currentChar = nextChar();  // 推进到下一个字符
        }
    }
