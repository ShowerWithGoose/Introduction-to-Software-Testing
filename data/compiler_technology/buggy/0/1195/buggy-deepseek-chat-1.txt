import java.io.File;
import java.io.FileNotFoundException;
import java.util.Scanner;
import frontend.Lexer;

public class Compiler {
    public static void main(String[] args) {
        String filePath = "testfile.txt";

        StringBuilder input = new StringBuilder();

        try(Scanner scanner = new Scanner(new File(filePath))){
            while(scanner.hasNextLine()){
                String line = scanner.nextLine();
                input.append(line); // @@ [Missing newline character: When reading lines with Scanner.nextLine(), the newline character '\n' is not included. This causes the lexer to lose line break information, making it impossible to correctly track line numbers and handle newline-dependent logic.]
            }
        }catch (FileNotFoundException e){
            System.out.printf("文件未找到！！！");
        }

        Lexer lexer = new Lexer(input.toString());
        lexer.lex();
    }
}
