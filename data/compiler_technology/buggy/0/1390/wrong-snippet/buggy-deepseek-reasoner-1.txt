    private String removeAnnotation() {
        // 正则表达式：匹配单行注释和多行注释
        String regex = "(//.*?$)|(/\\*[^*]*\\*+(?:[^/*][^*]*\\*+)*/)";

        // 使用 Pattern.DOTALL 使 '.' 匹配换行符，Pattern.MULTILINE 使 '^' 和 '$' 匹配每一行的开始和结束
        Pattern pattern = Pattern.compile(regex, Pattern.DOTALL | Pattern.MULTILINE);
        Matcher matcher = pattern.matcher(input);

        // 替换所有匹配到的注释，单行注释直接移除，多行注释替换为空行（保留换行符）
        StringBuffer result = new StringBuffer();
        while (matcher.find()) {
            if (matcher.group(1) != null) {
                // 这是单行注释，移除它但保留换行符
                matcher.appendReplacement(result, "");
            } else if (matcher.group(2) != null) {
                // 这是多行注释，保留注释中的换行符
                String multiLineComment = matcher.group(2);
                String newLines = multiLineComment.replaceAll("[^\n]", ""); // 只保留换行符
                matcher.appendReplacement(result, newLines);
            }
        }
        matcher.appendTail(result);
        return result.toString();
    }
