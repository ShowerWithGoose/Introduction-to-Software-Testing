case Start:  {
    if (Character.isDigit(current)) { // @@ NullPointerException if current is null (e.g., after reading end-of-file)
        state = State.Digit;
        token.append(current);
    }
    else if (Character.isAlphabetic(current) || current.equals('_')){ // @@ current.equals('_') throws NullPointerException if current is null
        state = State.NonDigit;
        token.append(current);
        peekChar(); // @@ If this peekChar() reaches end-of-file, current becomes null, causing NPE in NonDigit while loop
    }
    else if (current.equals(' ')
            || current.equals('\r')
            || current.equals('\t')
            || current.equals('\n')) {
        peekChar();
        break;
    }
    else {
        state = State.Operation;
    }
    break;
