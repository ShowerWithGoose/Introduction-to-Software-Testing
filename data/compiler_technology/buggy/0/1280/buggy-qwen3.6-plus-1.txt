    private void identLogical() throws IOException {
        int line = lineNumber;
        int column = columnNumber;
        if (currentChar == '&') {
            advance();
            if (currentChar != '&') {
                errors.add(new Error(line, column, 'a')); // @@ 报错后缺少 advance() 调用，未消耗当前字符，导致该字符保留在缓冲区中，可能在下一轮循环中被错误解析或引发状态不一致
            } else {
                words.add(new Word(line, column, "&&", Type.AND));
                advance();
            }
        } else if (currentChar == '|') {
            advance();
            if (currentChar != '|') {
                errors.add(new Error(line, column, 'a')); // @@ 报错后缺少 advance() 调用，未消耗当前字符，导致该字符保留在缓冲区中，可能在下一轮循环中被错误解析或引发状态不一致
            } else {
                words.add(new Word(line, column, "||", Type.OR));
                advance();
            }
        } else {
            System.out.println("Error : not Logical Char");
        }
    }
