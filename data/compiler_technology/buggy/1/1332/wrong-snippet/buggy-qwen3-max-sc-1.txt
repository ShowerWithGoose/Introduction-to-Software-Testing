The key problem is in the `parseCompUnit()` method in Parser.java, which incorrectly determines when to parse declarations vs function definitions vs main function.

    private CompUnit parseCompUnit() {
        // CompUnit -> {Decl} {FuncDef} MainFuncDef
        List<Decl> decls = null;
        List<FuncDef> funcDefs = null;
        MainFuncDef mainFuncDef = null;

        // Default program completeness, i.e., preReadToken(2) won't be null
        // While condition not satisfied, pay attention to infinite loop issues, often occurring when token is null
        while (!isEOF() && getPreReadTokenType(2) != TokenType.LPARENT
                && getPreReadTokenType(1) != TokenType.MAINTK) {
            // As long as the loop entry condition is satisfied, there will definitely be decls
            if (decls == null) {
                decls = new ArrayList<>();
            }
            decls.add(parseDecl());
        }
        // MAINTK is considered entering MainFuncDef
        while (!isEOF() && getPreReadTokenType(1) != TokenType.MAINTK) {
            if(funcDefs == null){
                funcDefs = new ArrayList<>();
            }
            funcDefs.add(parseFuncDef());
        }
        mainFuncDef = parseMainFuncDef();
        return new CompUnit(decls, funcDefs, mainFuncDef);
    }
