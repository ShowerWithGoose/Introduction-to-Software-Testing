    public Token match(LexType expect){
        Token tmp=now;
        if(now.getLexType()==expect){
            if(index<tokens.size()-1){
                now=tokens.get(++index);
            }
            return tmp;
        }
        else if (expect==LexType.SEMICN) {
            ErrorHandler.getInstance().print(new CompileException(tokens.get(index-1).getLineNumber(), ErrorType.i)); // @@ Potential IndexOutOfBoundsException if index == 0; should use now.getLineNumber() or check bounds.
            return new Token(LexType.SEMICN,";", tokens.get(index-1).getLineNumber()); // @@ Potential IndexOutOfBoundsException if index == 0.
        }
        else if (expect==LexType.RPARENT) {
            ErrorHandler.getInstance().print(new CompileException(tokens.get(index-1).getLineNumber(), ErrorType.j)); // @@ Potential IndexOutOfBoundsException if index == 0.
            return new Token(LexType.RPARENT,")", tokens.get(index-1).getLineNumber()); // @@ Potential IndexOutOfBoundsException if index == 0.
        } else if (expect==LexType.RBRACK) {
            ErrorHandler.getInstance().print(new CompileException(tokens.get(index-1).getLineNumber(), ErrorType.k)); // @@ Potential IndexOutOfBoundsException if index == 0.
            return new Token(LexType.RBRACK,"]", tokens.get(index-1).getLineNumber()); // @@ Potential IndexOutOfBoundsException if index == 0.
        }
        else{
//            ErrorHandler.getInstance().print(new CompileException(tokens.get(index-1).getLineNumber(), ErrorType.NoneMatch));
            return new Token(expect,lexerTypes.get(expect), tokens.get(index-1).getLineNumber()); // @@ Potential IndexOutOfBoundsException if index == 0.
        }
    }
