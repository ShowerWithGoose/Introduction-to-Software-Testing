    public CompUnit parseCompUnit() {
        CompUnit compUnit = new CompUnit();
        while (words.get(pos).getStr().equals("const") ||
            !words.get(pos + 2).getStr().equals("(")) { // @@ [This condition can cause IndexOutOfBoundsException when pos+2 exceeds the list size, and incorrectly assumes that non-const declarations must be followed by '(' at pos+2 to distinguish from FuncDef]
            compUnit.addDecl(parseDecl());
        }
        while (!words.get(pos + 1).getStr().equals("main")) { // @@ [This condition can cause IndexOutOfBoundsException when pos+1 exceeds the list size, and fails to properly identify MainFuncDef which should start with 'int main']
            compUnit.addFuncDef(parseFuncDef());
        }
        compUnit.setMainFuncDef(parseMainFuncDef());
        return compUnit;
    }
