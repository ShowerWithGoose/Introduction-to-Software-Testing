    public CompUnit parseCompUnit() {
        CompUnit compUnit = new CompUnit();
        while (pos < words.size() && (words.get(pos).getStr().equals("const") ||
            (pos + 2 < words.size() && !words.get(pos + 2).getStr().equals("(")))) {
            compUnit.addDecl(parseDecl());
        }
        while (pos < words.size() && !words.get(pos + 1).getStr().equals("main")) { // @@ [This condition can cause IndexOutOfBoundsException when pos+1 >= words.size(), and the logic for distinguishing FuncDef from MainFuncDef is incorrect because it assumes all non-Decl items before main are FuncDef, but doesn't properly check function signature structure]
