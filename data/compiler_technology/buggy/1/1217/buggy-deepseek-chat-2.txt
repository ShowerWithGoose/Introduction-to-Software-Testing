    public Block parse(Parser parser,BufferedWriter writer) throws IOException{

        // 匹配 '{'
        parser.match(LexType.LBRACE);
        writer.write("LBRACE {\n");

        // 解析 BlockItem
        while (parser.getCurrentToken().type != LexType.RBRACE) { // @@ This loop does not check for end-of-tokens, which can cause a NullPointerException if the RBRACE is missing. Also, it does not handle the case where the block is empty (e.g., "{}"), as it will still enter the loop and try to parse a BlockItem.
            new BlockItem().parse(parser,writer);
        }

        // 匹配 '}'
        writer.write("RBRACE }\n");
        parser.match(LexType.RBRACE);
        writer.flush();
        return this;
