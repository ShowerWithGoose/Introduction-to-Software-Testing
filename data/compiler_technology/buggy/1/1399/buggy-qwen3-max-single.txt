    // 处理终结符
    private Token handleTerminal(LexType lexType) {
        if (getSymType().equals(lexType)) {
            Token token = getNowToken();
            nextIndex();
            return token;
        }
        is_analyze_success = false;
        if (lexType.equals(LexType.SEMICN)) {
            ErrorHandler.recordError(ErrorType.SEMICN_MISSING, getLineNumber(-1)); @@ [Using getLineNumber(-1) may not correctly identify the line number of the last non-terminal before the missing semicolon; it should use the line number of the previous token instead]
        }
        else if (lexType.equals(LexType.RPARENT)) {
            ErrorHandler.recordError(ErrorType.RPARENT_MISSING, getLineNumber(-1)); @@ [Using getLineNumber(-1) may not correctly identify the line number of the last non-terminal before the missing right parenthesis; it should use the line number of the previous token instead]
        }
        else if (lexType.equals(LexType.RBRACK)) {
            ErrorHandler.recordError(ErrorType.RBRACK_MISSING, getLineNumber(-1)); @@ [Using getLineNumber(-1) may not correctly identify the line number of the last non-terminal before the missing right bracket; it should use the line number of the previous token instead]
        }
        else {
            throw new Error(ErrorType.OTHER_ERROR, getLineNumber());
        }
        return null;
    }
