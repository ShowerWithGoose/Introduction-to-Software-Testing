    // 处理终结符
    private Token handleTerminal(LexType lexType) {
        if (getSymType().equals(lexType)) {
            Token token = getNowToken();
            nextIndex();
            return token;
        }
        is_analyze_success = false;
        if (lexType.equals(LexType.SEMICN)) {
            ErrorHandler.recordError(ErrorType.SEMICN_MISSING, getLineNumber(-1)); @@ [The error line number should correspond to the line of the last non-terminal before the expected semicolon, but getLineNumber(-1) points to the previous token which may not be correct in all contexts]
        }
        else if (lexType.equals(LexType.RPARENT)) {
            ErrorHandler.recordError(ErrorType.RPARENT_MISSING, getLineNumber(-1)); @@ [The error line number should correspond to the line where the last non-terminal before the right parenthesis is located, but using getLineNumber(-1) may not accurately reflect this]
        }
        else if (lexType.equals(LexType.RBRACK)) {
            ErrorHandler.recordError(ErrorType.RBRACK_MISSING, getLineNumber(-1)); @@ [The error line number should correspond to the line where the last preceding non-terminal symbol appears before the right square bracket, but getLineNumber(-1) might point to an incorrect line]
        }
        else {
            throw new Error(ErrorType.OTHER_ERROR, getLineNumber());
        }
        return null;
    }
