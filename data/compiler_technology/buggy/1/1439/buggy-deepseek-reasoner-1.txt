import java.util.List;

public class Compiler {
    public static void main(String[] args) {
        // 调用 Lexer 获取词法单元
        List<Token> tokens = Lexer.main(args); // @@ 若 Lexer 的 main 方法未能正确返回 Token 列表，将导致空指针异常或类型错误
        // 调用 Parser 进行语法分析
        Parser parser = new Parser(tokens);
        parser.parse(); // @@ 若 tokens 为空或 Parser 内部状态未初始化，会引发 NullPointerException 或 IndexOutOfBoundsException
    }
}
