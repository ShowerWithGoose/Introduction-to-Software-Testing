    public Branch CompUnit() {
        Branch compUnit = new Branch("CompUnit");
        while (true) {
            while (now() == Lexer.Type.INTTK) {
                if (preRead() == Lexer.Type.MAINTK) {
                    compUnit.add(MainFuncDef());
                    return compUnit;
                }
                if (preRead() == Lexer.Type.IDENFR) {
                    if (prePreRead() == Lexer.Type.LPARENT) {
                        compUnit.add(FuncDef());
                    } else {
                        compUnit.add(VarDecl());
                    }
                } else {
                    // @@ Error: If preRead() is neither MAINTK nor IDENFR, no action is taken,
                    // @@ and the token is not consumed, causing an infinite loop on the same token.
                }
            }
            while (now() == Lexer.Type.CHARTK) {
                if (preRead() == Lexer.Type.IDENFR) {
                    if (prePreRead() == Lexer.Type.LPARENT) {
                        compUnit.add(FuncDef());
                    } else {
                        compUnit.add(VarDecl());
                    }
                } else {
                    // @@ Similar issue: no action for unexpected follow token leads to infinite loop.
                }
            }
            while (now() == Lexer.Type.CONSTTK) {
                compUnit.add(ConstDecl());
                // @@ After ConstDecl, the token advances, but if now() is still CONSTTK, loop continues;
                // @@ However, if now() becomes something else, this loop exits, but outer while(true) never breaks.
            }
            while (now() == Lexer.Type.VOIDTK) {
                compUnit.add(FuncDef());
                // @@ Same issue: no mechanism to exit outer loop when no more tokens.
            }
            // @@ No break condition: if the current token is not any of the above, the loop never ends.
        }
    }
