public class BlockParser extends Parser {
    private final Lexer lexer;
    public BlockParser(Lexer lexer) {
        this.lexer = lexer;
    }

    // Block → '{' { BlockItem } '}'
    @Override
    public SynNode parse() {
        ArrayList<SynNode> children = new ArrayList<>();
        children.add(new SynLeaf(lexer.getAndNext())); // {
        while(lexer.hasNext()) { // @@ [This loop condition is incorrect: it may exit before encountering '}', causing the parser to miss the closing brace. A better approach is to loop until a '}' token is seen or until end of tokens.]
            Token t = lexer.getAndNext();
            if (t.getType() == TokenType.RBRACE) {
                children.add(new SynLeaf(t));
                break;
            } else {
                lexer.move(-1);
                children.add(parseBlockItem());
            }
        }
        // must end with }, hasNext is False
        if (!lexer.hasNext()) { // @@ [If the loop exited because hasNext() became false without seeing '}', this line will attempt to fetch a token that does not exist, leading to an IndexOutOfBoundsException. The parser should instead report an error for missing '}'.]
            children.add(new SynLeaf(lexer.getAndNext())); // }
        }
        return new SynNonLeaf(SynType.Block, children);
    }
