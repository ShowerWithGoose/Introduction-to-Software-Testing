        }
        if (tokens.get(token_index + 1).type == TokenType.MAINTK) { //主函数 @@ [Should check tokens.get(token_index).type == TokenType.INTTK first, then verify the next token is MAINTK, according to MainFuncDef grammar 'int' 'main' '(' ')' Block]
