    public void traverse(BufferedWriter writer) throws IOException {
        // 遍历每个子节点
        for (SyntaxTreeNode child : children) {
            // 如果是词法单元（叶子节点），先输出词法单元
            if (child.getChildren().isEmpty()) {
                writer.write(child.getType() + " " + child.getValue());
                writer.newLine(); // 换行
            } else {
                // 递归遍历子节点
                child.traverse(writer); // @@ [The syntax component name should be output BEFORE analyzing its contents, but this code outputs it AFTER traversing children. According to requirements, non-terminal names like <Stmt> must be printed before their constituent tokens are processed.]
