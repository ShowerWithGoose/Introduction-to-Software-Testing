    //<CompUnit> ::= {<Decl>} {<FuncDef>} <MainFuncDef>
    //<Decl>:           [const] int/char <Ident>
    //<FuncDef>:        void/int/char func()
    //<MainFuncDef>:    int main()
    private CompUnitNode parseCompUnit() {
        //Decl
        ArrayList<DeclNode> declNodes = new ArrayList<>();
        while (peek(2) != TokenType.LPARENT) { // @@ [Incorrect lookahead condition: using peek(2) to detect function definitions is unreliable because LPARENT might appear in expressions or other contexts, and declarations don't necessarily end before the first LPARENT. This can cause premature termination of Decl parsing or infinite loops.]
            declNodes.add(parseDecl());
        }

        //FuncDef
        ArrayList<FuncDefNode> funcDefNodes = new ArrayList<>();
        while (peek(1) != TokenType.MAINTK) { // @@ [Incorrect lookahead condition: using peek(1) to detect MainFuncDef is unreliable because MAINTK might not be immediately after FuncDef, and this doesn't properly handle the case where there are no FuncDefs. Should check for FuncType tokens instead.]
            funcDefNodes.add(parseFuncDef());
        }

        //MainFuncDef
        MainFuncDefNode mainFuncDefNode = parseMainFuncDef();

        return new CompUnitNode(declNodes, funcDefNodes, mainFuncDefNode);
    }
