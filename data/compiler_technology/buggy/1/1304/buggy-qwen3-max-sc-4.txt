    //<CompUnit> ::= {<Decl>} {<FuncDef>} <MainFuncDef>
    //<Decl>:           [const] int/char <Ident>
    //<FuncDef>:        void/int/char func()
    //<MainFuncDef>:    int main()
    private CompUnitNode parseCompUnit() {
        //Decl
        ArrayList<DeclNode> declNodes = new ArrayList<>();
        while (peek(2) != TokenType.LPARENT) { // @@ [The condition peek(2) != TokenType.LPARENT is incorrect for determining Decl boundaries. It fails to properly distinguish between variable declarations and function definitions, as both can start with 'int' or 'char'. A function definition has '(' after the identifier, but checking only peek(2) doesn't account for the full structure and may cause misclassification.]
            declNodes.add(parseDecl());
        }

        //FuncDef
        ArrayList<FuncDefNode> funcDefNodes = new ArrayList<>();
        while (peek(1) != TokenType.MAINTK) { // @@ [The condition peek(1) != TokenType.MAINTK is also flawed. It assumes that after declarations, all non-main tokens belong to FuncDef, but doesn't properly validate the function definition structure. This can lead to parsing errors when encountering main function or other constructs.]
            funcDefNodes.add(parseFuncDef());
        }

        //MainFuncDef
        MainFuncDefNode mainFuncDefNode = parseMainFuncDef();

        return new CompUnitNode(declNodes, funcDefNodes, mainFuncDefNode);
    }
