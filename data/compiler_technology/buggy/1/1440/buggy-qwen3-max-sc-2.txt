    private int StmtType(){
        // 检查当前 token 是否为标识符
        if (token.getType() != tokenType.IDENFR) {
            return 0;
        }
        tokenType nextType = index2Token(1).getType();
        // 处理普通变量赋值
        if (nextType == tokenType.ASSIGN) {
            if (index2Token(2).getType() == tokenType.GETINTTK) {
                return 1;
            } else if (index2Token(2).getType() == tokenType.GETCHARTK) {
                return 2;
            } else {
                return 3;
            }
        }
        // 处理数组元素赋值
        if (nextType == tokenType.LBRACK) {
            int currentIndex = this.index;  // 保存当前索引
            token mytoken=this.token;
            match();   //匹配ident;
            match();  // 匹配 LBRACK
            parseExp();  // 解析括号内表达式 @@ [This consumes arbitrary number of tokens during lookahead without proper bounds checking, and the parser state restoration may be incomplete because parseExp() can consume multiple tokens but we only save/restore the initial index]
            if (token.getType() == tokenType.ASSIGN || index2Token(1).getType() == tokenType.ASSIGN) {
                if (index2Token(2).getType() == tokenType.GETINTTK) {
                    this.index = currentIndex;
                    this.token= mytoken;
                    return 1;
                } else if (index2Token(2).getType() == tokenType.GETCHARTK) {
                    this.index = currentIndex;
                    this.token= mytoken;
                    return 2;
                } else {
                    this.index = currentIndex;
                    this.token= mytoken;
                    return 3;
                }
            } else {
                this.index=currentIndex;
                this.token=mytoken;
                return 0;
            }
        }
        return 0;
    }
