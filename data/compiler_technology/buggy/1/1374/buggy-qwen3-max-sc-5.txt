    // CompUnit → {Decl} {FuncDef} MainFuncDef
    public static void compUnit() {
        ArrayList<Decl> declList = new ArrayList<Decl>();
        ArrayList<FuncDef> funcDefList = new ArrayList<FuncDef>();
        MainFuncDef mainFuncDef = null;
        while (!getType(curPos + 3).equals(LexerType.LPARENT)) { //decl @@ [The lookahead of 3 tokens is arbitrary and incorrect; it fails to properly distinguish between Decl and FuncDef since both can start with 'int' or 'char', and the presence of LPARENT doesn't reliably indicate the end of declarations]
            declList.add(decl());
        }
        while (!getType(curPos + 2).equals(LexerType.MAINTK)) { //fundef @@ [The lookahead of 2 tokens is also arbitrary; FuncDef starts with FuncType (void/int/char) followed by Ident, not necessarily followed by MAINTK in 2 tokens. This condition may cause infinite loops or premature termination]
            funcDefList.add(funcDef());
        }
        mainFuncDef = mainFuncDef();
        set(declList, funcDefList, mainFuncDef);
    }
