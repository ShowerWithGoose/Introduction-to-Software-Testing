    public void checkError(char code, Node node) throws IOException {
        switch (code) {
            case 'i' -> {
                if (curType == LexType.SEMICN) {
                    node.addChild(new Node(";"));
                    next();
                } else {
                    printError('i');
                }
            }
            case 'j' -> {
                if (curType == LexType.RPARENT) {
                    node.addChild(new Node(")"));
                    next();
                } else {
                    printError('j'); // @@ [For error type 'j', the line number should be the line of the last non-terminal before the expected ')', but using lastWord.getLineNum() may not capture the correct line]
                }
            }
            case 'k' -> {
                if (curType == LexType.RBRACK) {
                    node.addChild(new Node("]"));
                    next();
                } else {
                    printError('k'); // @@ [For error type 'k', the line number should be the line of the last preceding non-terminal before the expected ']', but lastWord.getLineNum() may not be accurate]
                }
            }
        }
    }
