    private SyntaxNode parseBlock() {
        SyntaxNode blockNode = new SyntaxNode("<Block>");
        match("LBRACE", blockNode);
        while (!currentToken().getType().equals("RBRACE")) { // @@ potential NullPointerException when currentToken() is null (e.g., empty token list or missing closing brace)
            blockNode.addChild(parseBlockItem());
        }
        match("RBRACE", blockNode);
        return blockNode;
    }

    private boolean isDecl() {
        if (currentToken().getType().equals("CONSTTK")) { // @@ potential NullPointerException if currentToken() is null
            if (preRead(1).getType().equals("INTTK") || preRead(1).getType().equals("CHARTK")) { // @@ potential NullPointerException if preRead returns null (index out of bounds)
                if (preRead(2).getType().equals("SEMICN")) { // @@ incorrect: for const declarations like "const int a;" (no initializer), preRead(2) is IDENFR, not SEMICN; also missing null check
                    return true;
                }
                if (preRead(2).getType().equals("IDENFR") && // @@ potential NullPointerException
                        (preRead(3).getType().equals("ASSIGN") || preRead(3).getType().equals("LBRACK") || preRead(3).getType().equals("COMMA"))) { // @@ potential NullPointerException
                    return true;
                } else {
                    return false;
                }
            } else {
                return false;
            }
        } else if (currentToken().getType().equals("INTTK") || currentToken().getType().equals("CHARTK")) { // @@ potential NullPointerException
            if (preRead(1).getType().equals("IDENFR") && !preRead(2).getType().equals("LPARENT")) { // @@ potential NullPointerException; this condition is insufficient to distinguish function definitions from variable declarations (e.g., "int f()" vs "int f;")
                return true;
            } else {
                return false;
            }
        }
        return false;
    }
