    private ParsedTree parseVarDecl() throws IOException {
        //VarDecl → BType VarDef { ',' VarDef } ';' // i
        LinkedList<ParsedTree> subTree = new LinkedList<>();
        peekAndAdd(subTree, INTTK, CONSTTK); // @@ [BType only includes 'int' and 'char', but CONSTTK ('const') is incorrectly included here, which violates the grammar rule VarDecl → BType VarDef... since 'const' is not part of BType]
        subTree.add(parseVarDef());
        while (peekToken(COMMA)) {
            subTree.add(curToken);
            subTree.add(parseVarDef());
        }
        if (peekToken(SEMICN)){
            subTree.add(curToken);
        } else {
            addError("i");
        }
        return new ParsedTree("VarDecl", subTree);
    }
