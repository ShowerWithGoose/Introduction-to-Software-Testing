    private void CompUnit() {
        // CompUnit → {Decl} {FuncDef} MainFuncDef
        if(isEOF()) { return; }
        // {Decl}
        while(true) {
            if (peekSymbol(1).getType() == SymbolType.CONSTTK) {
                Decl();
            } else if(peekSymbol(3).getType() != SymbolType.LPARENT && !isEOF()) { // @@ [Incorrect lookahead logic: using peekSymbol(3) to decide Decl parsing is flawed because it doesn't properly check if the current token starts a VarDecl. Should use FirstSet.DeclFirst instead.]
                Decl();
            } else {
                break;
            }
        }
        // {FuncDef}
        while(true) {
            if(peekSymbol(2).getType() != SymbolType.MAINTK && !isEOF()) { // @@ [Incorrect lookahead logic: using peekSymbol(2) to decide FuncDef parsing is unreliable. Should check if current token is in FuncDefFirst set and ensure we're not at main function yet.]
                FuncDef();

                break;
            }
        }
        //MainFuncDef
        MainFuncDef();
        if(shouldPrint) System.out.println("<CompUnit>");
    }
