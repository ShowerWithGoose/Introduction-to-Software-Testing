    // CompUnit -> {Decl} {FuncDef} MainFuncDef
    private CompUnitNode CompUnit() {
        List<DeclNode> declNodes = new ArrayList<>();
        List<FuncDefNode> funcDefNodes = new ArrayList<>();
        MainFuncDefNode mainFuncDefNode;
        while (lexer.tokenList.get(pos + 1).getType() != Token.LexType.MAINTK && lexer.tokenList.get(pos + 2).getType() != Token.LexType.LPARENT) { // @@ [This condition can cause IndexOutOfBoundsException when pos+2 exceeds token list size, and the logic for detecting MainFuncDef is incorrect because it doesn't check the current token but looks ahead]
            DeclNode declNode = Decl();
            declNodes.add(declNode);
        }
        while (lexer.tokenList.get(pos + 1).getType() != Token.LexType.MAINTK) { // @@ [This condition can also cause IndexOutOfBoundsException when pos+1 exceeds token list size, and it fails to properly detect the start of MainFuncDef which begins with 'int' not directly with 'main']
            FuncDefNode funcDefNode = FuncDef();
            funcDefNodes.add(funcDefNode);
        }
        mainFuncDefNode = MainFuncDef();
        return new CompUnitNode(declNodes, funcDefNodes, mainFuncDefNode);
    }
