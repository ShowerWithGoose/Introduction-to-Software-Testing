package parse;

import java.util.ArrayList;

public class TokenList {
    private ArrayList<Token> tokens;
    private int place;
    private Token now;

    public TokenList(ArrayList<Token> tokens){
        this.tokens = tokens;
        place = -1;
        now = null;
    }

    public Token readNextToken() {
        place++;
        return now = this.tokens.get(place);
    }

    public boolean hasNext() { // @@ This method can throw IndexOutOfBoundsException because it does not check if place+1 is within bounds before accessing tokens.get(place+1). It should be: return place + 1 < tokens.size();
        return (this.tokens.get(place+1) != null);
    }

    public void unReadToken(int k) {
        place -= k;
    }

    public void reset(){
        place = -1;
    }
}
