    public GrammarNode parseConstInitVal() {
        GrammarNode node = new GrammarNode("ConstInitVal");
        if (curType == LexType.LBRACE) {
            node.addChild(new GrammarNode("{"));
            getWord();

            if (curType == LexType.RBRACE) {
                node.addChild(new GrammarNode("}"));
                getWord();  
            } else {
                node.addChild(parseConstInitVal());  // @@ [Should call parseConstExp() instead of recursively calling parseConstInitVal(), as array elements in ConstInitVal must be ConstExp, not nested ConstInitVal]
                while (curType == LexType.COMMA) {
                    node.addChild(new GrammarNode(","));
                    getWord();
                    node.addChild(parseConstInitVal());  // @@ [Same error: should be parseConstExp() to match grammar rule ConstInitVal → '{' [ ConstExp { ',' ConstExp } ] '}' ]
                }
                if (curType != LexType.RBRACE) {
                    Error.syntaxError(lastWord.getLineNum(), "j");
                    return node;
                } else {
                    node.addChild(new GrammarNode("}"));
                    getWord();
                }
            }
        } else if (curType == LexType.STRCON) {
            node.addChild(new GrammarNode(curToken));
            getWord();
        } else {
            node.addChild(parseConstExp());
        }
        // printToFile("ConstInitVal");
        return node;
    }
