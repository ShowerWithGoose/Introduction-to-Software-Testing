    private Stmt parseStmt() {
        Stmt stmt = new Stmt();
        switch (currentToken().getType()) {
            case IDENFR:
                Exp tmpExp = parseExp(); // @@ [This parses the identifier as a full expression first, but if it's followed by '=', it should be treated as LVal directly. The current approach may fail when trying to extract LVal from a parsed Exp that wasn't constructed as an LVal path.]
                if (!match(Token.Type.ASSIGN)) {
                    stmt.setType(2);
                    stmt.addExp(tmpExp); // Exp
                    if (match(Token.Type.SEMICN)) {
                        stmt.addSemicn(currentToken()); // ';'
                        advance();
                    } else {
                        peekToken(-1).setErrorCode('i');
                    }
                    break;
                }
                stmt.addLVal(tmpExp.getAddExp().getMulExp().getUnaryExp().getPrimaryExp().getLVal()); // @@ [This chain of getters assumes the Exp was parsed as an LVal through PrimaryExp, but parseExp() may have parsed it as a different expression type, leading to null pointer or incorrect structure.]
                stmt.addAssign(currentToken()); // '='
                advance();
                if (match(Token.Type.GETINTTK)) {
                    stmt.setType(9);
                    stmt.addToken(currentToken()); // 'getint'
                    advance();
                    stmt.addLParent(currentToken()); // '('
                    advance();
                    if (match(Token.Type.RPARENT)) {
                        stmt.addRParent(currentToken()); // ')'
                        advance();
                    } else {
                        peekToken(-1).setErrorCode('j');
                    }
                } else if (match(Token.Type.GETCHARTK)) {
                    stmt.setType(10);
                    stmt.addToken(currentToken()); // 'getchar'
                    advance();
                    stmt.addLParent(currentToken()); // '('
                    advance();
                    if (match(Token.Type.RPARENT)) {
                        stmt.addRParent(currentToken()); // ')'
                        advance();
                    } else {
                        peekToken(-1).setErrorCode('j');
                    }
                } else {
                    stmt.setType(1);
                    stmt.addExp(parseExp()); // Exp
                }
                if (match(Token.Type.SEMICN)) {
                    stmt.addSemicn(currentToken()); // ';'
                    advance();
                } else {
                    peekToken(-1).setErrorCode('i');
                }
                break;
