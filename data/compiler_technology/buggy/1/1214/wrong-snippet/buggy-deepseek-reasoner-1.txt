    public AST.AddExp parseAddExp() {

        ArrayList<String> ops = new ArrayList<>();

        writeType("<AddExp>"); // Error: should be called at the beginning of the method
        while (tokens.get(index).tokenType == TokenType.PLUS || tokens.get(index).tokenType ==
                TokenType.MINU) {

            writeToken(1);
            mulExps.add(parseMulExp());
            writeType("<AddExp>"); // Error: also called after each iteration instead of at start
        }
        AST.AddExp addExp = new AST.AddExp(mulExps, ops);
