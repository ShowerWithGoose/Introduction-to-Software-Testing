    // 解析常量声明
    private void parseConstDecl() throws IOException {
        // ConstDecl → 'const' BType ConstDef { ',' ConstDef } ';' // i
        outputToken(currentToken); // const
        nextToken();
        int currentLine = getPreviousTokenLineNumber(); // @@ [This captures the line number of 'const', but according to the spec, error 'i' should report the line of the last non-terminal before the semicolon, which would be the last ConstDef]
        parseBType();
        parseConstDef();
        while (currentToken.getType() == TokenType.COMMA) {
            outputToken(currentToken);
            nextToken();
            parseConstDef();
        }
        if (currentToken.getType() == TokenType.SEMICN) {
            outputToken(currentToken);
            nextToken();
        } else {
            addError(currentLine, 'i'); // 缺少分号
        }
        outputGrammar("<ConstDecl>");
    }

    private void parseVarDecl() throws IOException {
        // VarDecl → BType VarDef { ',' VarDef } ';' // i
        parseBType();
        int currentLine = getPreviousTokenLineNumber(); // @@ [This captures the line number of the BType token, but according to the spec, error 'i' should report the line of the last non-terminal before the semicolon, which would be the last VarDef]
        parseVarDef();
        while (currentToken.getType() == TokenType.COMMA) {
            outputToken(currentToken);
            nextToken();
            parseVarDef();
        }
        if (currentToken.getType() == TokenType.SEMICN) {
            outputToken(currentToken);
            nextToken();
        } else {
            addError(currentLine, 'i'); // 缺少分号
        }
        // 需要输出 <VarDecl>
        outputGrammar("<VarDecl>");
    }
