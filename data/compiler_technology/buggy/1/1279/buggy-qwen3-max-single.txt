    private void parseExpressionStmt(int currentLine) throws IOException {
        // 无法预读下一个 Token，按表达式语句处理
        // 其他情况，可能是表达式语句
        if (currentToken.getType() != TokenType.SEMICN) {
            parseExp();
        }
        if (currentToken.getType() == TokenType.SEMICN) {
            outputToken(currentToken);
            nextToken();
        } else {
            addError(currentLine, 'i'); // 缺少 ';' @@ [The error line number should correspond to the line where the last non-terminal before the semicolon is located, but currentLine may not be accurate if the expression spans multiple lines]
        }
        outputGrammar("<Stmt>");
    }

    private void parseAssignmentStmt(int currentLine) throws IOException {
        parseLVal(); // 解析左值
        if (currentToken.getType() == TokenType.ASSIGN) {
            outputToken(currentToken); // 输出 '='
            nextToken();

            if (currentToken.getType() == TokenType.GETCHARTK || currentToken.getType() == TokenType.GETINTTK) {
                parseGetExp(); // 解析 getint/getchar 表达式
            } else {
                parseExp(); // 解析表达式
            }

            if (currentToken.getType() == TokenType.SEMICN) {
                outputToken(currentToken); // 输出 ';'
                nextToken();
            } else {
                addError(currentLine, 'i'); // 缺少 ';' @@ [The error line number should be the line of the last token in the expression, not the initial token line. Using currentLine (from the beginning of statement) is incorrect when expression spans multiple lines]
            }
        }
        outputGrammar("<Stmt>");
    }
