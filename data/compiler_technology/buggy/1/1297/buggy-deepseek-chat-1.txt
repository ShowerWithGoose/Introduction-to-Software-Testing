    public static void nodeOut(TokenStream tokenStream,Node node) {

            writer.write(node.toString());
        } catch (IOException e) {
            e.printStackTrace();
        }
        try (BufferedWriter writer = new BufferedWriter(new FileWriter(errorfilePath))){
            String error = tokenStream.toStringWithErr() + Paser.error; // @@ The error string is constructed by concatenating tokenStream.toStringWithErr() and Paser.error, but if both are non-empty, there is no newline separator between them, which may cause the line number and error code to be incorrectly concatenated on the same line.
            writer.write(sortByLineNumber(error));
        } catch (IOException e) {

    }
