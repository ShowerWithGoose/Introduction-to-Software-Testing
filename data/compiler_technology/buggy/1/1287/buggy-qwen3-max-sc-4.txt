    public Node CompUnit() {
        Element root = new Element(ParseType.CompUnit);
        while (pos < tokens.size() && (nowRead().equals(LexType.CONSTTK) ||
                nowRead().equals(LexType.VOIDTK) || isBType(nowRead()))) {
            if (nowRead().equals(LexType.CONSTTK)) {
                root.addNode(ConstDecl());
            }
            else {
                if (preRead(1).equals(LexType.MAINTK)) { @@ [This check assumes there is at least one more token after current position, but may cause IndexOutOfBoundsException if pos+1 >= tokens.size(); also, 'int main' must appear together as MainFuncDef, but current logic doesn't ensure the current token is INTTK]
                    root.addNode(MainFuncDef());
                }
                else if (preRead(1).equals(LexType.IDENFR)) { @@ [Similarly, this uses preRead(1) without checking bounds; also, VarDecl starts with BType followed by IDENFR not in parentheses, but FuncDef has '(' after IDENFR. However, peeking ahead like this without proper boundary checks is unsafe and logically incomplete]
                    if (preRead(2).equals(LexType.LPARENT)) { @@ [preRead(2) can easily go out of bounds when fewer than 3 tokens remain; also, this approach fails to correctly distinguish VarDecl from FuncDef in all cases because it doesn't consume or validate structure properly]
                        root.addNode(FuncDef());
                    }
                    else {
                        root.addNode(VarDecl());
                    }
                }
                else {
                    error();
                }
            }
        }
        return root;
    }
