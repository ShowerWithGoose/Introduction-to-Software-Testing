    public StmtLValjudgeType StmtLValjuage() {
        int i = 1;
        int tempRow = WordList.get(Syntaxcur + 1).getRow();
        while (preRead(i) != WordType.ASSIGN
                && tempRow == WordList.get(Syntaxcur + i).getRow()) { // @@ Error: If preRead(i) returns null (end of tokens), calling .getRow() on null will cause NullPointerException. Also, if the '=' is not found on same line, the loop may attempt to access index out of bounds leading to IndexOutOfBoundsException.
            i++;
        }
        if (preRead(i + 1) == WordType.GETCHARTK) { // @@ Error: preRead(i+1) may also return null if out of bounds; comparing null to enum is safe (false) but subsequent logic relies on correct return.
            return StmtLValjudgeType.GETCHAR;
        } else if (preRead(i + 1) == WordType.GETINTTK) {
            return StmtLValjudgeType.GETINT;
        } else {
            return StmtLValjudgeType.EQ;
        }
    }
