    public void parseCompUnit() {
        curpos = 0;
        curToken = tokens.get(0);    // now in 1
        while (readAfterToken(1).getLexType() != LexType.MAINTK) { // @@ [This will cause NullPointerException if there's no MAINTK or if token list is too short; also fails to handle empty CompUnit correctly]
            Token thirdtoken = readAfterToken(2); // get 3
            if (thirdtoken.getLexType() == LexType.LPARENT) { // @@ [thirdtoken could be null, leading to NullPointerException]
                parseFuncDef();
            } else {
                parseDecl();
            }
        }
        curToken = nextToken(); // skip int
        curToken = nextToken(); // skip main
        curToken = nextToken(); // skip (
        if (curToken.getLexType() != LexType.RPARENT) {
            dealError('j');
        } else {
            curToken = nextToken(); // skip )
        }
        parseBlock();
        outputType("<MainFuncDef>");
        outputType("<CompUnit>");
        output2File();
    }

    private void parseConstDef() {
        curToken = nextToken(); // skip Ident
        if (curToken.getLexType() == LexType.LBRACK) { // if now is [
            curToken = nextToken(); // skip [
            parseConstExp();
            if (curToken.getLexType() != LexType.RBRACK) {
                dealError('k');
            } else {
                curToken = nextToken(); // skip [ @@ [Should skip ']' not '[', this is a typo that causes incorrect token consumption and potential parsing errors]
            }
        }
        curToken = nextToken(); //skip =,jump in InitiVal
        parseConstInitVal();
        outputType("<ConstDef>");
    }
