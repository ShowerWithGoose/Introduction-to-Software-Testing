    public Token getNextToken() {
        pos++;
        if (pos < tokens.size()) {
            nowToken = tokens.get(pos);
            return nowToken;
        }
        while (true) {
            Token token = lexer.nextToken();
            nowToken = token;
            tokens.add(token);
            break;
        }
        return nowToken;
    }
