        while (curToken != null) {
            Node child = null;
            if (peekOffset(1).getLexType() == LexType.MAINTK) {   // @@ potential NullPointerException: peekOffset(1) may return null if curTokenIndex is near the end of tokens list, causing .getLexType() to be called on null.

                child = parseFuncDef();
            } else if (curToken.getLexType() == LexType.CONSTTK) {
                child = parseConstDecl();
            } else if (isBType(curToken.getLexType())) {
                child = parseVarDecl();
