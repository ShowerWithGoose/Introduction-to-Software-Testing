    public Parser(ArrayList<Token> tokens) {
        this.tokens = tokens;
        this.pos = 0;
        this.compUnit = new CompUnit();
        while (this.pos < tokens.size()) {
            if (this.tokens.get(this.pos).getType() == TokenType.CONST) {
                this.compUnit.addDecl(parseConstDecl());
            } else if (this.tokens.get(this.pos + 1).getType() == TokenType.MAIN) { // @@ [Accessing pos+1 without checking bounds; may cause IndexOutOfBoundsException when pos is the last index]
                this.compUnit.setMainFuncDef(parseMainFuncDef());
            } else if (this.tokens.get(this.pos + 2).getType() == TokenType.LPARENT) { // @@ [Accessing pos+2 without checking bounds; may cause IndexOutOfBoundsException when pos is near the end of token list]
                this.compUnit.addFuncDef(parseFuncDef());
            } else {
                this.compUnit.addDecl(parseVarDecl());
            }
        }
    }
