node *multiplypolynomials(node *poly1, node *poly2) {
    node *result = NULL;

    for (node *p1 = poly1; p1 != NULL; p1 = p1->next) {
        for (node *p2 = poly2; p2 != NULL; p2 = p2->next) {

            int exp = p1->exponent + p2->exponent;
