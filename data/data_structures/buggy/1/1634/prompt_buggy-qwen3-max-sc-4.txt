void permutation(int *number, int *is_done, int n, int depth)
{
    int i;
    if (depth == n + 1)
    {
        for (i = 1; i <= n; i++)
        {
            printf("%d ", number[i]);
        }
        printf("\n");
        return;
    }
    for (i = 1; i <= n; i++)
    {
        if (is_done[i] == 0)
        {
            is_done[i] = 1;
            number[depth] = i;
            permutation(number, is_done, n, depth + 1);
