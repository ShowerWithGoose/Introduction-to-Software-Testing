Status InitQueue(Queue *Q){
    Q->front=Q->rear=(QueuePtr)malloc(sizeof(QNode));
    Q->length=0;
    return OK;
}

Status DestroyQueue(Queue *Q){
    QueuePtr p=Q->front;
    while(p){
        QueuePtr q=p;
        p=p->next;
        free(q);
    }
    Q->front=Q->rear=NULL;
    Q->length=0;
    return OK;
}

Status ClearQueue(Queue *Q){
    QueuePtr p=Q->front;
    while(p->next){
        QueuePtr q=p;
        p=p->next;
        free(q);
    }
    p->data=0;
    p->next=NULL;
    Q->front=Q->rear=p;
    Q->length=0;
    return OK;
}

Bool QueueEmpty(Queue Q){
    return Q.length==0;
}

int QueueLength(Queue Q){
    return Q.length;
}




Status EnQueue(Queue *Q,QueueElem e){
    QueuePtr p=(QueuePtr)malloc(sizeof(QNode));
    p->data=e;
    p->next=NULL;
    Q->rear->next=p;
    Q->rear=p;
    Q->length++;
    return OK;
}

Status DeQueue(Queue *Q,QueueElem *e){
    QueuePtr q=Q->front;
    Q->front=Q->front->next;
    (*e)=q->next->data;
    free(q);
    Q->length--;
    return OK;
}
