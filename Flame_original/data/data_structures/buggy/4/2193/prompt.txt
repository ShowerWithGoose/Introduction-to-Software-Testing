int getword(FILE *fp, char *w)
{
    char c;
    int num = 0;
    while( (c=fgetc(fp))!=EOF ) {
        if( !isalpha(c) && num==0 ) continue;
        else if( !isalpha(c) ) break;
        else {
            w[num++] = tolower(c);
        }
    }
    w[num] = '\0';
    if( num > 0 ) return 1;
    else return EOF;
