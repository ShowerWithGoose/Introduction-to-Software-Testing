typedef struct node BTNode,*BTNodeptr;

{
    if(p==NULL)//结点为空 
    {
        p=(BTNodeptr)malloc(sizeof(BTNode));
        p->data=item;
        p->left=p->right=NULL;
        p->depth=depth;
    }
    else if(item<p->data)//小于结点

        p->left = insertBST(p->left,item,depth+1);
    }
    else if(item>=p->data)//大于等于节点

}
