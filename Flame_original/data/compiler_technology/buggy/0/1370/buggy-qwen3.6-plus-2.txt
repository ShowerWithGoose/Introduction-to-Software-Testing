    public Character getAChar(){
        Character ans;
        if(cc == ll){
            try {
                line = br.readLine();
                if(line == null){
                    br.close();
                    return null;
                }
                ll = line.length();
                cc = 0;
                line = line+'\n';
                ll++;
                //System.out.println("GetChar Error:the last char of line is not '\n' \n");

            } catch (IOException e) {
                e.printStackTrace();
            }
        }
        ans = line.charAt(cc);
        cc++;
        if(ans!=null){ // @@ [The condition is always true because line.charAt(cc) returns a primitive char which is autoboxed to a non-null Character object, making the subsequent return null unreachable and breaking EOF handling.]
            return ans;
        }
        return null;
    }
