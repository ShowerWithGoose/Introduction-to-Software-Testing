char Lexer::getChar(std::string &str) {
    char result;
    std::stringstream ss;
    ss << '\'';
    if (cur_char == '\\') {
        ss << '\\';
        getNext();
        // 转义
        switch (cur_char) {
            case 'a':
                // 007
                result = 7;
                break;
            case 'b':
                // 008
                result =  8; // @@ Missing break statement causes fall-through execution, incorrectly overwriting the result value
            case 't':
                // 009
                result = 9; // @@ Missing break statement causes fall-through execution, incorrectly overwriting the result value
            case 'n':
                // 010
                result = 10; // @@ Missing break statement causes fall-through execution, incorrectly overwriting the result value
            case 'v':
                // 011
                result = 11; // @@ Missing break statement causes fall-through execution, incorrectly overwriting the result value
            case 'f':
                // 012
                result = 12; // @@ Missing break statement causes fall-through execution, incorrectly overwriting the result value
            case '\"':
                // 034
                result = 34; // @@ Missing break statement causes fall-through execution, incorrectly overwriting the result value
            case '\'':
                // 039
                result = 39; // @@ Missing break statement causes fall-through execution, incorrectly overwriting the result value
            case '\\':
                // 092
                result = 92; // @@ Missing break statement causes fall-through execution, incorrectly overwriting the result value
            case '0':
                // 000
                result = 0; // @@ Missing break statement causes fall-through execution, incorrectly overwriting the result value
            default:
                result = -1;
        }
        ss << cur_char;
    } else {
        // what ever
        result = cur_char;
        ss << cur_char;
    }
    getNext();
    ss << '\'';
    str = ss.str();
    return result;
}
