    private int getChar() throws IOException {
        // 检查当前字符是否有效，如果无效则读取新字符
        if (currentChar != -1) {
            int temp = currentChar; // 保存当前字符
            currentChar = -1; // 重置当前字符
            return temp; // 返回之前保存的字符
        }

        // 读取下一个字符并更新行号和字符位置
        int c = reader.read();
        // System.out.println(c+" "+(char)c);
        updatePosition(c);

        return c; // 返回读取的字符
    }
