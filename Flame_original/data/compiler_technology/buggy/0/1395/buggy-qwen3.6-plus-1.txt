    public ArrayList<Token> lex() {
        while (pos < sourceCode.length()) {
            Token token = getToken();
            if (token != null) {
                pos += token.getContent().length();
                line += token.getContent().split("\n").length - 1; // @@ Java's String.split() discards trailing empty strings. When a token consists solely of a newline "\n", split returns an empty array of length 0, causing the line counter to decrement by 1 instead of incrementing. This results in incorrect line numbers for all subsequent tokens and error reports.
                if (token.getType() != TokenType.BLANK &&
                        token.getType() != TokenType.LINECOMMENT &&
                        token.getType() != TokenType.BLOCKCOMMENT) {
                    tokens.add(token);
                }
            } else {  // 这儿关于出错的情况，我只考虑了“&”和“|”这两种（毕竟评测说了就这两种错误）
                pos++;
            }
        }
        return tokens;
    }
