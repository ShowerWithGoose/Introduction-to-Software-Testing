package frontend.tokens;

public class BlockComment extends Token {
    // constructor
    public BlockComment(String comment, int line) {
        super(comment, line);
        type = TokenType.BLOCKCOMMENT;
    }

    public static int getBlockComment(String sourceCode, int pos) {
        // return the position of the end of the comment
        for (int i = pos + 2; i < sourceCode.length(); i++) // @@ [Potential index out-of-bounds: if sourceCode length is less than pos+2, loop condition may be invalid; also, access to i+1 below can exceed length]
            if (sourceCode.charAt(i) == '*' && sourceCode.charAt(i + 1) == '/')
                return i + 2;
        return -1; // @@ [Returns -1 if no closing "*/", leading to substring error in Lexer.getToken]
    }
}
