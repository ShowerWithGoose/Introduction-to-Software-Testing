    /* 获取下一个字符，更新curPos, lineNum, colNum */
    private char readNextChar() {
        if (curPos < source.length()) { // 成功读取
            char ch = source.charAt(curPos++);
            if (ch == '\n') {
                lineNum++;
                colNum = 1; // @@ [Should be colNum = 0, because the else branch will increment colNum++ for the next character read, making the first character of a new line start at colNum=2 instead of 1]
            } else {
                colNum++;
            }
            return ch;
        } else {  // source已被读完
            return '\0';
        }
    }
