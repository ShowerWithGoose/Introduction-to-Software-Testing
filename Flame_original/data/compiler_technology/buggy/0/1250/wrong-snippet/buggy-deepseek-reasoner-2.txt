    public void toTokens() {
        while (charIndex < sourceCode.length()) {
            if (skipSpace()) {
                continue;
            }
            if (skipComment()) {
                continue;
            }
            if (getIdentifierOrReservedWord()) {
                continue;
            }
            if (getIntConst()) {
                continue;
            }
            if (getCharConst()) {
                continue;
            }
            if (getStringConst()) {
                continue;
            }
            if (getOperator()) {
                continue;
            }
            if (getSeparator()) {
                continue;
            }
        }
    }
