    public Token nextToken() {
        skipWhiteSpace();
        while (skipComment()) {
            skipWhiteSpace();
        }
        if (isEndOfInput()) {
            return null;
        }
        String currentWord = source.readCurrentWord(); // @@ Consumes the entire contiguous non-whitespace sequence, but the source pointer is not rolled back for the unmatched suffix, causing subsequent tokens to be skipped or misaligned.
        int currentLine = source.getLineNum(); // @@ Returns 0-based index, but problem requires 1-based line numbers for error output.
        for (TokenType tokentype : TokenType.values()) {
            Pattern pattern = tokentype.getPattern();
            Matcher matcher = pattern.matcher(currentWord);

            if (matcher.find() && matcher.start() == 0) {
                String value = matcher.group(); // 会不会出现value和Word实际长度不一样的情况？
                return new Token(tokentype, value, currentLine);
            }
        }
        return null;
    }
