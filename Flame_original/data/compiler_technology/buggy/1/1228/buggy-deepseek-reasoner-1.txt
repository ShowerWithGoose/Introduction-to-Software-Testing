    public CompUnit parseCompUnit() throws IOException {
        ArrayList<Decl> decls = new ArrayList<>();
        ArrayList<FuncDef> funcDefs = new ArrayList<>();
        ArrayList<LexType> lexTypes = lexer.preRead(2);
        while (lexTypes.get(0) != LexType.MAINTK) { // @@ Error: The first token of MainFuncDef is 'int' (INTTK), not MAINTK. This condition will never be true for a correct main function, causing the parser to incorrectly attempt to parse it as a Decl and eventually fail.
            if (lexTypes.get(1) == LexType.LPARENT) {
                FuncDef funcDef = parseFuncDef();
                funcDefs.add(funcDef);
            } else {
                Decl decl = parseDecl();
                decls.add(decl);
            }
            lexTypes = lexer.preRead(2);
        }
        MainFuncDef mainFuncDef = parseMainFuncDef();
        CompUnit compUnit = new CompUnit(decls, funcDefs, mainFuncDef);
        outputP("CompUnit");
        return compUnit;
    }
