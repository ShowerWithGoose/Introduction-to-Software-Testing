    private void handleError(ErrorType type, Token token) {
        if (type == i) {
            if (next(0).getType() != SEMICN) { // @@ [Potential IndexOutOfBoundsException if current is at end of tokens]
                Error error = new Error(i, token.getLine());
                errors.add(error);
            } else {
                move();
            }
        } else if (type == j) {
            if (next(0).getType() != RPARENT) { // @@ [Same issue: next(0) may be out of bounds]
                Error error = new Error(j, token.getLine());
                errors.add(error);
            } else {
                move();
            }
        } else if (type == k) {
            if (next(0).getType() != RBRACK) { // @@ [Same issue]
                Error error = new Error(k, token.getLine());
                errors.add(error);
            } else {
                move();
            }
        }
    }

    private Node funcDef() {
        // FuncDef → FuncType Ident '(' [FuncFParams] ')' Block // j
        Node node = new Node(FuncDef);
        Node funcTypeNode = funcType();
        node.addChild(funcTypeNode);
        Node identNode = new Node(next(0));
        node.addChild(identNode);
        move();
        Node lparentNode = new Node(next(0));
        node.addChild(lparentNode);
        move();
        if (next(0).getType() != RPARENT) {
            Node funcFParamsNode = funcFParams();
            node.addChild(funcFParamsNode);
        }
        Node rparenNode = new Node(new Token(RPARENT, ")", next(-1).getLine())); // @@ [Incorrect line number: next(-1) refers to last token of funcFParams, not the actual RPARENT]
        node.addChild(rparenNode);
        handleError(j, next(-1)); // @@ [Passes token with possibly wrong line; also handleError may double-move if RPARENT exists]
        Node blockNode = block();
        node.addChild(blockNode);
        return node;
    }
