public Unit.CompUnit run() {
    ArrayList<Unit.Decl> decls = new ArrayList<>();
    ArrayList<Unit.FuncDef> funcDefs = new ArrayList<>();
    while (tokenList.hasNext()) {
        if (tokenList.ahead(2).getType().equals(TokenType.LPARENT)) { // @@ [Incorrect lookahead logic: using ahead(2) to detect function definitions is unreliable and doesn't follow the grammar. FuncDef starts with FuncType (INTTK/CHARTK/VOIDTK) followed by IDENFR and LPARENT, but ahead(2) may go out of bounds or misidentify tokens.]
            break;
        } else {
            decls.add(parseDecl());
        }
    }
    while (tokenList.hasNext()) {
        if (tokenList.ahead(1).getType().equals(TokenType.MAINTK)) { // @@ [Incorrect lookahead logic: using ahead(1) to detect main function is fragile. MainFuncDef specifically starts with "int main()", but this check only looks at the next token being MAINTK, which could be preceded by other tokens incorrectly. Also, it should directly look for INTTK followed by MAINTK.]
            break;
        } else {
            funcDefs.add(parseFuncDef());
        }
    }
    Unit.MainFuncDef mainFuncDef = parseMainFuncDef();
    printSyntax("CompUnit");
    return new Unit.CompUnit(decls, funcDefs, mainFuncDef);
}
