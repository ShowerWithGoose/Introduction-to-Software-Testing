    public Node parseStmt() throws IOException {
        switch (curToken.getType()) {
            case LBRACE:
                return parseBlockStmt();
            case IFTK:
                return parseIfStmt();
            case FORTK:
                return parseForStmt();
            case BREAKTK:
                return parseBreakStmt();
            case CONTINUETK:
                return parseContinueStmt();
            case RETURNTK:
                return parseReturnStmt();
            case PRINTFTK:
                return parsePrintfStmt();
            case SEMICN://[Exp]; 解决单个;作为语句的情况
                return parseExpStmt();
        }

        tokenStream.setBackPoint();//将当前 pos 的值保存到 watchPoint 中
        Printer.getInstance().turnOff();
        parseExp(); // @@ [This call to parseExp() advances the token stream and consumes tokens, but the result is discarded. The subsequent logic uses backToBackPoint() to restore the position, but the turnOff()/turnOn() around this call suppresses output, which may cause incorrect output order or missing syntax component names. Additionally, the parseExp() call here is used purely for lookahead, which is inefficient and error-prone.]
        Printer.getInstance().turnOn();

        if (curToken.getType() == TokenType.ASSIGN) {
            if (tokenStream.detect(1).getType() == TokenType.GETINTTK) {
                curToken = tokenStream.backToBackPoint();
                return parseGetIntStmt();
            } else if (tokenStream.detect(1).getType() == TokenType.GETCHARTK) {
                curToken = tokenStream.backToBackPoint();
                return parseGetCharStmt();
            } else {
                curToken = tokenStream.backToBackPoint();
                return parseAssignStmt();
            }
        } else {
            curToken = tokenStream.backToBackPoint();
            return parseExpStmt();
        }
    }
