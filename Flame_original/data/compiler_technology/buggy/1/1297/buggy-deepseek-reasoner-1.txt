public Node parseUnitNode() {
    ArrayList<Node> children = new ArrayList<>();
    Node node = null;
    int startLine = tokenStream.get(p).getLineNumber();
    while (true) {
        if (p >= tokenStream.size() || tokenStream.get(p) == null) {
            break;
        } else if (tokenStream.get(p + 1).getType() == Token.TokenType.MAINTK) { // @@ Error: Accesses p+1 without checking bounds; also, condition is insufficient because it does not verify that the current token is 'int'. The grammar requires 'int main' at the start, so should check tokenStream.get(p).getType() == INTTK and tokenStream.get(p+1).getType() == MAINTK.
            node = parseMainFuncDef();
        }
        // parse FuncDef
        else if (tokenStream.get(p + 2).getType() == Token.TokenType.LPARENT) { // @@ Error: Accesses p+2 without bounds check; also, this lookahead is fragile and can misidentify constructs. For FuncDef, the first token is a type (int/char/void) followed by an identifier and then '('; a safer approach is to check the current token(s) appropriately.
            node = parseFuncDef();
        }
        // parse ConstDecl
        else if (tokenStream.get(p).getType() == Token.TokenType.CONSTTK) {
            node = parseConstDecl();
        }
        // parse VarDecl
        else if (tokenStream.get(p).getType() == Token.TokenType.INTTK || tokenStream.get(p).getType() == Token.TokenType.CHARTK) {
            node = parseVarDecl();
        } else {
            System.err.println("Syntax Error when parse Unit: " + tokenStream.get(p).getLineNumber() + " " + tokenStream.get(p).getDescription());
            break;
        }
        children.add(node);
    }
    int endLine = tokenStream.get(p - 1).getLineNumber();
    return NodeFactory.createNode(startLine, endLine, SyntaxVarType.COMP_UNIT, children);
}
