    private void errorI(SyntaxNode node) {
        if (!match("SEMICN", node)) {
            int lineNum = tokens.get(currentTokenIndex - 1).getLineNum(); // @@ [When a semicolon is missing, currentTokenIndex points to the next token after the expected semicolon position, but currentTokenIndex - 1 may not be the correct token that should precede the semicolon. The line number should come from the last parsed non-terminal's actual line, not from the token stream index arithmetic.]
            ErrorToken errorToken = new ErrorToken(lineNum, "i");
            errorTokens.add(errorToken);
        }
    }
