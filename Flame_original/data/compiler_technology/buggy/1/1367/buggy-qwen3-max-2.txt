    private void errorI(SyntaxNode node) {
        if (!match("SEMICN", node)) {
            int lineNum = tokens.get(currentTokenIndex - 1).getLineNum(); // @@ [When SEMICN is missing, currentTokenIndex hasn't advanced past the last valid token, so currentTokenIndex - 1 may not point to the correct previous token. The error line should be based on the last successfully parsed token in the current syntax component, not currentTokenIndex - 1.]
            ErrorToken errorToken = new ErrorToken(lineNum, "i");
            errorTokens.add(errorToken);
        }
    }

    private void errorJ(SyntaxNode node) {
        if (!match("RPARENT", node)) {
            int lineNum = tokens.get(currentTokenIndex - 1).getLineNum(); // @@ [When RPARENT is missing, currentTokenIndex points to the unexpected token after the expected RPARENT, so currentTokenIndex - 1 may not correspond to the last token of the expression inside parentheses. The error line should be based on the last token that was successfully parsed within the parentheses context.]
            ErrorToken errorToken = new ErrorToken(lineNum, "j");
            errorTokens.add(errorToken);
        }
    }

    private void errorK(SyntaxNode node) {
        if (!match("RBRACK", node)) {
            int lineNum = tokens.get(currentTokenIndex - 1).getLineNum(); // @@ [When RBRACK is missing, currentTokenIndex points to the token following where RBRACK should be, so currentTokenIndex - 1 may not be the correct line for the error. The error line should correspond to the last token inside the brackets or the LBRACK token itself.]
            ErrorToken errorToken = new ErrorToken(lineNum, "k");
            errorTokens.add(errorToken);
        }
