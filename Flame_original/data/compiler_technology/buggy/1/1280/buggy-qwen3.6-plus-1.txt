    private void UnaryExp() {
        // UnaryExp → PrimaryExp | Ident '(' [FuncRParams] ')' | UnaryOp UnaryExp
        updateNumber();
        Type type = words.get(wordPoint).getType();
        if (type == Type.PLUS
                || type == Type.MINU
                || type == Type.NOT) {
            UnaryOp();
            UnaryExp();
        } else if (type == Type.IDENFR
                && words.get(wordPoint + 1).getType() == Type.LPARENT) { // @@ Unsafe lookahead without bounds checking, which may throw IndexOutOfBoundsException if the identifier is the last token in the input.
            //Ident
            addOut();
            //'('
            addOut();
            while (words.get(wordPoint).getType() != Type.RPARENT) { // @@ Incorrectly uses a while loop for the optional [FuncRParams]. Since FuncRParams already handles the comma-separated list internally, this should be an if statement. The while loop causes parsing failures on empty parameter lists (e.g., func()) and may lead to infinite loops or crashes.
                FuncRParams();
            }
            //')'
            if (words.get(wordPoint).getType() == Type.RPARENT) {
                addOut();
            } else {
                addError('j');
            }
        } else {
            PrimaryExp();
        }

        outStr.add("<UnaryExp>");
    }
