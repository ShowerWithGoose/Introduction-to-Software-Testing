    public UnaryExp parseUnaryExp() {
        if (Arrays.asList(UnaryOp.FIRST).contains(lexer.getLexType())) {
            // UnaryOp UnaryExp
            UnaryOp unaryOp = parseUnaryOp();
            UnaryExp unaryExp = parseUnaryExp();
            return new UnaryExp(unaryOp, unaryExp);
        } else if (lexer.getLexType() != LexType.IDENFR) {
            // PrimaryExp
            PrimaryExp primaryExp = parsePrimaryExp();
            return new UnaryExp(primaryExp);
        } else {
            // PrimaryExp | Ident '(' [FuncRParams] ')'
            String name = lexer.getToken();
            lexer.nextNoPrint(); // @@ Using nextNoPrint() for lookahead modifies the lexer's internal state (pos, lineno, history). Relying on prev() to rewind is fragile and can cause state inconsistencies or incorrect token printing order, especially when combined with Logger outputs.
            if (lexer.getLexType() == LexType.LPARENT) {
                lexer.prev();
                lexer.next();
                lexer.next(); // LPARENT // @@ The sequence of prev() followed by two next() calls to re-consume and print tokens is convoluted. It attempts to manually manage printing order but risks skipping tokens or misaligning the lexer state if prev() doesn't perfectly restore all necessary fields or if next() prints the previously consumed token instead of the current one.
                // TODO 真的是这样吗？
                // Ident '(' [FuncRParams] ')'
                FuncRParams funcRParams = null;
                if (Arrays.asList(FuncRParams.FIRST).contains(lexer.getLexType())) {
                    funcRParams = parseFuncRParams();
                }
                if (lexer.getLexType() == LexType.RPARENT) {
                    lexer.next(); // RPARENT
                } else {
                    Logger.logBad(lexer.getPrevLineno() + " j\n");
                }
                return new UnaryExp(name, funcRParams);
            } else {
                // PrimaryExp
                lexer.prev();
                PrimaryExp primaryExp = parsePrimaryExp();
