    private void parseCompUnit() {
        while (currentToken != null) { // @@ Potential infinite loop if no condition matches
            if(currentToken.getType() == Type.INTTK && checkNext().getType() == Type.MAINTK){ // @@ checkNext() may return null, causing NullPointerException
                ASTNode main = parseMain();
                compUnitNode.addChild(main);
            }else if (judge.isFuncDef(currentToken,checkNext(),checkNext2())) { // @@ checkNext()/checkNext2() may return null
                ASTNode funcDefNode = parseFuncDef();
                compUnitNode.addChild(funcDefNode);
            } else if (judge.isDecl(currentToken)) {
                ASTNode declNode = parseDecl();
                compUnitNode.addChild(declNode);
            } // @@ No else clause: if none match, loop runs forever without consuming currentToken
        }
    }
