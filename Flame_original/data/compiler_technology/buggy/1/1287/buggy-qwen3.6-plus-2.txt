    public Node RETURN() {
        Element ret = new Element(ParseType.Stmt);
        if (!nowRead().equals(LexType.RETURNTK)) {
            error();
        }
        else {
            ret.addNode(tokens.get(pos++));
            int tmp = pos;
            if (isExpFirst(nowRead())) {
                int top1 = errors.size();
                Exp();
                int top2 = errors.size();
                if (top2 > top1) {
                    errors.subList(top1, top2).clear();
                }
                if (nowRead().equals(LexType.ASSIGN)) {
                    errors.add(new Error(tokens.get(tmp - 1).getLineNum(), "i"));
                    ret.addNode(new Token(LexType.SEMICN, ";", tokens.get(tmp - 1).getLineNum()));
                    pos = tmp; // @@ Resets parser position after successfully parsing Exp(), discarding the parsed expression node from the syntax tree and causing missing output.
                    return ret;
                }
                else if (!nowRead().equals(LexType.SEMICN)) {
                    pos = tmp; // @@ Unnecessarily resets parser position, causing the expression to be parsed twice and leading to redundant processing and inconsistent parser state.
                    ret.addNode(Exp()); // @@ Re-parses the expression instead of adding the already parsed Exp node from the initial call, which should have been stored and added to ret.
                    errors.add(new Error(getPreLineNum(), "i"));
                    ret.addNode(new Token(LexType.SEMICN, ";", getPreLineNum()));
                    return ret;
                }
                else {
                    ret.addNode(tokens.get(pos++));
                    return ret;
                }
            }
            else if (!nowRead().equals(LexType.SEMICN)) {
                errors.add(new Error(getPreLineNum(), "i"));
                ret.addNode(new Token(LexType.SEMICN, ";", getPreLineNum()));
                return ret;
            }
            else {
                ret.addNode(tokens.get(pos++));
                return ret;
            }
        }
        return ret;
    }
