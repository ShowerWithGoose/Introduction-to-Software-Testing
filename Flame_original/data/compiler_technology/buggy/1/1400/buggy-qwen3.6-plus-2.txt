    public String removeSingleComment(String line) {
        for (int i = 0; i < line.length(); i++) {
            if (line.charAt(i) == '/' && line.charAt(i + 1) == '/') { // @@ [StringIndexOutOfBoundsException when i is the last index, as i+1 exceeds string bounds]
                line = line.substring(0, i);
                break;
            }
        }
        return line;
    }
