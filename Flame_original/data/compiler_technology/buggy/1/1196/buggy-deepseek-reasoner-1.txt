    public TreeNode FuncDef() {
        TreeNode funcdef = new TreeNode(ParsType.FuncDef,new ArrayList<>());
        funcdef.addNode(FuncType());
        if (isMatch(LexType.IDENFR)) {
            addLeaf(funcdef);
        }
        else {
            error();
        }
        if (isMatch(LexType.LPARENT)) {
            addLeaf(funcdef);
            if (isMatch(LexType.RPARENT)) {
                addLeaf(funcdef);
            }
            // 没有右括号，直接到了block的{
            else if (isMatch(LexType.LBRACE)) { // @@ This condition may be too specific; should check for RPARENT or error first, but it's acceptable if grammar allows missing ')'. However, the logic might miss case where next token is not LBRACE.
                addError('j');
            }
            else {
                funcdef.addNode(FuncFParams());
                RparentJudge(funcdef);
            }
            funcdef.addNode(Block());
        }
        addOut("<FuncDef>");
        return funcdef;
    }
