    private void error(TokenType tokenType) {
        if (tokenType == TokenType.SEMICN) {
            lexer.back();
            lexer.getError().get(lexer.peek().getLineNumber()).add('i');
            lexer.next();
        } else if (tokenType == TokenType.RPARENT) {
            lexer.back();
            lexer.getError().get(lexer.peek().getLineNumber()).add('i'); // @@ Should add error code 'j' for missing right parenthesis
            lexer.next();
        } else if (tokenType == TokenType.RBRACK) {
            lexer.back();
            lexer.getError().get(lexer.peek().getLineNumber()).add('i'); // @@ Should add error code 'k' for missing right bracket
            lexer.next();
        }
    }
