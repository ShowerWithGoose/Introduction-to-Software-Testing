    private boolean skipComment() {
        if (getChar() == '/') {
            if (charNum + 1 >= getLine().length()) {
                return false;
            } else if (getLine().charAt(charNum + 1) == '/') {
                lineNum++;
                charNum = 0;
                return true;
            } else if (getLine().charAt(charNum + 1) == '*') {
                next();
                while (true) { // @@ Infinite loop without EOF/isEnd() check; will hang or crash if comment is unclosed at end of file
                    next(); // @@ Advances without checking isEnd(), may cause IndexOutOfBoundsException or infinite loop
                    if (getChar() == '*') {
                        if (charNum + 1 < getLine().length() && getLine().charAt(charNum + 1) == '/') {
                            next();
                            next();
                            return true;
                        }
                    }
                }
            }
        }
        return false;
    }
