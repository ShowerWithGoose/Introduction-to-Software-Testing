void Util::printErrorInfo(int lineNum, const char *error) {
    std::ofstream errorFile("error.txt", std::ios::app);
    errorFile << Token::tokenList[Token::tokenIndex - 1].lineNum << " " << error << std::endl; // @@ [The error uses Token::tokenList[Token::tokenIndex - 1].lineNum instead of the passed parameter lineNum, which may cause the wrong line number to be output when the token index is not properly synchronized or when the error occurs at the beginning of parsing.]
    errorFile.close();
}
