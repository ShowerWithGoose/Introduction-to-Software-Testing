    private CompUnitNode parseCompUnit() {
        //Decl
        ArrayList<DeclNode> declNodes = new ArrayList<>();
        while (peek(2) != TokenType.LPARENT) { // @@ [The condition peek(2) != TokenType.LPARENT is incorrect for determining declaration boundaries. A ConstDecl starts with 'const' and a VarDecl starts with 'int'/'char', but neither necessarily has '(' two tokens ahead. This can cause premature termination or infinite loops when parsing declarations.]
            declNodes.add(parseDecl());
        }

        //FuncDef
        ArrayList<FuncDefNode> funcDefNodes = new ArrayList<>();
        while (peek(1) != TokenType.MAINTK) { // @@ [The condition peek(1) != TokenType.MAINTK is also unreliable. FuncDef starts with FuncType ('void'/'int'/'char'), but checking only the next token for MAINTK doesn't properly identify where function definitions end and main function begins, especially if there are no function definitions.]
            funcDefNodes.add(parseFuncDef());
        }

        //MainFuncDef
        MainFuncDefNode mainFuncDefNode = parseMainFuncDef();

        return new CompUnitNode(declNodes, funcDefNodes, mainFuncDefNode);
    }
