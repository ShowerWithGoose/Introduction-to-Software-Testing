    //<CompUnit> ::= {<Decl>} {<FuncDef>} <MainFuncDef>
    //<Decl>:           [const] int/char <Ident>
    //<FuncDef>:        void/int/char func()
    //<MainFuncDef>:    int main()
    private CompUnitNode parseCompUnit() {
        //Decl
        ArrayList<DeclNode> declNodes = new ArrayList<>();
        while (peek(2) != TokenType.LPARENT) { // @@ [This condition is incorrect because it assumes that any declaration will not have '(' at position 2, but variable declarations can be followed by other tokens, and function definitions are identified by 'type ID (' pattern. The parser should check for the complete function definition pattern (BType/FuncType + IDENFR + LPARENT) rather than just peeking at position 2.]
            declNodes.add(parseDecl());
        }

        //FuncDef
        ArrayList<FuncDefNode> funcDefNodes = new ArrayList<>();
        while (peek(1) != TokenType.MAINTK) { // @@ [This condition is also problematic because it assumes that FuncDef continues until MAINTK is found at position 1, but it doesn't properly validate that the current position actually represents a valid FuncDef. It should check for the proper function definition starting tokens (VOIDTK/INTTK/CHARTK followed by IDENFR and LPARENT).]
            funcDefNodes.add(parseFuncDef());
        }

        //MainFuncDef
        MainFuncDefNode mainFuncDefNode = parseMainFuncDef();

        return new CompUnitNode(declNodes, funcDefNodes, mainFuncDefNode);
    }
