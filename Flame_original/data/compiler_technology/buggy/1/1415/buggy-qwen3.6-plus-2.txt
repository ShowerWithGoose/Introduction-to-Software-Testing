    public boolean isLval() {

    public String syntaxOutput() {

        if (mulExps.size() == 1) {
            sb.append(mulExps.get(0).syntaxOutput());
            sb.append("\n");
            sb.append(name);
        } else if (mulExps.size() > 1) {
            MulExp mulExp = mulExps.get(mulExps.size() - 1);
            mulExps.remove(mulExps.size() - 1);
            Token op = ops.get(ops.size() - 1);
            ops.remove(ops.size() - 1);
            AddExp addExp = new AddExp(mulExps, ops);
            sb.append(addExp.syntaxOutput());
            sb.append("\n");
            sb.append(op);
            sb.append("\n");
            sb.append(mulExp.syntaxOutput());
            sb.append("\n");
            sb.append(name); // @@ The removed elements (mulExp and op) are not restored to their respective lists after recursion, permanently mutating the AST node's state and causing incorrect output or crashes on subsequent calls
        }
        return sb.toString();
