    public void parseDecl(){
        //如果预读的第二个是(结束
        while (tokenList.hasNext()){
            if (tokenList.preRead(2).getType().equals(TokenType.LPARENT)){ // @@ [Error: preRead(2) may throw IndexOutOfBoundsException if fewer than 3 tokens remain, because hasNext() only guarantees at least 1 token exists.]
                //有(，说明是函数了
                return;
            }
            Decl decl = new Decl(tokenList);
            decl.parseDecl();
            declArrayList.add(decl);
        }
    }
    public void parseFuncDef(){
        //如果预读的第一个是main结束
        while (tokenList.hasNext()){
            if (tokenList.preRead(1).getType().equals(TokenType.MAINTK)){ // @@ [Error: preRead(1) may throw IndexOutOfBoundsException if fewer than 2 tokens remain, because hasNext() only guarantees at least 1 token exists.]
                //有main，说明是主函数了
                return;
            }
            FuncDef funcDef = new FuncDef(tokenList);
            funcDef.parseFuncDef();
            funcDefArrayList.add(funcDef);
        }
    }
