struct tree

    // int floor;

// 该函数用于创建新节点
struct tree *creat_new_leaf(int ans)
{
    struct tree *p;
    p = malloc(sizeof(struct tree));
    p->lf = NULL;
    p->ri = NULL;
    return p;
}

// 该函数用于将创建的节点添加在树中
struct plus_tree tree(struct tree *q, struct tree *p, int f1, int ans) // p是新节点，q是老节点
{
    struct plus_tree answer;
    if (ans < q->num)
    {
        if (q->lf == NULL)
        {
            q->lf = p;
            p->flap = f1;
            answer.if_break = 1;
        }
        else
        {
            f1++;
            q = q->lf;
            answer.p = q;
            answer.if_break = 0;
        }
    }
    else
    {
        if (q->ri == NULL)
        {
            q->ri = p;
            p->flap = f1;
            answer.if_break = 1;
        }
        else
        {
            f1++;
            q = q->ri;
            answer.p = q;
            answer.if_break = 0;
        }
    }
    return answer;
}

// 该函数用于打印树的叶子节点
void print_tree(struct tree *q)
{
    if (q == NULL)
    {
        return;
    }
    else if (q->lf == NULL && q->ri == NULL)

        return;

}
