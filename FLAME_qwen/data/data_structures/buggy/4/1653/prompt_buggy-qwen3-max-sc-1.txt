int structcmp(const void *a, const void *b)
{
    const word *aa = *(word **)a;
    const word *bb = *(word **)b;
    for (int i = 0; i < maxlen; i++)
    {
        if (aa->w[i] < bb->w[i])
        {
            return -1;
        }
        if (aa->w[i] > bb->w[i])
        {
            return 1;
        }
    }
}
