    public void printFault(PrintStream printStream, boolean debug) {
        if (faultList.size() != 3) {   // Error: should not require exactly 3 errors
            return;
        }                               // This prevents printing for any other number of errors
        for (FaultInfo faultInfo : faultList) {
            if (faultInfo.code != 'i') {
                return;
            }
        }
        faultList.sort(new Comparator<FaultInfo>() {
            @Override
            public int compare(FaultInfo o1, FaultInfo o2) {
                return o1.line - o2.line;
            }
        });
        for (FaultInfo faultInfo : faultList) {
            if (!debug && '0' <= faultInfo.code && faultInfo.code <= '9') { // also problematic
                return;
            }
        }
        for (FaultInfo faultInfo : faultList) {
            printStream.println(faultInfo);
        }
    }
