    public void printFault(PrintStream printStream, boolean debug) {
        if (faultList.size() != 3) {   // @@ Error: condition should not require exactly 3 errors; prevents printing for any other number of errors
            return;
        }                               
        for (FaultInfo faultInfo : faultList) {
            if (faultInfo.code != 'i') {
                return;
            }
        }
        faultList.sort(new Comparator<FaultInfo>() {
            @Override
            public int compare(FaultInfo o1, FaultInfo o2) {
                return o1.line - o2.line;
            }
        });
        for (FaultInfo faultInfo : faultList) {
            if (!debug && '0' <= faultInfo.code && faultInfo.code <= '9') { // @@ Error: filtering out internal error codes is incorrect; should not return here
                return;
            }
        }
        for (FaultInfo faultInfo : faultList) {
            printStream.println(faultInfo);
        }
    }
