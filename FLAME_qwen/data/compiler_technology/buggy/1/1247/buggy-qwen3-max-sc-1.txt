static void parseCompUnit(CompUnit &it){
    bool fl=false;
    while(index<size){
        if(getType(2)==LPARENT){ // @@ [This condition is insufficient: it assumes that any token sequence where the third token is '(' must be a FuncDef, but doesn't verify that the first token is a valid FuncType (VOIDTK/INTTK/CHARTK). This can misclassify variable declarations or other constructs as function definitions if they coincidentally have '(' two tokens ahead.]
            if(isWordType(words[index+1].content,MAINTK)){
                fl=true;
                parseFuncDef(it.mainFuncDef);
            }
            else{
                it.funcDefs.emplace_back();
                parseFuncDef(it.funcDefs.back());
            }
        }
        else{
            it.decls.emplace_back();
            parseBlockItem(it.decls.back());
        }
    }
    ast(fl);
}
