static void parseCompUnit(CompUnit &it){
    bool fl=false;
    while(index<size){
        if(getType(2)==LPARENT){
            if(isWordType(words[index+1].content,MAINTK)){ // @@ [This check is incorrect because it assumes the current token (index) is 'int' but doesn't verify it. According to the grammar, MainFuncDef must start with 'int main()', so we need to check that words[index].code == INTTK and words[index+1].code == MAINTK, not just check the content of index+1.]
                fl=true;
                parseFuncDef(it.mainFuncDef);
            }
            else{
                it.funcDefs.emplace_back();
                parseFuncDef(it.funcDefs.back());
            }
        }
        else{
            it.decls.emplace_back();
            parseBlockItem(it.decls.back());
        }
    }
    ast(fl);
}
