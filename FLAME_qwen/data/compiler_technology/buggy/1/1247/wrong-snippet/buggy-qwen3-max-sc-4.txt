Based on the problem description and code review, the error appears to be in the `parseCompUnit` function in parser.cpp. The issue is with how it identifies MainFuncDef versus regular FuncDef. According to the grammar, MainFuncDef specifically starts with "int main()", but the current implementation checks `getType(2)==LPARENT` and then looks at `words[index+1].content`, which doesn't properly handle the "int" token that should precede "main".

static void parseCompUnit(CompUnit &it){
    bool fl=false;
    while(index<size){
        if(getType(2)==LPARENT){
            if(isWordType(words[index+1].content,MAINTK)){
                fl=true;
                parseFuncDef(it.mainFuncDef);
            }
            else{
                it.funcDefs.emplace_back();
                parseFuncDef(it.funcDefs.back());
            }
        }
        else{
            it.decls.emplace_back();
            parseBlockItem(it.decls.back());
        }
    }
    ast(fl);
}
