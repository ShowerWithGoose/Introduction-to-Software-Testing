    public Parser(ArrayList<Token> tokens) {
        this.tokens = tokens;
        this.pos = 0;
        this.compUnit = new CompUnit();
        while (this.pos < tokens.size()) {
            if (this.tokens.get(this.pos).getType() == TokenType.CONST) {
                this.compUnit.addDecl(parseConstDecl());
            } else if (this.tokens.get(this.pos + 1).getType() == TokenType.MAIN) { // @@ Potential IndexOutOfBoundsException when tokens list has insufficient elements; also does not check current token type.
                this.compUnit.setMainFuncDef(parseMainFuncDef());
            } else if (this.tokens.get(this.pos + 2).getType() == TokenType.LPARENT) { // @@ Potential IndexOutOfBoundsException when tokens list has fewer than 3 elements.
                this.compUnit.addFuncDef(parseFuncDef());
            } else {
                this.compUnit.addDecl(parseVarDecl());
            }
        }
    }
