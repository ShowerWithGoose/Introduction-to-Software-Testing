    public Parser(ArrayList<Token> tokens) {
        this.tokens = tokens;
        this.pos = 0;
        this.compUnit = new CompUnit();
        while (this.pos < tokens.size()) {
            if (this.tokens.get(this.pos).getType() == TokenType.CONST) {
                this.compUnit.addDecl(parseConstDecl());
            } else if (this.tokens.get(this.pos + 1).getType() == TokenType.MAIN) { // @@ Potential IndexOutOfBoundsException when pos is at or near the end of the token list; lacks bounds checking and does not verify that the current token is INTTK before looking ahead.
                this.compUnit.setMainFuncDef(parseMainFuncDef());
            } else if (this.tokens.get(this.pos + 2).getType() == TokenType.LPARENT) { // @@ Potential IndexOutOfBoundsException when pos is near the end of the token list; unsafe lookahead without bounds checking.
                this.compUnit.addFuncDef(parseFuncDef());
            } else {
                this.compUnit.addDecl(parseVarDecl());
            }
        }
    }
