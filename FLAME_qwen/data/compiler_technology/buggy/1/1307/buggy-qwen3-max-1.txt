    public ParentExp parseParentExp() {
        Token leftParent = tokenList.getNextToken();
        ExpParser expParser = new ExpParser(tokenList, symbolList);
        Exp exp = expParser.parseExp();
        Token rightParent = tokenList.getNextToken();
        if (rightParent.getType() != Token.TokenType.RPARENT) {
            tokenList.moveBack(2);
            ErrorList.addError(new Error(tokenList.getNextToken().getLineIndex(), Error.ErrorType.LACK_RIGHT_PARENT)); // @@ [The error line number should correspond to the line where the last non-terminal before the right parenthesis is located, but this code gets the line index of the current token after moving back, which may not be the correct location. According to the problem specification, for missing right parenthesis error, the line number should be where the last preceding non-terminal symbol appears before the expected right parenthesis.]
        }
        return new ParentExp(leftParent, exp, rightParent);
    }
