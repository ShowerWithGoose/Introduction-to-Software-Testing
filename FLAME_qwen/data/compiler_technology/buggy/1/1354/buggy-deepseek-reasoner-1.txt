    public Ast.CompUnit parseCompUnit() {
        ArrayList<Ast.Decl> decls = new ArrayList<>();
        ArrayList<Ast.FuncDef> funcDefs = new ArrayList<>();
        Ast.MainFuncDef mainFuncDef = null;
        while (tokenList.hasNext()) {
            if (tokenList.checkAheadChar(2).type == TokenType.LPARENT && tokenList.checkAheadChar(1).type == TokenType.MAINTK) { // @@ IndexOutOfBounds if tokenList has fewer than 3 tokens; also, after parsing main, loop continues without break
                mainFuncDef = parseMainFuncDef();
            } else if (tokenList.checkAheadChar(2).type == TokenType.LPARENT) { // @@ Same bounds issue; also, this condition may match non-function tokens mistakenly, e.g., function call inside an expression, but at top level it's safe
                funcDefs.add(parseFuncDef());
            } else {
                decls.add(parseDecl());
            }
        }
        return new Ast.CompUnit(decls,funcDefs,mainFuncDef);
    }
