    public CompUnit compUnit(){
        List<Decl> declList=new ArrayList<>();
        List<FuncDef> funcDefList=new ArrayList<>();
        MainFuncDef mainFuncDef=null;
        while(tokens.get(index+1).getLexType()!=LexType.MAINTK && tokens.get(index+2).getLexType()!=LexType.LPARENT){ // @@ [This condition can cause IndexOutOfBoundsException when index+1 or index+2 exceeds tokens size, and the logic for detecting MainFuncDef is incorrect as it doesn't properly check the sequence 'int' 'main' '(' ')']
            Decl decl=decl();
            declList.add(decl);
        }
        while(tokens.get(index+1).getLexType()!=LexType.MAINTK){ // @@ [This condition can cause IndexOutOfBoundsException when index+1 exceeds tokens size, and it doesn't properly verify that MAINTK is preceded by INTTK to identify MainFuncDef]
            FuncDef funcDef=funcDef();
            funcDefList.add(funcDef);
        }
        mainFuncDef=mainFuncDef();
        return new CompUnit(declList,funcDefList,mainFuncDef);
    }
