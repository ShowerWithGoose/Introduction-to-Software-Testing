void outputComponent(Component *src, FILE *out)
{
    if (src->getCType() >= ComponentEnum::TokenConst && src->getCType() <= ComponentEnum::StringConst)
    {
        fprintf(out, "%s %s\n", TokenStrings[src->getTk().key].c_str(), src->getTk().content.c_str());
    }
    else
    {
        if (src->getCType() == ComponentEnum::MulExp || src->getCType() == ComponentEnum::AddExp ||
            src->getCType() == ComponentEnum::RelExp || src->getCType() == ComponentEnum::EqExp ||
            src->getCType() == ComponentEnum::LAndExp || src->getCType() == ComponentEnum::LOrExp)
        {
            for (int i = 0; i < src->getParts().size(); i += 1)
            {
                outputComponent(src->getParts()[i], out);
                if (src->getParts()[i]->getCType() != ComponentEnum::TokenConst) // @@ Incorrectly prints the component name multiple times inside the loop for each non-token child, violating the requirement to output it only once after complete analysis.
                {
                    fprintf(out, "<%s>\n", ComponentStrings[src->getCType()].c_str()); // @@ Causes duplicate and premature output of the syntax component name.
                }
            }
            return; // @@ Prematurely exits the function, bypassing the standard logic that correctly outputs the component name exactly once after processing all children.
        }
        for (int i = 0; i < src->getParts().size(); i += 1)
        {
            outputComponent(src->getParts()[i], out);
        }
        if (src->getCType() == ComponentEnum::BlockItem || src->getCType() == ComponentEnum::Decl ||
            src->getCType() == ComponentEnum::BType)
        {
            return;
        }
        fprintf(out, "<%s>\n", ComponentStrings[src->getCType()].c_str());
    }
}
