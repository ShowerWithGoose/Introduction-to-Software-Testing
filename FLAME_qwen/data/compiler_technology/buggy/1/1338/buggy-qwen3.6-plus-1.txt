    public boolean tryParseLVal() throws IOException {
        this.isOutput = false;
        int pos = this.curPos;
        if (this.curToken.getType() == TokenType.IDENFR) {
            getNextToken();
            if (this.curToken.getType() == TokenType.LBRACK) {
                getNextToken();
                parseExp(); // @@ Calling parseExp() during lookahead consumes tokens and alters the parser's internal state, causing incorrect parsing or exceptions when the state is reset.
            }
            if(this.curToken.getType() == TokenType.RBRACK) {
                getNextToken();
            }
            if(this.curToken.getType() == TokenType.ASSIGN) {
                this.curPos = pos;
                this.curToken = this.tokens.get(this.curPos);
                this.isOutput = true;
                return true;
            }
        }
        this.curPos = pos;
        this.curToken = this.tokens.get(this.curPos);
        this.isOutput = true;
        return false;
    }
