    private static Stmt parseLValOrExpStmt() {
        int pointer = handler.getPointer();
        LVal lVal = ExpParser.parseLVal();
        if (handler.getNext().getType() != TokenType.ASSIGN) {
            handler.setPointer(pointer);
            Exp exp = ExpParser.parseExp(false);
            Token semicn = null;
            if (handler.getNext().getType() == TokenType.SEMICN) {
                semicn = handler.getNextAndMove();
            } else {
                handler.addError(ErrorType.MISS_SEMICN); // @@ [When semicolon is missing, the parser should still continue parsing but this error handling doesn't ensure proper recovery for subsequent parsing steps]
            }
            return new Stmt(exp, semicn);
        } else {
            Token assign = handler.getNextAndMove();
            if (handler.getNext().getType() != TokenType.GETCHARTK && handler.getNext().getType() != TokenType.GETINTTK) {
                Exp exp = ExpParser.parseExp(false);
                Token semicn = null;
                if (handler.getNext().getType() == TokenType.SEMICN) {
                    semicn = handler.getNextAndMove();
                } else {
                    handler.addError(ErrorType.MISS_SEMICN); // @@ [Same issue: missing semicolon error is recorded but parser may not handle recovery properly, potentially causing cascading errors]
                }
                return new Stmt(lVal, assign, exp, semicn);
            } else {
                Token inputTK = handler.getNextAndMove();
                Token lParent = handler.getNextAndMove();
                Token rParent = null;
                if (handler.getNext().getType() == TokenType.RPARENT) {
                    rParent = handler.getNextAndMove();
                } else {
                    handler.addError(ErrorType.MISS_RPARENT); // @@ [Missing right parenthesis error is handled but the parser continues with rParent as null, which may cause issues in toString() or later processing]
                }
                Token semicn = null;
                if (handler.getNext().getType() == TokenType.SEMICN) {
                    semicn = handler.getNextAndMove();
                } else {
                    handler.addError(ErrorType.MISS_SEMICN); // @@ [Again, missing semicolon error without proper recovery mechanism]
                }
                return new Stmt(lVal, assign, inputTK, lParent, rParent, semicn);
            }
        }
    }
