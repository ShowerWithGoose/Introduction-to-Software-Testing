    // 解析变量声明
    private void parseVarDecl() throws IOException {
        // VarDecl → BType VarDef { ',' VarDef } ';' // i
        parseBType();
        int currentLine = getPreviousTokenLineNumber(); // @@ [This captures the line number of the BType token (e.g., 'int'), but according to error type 'i' specification, the error line should correspond to the line where the last non-terminal before the semicolon is located—likely the last VarDef or ConstDef. Using BType's line number is incorrect for multi-line declarations.]
        parseVarDef();
        while (currentToken.getType() == TokenType.COMMA) {
            outputToken(currentToken);
            nextToken();
            parseVarDef();
        }
        if (currentToken.getType() == TokenType.SEMICN) {
            outputToken(currentToken);
            nextToken();
        } else {
            addError(currentLine, 'i'); // 缺少分号
        }
        // 需要输出 <VarDecl>
        outputGrammar("<VarDecl>");
    }
