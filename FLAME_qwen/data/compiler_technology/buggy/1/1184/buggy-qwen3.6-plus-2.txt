    public boolean isLValBegin(Token token) {
        int index = tokens.getIndex();
        Token t = tokens.getNext();
        int check = 0;
        while (t.getTokenType() != TokenType.ASSIGN) { // @@ [Infinite loop risk if ASSIGN is never encountered; lacks bounds checking and fails to terminate on other valid expression/statement delimiters like RPARENT, COMMA, or RBRACE]
            if (t.getTokenType() == TokenType.SEMICN) {
                tokens.setIndex(index);
                return false;
            }
            if (t.getTokenType() == TokenType.LBRACK ){
                check = 1;
            }

            if (t.getTokenType() != TokenType.LBRACK && t.getTokenType() != TokenType.RBRACK && check == 0) {
                tokens.setIndex(index);
                return false;
            }
            if (t.getTokenType() == TokenType.RBRACK ){
                check = 0;
            }
            t = tokens.getNext(); // @@ [May throw IndexOutOfBoundsException when scanning past the end of the token list]
        }
        tokens.setIndex(index);
        return true;
    }
