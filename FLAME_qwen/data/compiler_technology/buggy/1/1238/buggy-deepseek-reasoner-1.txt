    private Node compUnit() {
        // CompUnit → {Decl} {FuncDef} MainFuncDef
        Node node = new Node(CompUnit);
        while (next(1).getType() != MAINTK && next(2).getType() != LPARENT) { // @@ Potential IndexOutOfBoundsException if token list is too small; no bounds checking in next()
            Node declNode = decl();
            node.addChild(declNode);
        }
        while (next(1).getType() != MAINTK) { // @@ Same issue: next(1) may throw exception if current is near end
            Node funcNode = funcDef();
            node.addChild(funcNode);
        }
        Node mainFuncNode = mainFuncDef();
        node.addChild(mainFuncNode);
        return node;
    }
