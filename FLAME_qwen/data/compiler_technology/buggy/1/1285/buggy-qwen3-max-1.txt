Node *CompUnit(int depth){
    Node *root = new Node;
    strcpy(root->GrammarItem, "CompUnit");
    root->parent = NULL;
    root->depth = depth;
    Node *child = NULL;
    while(strcmp(tokens[tokenPos + 2]->symbol, "LPARENT") != 0){ // @@ [Accessing tokens[tokenPos + 2] without checking bounds may cause out-of-range access; also logic is flawed because it assumes presence of '(' to distinguish Decl from FuncDef/MainFuncDef, but this fails when there are no declarations or when token stream ends]
        child = Decl(depth + 1);
        root->children.push_back(child);
        child->parent = root;
    }
    while(strcmp(tokens[tokenPos + 1]->symbol, "MAINTK") != 0){ // @@ [Similarly, accessing tokens[tokenPos + 1] without bound check; also incorrect condition—should parse FuncDef only if next token is a valid FuncType (VOIDTK/INTTK/CHARTK) followed by IDENFR and '(', not based on absence of MAINTK in lookahead of 1]
        child = FuncDef(depth + 1);
        root->children.push_back(child);
        child->parent = root;
    }
    child = MainFuncDef(depth + 1);
    root->children.push_back(child);
    child->parent = root;
    return root;
}
