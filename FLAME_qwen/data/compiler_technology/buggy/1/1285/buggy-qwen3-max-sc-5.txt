    strcpy(root->GrammarItem, "CompUnit");
    root->parent = NULL;
    root->depth = depth;
    Node *child = NULL;
    while(strcmp(tokens[tokenPos + 2]->symbol, "LPARENT") != 0){ // @@ [This condition is unsafe: it assumes at least 3 tokens exist ahead and that Decl ends before any '('; also, it may access out-of-bounds tokens when input is short or malformed. Moreover, the logic for distinguishing Decl from FuncDef/MainFuncDef based on lookahead is incorrect—'(' after 'main' is part of MainFuncDef, but Decl doesn't necessarily end before any '(' elsewhere.]
        child = Decl(depth + 1);

        child->parent = root;
    }

    }
    child = MainFuncDef(depth + 1);

    return root;
}
