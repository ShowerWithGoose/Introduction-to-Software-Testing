            String filePath = "lexer.txt";  // @@ File should be "parser.txt" as per problem description.
            try (BufferedWriter writer = new BufferedWriter(new FileWriter(filePath, true))) {
                for (Token word : words) {
                    writer.write(word + "\n");
                }
                writer.flush();
            } catch (IOException e) {
                e.printStackTrace();
                System.out.println("写入文件时发生错误。");
            }
        } else {

            } catch (IOException e) {
                e.printStackTrace();
                System.out.println("写入文件时发生错误。");
            }
        }
