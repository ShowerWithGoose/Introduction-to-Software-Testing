    public Parser(String filename, BufferedWriter out) {
        this.out = out;
        Lexer lexer = new Lexer(filename, out);
        while (lexer.getLexType() != LexType.EOF) {
            lexer.next();
            if (lexer.getLexType() != LexType.ERR) {
                words.add(new Word(lexer.getToken(), lexer.getLexType(), lexer.getNumber(), lexer.getLineNum()));
            }
        }
        lastWord = words.get(0);  // @@ [Incorrectly initializes lastWord to the first token before parsing starts; lastWord should represent the previously consumed token during parsing, not be preset. This leads to wrong line numbers in error reporting when the first token causes or precedes an error.]
    }

    public void getWord() {
        if (curPos < words.size()) {
            lastWord = curWord;  // @@ [On the first call, curWord is null (since it's only set after this line in earlier versions), so lastWord becomes null, causing potential NullPointerException or incorrect line number tracking in error reporting. Even if curWord were set, the initial state is broken due to prior misinitialization.]
            curWord = words.get(curPos);
            curType = words.get(curPos).getType();
            curToken = words.get(curPos).getToken();
            curPos++;
            if (print && lastWord != null) {
                try {
                    if (lastWord.getType() != LexType.ERR && lastWord.getType() != LexType.EOF) {
                        out.write(lastWord.getType().toString() + " " + lastWord.getToken() + "\n");
                    }
                } catch (Exception e) {
                    throw new RuntimeException(e);
                }
            }
        } else {
            curType = LexType.EOF;
        }
    }
