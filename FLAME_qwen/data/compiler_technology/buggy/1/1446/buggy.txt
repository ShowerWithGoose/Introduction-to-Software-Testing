        flag = lexer();

                lexer.write(s + "\n");
            }
        } else {
            wrongOutput.sort(Comparator.comparingInt(s -> Integer.parseInt(s.split(" ")[0])));
            for (String s : wrongOutput) {
                error.write(s + "\n");
            }
        }
        in.close();
        lexer.close();
        error.close();
    }

        rightOutput.add("<CompUnit>");
    }

    private static void reserve(String token,int line) {
        switch (token) {
            case "main" -> category.add("MAINTK");
            case "const" -> category.add("CONSTTK");
            case "int" -> category.add("INTTK");
            case "char" -> category.add("CHARTK");
            case "break" -> category.add("BREAKTK");
            case "continue" -> category.add("CONTINUETK");
            case "if" -> category.add("IFTK");
            case "else" -> category.add("ELSETK");
            case "for" -> category.add("FORTK");
            case "getint" -> category.add("GETINTTK");
            case "getchar" -> category.add("GETCHARTK");
            case "printf" -> category.add("PRINTFTK");
            case "return" -> category.add("RETURNTK");
            case "void" -> category.add("VOIDTK");
            default -> category.add("IDENFR");
        }
        value.add(token);
        lines.add(line);
    }

    private static int lexer() throws IOException {
        String str;
        int flag = 0;
        int line = 1;
        while ((str = in.readLine()) != null) {
            for (int curPos = 0; curPos < str.length(); curPos++) {
                char c = str.charAt(curPos);
                if(Character.isLetter(c) || c == '_') {
                    StringBuilder token = new StringBuilder();
                    token.append(c);
                    curPos++;
                    while(curPos < str.length() && (Character.isLetter(str.charAt(curPos)) || str.charAt(curPos) == '_' || Character.isDigit(str.charAt(curPos)))) {
                        c = str.charAt(curPos);
                        token.append(c);
                        curPos++;
                    }
                    reserve(String.valueOf(token),line);
                    curPos--;
                } else if(Character.isDigit(c)) {
                    StringBuilder token = new StringBuilder();
                    token.append(c);
                    curPos++;
                    while(curPos < str.length() && Character.isDigit(str.charAt(curPos))) {
                        c = str.charAt(curPos);
                        token.append(c);
                        curPos++;
                    }
                    curPos--;
                    category.add("INTCON");
                    value.add(String.valueOf(token));
                    lines.add(line);
                } else if(c == '\'') {
                    StringBuilder token = new StringBuilder();
                    token.append(c);
                    curPos++;
                    while(curPos < str.length() && str.charAt(curPos) != '\'') {
                        c = str.charAt(curPos);
                        token.append(c);
                        curPos++;
                    }
                    token.append("'");
                    category.add("CHRCON");
                    value.add(String.valueOf(token));
                    lines.add(line);
                } else if(c == '\"') {
                    StringBuilder token = new StringBuilder();
                    token.append(c);
                    curPos++;
                    while(curPos < str.length() && str.charAt(curPos) != '\"') {
                        c = str.charAt(curPos);
                        token.append(c);
                        curPos++;
                    }
                    token.append('\"');
                    category.add("STRCON");
                    value.add(String.valueOf(token));
                    lines.add(line);
                } else if (c == '/') {
                    if (str.charAt(curPos + 1) == '/') {
                        curPos = str.length() - 1;
                    } else if (str.charAt(curPos + 1) == '*') {
                        curPos += 2;
                        int end = 0;
                        while (end == 0) {
                            for (;curPos < str.length() - 1;curPos++) {
                                if (str.charAt(curPos) == '*' && str.charAt(curPos + 1) == '/') {
                                    end = 1;
                                    curPos += 2;
                                    break;
                                }
                            }
                            if (end == 0) {
                                str = in.readLine();
                                line++;
                                curPos = 0;
                            }
                        }
                        curPos--;
                    } else {
                        category.add("DIV");
                        value.add("/");
                        lines.add(line);
                    }
                } else {
                    switch (c) {
                        case '!' -> {
                            if (curPos < str.length() - 1 && str.charAt(curPos + 1) == '=') {
                                category.add("NEQ");
                                value.add("!=");
                                lines.add(line);
                                curPos++;
                            } else {
                                category.add("NOT");
                                value.add("!");
                                lines.add(line);
                            }
                        }
                        case '&' -> {
                            if (curPos < str.length() - 1 && str.charAt(curPos + 1) == '&') {
                                curPos++;
                            } else {
                                flag = 1;
                                wrongOutput.add(line + " a");
                            }
                            category.add("AND");
                            value.add("&&");
                            lines.add(line);
                        }
                        case '|' -> {
                            if (curPos < str.length() - 1 && str.charAt(curPos + 1) == '|') {
                                curPos++;
                            } else {
                                flag = 1;
                                wrongOutput.add(line + " a");
                            }
                            category.add("OR");
                            value.add("||");
                            lines.add(line);
                        }
                        case '+' -> {
                            category.add("PLUS");
                            value.add("+");
                            lines.add(line);
                        }
                        case '-' -> {
                            category.add("MINU");
                            value.add("-");
                            lines.add(line);
                        }
                        case '*' -> {
                            category.add("MULT");
                            value.add("*");
                            lines.add(line);
                        }
                        case '%' -> {
                            category.add("MOD");
                            value.add("%");
                            lines.add(line);
                        }
                        case '<' -> {
                            if (curPos < str.length() - 1 && str.charAt(curPos + 1) == '=') {
                                category.add("LEQ");
                                value.add("<=");
                                lines.add(line);
                                curPos++;
                            } else {
                                category.add("LSS");
                                value.add("<");
                                lines.add(line);
                            }
                        }
                        case '>' -> {
                            if (curPos < str.length() - 1 && str.charAt(curPos + 1) == '=') {
                                category.add("GEQ");
                                value.add(">=");
                                lines.add(line);
                                curPos++;
                            } else {
                                category.add("GRE");
                                value.add(">");
                                lines.add(line);
                            }
                        }
                        case '=' -> {
                            if (curPos < str.length() - 1 && str.charAt(curPos + 1) == '=') {
                                category.add("EQL");
                                value.add("==");
                                lines.add(line);
                                curPos++;
                            } else {
                                category.add("ASSIGN");
                                value.add("=");
                                lines.add(line);
                            }
                        }
                        case ';' -> {
                            category.add("SEMICN");
                            value.add(";");
                            lines.add(line);
                        }
                        case ',' -> {
                            category.add("COMMA");
                            value.add(",");
                            lines.add(line);
                        }
                        case '(' -> {
                            category.add("LPARENT");
                            value.add("(");
                            lines.add(line);
                        }
                        case ')' -> {
                            category.add("RPARENT");
                            value.add(")");
                            lines.add(line);
                        }
                        case '[' -> {
                            category.add("LBRACK");
                            value.add("[");
                            lines.add(line);
                        }
                        case ']' -> {
                            category.add("RBRACK");
                            value.add("]");
                            lines.add(line);
                        }
                        case '{' -> {
                            category.add("LBRACE");
                            value.add("{");
                            lines.add(line);
                        }
                        case '}' -> {
                            category.add("RBRACE");
                            value.add("}");
                            lines.add(line);
                        }
                    }
                }
            }
            line++;
        }
        return flag;
    }
