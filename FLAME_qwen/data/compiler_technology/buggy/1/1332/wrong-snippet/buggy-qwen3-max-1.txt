    private CompUnit parseCompUnit() {
        // CompUnit -> {Decl} {FuncDef} MainFuncDef
        List<Decl> decls = null;
        List<FuncDef> funcDefs = null;
        MainFuncDef mainFuncDef = null;

        // Default program integrity, i.e., preReadToken(2) will not be null
        // While condition not satisfied, pay attention to infinite loop issues, which often occur when token is null
        while (!isEOF() && getPreReadTokenType(2) != TokenType.LPARENT
                && getPreReadTokenType(1) != TokenType.MAINTK) {
            // As long as the loop entry condition is met, there will definitely be decls
            if (decls == null) {
                decls = new ArrayList<>();
            }
            decls.add(parseDecl());
        }
        // MAINTK is considered to have entered MainFuncDef
        while (!isEOF() && getPreReadTokenType(1) != TokenType.MAINTK) {
            if(funcDefs == null){
                funcDefs = new ArrayList<>();
            }
            funcDefs.add(parseFuncDef());
        }
        mainFuncDef = parseMainFuncDef();
        return new CompUnit(decls, funcDefs, mainFuncDef);
    }
