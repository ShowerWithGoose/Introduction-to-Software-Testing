    public boolean getToken(Word... words) {

        curToken = tokens.get(pos); // Potential IndexOutOfBoundsException when tokens list is empty
        if (words.length == 0) {
            return true;
        }
        for (Word word : words) {
            if (curToken.getWord().equals(word)) return true;
        }

        return false;
