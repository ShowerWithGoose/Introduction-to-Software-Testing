Based on the problem description and code review, the issue appears to be in the AST dumping logic where syntax component names are not being output correctly according to the specification. The problem is in the `DumpAST.java` file where the order of operations is incorrect - it should output the syntax component name BEFORE processing its children, not after.

    // 语法分析成分分析结束前，另起一行输出当前语法成分的名字
    // <BlockItem>, <Decl>, <BType> 不用输出
    private static void dump(ArrayList<Object> nodes) throws IOException {
        for (Object node : nodes) {
            if (node != null) {
                if (node instanceof ASTNode branchNode) {
                    String simpleName = branchNode.getClass().getSimpleName();
                    if (!simpleName.equals("BlockItem") && 
                        !simpleName.equals("Decl") && 
                        !simpleName.equals("BType")) {
                        out.write("<" + simpleName + ">" + "\n");
                    }
                    dump(branchNode.explore());
                } else if (node instanceof ASTNodeWithOption<?> branchNodeWithOption) {
                    String simpleName = branchNodeWithOption.getClass().getSimpleName();
                    if (!simpleName.equals("BlockItem") && 
                        !simpleName.equals("Decl") && 
                        !simpleName.equals("BType")) {
                        out.write("<" + simpleName + ">" + "\n");
                    }
                    dump(branchNodeWithOption.extract().explore());
                } else if (node instanceof Token leafNode) {
                    out.write(leafNode.type().toString() + " " + leafNode.strVal() + "\n");
                }
            }
        }
    }
