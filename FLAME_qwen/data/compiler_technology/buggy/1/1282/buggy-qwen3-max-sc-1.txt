    public CompUnit(TokenStream stream) {
        decls = new ArrayList<>();
        funcDefs = new ArrayList<>();
        // ConstDecl → 'const' BType ConstDef { ',' ConstDef } ';'
        //                ConstDef → $Ident$ [ '[' ConstExp ']' ] '=' ConstInitVal
        // VarDecl → BType VarDef { $','$ VarDef } $';'$ // 该分号也有可能因为i类错误缺失，但是如果缺失后变为'('那就无法确定是否定义函数
        //        VarDef → Ident [ $'['$ ConstExp ']' ] [ $'='$ InitVal ]
        // FuncDef → FuncType Ident $'('$ [FuncFParams] ')' Block
        // MainFuncDef → 'int' 'main' $'('$ ')' Block
        while (stream.getNext(2).type() != TokenType.LPARENT) { // @@ [This lookahead logic is flawed because it assumes that any declaration must not have '(' two tokens ahead, but this fails when a VarDecl is followed by a function definition or when there are empty declarations. It may incorrectly stop parsing declarations too early or consume tokens meant for FuncDef.]
            decls.add(Decl.parse(stream));
        }
        // FuncDef → FuncType $Ident$ '(' [FuncFParams] ')' Block
        // MainFuncDef → 'int' $'main'$ '(' ')' Block
        while (stream.getNext(1).type() != TokenType.MAINTK) { // @@ [This lookahead is also unreliable because it assumes the next token after current position being MAINTK indicates the start of main function, but if there are no FuncDefs, this may skip valid tokens or cause incorrect parsing. Also, if MAINTK appears unexpectedly, it may lead to infinite loops or missed function definitions.]
            funcDefs.add(new FuncDef(stream));
        }
        mainFuncDef = new MainFuncDef(stream);
        if (stream.getNow().type() != TokenType.EOF) {
            if (Config.parserThrowable) {
                throw new RuntimeException("When CompUnit() end, didn't reach the end of the file.");
            }
        }
    }
