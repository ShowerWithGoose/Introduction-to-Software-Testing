    public CompUnit(TokenStream stream) {
        decls = new ArrayList<>();
        funcDefs = new ArrayList<>();
        // ConstDecl → 'const' BType ConstDef { ',' ConstDef } ';'
        //                ConstDef → $Ident$ [ '[' ConstExp ']' ] '=' ConstInitVal
        // VarDecl → BType VarDef { $','$ VarDef } $';'$ // 该分号也有可能因为i类错误缺失，但是如果缺失后变为'('那就无法确定是否定义函数
        //        VarDef → Ident [ $'['$ ConstExp ']' ] [ $'='$ InitVal ]
        // FuncDef → FuncType Ident $'('$ [FuncFParams] ')' Block
        // MainFuncDef → 'int' 'main' $'('$ ')' Block
        while (stream.getNext(2).type() != TokenType.LPARENT) { // @@ [This lookahead logic is flawed because it assumes that any declaration must not be followed by '(' two tokens ahead, but variable declarations can be followed by function definitions that start with 'int'/'char' and then an identifier and '(', which would break this condition prematurely]
            decls.add(Decl.parse(stream));
        }
        // FuncDef → FuncType $Ident$ '(' [FuncFParams] ')' Block
        // MainFuncDef → 'int' $'main'$ '(' ')' Block
        while (stream.getNext(1).type() != TokenType.MAINTK) { // @@ [This lookahead is also incorrect because it assumes all function definitions come before main, but the grammar allows functions to appear before or after declarations, and this condition may skip valid function definitions or include non-function tokens]
            funcDefs.add(new FuncDef(stream));
        }
        mainFuncDef = new MainFuncDef(stream);
        if (stream.getNow().type() != TokenType.EOF) {
            if (Config.parserThrowable) {
                throw new RuntimeException("When CompUnit() end, didn't reach the end of the file.");
            }
        }
    }
