    public boolean isLValAndEqual(int lineNum, int colNum) {
        Pair pair = getToken(lineNum, colNum);
        if (pair.getKey().equals(LexType.IDENFR)) {
            colNum++;
            boolean flag = false;
            while (colNum < this.tokens.get(lineNum).size() && !getToken(lineNum, colNum).getValue().equals(";")) {
                if (getToken(lineNum, colNum).getValue().equals("=")) {
                    flag = true;
                }
                colNum++;
            }
            return flag; // @@ [This method only checks within the same line for '=', but LVal assignment may span multiple lines or contain complex expressions; it also doesn't verify that '=' appears immediately after the LVal or its optional array index, leading to incorrect Stmt parsing decisions]
        }
        return false;
    }
