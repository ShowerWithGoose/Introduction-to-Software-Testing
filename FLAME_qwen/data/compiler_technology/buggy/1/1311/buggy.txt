void init_Word2Type(){
    vector<string>key={"main","const","int","char","break","continue","if","else","!","&&","||","for","getint","getchar","printf","return","+","-","void","*","/","%","<","<=",">",">=","==","!=","=",";",",","(",")","[","]","{","}"};
    vector<string>value={"MAINTK","CONSTTK","INTTK","CHARTK","BREAKTK","CONTINUETK","IFTK","ELSETK","NOT","AND","OR","FORTK","GETINTTK","GETCHARTK","PRINTFTK","RETURNTK","PLUS","MINU","VOIDTK","MULT","DIV","MOD","LSS","LEQ","GRE","GEQ","EQL","NEQ","ASSIGN","SEMICN","COMMA","LPARENT","RPARENT","LBRACK","RBRACK","LBRACE","RBRACE"};
    while(!key.empty()){
        Word2Type.emplace(key.back(),value.back());
        key.pop_back();
        value.pop_back();
    }
}
void Read(string buf){
    int l = buf.length();
    for(int i=0;i<l;i++){
        if(isAnnot){
            if(buf[i]!='*') continue;
            else if(i+1<=l-1&&buf[i+1]=='/'){
                i++;
                isAnnot = 0;
                continue;
            }
            else continue;         
        }
        if(buf[i]=='"'){
            token+='"';
            while(buf[++i]!='"'){
                token+=buf[i];
            }
            token+='"';
            pushWord();
        }
        else if(buf[i]=='\''){
            token+='\'';     
            if(buf[i+3]=='\''){
                token+=buf[i+1];
                token+=buf[i+2];
                token+='\'';
                i+=3;
            }
            else if(buf[i+2]=='\''){
                token+=buf[i+1];
                token+='\'';
                i+=2;
            }
            pushWord();
        }
        else if(sym.find(buf[i])!=sym.end()){
            pushWord();
            if(buf[i]!=' ') token+=buf[i];
            
            if(buf[i]=='&'||buf[i]=='|') {
                if(buf[i+1]!=buf[i]){
                    token+=buf[i];
                    err.push_back({nowLineNum,'a'});//0
                }
                else token+=buf[++i];
                pushWord();
            }
            else if(buf[i]=='!'||buf[i]=='<'||buf[i]=='='||buf[i]=='>'){
                if(buf[i+1]=='=') token+=buf[++i];
                pushWord();
            }
            else if(buf[i]=='/'&&buf[i+1]=='*'){
                isAnnot = 1;
                token = "";
            }
            else if(buf[i]=='/'&&buf[i+1]=='/'){
                token = "";
                break;
            }
            else if(buf[i]!=' ') pushWord();
        }
        else if(buf[i]>=32&&buf[i]<=126) token+=buf[i];
    }
    pushWord();
}
string getLexType(string word){
    auto it = Word2Type.find(word);
    if(it!=Word2Type.end()) return it->second;
    else if(isNum(word)) return "INTCON";
    else if(word[0]=='"') return "STRCON";
    else if(word[0]=='\'') return "CHRCON";
    else if(isIdent(word)) return "IDENFR";
    else return "123";
}
void pushWord(){
    if(token == "") return;
    word.push_back({getLexType(token),token});
    lineNumber.push_back(nowLineNum);
    token = "";
}
bool isNum(string token){
    int l = token.length();
    for(int i=0;i<l;i++){
        if(!isdigit(token[i])) return 0;
        }
    return 1;
}
bool isIdent(string token){
    int l = token.length();
    if(token[0]!='_'&&!isalpha(token[0])) return 0;
    for(int i=1;i<l;i++){
        char ch = token[i];
        if((!isalpha(ch))&&(!isdigit(ch))&&(ch!='_')) return 0;
    }
    return 1;
}

void Grammer(){
    now = word[0].second;//目前的单词
    preRead = word[1].second;//预读
    prePreRead = word[2].second;
    word.resize(word.size() + 2);
    int t = word.size() - 2;
    while(Index < t){
        //cout<<Index+1<<" "<<now<<"\n";
        if(isConst()) ConstDecl();
        else if(now == "char"){
            if((word[Index+1].first)=="IDENFR"){
                if(prePreRead  == "=" || prePreRead == ";" || prePreRead == ","|| prePreRead == "[") VarDecl();
                else if(prePreRead == "(") FuncDef();
                else E();
            }
        } 
        else if(now == "int"){
            if(preRead == "main"){
                MainFuncDef();
            }
            else if((word[Index+1].first)=="IDENFR"){
                if(prePreRead  == "=" || prePreRead == ";" || prePreRead == "," || prePreRead == "[") VarDecl();
                else if(prePreRead  == "(") FuncDef();
                else E();
            }
        }
        else if(now == "void"){
            FuncDef();
        }
        else if(now == "if" || now == "for" || now == "break" || now == "return") Stmt();
    }
    res.push_back("<CompUnit>");
}
void next(){
    res.push_back(word[Index].first + " " + word[Index].second);
    //cout<<Index+1<<" "<<word[Index].second<<"\n";
    Index++;
    now = preRead;
    preRead = prePreRead;
    prePreRead = word[Index+2].second;
}

void Stmt(){
    if(now == "if"){
        next();
        if(now == "("){
            next();
            Cond();
            if(now == ")"){
                next();
                Stmt();
                if(now == "else"){
                    next();
                    Stmt();
                }
            }
            else{
                err.push_back({lineNumber[Index-1],'j'});//testcase 18
                Stmt();
                if(now == "else"){
                    next();
                    Stmt();
                }
            }
        }
    }
    else if(now == "for"){
        next();
        if(now == "("){
            next();
            if(now != ";"){
                ForStmt();
            }
            if(now == ";"){
                next();
                if(now != ";"){
                    Cond();
                }
                if(now == ";"){
                    next();
                    if(now != ")"){
                        ForStmt();
                    }
                    if(now == ")"){
                        next();
                        Stmt();
                    }
                }
            }
        }
    }
    else if(now == "break" || now == "continue"){
        next();
        if(now == ";") next();
        else err.push_back({lineNumber[Index-1],'i'});//0
    }
    else if(now == "return"){
        next();
        if(now != ";"){
            Exp();
        }
        if(now == ";") next();
        else err.push_back({lineNumber[Index-1],'i'});//testcase 16
    }
    else if(now == "printf"){
        next();
        if(now == "("){
            next();
            if(word[Index].first == "STRCON"){
                next();
                while(now == ","){
                    next();
                    Exp();
                }
                if(now == ")"){
                    next();
                    if(now == ";") next();
                    else err.push_back({lineNumber[Index-1],'i'});//testcase 16
                }
                else {
                    err.push_back({lineNumber[Index-1],'j'});//testcase 20
                    if(now == ";") next();
                    //else err.push_back({lineNumber[Index-1],'i'});
                }
            }
        }
    }
    else if(now == "{") Block();
    else if(now == ";") next();
    else{//预读到 ;之前的=
        j = Index;
        flag = 0;
        while(word[j].second != ";"){
            if(word[j].second == "="){
                flag = 1;
                break;
            }
            j++;
        }
        if(!flag){
            Exp();
            if(now == ";") next();
            else err.push_back({lineNumber[Index-1],'i'});
        }
        else if(word[j+1].second == "getint" || word[j+1].second == "getchar"){
            LVal();
            if(now == "="){
                next();
                next();
                if(now == "("){
                    next();
                    if(now == ")"){
                        next();
                        if(now == ";") next();
                        else{
                            err.push_back({lineNumber[Index-1],'i'});
                        }
                    }
                    else{
                        err.push_back({lineNumber[Index-1],'j'});
                        if(now == ";") next();
                        else{
                            err.push_back({lineNumber[Index-1],'i'});
                        }
                    }
                }
            }
        }
        else{
            LVal();
            if(now == "="){
                next();
                Exp();
                if(now == ";") next();
                else err.push_back({lineNumber[Index-1],'i'});
            }
        }

void UnaryExp(){
    if(isIdent1()&&preRead == "("){
        next();

        if(now == ")") {
            next();
        }
        else err.push_back({lineNumber[Index-1],'j'});
    }
    else if(isUnaryOp()){
        next();
        res.push_back("<UnaryOp>");
        UnaryExp();
    }
    else PrimaryExp();
    res.push_back("<UnaryExp>");
}
