    private CompUnitNode compUnit() {
        // CompUnit -> {Decl} {FuncDef} MainFuncDef
        List<DeclNode> declNodes = new ArrayList<>();
        List<FuncDefNode> funcDefNodes = new ArrayList<>();
        MainFuncDefNode mainFuncDefNode = null;
        // Decl
        while ((!tokens.get(index + 1).getType().equals(TokenType.MAINTK)) && (!tokens.get(index + 2).getType().equals(TokenType.LPARENT))) { // @@ The condition uses hardcoded indices (index+1, index+2) which can cause IndexOutOfBoundsException when the token list is short, and the logic is incorrect for distinguishing Decl from FuncDef/MainFuncDef. It should check the current token and look ahead properly based on the grammar.
            // 下一个不是main也不是(，就保证了不是函数定义也不是主函数
            DeclNode decl = Decl();
            declNodes.add(decl);
        }
        // FuncDef
        while ((!tokens.get(index + 1).getType().equals(TokenType.MAINTK))) { // @@ Same issue: using index+1 without bounds checking, and the condition is insufficient to correctly identify FuncDef (e.g., it doesn't check for function type keywords like 'void', 'int', 'char').
            // 没有Decl之后，下一个不是main就是普通函数定义
            FuncDefNode funcDefNode = FuncDef();
            funcDefNodes.add(funcDefNode);
        }
        // MainFuncDef
        mainFuncDefNode = MainFuncDef();
        return new CompUnitNode(declNodes, funcDefNodes, mainFuncDefNode);
    }
