    /**
     * CompUnit → {Decl} {FuncDef} MainFuncDef
     *
     * @return Node
     */
    private Node CompUnit() {
        GComponent currentComponent = GComponent.CompUnit;
        NLNode result = new NLNode(currentComponent);

        while (!peek(2).getType().equals(TokenType.LPARENT)) { // @@ [Incorrect lookahead condition: using peek(2) to detect Decl is unreliable because Decl starts with 'const', 'int', or 'char', not necessarily followed by '(' at offset 2. This causes premature termination or infinite loops.]
            //CompUnit → {Decl}
            Node tmp = Decl();
            result.addChild(tmp);
        }

        while (!peek(1).getType().equals(TokenType.MAINTK)) { // @@ [Incorrect lookahead condition: FuncDef starts with FuncType ('void'/'int'/'char'), but this check assumes the next token after FuncDef is MAINTK, which fails when there are multiple FuncDefs or when MainFuncDef hasn't started yet. Should check current token for FuncType instead.]
            //CompUnit → {FuncDef}
            Node tmp = FuncDef();
            result.addChild(tmp);
        }
        //CompUnit → MainFuncDef
        Node mustNode = MainFuncDef();
        result.addChild(mustNode);



        return result;
    }
