    /**
     * CompUnit → {Decl} {FuncDef} MainFuncDef
     *
     * @return Node
     */
    private Node CompUnit() {
        GComponent currentComponent = GComponent.CompUnit;
        NLNode result = new NLNode(currentComponent);

        while (!peek(2).getType().equals(TokenType.LPARENT)) { // @@ [Incorrect lookahead condition: using peek(2) to decide Decl parsing is not aligned with grammar; Decl can appear as long as the token is 'const', 'int', or 'char', regardless of what comes two tokens ahead]
            //CompUnit → {Decl}
            Node tmp = Decl();
            result.addChild(tmp);
        }

        while (!peek(1).getType().equals(TokenType.MAINTK)) { // @@ [Incorrect lookahead condition: FuncDef parsing should continue while valid function definitions exist, not just until MAINTK appears in the next token; this fails when there are no FuncDefs or when FuncDef structure doesn't match this arbitrary lookahead]
            //CompUnit → {FuncDef}
            Node tmp = FuncDef();
            result.addChild(tmp);
        }
        //CompUnit → MainFuncDef
        Node mustNode = MainFuncDef();
        result.addChild(mustNode);



        return result;
    }
