    /**
     * CompUnit → {Decl} {FuncDef} MainFuncDef
     *
     * @return Node
     */
    private Node CompUnit() {
        GComponent currentComponent = GComponent.CompUnit;
        NLNode result = new NLNode(currentComponent);

        while (!peek(2).getType().equals(TokenType.LPARENT)) { // @@ [Incorrect lookahead condition: using peek(2) to detect Decl is unreliable because it assumes the third token ahead is LPARENT, but Decl starts with CONSTTK/INTTK/CHARTK and doesn't necessarily relate to LPARENT two tokens ahead. This may cause premature termination or infinite loop.]
            //CompUnit → {Decl}
            Node tmp = Decl();
            result.addChild(tmp);
        }

        while (!peek(1).getType().equals(TokenType.MAINTK)) { // @@ [Incorrect lookahead condition: using peek(1) to detect FuncDef termination is flawed because MAINTK may not appear immediately after FuncDef; also, if there are no FuncDefs, this condition might incorrectly try to parse a FuncDef when next token isn't MAINTK but something else valid like RBRACE or EOF.]
            //CompUnit → {FuncDef}
            Node tmp = FuncDef();
            result.addChild(tmp);
        }
        //CompUnit → MainFuncDef
        Node mustNode = MainFuncDef();
        result.addChild(mustNode);



        return result;
    }
