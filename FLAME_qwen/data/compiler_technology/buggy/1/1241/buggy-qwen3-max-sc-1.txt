    private Node CompUnit() {
        GComponent currentComponent = GComponent.CompUnit;
        NLNode result = new NLNode(currentComponent);

        while (!peek(2).getType().equals(TokenType.LPARENT)) { // @@ [Incorrect lookahead condition: using peek(2) to detect Decl is unreliable and doesn't match grammar rules; Decl starts with 'const', 'int', or 'char', not based on LPARENT two tokens ahead]

            Node tmp = Decl();
            result.addChild(tmp);
        }

        while (!peek(1).getType().equals(TokenType.MAINTK)) { // @@ [Incorrect lookahead condition: FuncDef parsing should continue until MainFuncDef is found, but checking only next token for MAINTK may skip valid FuncDefs or include invalid ones; also fails if tokens are exhausted]
            //CompUnit → {FuncDef}

        Node mustNode = MainFuncDef();
        result.addChild(mustNode);



        return result;
