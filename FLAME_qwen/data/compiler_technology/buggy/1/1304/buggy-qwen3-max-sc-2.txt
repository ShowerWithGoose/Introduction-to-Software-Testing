    private CompUnitNode parseCompUnit() {
        //Decl
        ArrayList<DeclNode> declNodes = new ArrayList<>();
        while (peek(2) != TokenType.LPARENT) { @@ [Incorrect condition for detecting Decl vs FuncDef: peek(2) may go out of bounds and doesn't correctly identify function definitions which start with FuncType (VOIDTK/INTTK/CHARTK) followed by IDENFR and LPARENT. This can cause misclassification of FuncDef as Decl or infinite loops.]
            declNodes.add(parseDecl());
        }

        //FuncDef
        ArrayList<FuncDefNode> funcDefNodes = new ArrayList<>();
        while (peek(1) != TokenType.MAINTK) { @@ [Incorrect condition for detecting FuncDef vs MainFuncDef: peek(1) assumes the next token after current position determines main function, but doesn't account for proper function definition structure. Should check current token type instead.]
            funcDefNodes.add(parseFuncDef());
        }

        //MainFuncDef
        MainFuncDefNode mainFuncDefNode = parseMainFuncDef();

        return new CompUnitNode(declNodes, funcDefNodes, mainFuncDefNode);
    }
