    //<CompUnit> ::= {<Decl>} {<FuncDef>} <MainFuncDef>
    //<Decl>:           [const] int/char <Ident>
    //<FuncDef>:        void/int/char func()
    //<MainFuncDef>:    int main()
    private CompUnitNode parseCompUnit() {
        //Decl
        ArrayList<DeclNode> declNodes = new ArrayList<>();
        while (peek(2) != TokenType.LPARENT) { // @@ [This condition is incorrect because it assumes that any declaration must be followed by two tokens before a '(' appears, but this doesn't properly distinguish between Decl and FuncDef. A VarDecl like "int x;" has no '(' at all, and the lookahead logic fails to handle the grammar structure correctly.]
            declNodes.add(parseDecl());
        }

        //FuncDef
        ArrayList<FuncDefNode> funcDefNodes = new ArrayList<>();
        while (peek(1) != TokenType.MAINTK) { // @@ [This condition is also flawed because it assumes that FuncDef continues until MAINTK is found one token ahead, but this doesn't account for cases where there are no FuncDefs or when the token stream doesn't align with this expectation. The proper approach should check the current token type to decide what to parse next.]
            funcDefNodes.add(parseFuncDef());
        }

        //MainFuncDef
        MainFuncDefNode mainFuncDefNode = parseMainFuncDef();

        return new CompUnitNode(declNodes, funcDefNodes, mainFuncDefNode);
    }
