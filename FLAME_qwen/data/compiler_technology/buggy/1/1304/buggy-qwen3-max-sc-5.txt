    private CompUnitNode parseCompUnit() {
        //Decl
        ArrayList<DeclNode> declNodes = new ArrayList<>();
        while (peek(2) != TokenType.LPARENT) { @@ [Incorrect condition for distinguishing Decl from FuncDef. Decl starts with 'const', 'int', or 'char', but FuncDef can also start with 'int'/'char' followed by identifier and '(', so peek(2) check is unreliable and will cause misclassification]
            declNodes.add(parseDecl());
        }

        //FuncDef
        ArrayList<FuncDefNode> funcDefNodes = new ArrayList<>();
        while (peek(1) != TokenType.MAINTK) { @@ [Incorrect condition for distinguishing FuncDef from MainFuncDef. Should check for function signature pattern rather than just next token not being MAINTK]
            funcDefNodes.add(parseFuncDef());
        }

        //MainFuncDef
        MainFuncDefNode mainFuncDefNode = parseMainFuncDef();

        return new CompUnitNode(declNodes, funcDefNodes, mainFuncDefNode);
    }
