    private void parseInitVal() {
        if (curToken.getCategory() == Category.STRCON) {
            skipTokens(1);
        } else if (curToken.getCategory() == Category.LBRACE) {
            skipTokens(1); // skip {
            parseExp(); // @@ The grammar rule InitVal → '{' [ Exp { ',' Exp } ] '}' specifies that the expression inside curly braces is optional. Unconditionally calling parseExp() here will cause a parsing error or incorrect output when encountering an empty initializer list like '{}'.
            while (curToken.getCategory() == Category.COMMA) {
                skipTokens(1); // skip ,
                parseExp();
            }
            skipTokens(1); // skip }
        } else {
            parseExp();
        }
        outputList.add("<InitVal>\n");
    }
