        private Exp parseExp(boolean isLval) {
            Exp exp = new Exp();
            AddExp addExp = parseAddExp(false, isLval); // @@ The first argument 'false' incorrectly disables the '!' (NOT) operator for expressions, but the grammar allows UnaryOp (including '!') in Exp. It should be 'true'.
            exp.setAddExp(addExp);

            this.parseOut.add(exp.returnSymbolType());

            return exp;
        }

        private OpUnaryExp parseOpUnaryExp(boolean type) {
            ArrayList<String> ops = new ArrayList<>();
            while (judgeUnaryOpFirst(curToken, type)) {
                ops.add(parseUnaryOp().get(0));
            }
            UnaryExp unaryExp = parseUnaryExp(type, false);


            OpUnaryExp opUnaryExp = new OpUnaryExp(ops, unaryExp);

            for (int i = 0; i < ops.size() - 1; i++) { // @@ The loop condition 'ops.size() - 1' incorrectly skips outputting '<UnaryExp>' for the outermost unary operation. Each unary operator application should output '<UnaryExp>', so the condition should be 'i < ops.size()'.
                this.parseOut.add(opUnaryExp.returnSymbolType());
            }

            return opUnaryExp;
        }
