    private void getNextToken() {
        if (buffer.isEmpty()) {
            if (lexer.getArriveEnd()) {
                Token token = new Token(LexType.ISEND, "error", -1);
                oldToken = curToken;
                curToken = token;
            } else {
                lexer.next();
                oldToken = curToken;
                curToken = lexer.getNewToken();
            }
        } else {
            oldToken = curToken;
            curToken = buffer.poll();
        }
        this.parseOut.add(oldToken.getLexType() + " " + oldToken.getName());
    }
