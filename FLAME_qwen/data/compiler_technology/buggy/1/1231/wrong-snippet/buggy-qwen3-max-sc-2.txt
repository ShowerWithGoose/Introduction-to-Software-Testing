    public static boolean handleIError(TokenIterator tokens, ArrayList<SyntaxNode> children) {
        Token token;
        token = tokens.next();
        // handle I_MISSING_SEMICOLON
        if (token.getType() != TokenType.SEMICN) {
            tokens.stepBack(2);
            token = tokens.next();
            children.add(new Token(TokenType.SEMICN, ";", token.getLineNum(), 0));
            ErrorList.addError(MyErrorType.I_MISSING_SEMICOLON, token.getLineNum());
            return false;
        } else {
            children.add(token);
            return true;
        }
    }
