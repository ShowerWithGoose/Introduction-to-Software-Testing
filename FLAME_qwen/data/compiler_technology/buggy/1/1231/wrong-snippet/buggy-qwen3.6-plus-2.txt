    @Override
    public boolean parse() {
        token = tokens.next();
        if (token.getType().equals(TokenType.SEMICN)) {
            return true;
        } else {
            tokens.stepBack(2);
            Token tokenBefore = tokens.next();
            ErrorList.addError(MyErrorType.I_MISSING_SEMICOLON, tokenBefore.getLineNum());
            token = new Token(TokenType.SEMICN, ";", tokenBefore.getLineNum(), 0);
            return false;
        }
    }
