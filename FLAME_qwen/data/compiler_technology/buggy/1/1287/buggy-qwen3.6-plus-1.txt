    public Node Block() {
        Element ret = new Element(ParseType.Block);
        if (!nowRead().equals(LexType.LBRACE)) {
            error();
        }
        else {
            ret.addNode(tokens.get(pos++));
            while (isDeclFirst(nowRead()) || isStmtFisrt(nowRead())) { // @@ Missing boundary check for pos < tokens.size() before calling nowRead(), which can cause IndexOutOfBoundsException if the token stream ends unexpectedly.
                if (isBType(nowRead())) {
                    ret.addNode(VarDecl());
                }
                else if (nowRead().equals(LexType.CONSTTK)) {
                    ret.addNode(ConstDecl());
                }
                else {
                    ret.addNode(Stmt());
                }
            }
            if (!nowRead().equals(LexType.RBRACE)) {
                error();
            }
            else {
                ret.addNode(tokens.get(pos++));
            }
        }
        return ret;
    }
