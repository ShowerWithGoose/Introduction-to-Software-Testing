    public Node handleCompUnit() {
        Node node = new Node(GrammarType.COMPUNIT);
        while (true) {
            if (tokens.preview(1).getLexType() == LexType.MAINTK) {  // potential NullPointerException if preview(1) is null
                node.add(handleMainFuncDef());
                break;
            } else if (tokens.preview(2).getLexType() == LexType.LPARENT) {  // potential NullPointerException if preview(2) is null
                node.add(handleFuncDef());
            } else {
                node.add(handleDecl());
            }
        }
        return node;
    }
