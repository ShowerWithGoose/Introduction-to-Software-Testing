    public Node parseCompUnit() {
        ArrayList<Node> children = new ArrayList<>();
        int startLine = curToken.getLineNumber();
        Node node = null;
        while (true) {
            if (curToken == null) {
                break;
            }
            // parse MainFunDef
            else if (tokenStream.look(1).getType() == TokenType.MAINTK) { // @@ Syntax error: 'else if' cannot follow a closed 'if' block. The preceding 'if' statement ends with a closing brace and 'break;', leaving this 'else' without a matching 'if'.
                node = parseMainFuncDef();
            }
            // parse FuncDef
            else if (tokenStream.look(2).getType() == TokenType.LPARENT) {
                node = parseFuncDef();
            }
            // parse ConstDecl
            else if (curToken.getType() == TokenType.CONSTTK) {
                node = parseConstDecl();
            }
            // parse VarDecl
            else if (curToken.getType() == TokenType.INTTK || curToken.getType() == TokenType.CHARTK) {
                node = parseVarDecl();
            }
            else break;
            children.add(node);
        }
