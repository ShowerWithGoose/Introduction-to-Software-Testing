    public int getPrevTokenLineNum() {
        if (curTokenIndex == 0) {
            int lineNum = curLineNum - 1;
            while (lexerTokens.containsKey(lineNum) && lexerTokens.get(lineNum).isEmpty()) {
                lineNum--;
            }
            return lineNum;
        } else {
            return curLineNum;
        }
    }

    public void checkSEMICN() {
        if (peek().getType() != Token.TokenType.SEMICN) {
