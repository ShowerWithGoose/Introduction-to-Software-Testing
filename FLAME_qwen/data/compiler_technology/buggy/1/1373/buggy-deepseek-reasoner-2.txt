    public SyntaxTreeNode parseCompUnit() {
        SyntaxTreeNode compUnitNode = new SyntaxTreeNode(SyntaxType.CompUnit, null);

        // 解析零个或多个 Decl
        while (currentTokenIndex < tokens.size()) {
            Token currentToken = tokens.get(currentTokenIndex);

            if (currentToken.getType().equals("CONSTTK")
                    || currentToken.getType().equals("INTTK") && !getNext2TokeType().equals("LPARENT")
                    || currentToken.getType().equals("CHARTK") && !getNext2TokeType().equals("LPARENT")) { // @@ [Condition for CHARTK in Decl uses !getNext2TokeType().equals("LPARENT") without checking next token is identifier; also lacks getNextTokeType() safeguard; may misclassify function definitions]
                compUnitNode.addChild(parseDecl());
            } else {
                break;
            }
        }

        // 解析零个或多个 FuncDef
        while (currentTokenIndex < tokens.size()) {
            Token currentToken = tokens.get(currentTokenIndex);

            if (currentToken.getType().equals("VOIDTK")
                    || currentToken.getType().equals("INTTK") && getNextTokeType().equals("IDENFR") && getNext2TokeType().equals("LPARENT")
                    || currentToken.getType().equals("CHARTK") && getNext2TokeType().equals("LPARENT")) { // @@ [Condition for CHARTK in FuncDef only checks getNext2TokeType() but does not verify that the next token is an identifier (IDENFR); inconsistent with INTTK handling; may cause parsing errors]
                compUnitNode.addChild(parseFuncDef());
            } else {
                break;
            }
        }

        // 解析 MainFuncDef
        compUnitNode.addChild(parseMainFuncDef());

        return compUnitNode;
    }
