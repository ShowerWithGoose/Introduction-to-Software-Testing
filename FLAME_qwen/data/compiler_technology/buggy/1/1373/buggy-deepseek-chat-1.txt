    public SyntaxTreeNode parseCompUnit() {
        SyntaxTreeNode compUnitNode = new SyntaxTreeNode(SyntaxType.CompUnit, null);

        // 解析零个或多个 Decl
        while (currentTokenIndex < tokens.size()) {
            Token currentToken = tokens.get(currentTokenIndex);

            if (currentToken.getType().equals("CONSTTK")
                    || currentToken.getType().equals("INTTK") && !getNext2TokeType().equals("LPARENT")
                    || currentToken.getType().equals("CHARTK") && !getNext2TokeType().equals("LPARENT")) { // @@ [Error: The condition for CHARTK uses getNext2TokeType() instead of getNextTokeType() to check if the next token is IDENFR, which is inconsistent with the logic for INTTK and may cause incorrect parsing of char function definitions.]
                compUnitNode.addChild(parseDecl());
            } else {
                break;
            }
        }

        // 解析零个或多个 FuncDef
        while (currentTokenIndex < tokens.size()) {
            Token currentToken = tokens.get(currentTokenIndex);

            if (currentToken.getType().equals("VOIDTK")
                    || currentToken.getType().equals("INTTK") && getNextTokeType().equals("IDENFR") && getNext2TokeType().equals("LPARENT")
                    || currentToken.getType().equals("CHARTK") && getNext2TokeType().equals("LPARENT")) { // @@ [Error: The condition for CHARTK uses getNext2TokeType() directly to check for LPARENT, skipping the check for IDENFR. This is inconsistent with the INTTK case and may incorrectly treat char variable declarations as function definitions.]
                compUnitNode.addChild(parseFuncDef());
            } else {
                break;
            }
        }

        // 解析 MainFuncDef
        compUnitNode.addChild(parseMainFuncDef());

        return compUnitNode;
    }
