    private void CompUnit() {
        // CompUnit → {Decl} {FuncDef} MainFuncDef
        if(isEOF()) { return; }
        // {Decl}
        while(true) {
            if (peekSymbol(1).getType() == SymbolType.CONSTTK) {
                Decl();
            } else if(peekSymbol(3).getType() != SymbolType.LPARENT && !isEOF()) { // @@ [Incorrect lookahead logic: checking peekSymbol(3) without ensuring there are at least 3 tokens ahead can cause IndexOutOfBoundsException or wrong parsing decisions. Also, the condition for identifying VarDecl is flawed since BType starts with INTTK/CHARTK, not determined by absence of LPARENT at position 3]
                Decl();
            } else {
                break;
            }
        }
        // {FuncDef}
        while(true) {
            if(peekSymbol(2).getType() != SymbolType.MAINTK && !isEOF()) { // @@ [Incorrect lookahead logic: checking peekSymbol(2) without proper bounds and using negative condition to detect FuncDef is unreliable. FuncDef starts with FuncType (VOIDTK/INTTK/CHARTK), but this check may incorrectly include or exclude valid cases]
                FuncDef();
            } else {
                break;
            }
        }
        //MainFuncDef
        MainFuncDef();
        if(shouldPrint) System.out.println("<CompUnit>");
    }
