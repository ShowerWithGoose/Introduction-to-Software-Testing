    private void CompUnit() {
        // CompUnit → {Decl} {FuncDef} MainFuncDef
        if(isEOF()) { return; }
        // {Decl}
        while(true) {
            if (peekSymbol(1).getType() == SymbolType.CONSTTK) {
                Decl();
            } else if(peekSymbol(3).getType() != SymbolType.LPARENT && !isEOF()) { // @@ [Incorrect lookahead logic: peekSymbol(3) may go out of bounds and doesn't properly distinguish between VarDecl and FuncDef/MainFuncDef since both can start with 'int' or 'char']
                Decl();
            } else {
                break;
            }
        }
        // {FuncDef}
        while(true) {
            if(peekSymbol(2).getType() != SymbolType.MAINTK && !isEOF()) { // @@ [Incorrect lookahead logic: peekSymbol(2) assumes the next token after identifier is LPARENT for FuncDef, but doesn't verify the first token is actually a valid FuncType followed by IDENFR; also fails to handle case where MainFuncDef appears]
                FuncDef();
            } else {
                break;
            }
        }
        //MainFuncDef
        MainFuncDef();
        if(shouldPrint) System.out.println("<CompUnit>");
    }
