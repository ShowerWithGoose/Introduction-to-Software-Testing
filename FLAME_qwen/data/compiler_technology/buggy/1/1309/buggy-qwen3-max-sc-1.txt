    private void CompUnit() {
        // CompUnit → {Decl} {FuncDef} MainFuncDef
        if(isEOF()) { return; }
        // {Decl}
        while(true) {
            if (peekSymbol(1).getType() == SymbolType.CONSTTK) {
                Decl();
            } else if(peekSymbol(3).getType() != SymbolType.LPARENT && !isEOF()) { @@ [Incorrect lookahead logic: peekSymbol(3) may go out of bounds and the condition doesn't properly check for VarDecl which starts with INTTK/CHARTK followed by IDENFR, not based on LPARENT presence]
                Decl();
            } else {
                break;
            }
        }
        // {FuncDef}
        while(true) {
            if(peekSymbol(2).getType() != SymbolType.MAINTK && !isEOF()) { @@ [Incorrect lookahead logic: peekSymbol(2) may go out of bounds and the condition doesn't properly identify FuncDef which starts with FuncType (VOIDTK/INTTK/CHARTK) followed by IDENFR and LPARENT, not just absence of MAINTK at position 2]
                FuncDef();
            } else {
                break;
            }
        }
        //MainFuncDef
        MainFuncDef();
        if(shouldPrint) System.out.println("<CompUnit>");
    }
