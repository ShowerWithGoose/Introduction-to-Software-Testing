    public FuncDef parseFuncDef() {
        FuncDef funcDef = new FuncDef(tokens.next(), tokens.next());
        funcDef.setLparent(tokens.next());//(
        if (!tokens.peek().is(TokenType.RPARENT)) { // @@ [Missing null check for tokens.peek(), which may cause NullPointerException if the token stream ends unexpectedly before finding a right parenthesis or brace]
            if (tokens.peek().is(TokenType.LBRACE)) {
                //error j
                addError("j");
                funcDef.setBlock(parseBlock());
                return funcDef;
            } else {
                funcDef.setFuncFParams(parseFuncFParams());
            }
        }
        //error j
        if (!tokens.peek().is(TokenType.RPARENT)) { // @@ [Missing null check for tokens.peek(), which may cause NullPointerException if the token stream ends unexpectedly]
            addError("j");
        } else {
            funcDef.setRparent(tokens.next());
        }
        funcDef.setBlock(parseBlock());
        return funcDef;
