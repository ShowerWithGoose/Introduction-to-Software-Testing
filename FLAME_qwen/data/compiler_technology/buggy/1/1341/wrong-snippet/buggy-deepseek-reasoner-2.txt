    public void parse() {
        //int a
        //const int a
        //int a()
        while (!tokens.peek(2).is(TokenType.LPARENT)) {
            if (tokens.peek().is(TokenType.CONSTTK)) {
                compUnit.addDecl(parseConstDecl());
            } else {
                compUnit.addDecl(parseVarDecl());
            }
        }
        while (tokens.peek() != null) {
            compUnit.addFuncDef(parseFuncDef());
        }
    }
