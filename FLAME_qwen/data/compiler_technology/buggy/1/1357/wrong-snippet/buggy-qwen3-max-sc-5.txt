    public CompUnit parseCompUnit() {
        lexer.next();

        ArrayList<Decl> decls = new ArrayList<>();
        boolean hasDecl;
        do {
            hasDecl = false;
            if (lexer.getLexType() == LexType.CONSTTK) {
                decls.add(parseDecl());
                hasDecl = true;
            } else {
                lexer.nextNoPrint();
                lexer.nextNoPrint();
                if (lexer.getLexType() != LexType.LPARENT) {
                    lexer.prev();
                    lexer.prev();
                    decls.add(parseDecl());
                    hasDecl = true;
                } else {
                    lexer.prev();
                    lexer.prev();
                }
            }
        } while (hasDecl);

        ArrayList<FuncDef> funcDefs = new ArrayList<>();
        boolean hasFuncDef;
        do {
            hasFuncDef = false;
            lexer.nextNoPrint();
            if (lexer.getLexType() != LexType.MAINTK) {
                lexer.prev();
                funcDefs.add(parseFuncDef());
                hasFuncDef = true;
            } else {
                lexer.prev();
            }
        } while (hasFuncDef);

        MainFuncDef mainFuncDef = parseMainFuncDef();
        return new CompUnit(decls.toArray(new Decl[0]), funcDefs.toArray(new FuncDef[0]), mainFuncDef);
    }
