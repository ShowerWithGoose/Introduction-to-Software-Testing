    public TreeNode Stmt() {
        TreeNode stmt = new TreeNode(ParsType.Stmt,new ArrayList<>());
        if (isMatch(LexType.IFTK)) {
            addLeaf(stmt);
            if (isMatch(LexType.LPARENT)) {
                addLeaf(stmt);
            }
            stmt.addNode(Cond());
            RparentJudge(stmt); // @@ [If RPARENT is missing, error 'j' is added but token is not consumed, causing potential infinite loop or incorrect subsequent parsing]
            stmt.addNode(Stmt());
            if (isMatch(LexType.ELSETK)) {
                addLeaf(stmt);
                stmt.addNode(Stmt());
            }
        }
        else if (isMatch(LexType.FORTK)) {
            addLeaf(stmt);
            if (isMatch(LexType.LPARENT)) {
                addLeaf(stmt);
            }
            if (isMatch(LexType.SEMICN)) {
                addLeaf(stmt);
            }
            else {
                stmt.addNode(ForStmt());
                addLeaf(stmt);
            }
            if (isMatch(LexType.SEMICN)) {
                addLeaf(stmt);
            }
            else {
                stmt.addNode(Cond());
                addLeaf(stmt);
            }
            if (isMatch(LexType.RPARENT)) {
                addLeaf(stmt);
            }
            else {
                stmt.addNode(ForStmt());
                addLeaf(stmt); // @@ [This addLeaf(stmt) consumes a token that may not be RPARENT; if RPARENT is missing, this incorrectly consumes the next token as SEMICN or something else, breaking parsing]
            }
            stmt.addNode(Stmt());
        }
        else if (isMatch(LexType.BREAKTK) || isMatch(LexType.CONTINUETK)) {
            addLeaf(stmt);
            SemicnJudge(stmt); // @@ [If SEMICN is missing, error 'i' is recorded but no token is consumed, which may cause the parser to get stuck on the same token]
        }
        else if (isMatch(LexType.RETURNTK)) {
            addLeaf(stmt);
            if (isMatch(LexType.SEMICN)) {
                addLeaf(stmt);
            }
            else if (isExp()){
                stmt.addNode(Exp());
                SemicnJudge(stmt); // @@ [Same issue: missing SEMICN leads to error 'i' but no token consumption, risking infinite loop]
            }
            else {
                SemicnJudge(stmt); // @@ [Same issue: missing SEMICN leads to error 'i' but no token consumption]
            }
        }
        else if (isMatch(LexType.PRINTFTK)) {
            addLeaf(stmt);
            addLeaf(stmt);
            addLeaf(stmt);
            while (isMatch(LexType.COMMA)) {
                addLeaf(stmt);
                stmt.addNode(Exp());
            }
            RparentJudge(stmt); // @@ [Missing RPARENT triggers error 'j' but doesn't consume a token, possibly causing repeated errors or hang]
            SemicnJudge(stmt); // @@ [Missing SEMICN triggers error 'i' but doesn't consume a token]
        }
        else if (isMatch(LexType.LBRACE)) {
            stmt.addNode(Block());
        }
        else {
            if (isMatch(LexType.IDENFR)) {
                // 回溯
                LexType preType = type;
                int preline = lexer.getlineNUM();
                int lexpre = lexer.getCurPos();
                int pre = curPos;
                int outSize = outputTable.size();
                int errSize = errorTable.size();
                int tokenSize = tokens.size();
                TreeNode lval = LVal(); //叶子节点在lval
                // LVal
                if (isMatch(LexType.ASSIGN)) {
                    stmt.addNode(lval);
                    addLeaf(stmt);
                    if (isMatch(LexType.GETINTTK) || isMatch(LexType.GETCHARTK)) {
                        addLeaf(stmt);
                        addLeaf(stmt);
                        RparentJudge(stmt); // @@ [Missing RPARENT in getchar()/getint() call triggers error 'j' but doesn't advance token stream]
                        SemicnJudge(stmt); // @@ [Missing SEMICN triggers error 'i' without consuming token]
                    }
                    else {
                        stmt.addNode(Exp());
                        SemicnJudge(stmt); // @@ [Missing SEMICN triggers error 'i' without consuming token]
                    }
                }
                // Exp
                else {
                    lexer.setLineNum(preline);
                    type = preType;
                    lexer.reStart(lexpre);
                    curPos = pre;
                    reStart(outputTable,outSize);
                    reStart(errorTable,errSize);
                    reStartT(tokens,tokenSize);
                    stmt.addNode(Exp());
                    SemicnJudge(stmt); // @@ [Missing SEMICN triggers error 'i' without consuming token]
                }
            }
            else {
                if (isMatch(LexType.SEMICN)) {
                    addLeaf(stmt);
                }
                else {
                    stmt.addNode(Exp());
                    SemicnJudge(stmt); // @@ [Missing SEMICN triggers error 'i' without consuming token, and if current token isn't SEMICN or valid Exp start, may cause cascading errors]
                }
            }
        }
        addOut("<Stmt>");
        return stmt;
    }
