    public CompUnit parseCompUnit() {
        ArrayList<Decl> decls = new ArrayList<>();
        ArrayList<FuncDef> funcDefs = new ArrayList<>();
        MainFuncDef mainFuncDef = null;

        while (true) {
            if (curToken == null) {
                break;
            } else if (movedTkEquals(1, TokenType.MAINTK)) { // @@ [Incorrect lookahead logic: checking if next token is MAINTK without verifying current token is INTTK, which violates MainFuncDef → 'int' 'main' '(' ')' Block grammar rule]
                mainFuncDef = parseMainFuncDef();
                break;
            } else if (movedTkEquals(2, TokenType.LPARENT)) { // @@ [Flawed function detection: using movedTkEquals(2, LPARENT) to identify FuncDef can misclassify VarDecl or ConstDecl as FuncDef when they have array declarations with brackets followed by parentheses-like patterns]
                funcDefs.add(parseFuncDef());
            } else if (curEquals(TokenType.CONSTTK) ||
                    curEquals(TokenType.INTTK) ||
                    curEquals(TokenType.CHARTK)) {
                decls.add(parseDecl());
            } else {
                break;
            }
        }

        return new CompUnit(decls, funcDefs, mainFuncDef);
    }
