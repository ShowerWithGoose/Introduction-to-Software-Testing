    public CompUnit parseCompUnit() {
        ArrayList<Decl> decls = new ArrayList<>();
        ArrayList<FuncDef> funcDefs = new ArrayList<>();
        MainFuncDef mainFuncDef = null;

        while (true) {
            if (curToken == null) {
                break;
            } else if (movedTkEquals(1, TokenType.MAINTK)) { // @@ [Incorrect lookahead logic: checking if next token is MAINTK without verifying current token is INTTK, which violates MainFuncDef grammar 'int' 'main' '(' ')' Block]
                mainFuncDef = parseMainFuncDef();
                break;
            } else if (movedTkEquals(2, TokenType.LPARENT)) { // @@ [Flawed lookahead: assumes any sequence with LPARENT at offset 2 is a FuncDef, but doesn't verify proper FuncType and Ident structure, leading to misclassification of declarations as function definitions]
                funcDefs.add(parseFuncDef());
            } else if (curEquals(TokenType.CONSTTK) ||
                    curEquals(TokenType.INTTK) ||
                    curEquals(TokenType.CHARTK)) {
                decls.add(parseDecl());
            } else {
                break;
            }
        }

        return new CompUnit(decls, funcDefs, mainFuncDef);
    }
