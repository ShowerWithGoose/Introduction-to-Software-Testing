    public CompUnit parseCompUnit() {
        ArrayList<Decl> decls = new ArrayList<>();
        ArrayList<FuncDef> funcDefs = new ArrayList<>();
        MainFuncDef mainFuncDef = null;

        while (true) {
            if (curToken == null) {
                break;
            } else if (movedTkEquals(1, TokenType.MAINTK)) { // @@ [Incorrect lookahead logic: checking if next token is MAINTK assumes current token is INTTK, but doesn't verify this. This can misidentify other constructs as MainFuncDef.]
                mainFuncDef = parseMainFuncDef();
                break;
            } else if (movedTkEquals(2, TokenType.LPARENT)) { // @@ [Fragile lookahead: assuming FuncDef based on LPARENT two tokens ahead doesn't account for all valid function definitions and may incorrectly classify VarDecl as FuncDef.]
                funcDefs.add(parseFuncDef());
            } else if (curEquals(TokenType.CONSTTK) ||
                    curEquals(TokenType.INTTK) ||
                    curEquals(TokenType.CHARTK)) {
                decls.add(parseDecl());
            } else {
                break;
            }
        }

        return new CompUnit(decls, funcDefs, mainFuncDef);
    }
