    public CompUnit parseCompUnit() {
        ArrayList<Decl> decls = new ArrayList<>();
        ArrayList<FuncDef> funcDefs = new ArrayList<>();
        MainFuncDef mainFuncDef = null;

        while (true) {
            if (curToken == null) {
                break;
            } else if (movedTkEquals(1, TokenType.MAINTK)) { // @@ [Incorrect lookahead: checking if next token is MAINTK, but should check current token is INTTK and next is MAINTK for MainFuncDef]
                mainFuncDef = parseMainFuncDef();
                break;
            } else if (movedTkEquals(2, TokenType.LPARENT)) { // @@ [Unreliable lookahead: function detection based on token 2 positions ahead may fail with complex declarations or empty input]
                funcDefs.add(parseFuncDef());
            } else if (curEquals(TokenType.CONSTTK) ||
                    curEquals(TokenType.INTTK) ||
                    curEquals(TokenType.CHARTK)) {
                decls.add(parseDecl());
            } else {
                break;
            }
        }

        return new CompUnit(decls, funcDefs, mainFuncDef);
    }
