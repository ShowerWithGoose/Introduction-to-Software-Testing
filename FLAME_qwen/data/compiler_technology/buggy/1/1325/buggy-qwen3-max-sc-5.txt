    public ParsedTree parseVarDecl() throws IOException {
        //VarDecl → BType VarDef { ',' VarDef } ';' // i
        LinkedList<ParsedTree> subTree = new LinkedList<>();
        peekAndAdd(subTree, INTTK, CONSTTK); // @@ [BType is 'int' or 'char', but CONSTTK is not part of BType; this incorrectly allows 'const' as a type in VarDecl, which violates the grammar]
        subTree.add(parseVarDef());
        while (peekToken(COMMA)) {
            subTree.add(curToken);
            subTree.add(parseVarDef());
        }
        if (peekToken(SEMICN)){
            subTree.add(curToken);
        } else {
            addError("i");
        }
        return new ParsedTree("VarDecl", subTree);
    }
