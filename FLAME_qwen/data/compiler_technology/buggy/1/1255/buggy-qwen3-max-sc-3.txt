    public CompUnit parseCompUnit() {
        CompUnit compUnit = new CompUnit();
        MainFuncDef mainFuncDef;
        getNextToken(); // @@ [This consumes the first token but doesn't check if it's valid; may skip necessary tokens or cause misalignment]
        while (true) {
            Token token1 = getNextToken(); // @@ [Advances token stream without ensuring there are enough tokens; may return null leading to NPE]
            Token token2 = getNextToken(); // @@ [Same issue as above; also, lookahead logic is flawed for detecting MainFuncDef start]
            getBack(2);
            if (token1.getTokenCode() == TokenCode.MAINTK || token2.getTokenCode() == TokenCode.LPARENT) { // @@ [Incorrect condition: 'main(' should be detected by MAINTK followed by LPARENT, but this checks either MAINTK alone OR any LPARENT, which can falsely break on unrelated '(']
                break;
            }
            compUnit.addDecl(parseDecl());
        }
        while (true) {
            if (getNextToken().getTokenCode() == TokenCode.MAINTK) { // @@ [Consumes a token and checks for MAINTK, but if not MAINTK, it backs up and tries to parse FuncDef; however, FuncDef starts with FuncType (VOIDTK/INTTK/CHARTK), not necessarily distinguishable from VarDecl which also starts with BType (INTTK/CHARTK), leading to ambiguity and incorrect parsing]
                getBack(1);
                break;
            }
            getBack(1);
            compUnit.addFuncDef(parseFuncDef());
        }
        mainFuncDef = parseMainFuncDef();
        compUnit.setMainFuncDef(mainFuncDef);
        return compUnit;
