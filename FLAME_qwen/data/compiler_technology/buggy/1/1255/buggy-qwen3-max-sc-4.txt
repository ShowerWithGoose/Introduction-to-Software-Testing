    public CompUnit parseCompUnit() {
        CompUnit compUnit = new CompUnit();
        MainFuncDef mainFuncDef;
        getNextToken();
        while (true) {
            Token token1 = getNextToken();
            Token token2 = getNextToken();
            getBack(2);
            if (token1.getTokenCode() == TokenCode.MAINTK || token2.getTokenCode() == TokenCode.LPARENT) { // @@ [Incorrect lookahead logic: checking token2 for LPARENT without ensuring token1 is an identifier or function type; also, MAINTK alone doesn't guarantee MainFuncDef since 'int main()' requires INTTK before MAINTK]
                break;
            }
            compUnit.addDecl(parseDecl());
        }
        while (true) {
            if (getNextToken().getTokenCode() == TokenCode.MAINTK) {
                getBack(1);
                break;
            }
            getBack(1);
            compUnit.addFuncDef(parseFuncDef()); // @@ [This may attempt to parse a FuncDef even when the next tokens don't form a valid function definition, because the lookahead only checks for MAINTK but not for preceding function types like VOIDTK/INTTK/CHARTK followed by IDENFR and LPARENT]
        }
        mainFuncDef = parseMainFuncDef();
        compUnit.setMainFuncDef(mainFuncDef);
        return compUnit;
    }
