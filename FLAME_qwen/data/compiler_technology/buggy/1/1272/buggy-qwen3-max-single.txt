    public static ExpStmt parse(TokenList list) {
        Exp exp = null;
        if (list.now().getType().equals(TokenType.Type.SEMICN)) {
            list.next();
        } else {
            exp = Exp.parse(list);
            if (list.now().getType().equals(TokenType.Type.SEMICN)) {
                list.next();
            } else {
                ErrorList.addError(list.before().getLine(),'i'); // @@ [This error reporting uses list.before().getLine() which may not point to the correct line for the missing semicolon. According to the problem description, for error type 'i' (missing semicolon), the error line number should correspond to "the line where the last non-terminal before the semicolon is located". However, after parsing Exp, the current token is whatever comes after the expression, and list.before() points to the last token of the expression, which might not be the correct location for the error report.]
            }
        }
        return new ExpStmt(exp);
    }
