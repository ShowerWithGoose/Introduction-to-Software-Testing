    public static ForStatement parse(TokenList list) {
        if (list.now().getType().equals(TokenType.Type.FORTK)) {
            Token For = list.now();
            list.next();
            list.next();//( @@ Unconditionally consumes '(' without checking if the token exists or matches, causing misalignment or crash if missing.
            ForStmt stmt1 = null;
            if (!list.now().getType().equals(TokenType.Type.SEMICN)) {
                stmt1 = ForStmt.parse(list);
            }
            list.next();//; @@ Unconditionally consumes ';' without verifying its presence, leading to errors if the semicolon is missing.
            Cond cond = null;
            if (!list.now().getType().equals(TokenType.Type.SEMICN)){
                cond = Cond.parse(list);   
            }
            list.next();//; @@ Unconditionally consumes the second ';' without checking, risking misalignment or crash.
            ForStmt stmt2 = null;
            if (!list.now().getType().equals(TokenType.Type.RPARENT)) {
                stmt2 = ForStmt.parse(list); 
            }
            
            list.next();//) @@ Unconditionally consumes ')' without checking, causing errors if the parenthesis is missing.
            Stmt mainStmt = Stmt.parse(list);
            return new ForStatement(For,stmt1,cond,stmt2,mainStmt);
        } else {
            System.err.println("Error:ForStatement cannot find for");
            return null;
        }
    } //不做其他错误特判
}

    public static UnaryExp parse(TokenList list){
        PrimaryExp primaryExp = null;
        Token ident = null;
        FuncRParams funcRParams = null;
        UnaryOp unaryOp = null;
        UnaryExp unaryExp = null;
        if (list.now().getType().equals(TokenType.Type.IDENFR)
                && list.getAheadOne(1).getType().equals(TokenType.Type.LPARENT)) { @@ Calls getAheadOne(1) without checking hasAhead(1), causing IndexOutOfBoundsException if IDENFR is the last token.
            ident = list.now();
            list.next();
            if (list.now().getType().equals(TokenType.Type.LPARENT)) {
                list.next(); //(
                Token first = list.now(); @@ Calls now() without checking hasToken(), risking crash if the token list is exhausted.
                if (list.now().getType().equals(TokenType.Type.RPARENT)) {
                    list.next();
                } else if (first.getType().equals(TokenType.Type.LPARENT) ||
                        first.getType().equals(TokenType.Type.INTCON) ||
                        first.getType().equals(TokenType.Type.CHRCON) ||
                        (first.getType().equals(TokenType.Type.IDENFR)) ||
                        first.getType().equals(TokenType.Type.PLUS) ||
                        first.getType().equals(TokenType.Type.MINU) ||
                        first.getType().equals(TokenType.Type.NOT)){
                    //不需要考虑可能出现assignStmt & getIntCharStmt
                    funcRParams = FuncRParams.parse(list);
                    if (list.now().getType().equals(TokenType.Type.RPARENT)) {
                        list.next();
                    } else {
                        ErrorList.addError(list.before().getLine(),'j');
                    }
                }
            } else {
                System.err.println("Error:UnaryExp cannot find (");
            }
        } else if (list.now().getType().equals(TokenType.Type.PLUS)
                || list.now().getType().equals(TokenType.Type.MINU)
                || list.now().getType().equals(TokenType.Type.NOT)) {
            unaryOp = UnaryOp.parse(list);
            unaryExp = parse(list);
        } else {
            primaryExp = PrimaryExp.parse(list);
        }
        return new UnaryExp(primaryExp,ident,funcRParams,unaryOp,unaryExp);
    }
