    public void checkError(char code, Node node) throws IOException {
        switch (code) {
            case 'i' -> { // @@ The arrow syntax for switch expressions is only supported in Java 14+, causing a compilation error in older Java environments.
                if (curType == LexType.SEMICN) {
                    node.addChild(new Node(";"));
                    next();
                } else {
                    printError('i');
                }
            }
            case 'j' -> { // @@ The arrow syntax for switch expressions is only supported in Java 14+, causing a compilation error in older Java environments.
                if (curType == LexType.RPARENT) {
                    node.addChild(new Node(")"));
                    next();
                } else {
                    printError('j');
                }
            }
            case 'k' -> { // @@ The arrow syntax for switch expressions is only supported in Java 14+, causing a compilation error in older Java environments.
                if (curType == LexType.RBRACK) {
                    node.addChild(new Node("]"));
                    next();
                } else {
                    printError('k');
                }
            }
        }
    }
