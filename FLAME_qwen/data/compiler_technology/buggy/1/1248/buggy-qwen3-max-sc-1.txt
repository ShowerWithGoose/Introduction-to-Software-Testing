    public void checkError(char code, Node node) throws IOException {
        switch (code) {
            case 'i' -> {
                if (curType == LexType.SEMICN) {
                    node.addChild(new Node(";"));
                    next();
                } else {
                    printError('i'); // @@ [This uses lastWord.getLineNum(), but lastWord may not correspond to the actual location where the semicolon is expected because additional tokens may have been consumed after the relevant construct]
                }
            }
            case 'j' -> {
                if (curType == LexType.RPARENT) {
                    node.addChild(new Node(")"));
                    next();
                } else {
                    printError('j'); // @@ [Same issue: lastWord may point to a token far from where the missing ')' should be, leading to incorrect line number in error report]
                }
            }
            case 'k' -> {
                if (curType == LexType.RBRACK) {
                    node.addChild(new Node("]"));
                    next();
                } else {
                    printError('k'); // @@ [Same issue: lastWord does not necessarily represent the line where the unmatched '[' occurred or where ']' is expected]
                }
            }
        }
    }
