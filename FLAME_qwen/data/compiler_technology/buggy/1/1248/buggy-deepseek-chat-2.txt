    public void parseCompUnit() throws IOException {
       // CompUnit → {Decl} {FuncDef} MainFuncDef
       while (curType == LexType.CONSTTK || // ConstDecl
               (isBType(curType)) && getNextType() == LexType.IDENFR && getNext2Type() != LexType.LPARENT) { // @@ [Operator precedence issue: `||` has lower precedence than `&&`, causing the condition to be evaluated as `curType == LexType.CONSTTK || ((isBType(curType)) && getNextType() == LexType.IDENFR && getNext2Type() != LexType.LPARENT)`, which may incorrectly treat a function definition as a declaration when `curType` is `CONSTTK` but the next tokens indicate a function.]
           root.addChild(parseDecl());
       }
       while ((isFuncType(curType) && getNextType() != LexType.MAINTK)) { // @@ [Logic error: The condition `getNextType() != LexType.MAINTK` is always true when `curType` is a function type, because `MAINTK` is a separate token for "main", not a function type. This loop will never execute for `main` function definitions, but it may incorrectly skip other function definitions if the next token is `MAINTK` (which shouldn't happen). The intended logic should check if the next token is `IDENFR` (function name) rather than `MAINTK`.]
           root.addChild(parseFuncDef());
       }
       if (curType == LexType.INTTK && getNextType() == LexType.MAINTK) {
           root.addChild(parseMain());
       }
       // 输出
        printGrammarNode("CompUnit");
       printErrorToFile();
    }
