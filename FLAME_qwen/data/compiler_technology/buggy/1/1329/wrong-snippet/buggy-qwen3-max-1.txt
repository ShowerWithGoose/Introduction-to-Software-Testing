The logic for determining when to parse declarations vs function definitions is flawed - it uses `words.get(pos + 2).getStr().equals("(")` which can cause index out of bounds errors and incorrect parsing decisions.

    public CompUnit parseCompUnit() {
        CompUnit compUnit = new CompUnit();
        while (pos < words.size() && (words.get(pos).getStr().equals("const") ||
            (pos + 2 < words.size() && !words.get(pos + 2).getStr().equals("(")))) {
            compUnit.addDecl(parseDecl());
        }
        while (pos < words.size() && !words.get(pos + 1).getStr().equals("main")) {
            compUnit.addFuncDef(parseFuncDef());
        }
        compUnit.setMainFuncDef(parseMainFuncDef());
        return compUnit;
