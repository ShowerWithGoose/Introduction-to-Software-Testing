    public CompUnit parseCompUnit() {
        CompUnit compUnit = new CompUnit();
        while (words.get(pos).getStr().equals("const") ||
            !words.get(pos + 2).getStr().equals("(")) { // @@ [This condition can cause ArrayIndexOutOfBoundsException when pos+2 exceeds the list size, and the logic for distinguishing Decl from FuncDef is incorrect because it assumes all non-const declarations must be followed by a function pattern]
            compUnit.addDecl(parseDecl());
        }
        while (!words.get(pos + 1).getStr().equals("main")) { // @@ [This condition can also cause ArrayIndexOutOfBoundsException when pos+1 exceeds the list size, and fails to properly identify MainFuncDef when there are no FuncDefs before it]
            compUnit.addFuncDef(parseFuncDef());
        }
        compUnit.setMainFuncDef(parseMainFuncDef());
        return compUnit;
    }
