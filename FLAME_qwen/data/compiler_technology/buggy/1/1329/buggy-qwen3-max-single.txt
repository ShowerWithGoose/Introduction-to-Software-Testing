    public CompUnit parseCompUnit() {

            compUnit.addDecl(parseDecl());
        }
        while (pos < words.size() && !(pos + 1 < words.size() && words.get(pos + 1).getStr().equals("main"))) { // @@ [The condition lacks bounds checking and incorrectly assumes that any non-main function has its name at pos+1.
