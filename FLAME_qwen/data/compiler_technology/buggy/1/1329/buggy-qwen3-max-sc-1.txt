    public CompUnit parseCompUnit() {
        CompUnit compUnit = new CompUnit();
        while (pos < words.size() && (words.get(pos).getStr().equals("const") ||
            (pos + 2 < words.size() && !words.get(pos + 2).getStr().equals("(")))) { // @@ [Original condition can cause IndexOutOfBoundsException when accessing pos+2 without checking bounds, and logic incorrectly assumes all non-const declarations must be followed by '(' for functions]
            compUnit.addDecl(parseDecl());
        }
        while (pos < words.size() && (pos + 1 < words.size() && !words.get(pos + 1).getStr().equals("main"))) { // @@ [Original condition can cause IndexOutOfBoundsException when accessing pos+1 without checking bounds, and fails to properly identify MainFuncDef which starts with 'int main']
            compUnit.addFuncDef(parseFuncDef());
        }
        compUnit.setMainFuncDef(parseMainFuncDef());
        return compUnit;
    }
