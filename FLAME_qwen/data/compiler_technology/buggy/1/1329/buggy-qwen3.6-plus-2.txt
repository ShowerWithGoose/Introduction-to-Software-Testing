    public CompUnit parseCompUnit() {
        CompUnit compUnit = new CompUnit();
        while (words.get(pos).getStr().equals("const") ||
            !words.get(pos + 2).getStr().equals("(")) { // @@ Accessing index pos+2 without bounds checking causes IndexOutOfBoundsException on short inputs; logic also fails to correctly distinguish Decl from FuncDef for 'void' type.
            compUnit.addDecl(parseDecl());
        }
        while (!words.get(pos + 1).getStr().equals("main")) { // @@ Accessing index pos+1 without bounds checking causes IndexOutOfBoundsException if the token list ends before 'main'.
            compUnit.addFuncDef(parseFuncDef());
        }
        compUnit.setMainFuncDef(parseMainFuncDef());
        return compUnit;
    }
