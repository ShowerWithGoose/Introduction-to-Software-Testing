    // 处理终结符
    private Token handleTerminal(LexType lexType) {
        if (getSymType().equals(lexType)) {
            Token token = getNowToken();
            nextIndex();
            return token;
        }
        is_analyze_success = false;
        if (lexType.equals(LexType.SEMICN)) {
            ErrorHandler.recordError(ErrorType.SEMICN_MISSING, getLineNumber(-1)); @@ [The error line number should correspond to the line of the last non-terminal before the semicolon, but getLineNumber(-1) returns the previous token's line which may not be the correct non-terminal location]
        }
        else if (lexType.equals(LexType.RPARENT)) {
            ErrorHandler.recordError(ErrorType.RPARENT_MISSING, getLineNumber(-1)); @@ [The error line number should correspond to the line of the last non-terminal before the right parenthesis, but getLineNumber(-1) may not point to the correct non-terminal]
        }
        else if (lexType.equals(LexType.RBRACK)) {
            ErrorHandler.recordError(ErrorType.RBRACK_MISSING, getLineNumber(-1)); @@ [The error line number should correspond to the line where the last preceding non-terminal symbol appears before the right square bracket, but getLineNumber(-1) uses the previous token which might not be the non-terminal]
        }
        else {
            throw new Error(ErrorType.OTHER_ERROR, getLineNumber());
        }
        return null;
    }
