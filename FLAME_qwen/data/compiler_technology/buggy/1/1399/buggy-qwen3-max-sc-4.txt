    // 处理终结符
    private Token handleTerminal(LexType lexType) {
        if (getSymType().equals(lexType)) {
            Token token = getNowToken();
            nextIndex();
            return token;
        }
        is_analyze_success = false;
        if (lexType.equals(LexType.SEMICN)) {
            ErrorHandler.recordError(ErrorType.SEMICN_MISSING, getLineNumber(-1)); @@ [Using getLineNumber(-1) may not correctly identify the line where the semicolon is expected; it should use the current token's line number or the last valid token's line number before the missing semicolon]
        }
        else if (lexType.equals(LexType.RPARENT)) {
            ErrorHandler.recordError(ErrorType.RPARENT_MISSING, getLineNumber(-1)); @@ [Same issue: getLineNumber(-1) might not point to the correct line for the missing right parenthesis]
        }
        else if (lexType.equals(LexType.RBRACK)) {
            ErrorHandler.recordError(ErrorType.RBRACK_MISSING, getLineNumber(-1)); @@ [Same issue: getLineNumber(-1) might not point to the correct line for the missing right square bracket]
        }
        else {
            throw new Error(ErrorType.OTHER_ERROR, getLineNumber());
        }
        return null;
    }
