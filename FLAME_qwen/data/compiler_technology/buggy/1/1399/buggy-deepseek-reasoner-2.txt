    private MulExp handleMulExp() {
        // todo 消除左递归了吗？
        // MulExp -> UnaryExp | MulExp ('*' | '/' | '%') UnaryExp
        List<UnaryExp> unaryExpArrayList = new ArrayList<>();
        List<Token> opList = new ArrayList<>();
        while (getSymType() == LexType.MULT || getSymType() == LexType.DIV || getSymType() == LexType.MOD) { // @@ [Error: This loop should only process operators after first UnaryExp has been parsed; the first UnaryExp is never added to the list]
            unaryExpArrayList.add(handleUnaryExp()); // @@ [Error: This adds a UnaryExp but without having first added the initial UnaryExp; also the operator should be read before the next UnaryExp, not after]
            if (getSymType() == LexType.MULT) {
                opList.add(handleTerminal(LexType.MULT));
            }
            else if (getSymType() == LexType.DIV) {
                opList.add(handleTerminal(LexType.DIV));
            }
            else if (getSymType() == LexType.MOD) {
                opList.add(handleTerminal(LexType.MOD));
            }
        }
        return new MulExp(opList, unaryExpArrayList); // [Error: If no operator, unaryExpArrayList is empty, but should contain the first UnaryExp]
