Based on the problem description and code review, the main issue appears to be in the recursive descent parser implementation, specifically in the expression parsing methods which use direct left recursion instead of iterative parsing. This will cause infinite recursion and stack overflow during parsing.

The most problematic methods are in the Parser.java file, particularly:

    private MulExp handleMulExp() {
        // todo 消除左递归了吗？
        // MulExp -> UnaryExp | MulExp ('*' | '/' | '%') UnaryExp
        List<UnaryExp> unaryExpArrayList = new ArrayList<>();
        List<Token> opList = new ArrayList<>();
        while (getSymType() == LexType.MULT || getSymType() == LexType.DIV || getSymType() == LexType.MOD) {
            unaryExpArrayList.add(handleUnaryExp());
            if (getSymType() == LexType.MULT) {
                opList.add(handleTerminal(LexType.MULT));
            }
            else if (getSymType() == LexType.DIV) {
                opList.add(handleTerminal(LexType.DIV));
            }
            else if (getSymType() == LexType.MOD) {
                opList.add(handleTerminal(LexType.MOD));
            }
        }
        return new MulExp(opList, unaryExpArrayList);
    }

    private AddExp handleAddExp() throws Error {
        // todo 圆括号表示什么？

        MulExp mulExpNode = null;
        Token operator = null;
        AddExp addExpNode = null;
        // todo 解决左递归了吗？
        // MulExp
        mulExpNode = handleMulExp();
        if (getSymType().equals(LexType.PLUS)) {
            operator = handleTerminal(LexType.PLUS);
            addExpNode = handleAddExp();
        }
        else if (getSymType().equals(LexType.MINU)) {
            operator = handleTerminal(LexType.MINU);
            addExpNode = handleAddExp();
        }
        return new AddExp(mulExpNode, operator, addExpNode);
