    public ConstDef parseConstDef() throws IOException {

        getToken();
        if (lexType == LexType.LBRACK) {
            int line1 = lexer.getLineNum();

    public Stmt parseStmt() throws IOException {
        int line = lexer.getLineNum();
        if (lexType == LexType.IFTK) {
            getToken();
            getToken();
            Cond cond = parseCond();
            if (errorTool.checkJ(token, line)) { // @@ [Error: The line number for missing ')' should be the line of the '(' token, not the line of 'if'. Should capture the line after the first getToken() (the '(' token).]
                getToken();
            }
            Stmt stmt = parseStmt();
            Stmt stmt1 = null;
            if (lexType == LexType.ELSETK) {
                getToken();
                stmt1 = parseStmt();
            }
            outputP("Stmt");
            return new Stmt(cond, stmt, stmt1, StmtType.If);
        } else if (lexType == LexType.FORTK) {
            ForStmt forStmt = null;
            Cond cond = null;
            ForStmt forStmt1 = null;
            Stmt stmt;
            getToken();
            getToken();
            if (lexType != LexType.SEMICN) {
                forStmt = parseForStmt();
            }
            getToken();
            if (lexType != LexType.SEMICN) {
                cond = parseCond();
            }
            getToken();
            if (lexType != LexType.RPARENT) {
                forStmt1 = parseForStmt();
            }
            getToken();
            stmt = parseStmt();
            outputP("Stmt");
            return new Stmt(forStmt, cond, forStmt1, stmt, StmtType.For);
        } else if (lexType == LexType.BREAKTK) {
            getToken();
            if (errorTool.checkI(token, line)) { // @@ [Error: The line number for missing ';' should be the line of the last non-terminal before ';', which is 'break' itself. Since getToken() was called, the current token is the next one; the line should be the line of 'break' (which is the original line), but after getToken(), the line number may have changed if there were newlines. Should capture the line before getToken().]
                getToken();
            }
            outputP("Stmt");
            return new Stmt(StmtType.Break);
        } else if (lexType == LexType.CONTINUETK) {
            getToken();
            if (errorTool.checkI(token, line)) { // @@ [Error: Same issue as BREAKTK; should use the line of 'continue' before getToken().]
                getToken();
            }
            outputP("Stmt");
            return new Stmt(StmtType.Continue);
        } else if (lexType == LexType.RETURNTK) {
            getToken();
            Exp exp = null;
            if (lexType != LexType.SEMICN) {
                exp = parseExp();
            }
            if (errorTool.checkI(token, line)) { // @@ [Error: The line number for missing ';' should be the line of the last non-terminal before ';', which could be the return token or the expression. Since getToken() was called, the line should be the line of 'return' (original line), but after parsing exp, the line may have changed. Should capture the line before getToken() or after parsing exp appropriately.]
                getToken();
            }
            outputP("Stmt");
            return new Stmt(exp, StmtType.Return, StmtType.Return);
        } else if (lexType == LexType.PRINTFTK) {
            getToken();
            getToken();
            String s = token;
            getToken();
            ArrayList<Exp> exps = new ArrayList<>();
            while (lexType == LexType.COMMA) {
                getToken();
                exps.add(parseExp());
            }
            if (errorTool.checkJ(token, line)) { // @@ [Error: The line number for missing ')' should be the line of the '(' token, not the line of 'printf'. Should capture the line after the first getToken() (the '(' token).]
                getToken();
            }
            if (errorTool.checkI(token, line)) { // @@ [Error: The line number for missing ';' should be the line of the last non-terminal before ';', which is the ')' token. Should capture the line of ')' after the previous check.]
                getToken();
            }
            outputP("Stmt");
            return new Stmt(s, exps, StmtType.Printf);
        } else if (lexType == LexType.LBRACE) {
            Block block = parseBlock();
            outputP("Stmt");
            return new Stmt(block, StmtType.Block);
        } else if (lexer.preReadCompare(LexType.ASSIGN)) {
            LVal lVal = parseLVal();
            getToken();
            //System.out.println(lexType);
            if (lexType == LexType.GETINTTK) {
                getToken();
                getToken();
                if (errorTool.checkJ(token, line)) { // @@ [Error: The line number for missing ')' should be the line of the '(' token after getint, not the original line. Should capture the line after the first getToken() after getint.]
                    getToken();
                }
                if (errorTool.checkI(token, line)) { // @@ [Error: The line number for missing ';' should be the line of the last non-terminal before ';', which is the ')' token. Should capture the line of ')' after the previous check.]
                    getToken();
                }
                outputP("Stmt");
                return new Stmt(lVal, StmtType.Getint);
            } else if (lexType == LexType.GETCHARTK) {
                getToken();
                getToken();
                if (errorTool.checkJ(token, line)) { // @@ [Error: Same issue as GETINTTK; should use the line of '(' after getchar.]
                    getToken();
                }
                if (errorTool.checkI(token, line)) { // @@ [Error: Same issue as GETINTTK; should use the line of ')' after the previous check.]
                    getToken();
                }
                outputP("Stmt");
                return new Stmt(lVal, StmtType.Getchar);
            } else {
                Exp exp = parseExp();

                    getToken();
                }
                outputP("Stmt");
                return new Stmt(lVal, exp, StmtType.LValExp);
            }
        } else {
            Exp exp = null;
            if (lexType != LexType.SEMICN) {
                exp = parseExp();
            }
            if (errorTool.checkI(token, line)) { // @@ [Error: The line number for missing ';' should be the line of the last non-terminal before ';', which could be the expression or the start of the statement. Should capture the line after parsing exp or the original line appropriately.]
                getToken();
            }
            outputP("Stmt");
            return new Stmt(exp, StmtType.EXP);
        }

    public UnaryExp parseUnaryExp() throws IOException {
        int line = lexer.getLineNum();
        if (lexType == LexType.PLUS || lexType == LexType.MINU
                || lexType == LexType.NOT) {
            UnaryOp op = parseUnaryOp();
            UnaryExp unaryExp = parseUnaryExp();
            outputP("UnaryExp");
            return new UnaryExp(op, unaryExp);
        } else if (lexType == LexType.IDENFR) {
            ArrayList<LexType> lexTypes = lexer.preRead(1);
            if (lexTypes.get(0) == LexType.LPARENT) {
                String s = token;
                getToken();
                getToken();
                FuncRParams funcRParams = null;
                if (lexType != LexType.RPARENT&&lexer.preReadCompare(LexType.RPARENT)) {
                    funcRParams = parseFuncRParams();
                }
                if (errorTool.checkJ(token, line)) { // @@ [Error: The line number for missing ')' should be the line of the '(' token, not the identifier line. Should capture the line after the first getToken() (the '(' token).]
                    getToken();
                }
                outputP("UnaryExp");
                return new UnaryExp(getIdent(s,line), funcRParams);
            } else {
                PrimaryExp primaryExp = parsePrimaryExp();
                outputP("UnaryExp");
                return new UnaryExp(primaryExp);
            }
        } else {
            PrimaryExp primaryExp = parsePrimaryExp();
            outputP("UnaryExp");
            return new UnaryExp(primaryExp);
        }
    }
