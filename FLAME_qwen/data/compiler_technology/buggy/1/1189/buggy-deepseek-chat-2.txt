    private StmtEle parseStmtEle() {
        if (now.getType() == TokenType.IFTK) {
            return parseStmtIf();
        }
        if (now.getType() == TokenType.FORTK) {
            return parseStmtFor();
        }
        if (now.getType() == TokenType.BREAKTK) {
            return parseStmtBreak();
        }
        if (now.getType() == TokenType.CONTINUETK) {
            return parseStmtContinue();
        }
        if (now.getType() == TokenType.RETURNTK) {
            return parseStmtReturn();
        }
        if (now.getType() == TokenType.PRINTFTK) {
            return parseStmtPrintf();
        }
        if (now.getType() == TokenType.LBRACE) {
            return parseStmtBlock();
        }
        if (now.getType() == TokenType.SEMICN) {
            return parseStmtExp();
        }
        if (preRead.getType() != TokenType.LBRACK && preRead.getType() != TokenType.ASSIGN) { // @@ [The condition incorrectly uses preRead to check for LBRACK and ASSIGN, but preRead is the token after now, not the token before now. This logic is flawed and may lead to incorrect parsing decisions.]
            return parseStmtExp();
        }
        int tmpCur = cur;
        this.errorState = false;
        Exp exp = parseExp();
        this.errorState = true;
        if (now.getType() != TokenType.ASSIGN) {
            backTracking(tmpCur);
            return parseStmtExp();
            /*Token semicolon = checkSemicolon();
            return new StmtExp(exp, semicolon);*/
        } else {
            if (preRead.getType() == TokenType.GETINTTK) { // @@ [The condition uses preRead to check for GETINTTK, but preRead is the token after now, not the token after the assignment operator. This is incorrect because after parsing the LVal and '=', the next token should be checked, not preRead.]
                backTracking(tmpCur);
                return parseStmtGetInt();
            } else if (preRead.getType() == TokenType.GETCHARTK) { // @@ [Same issue as above: preRead is used incorrectly to check for GETCHARTK.]
                backTracking(tmpCur);
                return parseStmtGetChar();
            } else {
                backTracking(tmpCur);
                return parseStmtAssign();
            }
        }
    }
