    private void parseInitVal() {
        if (curToken.getLexType() == LexType.STRCON) {
            curToken = nextToken();
        } else if (curToken.getLexType() == LexType.LBRACE) {
            curToken = nextToken(); // skip {
            parseExp(); // @@ According to the grammar, the expression inside braces is optional ([ Exp { ',' Exp } ]). Unconditionally calling parseExp() will fail to parse empty initialization lists like '{}' and cause incorrect token consumption or parsing errors.
            while (curToken.getLexType() == LexType.COMMA) {
                curToken = nextToken(); // skip ,
                parseExp();
            }
            curToken = nextToken(); // skip }
        } else {
            parseExp();
        }
        outputType("<InitVal>");
    }
