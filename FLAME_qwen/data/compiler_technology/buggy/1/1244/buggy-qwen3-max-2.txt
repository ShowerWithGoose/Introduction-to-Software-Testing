    private CompUnit parseCompUnit() {
        ArrayList<Decl> decls = new ArrayList<>();
        ArrayList<FuncDef> funcDefs = new ArrayList<>();
        MainFuncDef mainFuncDef;

        while (true) {
            if (curToken.getType() == TokenType.CONSTTK) {
                decls.add(parseConstDecl());
            } else if (curToken.getType() == TokenType.INTTK) {
                if (tokens.get(curTokenIndex + 1).getType() == TokenType.MAINTK) {
                    mainFuncDef = parseMainFunc();
                    break;
                } else {
                    if (tokens.get(curTokenIndex + 2).getType() == TokenType.LPARENT) {
                        funcDefs.add(parseFuncDef());
                    } else {
                        decls.add(parseVarDecl());
                    }
                }
            } else if (curToken.getType() == TokenType.CHARTK) {
                if (tokens.get(curTokenIndex + 2).getType() == TokenType.LPARENT) {
                    funcDefs.add(parseFuncDef());
                } else {
                    decls.add(parseVarDecl());
                }
            } else if (curToken.getType() == TokenType.VOIDTK) {
                funcDefs.add(parseFuncDef());
            }
        }

        addNode("<CompUnit>"); // @@ [The <CompUnit> node is added after all parsing is complete, but it should be output before any of its constituent tokens are processed. According to the problem requirements, non-terminal components should be output at the point when their recognition begins, not after they are fully parsed. This causes incorrect ordering in the output file.]
        return new CompUnit(decls, funcDefs, mainFuncDef);
    }
