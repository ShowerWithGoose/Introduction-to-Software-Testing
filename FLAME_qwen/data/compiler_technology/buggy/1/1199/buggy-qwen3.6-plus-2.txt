    private void error(TokenType tokenType) {
        if (tokenType == TokenType.SEMICN) {
            lexer.back();
            lexer.getError().get(lexer.peek().getLineNumber()).add('i');
            lexer.next();
        } else if (tokenType == TokenType.RPARENT) {
            lexer.back();
            lexer.getError().get(lexer.peek().getLineNumber()).add('i'); // @@ Incorrect error category code 'i' is added for missing right parenthesis; according to the problem description, it should be 'j'.
            lexer.next();
        } else if (tokenType == TokenType.RBRACK) {
            lexer.back();
            lexer.getError().get(lexer.peek().getLineNumber()).add('i'); // @@ Incorrect error category code 'i' is added for missing right square bracket; according to the problem description, it should be 'k'.
            lexer.next();
        }
    }
