package Lexer;

import frontend.FileProcessor;
import base.Keywords;
import base.Token;

import java.util.*;

public class Lexer {
    private ArrayList<Token> tokens = new ArrayList<>(); // 用于存储词法分析结果
    private ArrayList<String> errors = new ArrayList<>(); // 用于存储错误信息
    private int lineNumber = 1;  // 当前处理行号
    private char currentChar;  // 当前字符
    private int currentIndex = 0;  // 当前字符在字符串中的索引
    private String code;  // 输入的代码
    private int length;  // 代码的总长度
    private TreeMap<Integer, String> errorInOrder = new TreeMap<>();

    // 初始化词法分析器，设置要分析的代码
    public Lexer(FileProcessor fileProcessor) {
        this.code = fileProcessor.getCode();  // 从FileProcessor获取代码
        this.length = code.length();
        this.currentChar = nextChar();  // 预读第一个字符
    }

    // 获取下一个字符
    private char nextChar() {
        if (currentIndex < length) {
            return code.charAt(currentIndex++);
        } else {
            return '\0';  // 文件结束符
        }
    }

    // 预读下一个字符
    private char peekChar() {
        if (currentIndex < length) {
            return code.charAt(currentIndex);
        } else {
            return '\0';  // 文件结束符
        }
    }

    // 分析代码
    public void analyze() {
        while (currentChar != '\0') {  // 直到 EOF 结束
            if (Character.isWhitespace(currentChar)) {
                if (currentChar == '\n') {
                    lineNumber++;  // 处理换行符，增加行号
                }
                currentChar = nextChar();  // 跳过空白字符
            } else if (currentChar == '"') {
                analyzeString();  // 处理字符串常量 STRCON
            } else if (currentChar == '\'') {
                analyzeChar();  // 处理字符常量 CHRCON
            } else if (Character.isLetter(currentChar) || currentChar == '_') {
                analyzeIdentifierOrKeyword();  // 处理标识符或关键字
            } else if (Character.isDigit(currentChar)) {
                analyzeNumber();  // 处理数字
            } else {
                analyzeSymbol();  // 处理符号
            }
        }
    }

    // 处理字符串常量
    private void analyzeString() {
        StringBuilder sb = new StringBuilder();
        sb.append(currentChar);  // 添加起始双引号
        currentChar = nextChar();

        while (currentChar != '"' && currentChar != '\0') {
            if (currentChar == '\\' && peekChar() == 'n') {
                sb.append("\\n");  // 处理转义字符
                nextChar();  // 跳过 'n'
            } else {
                sb.append(currentChar);
            }
            currentChar = nextChar();
        }

        if (currentChar == '"') {
            sb.append('"');  // 添加结束双引号
            tokens.add(new Token("STRCON", sb.toString(), lineNumber));
            currentChar = nextChar();  // 跳过结束双引号
        } else {
            errors.add(lineNumber + " a");  // 未闭合的字符串
        }
    }

    // 处理字符常量
    private void analyzeChar() {
        StringBuilder sb = new StringBuilder();
        sb.append(currentChar);  // 添加起始单引号
        currentChar = nextChar();

        if (currentChar == '\\' ) {
            sb.append(currentChar);  // 处理转义字符
            currentChar = nextChar();  // 跳过 'n'
            sb.append(currentChar);
            currentChar = nextChar();
        } else  {
            sb.append(currentChar);
            currentChar = nextChar();
        }

        if (currentChar == '\'') {
            sb.append('\'');  // 添加结束单引号
            tokens.add(new Token("CHRCON", sb.toString(), lineNumber));
            currentChar = nextChar();  // 跳过结束单引号
        } else {
            errors.add(lineNumber + " a");  // 未闭合的字符常量
        }
    }

    // 处理标识符或关键字
    private void analyzeIdentifierOrKeyword() {
        StringBuilder sb = new StringBuilder();

        // 读取标识符或关键字，允许字母和数字以及下划线
        while (Character.isLetterOrDigit(currentChar) || currentChar == '_') {
            sb.append(currentChar);
            currentChar = nextChar();
        }

        String word = sb.toString();

        // 检查是否为关键字，若不是则认为是标识符
        String tokenType = Keywords.getTokenType(word);
        if (tokenType.equals("IDENFR")) {
            tokens.add(new Token("IDENFR", word, lineNumber));  // 标识符
        } else {
            tokens.add(new Token(tokenType, word, lineNumber));  // 关键字
        }
    }

    // 处理数字
    private void analyzeNumber() {
        StringBuilder sb = new StringBuilder();
        while (Character.isDigit(currentChar)) {
            sb.append(currentChar);
            currentChar = nextChar();
        }
        tokens.add(new Token("INTCON", sb.toString(), lineNumber));
    }

    // 处理其他符号（如运算符，分隔符等）
    private void analyzeSymbol() {
        String tokenValue = Character.toString(currentChar);  // 当前字符作为符号

        // 处理复合符号
        if (currentChar == '<') {
            if (peekChar() == '=') {
                tokenValue = "<=";  // 识别到复合符号 <=
                tokens.add(new Token("LEQ", tokenValue, lineNumber));
                currentChar = nextChar();//多更新一次
            } else {
                tokens.add(new Token("LSS", "<", lineNumber));  // 识别到符号 <
            }
            currentChar = nextChar();// 更新到下一个字符
        } else if (currentChar == '>') {
            if (peekChar() == '=') {
                tokenValue = ">=";  // 识别到复合符号 >=
                tokens.add(new Token("GEQ", tokenValue, lineNumber));
                currentChar = nextChar();
            } else {
                tokens.add(new Token("GRE", ">", lineNumber));  // 识别到符号 >
            }
            currentChar = nextChar();  // 更新到下一个字符
        } else if (currentChar == '!') {
            if (peekChar()== '=') {
                tokenValue = "!=";  // 识别到复合符号 !=
                tokens.add(new Token("NEQ", tokenValue, lineNumber));
                currentChar = nextChar();
            } else {
                tokens.add(new Token("NOT", "!", lineNumber));  // 识别到符号 !
            }
            currentChar = nextChar();  // 更新到下一个字符
        } else if (currentChar == '&' && peekChar() == '&') {
            tokenValue = "&&";  // 识别到复合符号 &&
            tokens.add(new Token("AND", tokenValue, lineNumber));
            nextChar();  // 跳过第二个 &
            currentChar = nextChar();  // 更新到下一个字符
        } else if (currentChar == '|' && peekChar() == '|') {
            tokenValue = "||";  // 识别到复合符号 ||
            tokens.add(new Token("OR", tokenValue, lineNumber));
            nextChar();  // 跳过第二个 |
            currentChar = nextChar();  // 更新到下一个字符
        } else if (currentChar == '=') {
            if (peekChar() == '=') {
                tokenValue = "==";  // 识别到复合符号 ==
                tokens.add(new Token("EQL", tokenValue, lineNumber));
                currentChar = nextChar();
            } else {
                tokens.add(new Token("ASSIGN", "=", lineNumber));  // 识别到赋值符号 =
            }
            currentChar = nextChar();  // 更新到下一个字符
        } else if (currentChar == '%') {
            tokens.add(new Token("MOD", "%", lineNumber));  // 识别到符号 %
            currentChar = nextChar();  // 更新到下一个字符
        } else if (currentChar == '/') {
            if (peekChar() == '/')
            {
                while (currentChar != '\n' && currentChar != '\0') {
                    currentChar = nextChar();
                    //读取注释
                }
                return;
            }
            else if (peekChar() == '*')
            {
                while (!(currentChar == '*' && peekChar() == '/'))
                {
                    currentChar = nextChar();
                }
                currentChar = nextChar();
                currentChar = nextChar();
                return;
            }
            tokens.add(new Token("DIV", "/", lineNumber));  // 识别到符号 /
            currentChar = nextChar();  // 更新到下一个字符
        } else if (currentChar == '+') {
            tokens.add(new Token("PLUS", "+", lineNumber));  // 识别到符号 +
            currentChar = nextChar();  // 更新到下一个字符
        } else if (currentChar == '-') {
            tokens.add(new Token("MINU", "-", lineNumber));  // 识别到符号 -
            currentChar = nextChar();  // 更新到下一个字符
        } else if (currentChar == '*') {
            tokens.add(new Token("MULT", "*", lineNumber));  // 识别到符号 *
            currentChar = nextChar();  // 更新到下一个字符
        } else if (currentChar == ';') {
            tokens.add(new Token("SEMICN", ";", lineNumber));  // 识别到分隔符 ;
            currentChar = nextChar();  // 更新到下一个字符
        } else if (currentChar == ',') {
            tokens.add(new Token("COMMA", ",", lineNumber));  // 识别到分隔符 ,
            currentChar = nextChar();  // 更新到下一个字符
        } else if (currentChar == '(') {
            tokens.add(new Token("LPARENT", "(", lineNumber));  // 识别到左括号 (
            currentChar = nextChar();  // 更新到下一个字符
        } else if (currentChar == ')') {
            tokens.add(new Token("RPARENT", ")", lineNumber));  // 识别到右括号 )
            currentChar = nextChar();  // 更新到下一个字符
        } else if (currentChar == '{') {
            tokens.add(new Token("LBRACE", "{", lineNumber));  // 识别到左花括号 {
            currentChar = nextChar();  // 更新到下一个字符
        } else if (currentChar == '}') {
            tokens.add(new Token("RBRACE", "}", lineNumber));  // 识别到右花括号 }
            currentChar = nextChar();  // 更新到下一个字符
        } else if (currentChar == '[') {
            tokens.add(new Token("LBRACK", "[", lineNumber));  // 识别到左中括号 [
            currentChar = nextChar();  // 更新到下一个字符
        } else if (currentChar == ']') {
            tokens.add(new Token("RBRACK", "]", lineNumber));  // 识别到右中括号 ]
            currentChar = nextChar();  // 更新到下一个字符
        } else {
            String tokenType = Keywords.getTokenType(tokenValue);
            if (!tokenType.equals("IDENFR")) {
                tokens.add(new Token(tokenType, tokenValue, lineNumber));  // 存储当前符号
            } else {
                errors.add(lineNumber + " a");  // 非法字符
                errorInOrder.put(lineNumber, "a");
                if (currentChar == '&') {
                    tokens.add(new Token("AND", "&&", lineNumber));
                }
                else if (currentChar == '|') {
                    tokens.add(new Token("OR", "||", lineNumber));
                }
            }
            currentChar = nextChar();  // 推进到下一个字符
        }
    }

    // 获取词法分析的结果（供语法分析器使用）
    public ArrayList<Token> getTokens() {
        return tokens;

import java.io.*;

public class FileProcessor {
    private final FileReader fileReader;
    private final String code;
    private final FileWriter fileWriter;

    public FileProcessor(String inputFile, String outputFile) throws IOException {
        fileReader = new FileReader(inputFile);
        code = file2Code();
        fileWriter = new FileWriter(new File(outputFile));
    }

    //把代码读出来
    private String file2Code() throws IOException {
        BufferedReader bufferedReader = new BufferedReader(fileReader);
        StringBuilder stringBuilder = new StringBuilder();
        String temp;
        while ((temp = bufferedReader.readLine()) != null)
        {

    private Token nextToken()
    {
        this.thisToken = tokens.get(this.thisNum);
        grammar.add(thisToken.toString());
        //防止遗漏
        thisNum++;
        return thisToken;
    }

    private Token peekToken(int next)
    {
        //因为thisNum永远会快一个，所以next为0为下一个
        return tokens.get(this.thisNum + next);
    }

    private void analyseCompUnit()
    {
        //CompUnit → {Decl} {FuncDef} MainFuncDef
        while(thisToken.getType().equals("INTTK") ||
                thisToken.getType().equals("VOIDTK") || thisToken.getType().equals("CHARTK")
                || thisToken.getType().equals("CONSTTK"))
        {
            if (thisToken.getType().equals("INTTK") && peekToken(0).getType().equals("MAINTK"))
            {
                //进主函数了,MainFuncDef
                //creatAndGoToNode("MainFuncDef",currentNode);
                analyseMainFuncDef();
                break;
            }
            else if (peekToken(1).getType().equals("LPARENT"))
            {
                //进函数了{FuncDef}
                //creatAndGoToNode("FuncDef",currentNode);
                analyseFuncDef();
            }
            else
            {
                //{Decl}
                //creatAndGoToNode("Decl",currentNode);
                analyseDecl();
            }
            nextToken();//;下一位
        }
        /*currentNode = currentNode.getFather();
        if (currentNode != null)
        {
            System.out.println("tree Error");
        }*/
        grammar.add("<CompUnit>");
    }

    private void analyseMainFuncDef()
    {
        nextToken(); //main
        nextToken(); //(
        if (!peekToken(0).getType().equals("RPARENT"))
        {
            //错误j
            errors.put(thisToken.getLineNumber(),"j");
        }
        else
        {
            nextToken(); //)
        }
        nextToken();//进入block
        //creatAndGoToNode("Block",currentNode);
        analyseBlock();
        //退出时为;
        //currentNode = currentNode.getFather();
        grammar.add("<MainFuncDef>");
    }

            //creatAndGoToNode("VarDecl",currentNode);

        //currentNode = currentNode.getFather();
        //grammar.add("<Decl>");
    }

    private void analyseFuncDef()
    {
        //进入为类型
        //FuncDef → FuncType Ident '(' [FuncFParams] ')' Block // j
        grammar.add("<FuncType>");
        nextToken();//ident
        nextToken();//(
        if (peekToken(0).getType().equals("RPARENT"))
        {
            nextToken();//)
        }
        else
        {
            if (peekToken(0).getType().equals("LBRACE"))

                errors.put(thisToken.getLineNumber(),"j");
            }
            else
            {
                nextToken();//进下一个了

                    errors.put(thisToken.getLineNumber(),"j");
                }
                else
                {
                    nextToken();//)
                }
            }

        }
        nextToken();//{
        //creatAndGoToNode("Block",currentNode);

    private void analyseBlock()
    {

            nextToken();//ident
           // creatAndGoToNode("VarDef",currentNode);

    private void analyseVarDef() {
        //进入时为ident
        //变量定义 VarDef → Ident [ '[' ConstExp ']' ] | Ident [ '[' ConstExp ']' ] '=' InitVal // k

            else

            analyseInitVal();
        }
        //退出时没有向前
        //currentNode = currentNode.getFather();
        grammar.add("<VarDef>");
    }

    private void analyseInitVal() {

                //结束时为最后一个
            }
        }
        else if (thisToken.getType().equals("STRCON"))
        {
            grammar.add("<InitVal>");
        }
        else

            grammar.add("<InitVal>");
        }
        //currentNode = currentNode.getFather();

    private void analyseConstDef() { //Ident [ '[' ConstExp ']' ] '=' ConstInitVal // k

        {

            if (peekToken(0).getType().equals("RBRACK"))

        nextToken();//Ident

    private boolean isLVal()
    {
        for (int i = 0; ;i++)
        {
            if (peekToken(i).getType().equals("ASSIGN"))
            {
                return true;
            }
            if (peekToken(i).getType().equals("SEMICN"))
            {
                return false;
            }
        }
    }

    private void analyseStmt()
    {
        /*
            Stmt → *LVal '=' Exp ';' // i
            | ????存疑[Exp] ';' // i
            | *Block
            | *'if' '(' Cond ')' Stmt [ 'else' Stmt ] // j
            | *'for' '(' [ForStmt] ';' [Cond] ';' [ForStmt] ')' Stmt
            | 'break' ';' | 'continue' ';' // i
            | *'return' [Exp] ';' // i
            | *LVal '=' 'getint''('')'';' // i j
            | *LVal '=' 'getchar''('')'';' // i j
            | *'printf''('StringConst {','Exp}')'';' // i j */
        //进入时为第一位
        if (thisToken.getType().equals("IDENFR") && (peekToken(0).getType().equals("ASSIGN") || isLVal())) //LVal
        {
            analyseLVal();
            if (peekToken(0).getType().equals("SEMICN")) //[Exp] ';' // i
            {
                nextToken();//;
                grammar.add("<Stmt>");
                return;
            }
            else if (peekToken(0).getType().equals("ASSIGN"))
            {
                nextToken();//=
                if (peekToken(0).getType().equals("GETINTTK") || peekToken(0).getType().equals("GETCHARTK"))//get*2
                {
                    nextToken();//get
                    nextToken();//(
                    if (peekToken(0).getType().equals("SEMICN"))
                    {
                        errors.put(thisToken.getLineNumber(),"j");
                        nextToken();//;
                        grammar.add("<Stmt>");
                        return;
                    }
                    else
                    {
                        nextToken();//)
                        if (!peekToken(0).getType().equals("SEMICN"))
                        {
                            errors.put(thisToken.getLineNumber(),"i");
                        }
                        else
                        {
                            nextToken();//;
                        }
                        grammar.add("<Stmt>");
                        return;
                    }
                }
                else //exp
                {
                    nextToken();//进
                    analyseExp();
                    if (peekToken(0).getType().equals("SEMICN"))
                    {
                        nextToken();//;
                    }
                    else
                    {
                        errors.put(thisToken.getLineNumber(),"i");

                    }
                    grammar.add("<Stmt>");
                    return;
                }
            }
        }
        else if (thisToken.getType().equals("LBRACE")) // block
        {
            analyseBlock();
            grammar.add("<Stmt>");
        }
        else if (thisToken.getType().equals("IFTK")) //if
        {
            nextToken();//(
            nextToken();//进入
            analyseCond();
            if (!peekToken(0).getType().equals("RPARENT"))
            {
                errors.put(thisToken.getLineNumber(),"j");
            }
            else
            {
                nextToken();//)
            }
            nextToken();//进
            analyseStmt();
            if (peekToken(0).getType().equals("ELSETK"))
            {
                nextToken();//else
                nextToken();//进
                analyseStmt();
            }
            grammar.add("<Stmt>");
            return;
        }
        else if (thisToken.getType().equals("PRINTFTK"))
        {
            //'printf''('StringConst {','Exp}')'';' // i j
            nextToken();//(
            nextToken();//str
            while (peekToken(0).getType().equals("COMMA"))
            {
                nextToken();//,
                nextToken();//进入
                analyseExp();
            }
            if (peekToken(0).getType().equals("SEMICN"))
            {
                errors.put(thisToken.getLineNumber(),"j");
            }
            else
            {
                nextToken();//)
                if (!peekToken(0).getType().equals("SEMICN"))
                {
                    errors.put(thisToken.getLineNumber(),"i");
                }
                else
                {
                    nextToken();//;
                }
            }
            grammar.add("<Stmt>");
            return;
        }
        else if(thisToken.getType().equals("RETURNTK"))
        {
            if (peekToken(0).getType().equals("SEMICN"))
            {
                nextToken();//;
                grammar.add("<Stmt>");
                return;
            }
            else
            {
                if (isExp())
                {
                    nextToken();
                    analyseExp();
                    if (peekToken(0).getType().equals("SEMICN"))
                    {
                        nextToken();//;
                    }
                    else
                    {
                        errors.put(thisToken.getLineNumber(),"i");
                    }
                }
                else
                {
                    errors.put(thisToken.getLineNumber(),"i");
                }
            }
            grammar.add("<Stmt>");
            return;
        }
        else if(thisToken.getType().equals("FORTK")) //'for' '(' [ForStmt] ';' [Cond] ';' [ForStmt] ')' Stmt
        {
            nextToken();//(
            if (!peekToken(0).getType().equals("SEMICN"))
            {
                nextToken();//进入
                analyseForStmt();
                nextToken();//;
            }
            else
            {
                nextToken();//;
            }
            if (!peekToken(0).getType().equals("SEMICN"))
            {
                nextToken();//进入
                analyseCond();
                nextToken();//;
            }
            else
            {
                nextToken();//;
            }
            if (!peekToken(0).getType().equals("RPARENT"))
            {
                nextToken();//进入
                analyseForStmt();
                nextToken();//)
            }
            else
            {
                nextToken();//)
            }
            nextToken();//进入
            analyseStmt();
            grammar.add("<Stmt>");
            return;
        }
        else if(thisToken.getType().equals("BREAKTK") || thisToken.getType().equals("CONTINUETK"))
        {
            if (peekToken(0).getType().equals("SEMICN"))
            {
                nextToken();//;
            }
            else
            {
                errors.put(thisToken.getLineNumber(),"i");
            }
            grammar.add("<Stmt>");
            return;
        }
        else if (thisToken.getType().equals("SEMICN"))
        {
            grammar.add("<Stmt>");
            return;
        }
        else
        {
            analyseExp();
            if (peekToken(0).getType().equals("SEMICN"))
            {
                nextToken();//;
            }
            else
            {
                errors.put(thisToken.getLineNumber(),"i");
            }
            grammar.add("<Stmt>");
        }

    }

    public boolean isExp()
    {
        String target = peekToken(0).getType();
        if (target.equals("PLUS") || target.equals("MINU") || target.equals("NOT") ||
                target.equals("IDENFR") || target.equals("LPARENT") || target.equals("INTCON") || target.equals("CHRCON"))
        {
            for (int i = 1;;i++)
            {
                if (peekToken(i).getType().equals("SEMICN"))
                {
                    return true;
                }
                if (peekToken(i).getType().equals("ASSIGN") || peekToken(i).getType().equals("RBRACE"))
                {
                    return false;
                }
            }
        }
        return false;
    }

    private void analyseForStmt() {
        //进入为第一项
        analyseLVal();
        nextToken();//=
        nextToken();//进
        analyseExp();
        grammar.add("<ForStmt>");
    }

    private void analyseCond() {
        analyseLOrExp();
        grammar.add("<Cond>");
    }

    private void analyseLOrExp() {
        int i = 0;
        do {
            if (i++ != 0 && peekToken(0).getType().equals("OR"))
            {
                grammar.add("<LOrExp>");
                nextToken();//||
                nextToken();//进
            }
            analyseLAndExp();
        }while (peekToken(0).getType().equals("OR"));
        grammar.add("<LOrExp>");
    }

    private void analyseLAndExp() {
        int i = 0;
        do {
            if (i++ != 0 && peekToken(0).getType().equals("AND"))
            {
                grammar.add("<LAndExp>");
                nextToken();//&&
                nextToken();//进
            }
            analyseEqExp();
        }while (peekToken(0).getType().equals("AND"));
        grammar.add("<LAndExp>");
    }

    private void analyseEqExp() {
        int i = 0;
        do {
            if (i++ != 0 && (peekToken(0).getType().equals("EQL") || peekToken(0).getType().equals("NEQ")))
            {
                grammar.add("<EqExp>");
                nextToken();//== !=
                nextToken();//进
            }
            analyseRelExp();
        }while (peekToken(0).getType().equals("EQL") || peekToken(0).getType().equals("NEQ"));
        grammar.add("<EqExp>");
    }

    private void analyseRelExp() {
        int i = 0;
        do {
            if (i++ != 0 && (peekToken(0).getType().equals("LSS") || peekToken(0).getType().equals("LEQ")||
            peekToken(0).getType().equals("GRE") || peekToken(0).getType().equals("GEQ")))
            {
                grammar.add("<RelExp>");
                nextToken();//== !=
                nextToken();//进
            }
            analyseAddExp();
        }while (peekToken(0).getType().equals("LSS") || peekToken(0).getType().equals("LEQ")||
                peekToken(0).getType().equals("GRE") || peekToken(0).getType().equals("GEQ"));
        grammar.add("<RelExp>");
    }

    private void analyseExp() {
        //进入时为第一个
        analyseAddExp();
        grammar.add("<Exp>");
    }

    public void writeTokens(FileProcessor fileProcessor) throws IOException {
        for (String temp : grammar) {
            fileProcessor.writeByLine(temp);
        }
    }

    // 输出错误信息
    public void writeErrors(FileProcessor fileProcessor) throws IOException {
        StringBuilder error = new StringBuilder();
        for (int temp : errors.keySet()) {
            error.append(temp);
            error.append(" ");
            error.append(errors.get(temp));
            fileProcessor.writeByLine(error.toString());
            error.replace(0, error.length(), "");
        }
    }
