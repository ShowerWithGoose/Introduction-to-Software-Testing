    private void identLogical() throws IOException {
        int line = lineNumber;
        int column = columnNumber;
        if (currentChar == '&') {
            advance();
            if (currentChar != '&') {
                errors.add(new Error(line, column, 'a'));
                words.add(new Word(line, column, "&&", Type.AND)); // @@ [Error: When a single '&' appears, the token value should be "&", not "&&". Adding the doubled operator misrepresents the input and may cause parsing errors.]
            } else {
                words.add(new Word(line, column, "&&", Type.AND));
                advance();
            }
        } else if (currentChar == '|') {
            advance();
            if (currentChar != '|') {
                errors.add(new Error(line, column, 'a'));
                words.add(new Word(line, column, "||", Type.OR)); // @@ [Error: Same issue for single '|'; should not add a word with the doubled operator.]
            } else {
                words.add(new Word(line, column, "||", Type.OR));
                advance();
            }
        } else {
            System.out.println("Error : not Logical Char");
        }
    }
