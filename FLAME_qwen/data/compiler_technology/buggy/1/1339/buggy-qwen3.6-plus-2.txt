    // PrimaryExp → '(' Exp ')' | LVal | Number | Character
    public PrimaryExp tryParsePrimaryExp() {
        switch (tokens.get().getType()) {
            case LPARENT -> { // @@ [Switch expression syntax (case X ->) requires Java 14+. If the compilation environment uses Java 8 or 11, this will cause a compilation error.]
                tokens.consume();
                Exp exp = tryParseExp();
                if (exp == null) {
                    return null;
                }
                tokens.consumeExpect(TokenType.RPARENT);
                return new PrimaryExp(exp);
            }
            case INTCON -> {
                Number number = tryParseNumber();
                if (number == null) {
                    return null;
                }
                return new PrimaryExp(number);
            }
            case CHRCON -> {
                Character character = tryParseCharacter();
                if (character == null) {
                    return null;
                }
                return new PrimaryExp(character);
            }
            case IDENFR -> {
                LVal lVal = tryParseLVal();
                if (lVal == null) {
                    return null;
                }
                return new PrimaryExp(lVal);
            }
            default -> {
                return null;
            }
        }
    }
