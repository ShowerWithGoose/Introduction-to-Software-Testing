    public void parseDecl(){
        //如果预读的第二个是(结束
        while (tokenList.hasNext()){
            if (tokenList.preRead(2).getType().equals(TokenType.LPARENT)){ // @@ [Incorrect lookahead: checking preRead(2) assumes at least 3 tokens ahead, but Decl can be followed directly by FuncDef starting with 'int' or 'char', and the '(' belongs to FuncDef's parameter list. This causes premature termination of Decl parsing.]
                //有(，说明是函数了
                return;
            }
            Decl decl = new Decl(tokenList);
            decl.parseDecl();
            declArrayList.add(decl);
        }
    }
    public void parseFuncDef(){
        //如果预读的第一个是main结束
        while (tokenList.hasNext()){
            if (tokenList.preRead(1).getType().equals(TokenType.MAINTK)){ // @@ [Incorrect lookahead: checking preRead(1) assumes at least 2 tokens ahead, but MainFuncDef starts with 'int main', so when current token is 'int', preRead(1) is 'main'. However, this logic fails when there are no FuncDefs and CompUnit goes directly from Decl to MainFuncDef, causing FuncDef parsing to incorrectly consume the 'int' of MainFuncDef.]
                //有main，说明是主函数了
                return;
            }
            FuncDef funcDef = new FuncDef(tokenList);
            funcDef.parseFuncDef();
            funcDefArrayList.add(funcDef);
        }
    }
