    private Stmt parseAssignmentOrExpStmt(LexType ident) {
        int step = 0;
        ArrayList<LexType> symbols = new ArrayList<>();
        LexType token = tokens.getAndMove();
        step++;

        while (true) {
            token = tokens.getAndMove();
            step++;

            if (token.getType() == LexType.TokenType.ASSIGN) {
                tokens.retreat(step);
                LVal lVal = new ExpParser(tokens, errorLog).parserLVal();
                LexType assign = tokens.getAndMove();
                symbols.add(assign);

                token = tokens.getToken();
                if (token.getType() == LexType.TokenType.GETINTTK ||
                        token.getType() == LexType.TokenType.GETCHARTK) {
                    return parseGetStmt(lVal, symbols);
                } else {
                    return parseAssignStmt(lVal, assign);
                }
            } else if (token.getType() == LexType.TokenType.SEMICN) {
                tokens.retreat(step);
                ExpStmt expStmt = new ExpParser(tokens, errorLog).parserExp();
                return new Stmt(expStmt, tokens);
            } else if (!Objects.equals(token.getLine(), ident.getLine())) {
                tokens.retreat(step);
                ExpStmt expStmt = new ExpParser(tokens, errorLog).parserExp();
                errorLog.add(tokens.getlastToken(1).getLine(), 'i'); // @@ [This error reporting is incorrect because it adds error 'i' (missing semicolon) when encountering a line break, but the actual issue might be that the expression parsing should have consumed more tokens or the logic for detecting end of statement is flawed. The error line should correspond to the last non-terminal before the expected semicolon, not necessarily the previous token.]
                return new Stmt(expStmt, tokens);
            }
        }
    }
