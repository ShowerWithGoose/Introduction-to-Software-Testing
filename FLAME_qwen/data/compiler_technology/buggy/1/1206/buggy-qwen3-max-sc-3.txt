        ArrayList<FunDefNode> funDefNodes = new ArrayList<>();
        MainFuncDefNode mainFuncDefNode;
        while (this.tokens.get(i+1).type!= token.Type.MAINTK&&
        this.tokens.get(i+2).type!= token.Type.LPARENT) {//不为main或函数声明，即为变量声明 @@ [This condition is flawed: it accesses tokens beyond current index without checking bounds, and the logic for distinguishing declarations from function definitions is incorrect. Function definitions start with FuncType (void/int/char) followed by Ident '(', but this checks for MAINTK and LPARENT at i+1 and i+2 which doesn't properly identify function definitions.]
            DeclNode declNode=Decl();
            declNodes.add(declNode);
        }
        while (this.tokens.get(i+1).type!= token.Type.MAINTK) {//不为main，即为函数声明 @@ [This condition is also flawed: it only checks if the next token is MAINTK, but function definitions don't necessarily end right before main. Also accesses i+1 without bounds checking, which can cause IndexOutOfBoundsException when parsing programs that end with main function.]
            FunDefNode funDefNode=FuncDef();
            funDefNodes.add(funDefNode);
        }
        mainFuncDefNode=MainFuncDef();
        return new CompUnitNode(declNodes,funDefNodes,mainFuncDefNode);
    }
