    public Node parseCompUnit() {
        Node root = new CompUnit(); root.setBeginLine(curToken.getLineNum());
        while (curToken != null) {
            Node child = null;
            if (peekOffset(1).getLexType() == LexType.MAINTK) { // @@ [This condition is incorrect because it doesn't verify that the current token is 'int' before 'main', which is required by MainFuncDef → 'int' 'main' '(' ')' Block. This can misclassify other constructs as MainFuncDef.]
                child = parseMainFuncDef();
            } else if(peekOffset(2).getLexType() == LexType.LPARENT) { // @@ [This lookahead is unreliable for distinguishing FuncDef from VarDecl/ConstDecl, as it may incorrectly identify a variable declaration with a complex initializer as a function definition. The correct approach should check if the current token is a valid FuncType followed by an identifier and then '('.]
                child = parseFuncDef();
            } else if (curToken.getLexType() == LexType.CONSTTK) {
                child = parseConstDecl();
            } else if (isBType(curToken.getLexType())) {
                child = parseVarDecl();
            } else {
                System.out.println("Error: parseCompUnit error");
            }

            if (child != null) {
                root.addChild(child);
            }
        }
        root.setEndLine(peekOffset(-1).getLineNum());
        return root;
    }
