        Node root = new CompUnit(); root.setBeginLine(curToken.getLineNum());
        while (curToken != null) {
            Node child = null;
            if (peekOffset(1).getLexType() == LexType.MAINTK) {
                child = parseMainFuncDef();
            } else if(peekOffset(2).getLexType() == LexType.LPARENT) { // @@ [This lookahead logic is flawed: it assumes that any BType followed by an identifier and then '(' must be a FuncDef, but it doesn't verify that the current token is a BType or that the next token is an identifier. Also, it may skip over VarDecl that starts with BType when the following tokens accidentally match this pattern.]
                child = parseFuncDef();
            } else if (curToken.getLexType() == LexType.CONSTTK) {
                child = parseConstDecl();
            } else if (isBType(curToken.getLexType())) {
                child = parseVarDecl();
            } else {

        root.setEndLine(peekOffset(-1).getLineNum());
