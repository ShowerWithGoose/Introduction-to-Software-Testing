    public Token match(Token.Type type){
        //System.out.println(now.lineno+" "+now.type+" "+now.content+" "+type+"\n");
        if(now.type==type){
            Token re=now;
            if(curPos<tokens.size()-1){
                now=tokens.get(++curPos);
            }
            return re;
        }
        else{
            isError=true;
            if(type==Token.Type.SEMICN){
                errorLine.put(tokens.get(curPos-1).lineno,'i'); // @@ [Incorrect line number for missing semicolon: should be the line of the last token in the current statement, not necessarily curPos-1]
                Error.getError().errors.add(new Errortype(tokens.get(curPos-1).lineno,'i'));
            }
            else if(type==Token.Type.RPARENT){
                //System.out.println(now.type);
                errorLine.put(tokens.get(curPos-1).lineno,'j'); // @@ [Incorrect line number for missing right parenthesis: should be the line of the last token before the expected ')', which may not be curPos-1]
                Error.getError().errors.add(new Errortype(tokens.get(curPos-1).lineno,'j'));
            }
            else if(type==Token.Type.RBRACK){
                errorLine.put(tokens.get(curPos-1).lineno,'k'); // @@ [Incorrect line number for missing right square bracket: should be the line of the last token before the expected ']', which may not be curPos-1]
                Error.getError().errors.add(new Errortype(tokens.get(curPos-1).lineno,'k'));
            }
            return new Token(Token.Type.NONE,"",0);//没匹配到
        }
    }
