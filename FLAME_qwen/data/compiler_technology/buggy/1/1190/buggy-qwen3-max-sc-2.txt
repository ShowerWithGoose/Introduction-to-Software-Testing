    private Stmt parseStmt() {
        Stmt stmt = new Stmt();
        switch (currentToken().getType()) {
            case IDENFR:
                Exp tmpExp = parseExp();
                if (!match(Token.Type.ASSIGN)) {
                    stmt.setType(2);
                    stmt.addExp(tmpExp); // Exp
                    if (match(Token.Type.SEMICN)) {
                        stmt.addSemicn(currentToken()); // ';'
                        advance();
                    } else {
                        peekToken(-1).setErrorCode('i');
                    }
                    break;
                }
