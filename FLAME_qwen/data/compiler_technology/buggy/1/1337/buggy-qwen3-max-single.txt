    std::ofstream errorFile("error.txt", std::ios::app);
    errorFile << Token::tokenList[Token::tokenIndex - 1].lineNum << " " << error << std::endl; // @@ [The line number should be the passed 'lineNum' parameter, not Token::tokenList[Token::tokenIndex - 1].lineNum.
