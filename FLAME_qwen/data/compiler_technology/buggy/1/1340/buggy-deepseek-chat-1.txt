    private CompUnit analyseCompUnit(){
        CompUnit compUnit = new CompUnit();
        while(isEqualWordType(0,WordType.CONSTTK)
                    ||!isEqualWordType(2,WordType.LPARENT)){ // @@ The condition `!isEqualWordType(2,WordType.LPARENT)` is incorrect. It should check if the current token is not a function definition (i.e., not an identifier followed by '('), but this logic is flawed. For example, when encountering a variable declaration like `int a;`, `isEqualWordType(2,WordType.LPARENT)` will be false (since there is no '(' at position 2), so `!isEqualWordType(2,WordType.LPARENT)` is true, causing the loop to continue incorrectly. This leads to infinite loops or incorrect parsing of declarations.
                compUnit.declList.add(analyseDecl());
        }
        while(isEqualWordType(0,WordType.VOIDTK)
                    ||(isEqualWordType(1,WordType.IDENFR)
                        && isEqualWordType(2,WordType.LPARENT))) { // @@ The condition for function definitions is also problematic. It only checks for `void` or an identifier followed by '(', but does not handle cases like `int func()` where the return type is `int`. This will cause function definitions with return type `int` or `char` to be missed, leading to incorrect parsing.
                compUnit.funcDefList.add(analyseFuncDef());
        }
        if(isEqualWordType(0,WordType.INTTK)
            && isEqualWordType(1,WordType.MAINTK)
            && isEqualWordType(2,WordType.LPARENT)){
            compUnit.mainFuncDef = analyseMainFuncDef();
        }
        syntaxRes.add("<"+SyntaxType.CompUnit+">");
        return compUnit;
    }
