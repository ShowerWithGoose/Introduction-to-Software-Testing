    private CompUnit analyseCompUnit(){
        CompUnit compUnit = new CompUnit();
        while(isEqualWordType(0,WordType.CONSTTK)
                    ||!isEqualWordType(2,WordType.LPARENT)){ // @@ [This condition is incorrect: it assumes that any token sequence where the third token is not '(' must be a Decl, but a FuncDef starting with 'int' or 'char' (non-void) would also not have '(' at position 2 during lookahead, causing it to be misparsed as a Decl. The correct approach should check for BType followed by Ident and then '(', but this loop greedily consumes non-const tokens as Decl without verifying they are actually declarations.]
                compUnit.declList.add(analyseDecl());
        }
        while(isEqualWordType(0,WordType.VOIDTK)
                    ||(isEqualWordType(1,WordType.IDENFR)
                        && isEqualWordType(2,WordType.LPARENT))) {
                compUnit.funcDefList.add(analyseFuncDef());
        }
        if(isEqualWordType(0,WordType.INTTK)
            && isEqualWordType(1,WordType.MAINTK)
            && isEqualWordType(2,WordType.LPARENT)){
            compUnit.mainFuncDef = analyseMainFuncDef();
        }
        syntaxRes.add("<"+SyntaxType.CompUnit+">");
        return compUnit;
    }
