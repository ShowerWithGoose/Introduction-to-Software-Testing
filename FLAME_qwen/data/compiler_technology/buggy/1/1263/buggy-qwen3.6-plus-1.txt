    public void consume_step(Tokentype type, String value) {
        Token token = gettoken(pos);
        if (consume(token, type, value)) step();
        else {
            String err = type.equals(Tokentype.RPARENT) ? "j" : 
                type.equals(Tokentype.RBRACK) ? "k" : 
                type.equals(Tokentype.SEMICN) ? "i" : null; // @@ err is assigned null for other missing tokens, which causes a NullPointerException in the subsequent err.equals("i") check
            errcnt ++;
            int line = gettoken(pos - 1).getline(); // @@ Potential IndexOutOfBoundsException if pos is 0, and lacks bounds checking for safe token access
            if (errcnt == 1 && gettoken(pos).getline() - line == 2) { // @@ gettoken(pos) may return null if pos equals tokens.size(), causing a NullPointerException
                if (err.equals("i")) { // @@ NullPointerException because err can be null
                    return;
                }
            }

            Error e = new Error(err, line);
            errors.adderror(e);
        }
    }

    public boolean isLval() {

        if (!ty.equals(Tokentype.IDENFR)) return false;
        Tokentype next = gettoken(pos + 1).gettype(); // @@ IndexOutOfBoundsException or NullPointerException if pos + 1 exceeds the token list bounds
