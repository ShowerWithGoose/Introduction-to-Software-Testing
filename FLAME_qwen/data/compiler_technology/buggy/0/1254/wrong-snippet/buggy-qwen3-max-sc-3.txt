    private Token parseLogic() throws IOException {
        if (nextChar == '&') {
            getChar();
            if (nextChar == '&') {
                return new Token("&&", lineNum);
            } else {
                unGetChar();
                errors.add(lineNum);
                return new Token(TokenType.ERRORA, "&", lineNum);
            }
        } else if (nextChar == '|') {
            getChar();
            if (nextChar == '|') {
                return new Token("||", lineNum);
            } else {
                unGetChar();
                errors.add(lineNum);
                return new Token(TokenType.ERRORA, "|", lineNum);
            }
        }
        return null;
    }
