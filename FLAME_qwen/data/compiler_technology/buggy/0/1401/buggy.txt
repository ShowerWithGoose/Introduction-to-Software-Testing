            }
            words.back();
            Words words1 = new Words(cl.getName().toString().replace(",","").replace("[","").replace("]","").replace(" ",""));

            Class cl2 = new Class();
            cl2.add(character);

                case 'c':

                    if(i(cl2,words1)){

                        return;

                    cl.IDENFR();
                    phrase.addclass(cl);
                    return;

        }
        if (Character.isDigit(character)) {
            character = words.get();
            while (Character.isDigit(character)) {
                cl.add(character);
                character = words.get();

            words.back();

package src;

public class Words {
    public String string;
    public Integer i;

    public Words(String string) {
        this.string=string;
        i=0;
    }
    public Character get(){
        if (i<string.length()) {
            Character a = string.charAt(i);
            i++;
            return a;
        }return '\n';
    }
    public void back(){
        i--;
    }
    public boolean max(){
        return i >= string.length()-1;
    }
