enum TokenType {
    IDENFR, ELSETK, VOIDTK, SEMICN,

struct Token {
    TokenType type;
    std::string value;
    bool haveError() const {
        return type == ERROR;
    }
    void print(std::ostream& os) const {

class Lexer {
public:
    Lexer(const std::string &source) : source(source), position(0) {}

    std::vector<Token> tokenize() {
        std::vector<Token> tokens;
        while (position < source.size()) {
            skipWhitespace();
            if (position >= source.size()) break;

            char current = source[position];
            if (current == '/') {
                // Check for comments
                if (position + 1 < source.size() && source[position + 1] == '/') {
                    skipLineComment();  // Skip single-line comment
                } else if (position + 1 < source.size() && source[position + 1] == '*') {
                    skipBlockComment();  // Skip multi-line comment
                } else {
                    tokens.push_back(handleSpecialTokens());
                }
            } else if (isalpha(current)||current=='_') {
                tokens.push_back(identifier());
            } else if (isdigit(current)) {
                tokens.push_back(integer());
            } else if (current == '"') {
                tokens.push_back(stringLiteral());
            } else if (current == '\'') {
                tokens.push_back(charLiteral());
            } else {
                tokens.push_back(handleSpecialTokens());
            }
        }
        return tokens;
    }

private:
    std::string source;
    int position;

    void skipWhitespace() {
        while (position < source.size() && isspace(source[position])) {
            if(source[position]=='\n')
            {
                line++;
            }
            position++;
        }
    }
    void skipLineComment() {
        while (position < source.size() && (source[position] != '\n'&&source[position] != EOF)){
            position++;
        }
        line++;
    }

    void skipBlockComment() {
        position += 2;

        while (position + 1 < source.size() && !(source[position] == '*' && source[position + 1] == '/')) {
            if(source[position]=='\n')
            {
                line++;
            }
            position++;
        }
        position += 2;
    }
    Token identifier() {
        int start = position;
        while (position < source.size() && (isalnum(source[position]) || source[position] == '_')) {
            position++;
        }
        std::string id = source.substr(start, position - start);
        if (id == "const") return {CONSTTK, id};
        if (id == "char") return {CHARTK,id};
        if (id == "int") return {INTTK, id};
        if (id == "void") return {VOIDTK, id};
        if (id == "main") return {MAINTK, id};
        if (id == "for") return {FORTK, id};
        if (id == "break") return {BREAKTK, id};
        if (id == "continue") return {CONTINUETK, id};
        if (id == "if") return {IFTK, id};
        if (id == "getint") return {GETINTTK, id};
        if (id == "getchar") return {GETCHARTK, id};
        if (id == "printf") return {PRINTFTK, id};
        if (id == "return") return {RETURNTK, id};
        if (id == "else") return{ELSETK,id};
        return {IDENFR, id};
    }

    Token integer() {
        int start = position;
        while (position < source.size() && isdigit(source[position])) {
            position++;
        }
        return {INTCON, source.substr(start, position - start)};
    }

    Token stringLiteral() {
        int start = position;
        position++;
        while (position < source.size() && (source[position] != '"')) {
            position++;
        }
        if (position < source.size()) {
            position++; // Skip the closing quote
            return {STRCON, source.substr(start , position - start)}; // Exclude quotes
        }
        return {ERROR, ""};
    }

    Token charLiteral() {
        int start = position;
        position++;
        if(source[position]=='\\'&&source[position+1]=='0')
        {
            position += 3;
            return {CHRCON,"'\\0'"};
        }
        if(source[position]=='\\'&&source[position+1]=='\'')
        {
            position += 3;
            return {CHRCON,"'\\\''"};
        }
        if (position < source.size()) {
            position++;
            if (position < source.size() && source[position] == '\'') {
                position++;

                return {CHRCON, source.substr(start, position - start)};
            }
        }


    }

    Token handleSpecialTokens() {
        char current = source[position];
        switch (current) {
            case ';': position++; return {SEMICN, ";"};
            case ',': position++; return {COMMA, ","};
            case '{': position++; return {LBRACE, "{"};
            case '}': position++; return {RBRACE, "}"};
            case '(': position++; return {LPARENT, "("};
            case ')': position++; return {RPARENT, ")"};
            case '[': position++; return {LBRACK, "["};
            case ']': position++; return {RBRACK, "]"};
            case '+': position++; return {PLUS, "+"};
            case '-': position++; return {MINU, "-"};
            case '*': position++; return {MULT, "*"};
            case '/': position++; return {DIV, "/"};
            case '%': position++;return {MOD, "%"};
            case '<': position++;
                if (position < source.size() && source[position] == '=') {
                    position++;
                    return {LEQ, "<="};
                }
                return {LSS, "<"};
            case '>': position++;
                if (position < source.size() && source[position] == '=') {
                    position++;
                    return {GEQ, ">="};
                }
                return {GRE, ">"};

            case '&': position++;
                if (position  < source.size() && source[position] == '&') {
                    position += 1;
                    return {AND, "&&"};
                }
                return {ERROR,std::to_string(line+1) };
            case '|': position++;
                if (position  < source.size() && source[position] == '|') {
                    position += 1;
                    return {OR, "||"};
                }
                return {ERROR, std::to_string(line+1)};
            case '=': position++;
                if (position + 1 < source.size() && source[position] == '=') {
                    position += 1;
                    return {EQL, "=="};
                }
                return {ASSIGN, "="};
            case '!': position++;
                if (position + 1 < source.size() && source[position] == '=') {
                    position += 1;
                    return {NEQ, "!="};
                }
                return {NOT, "!"};

        }
    }
};
