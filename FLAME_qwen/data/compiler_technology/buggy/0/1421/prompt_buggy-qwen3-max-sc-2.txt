package frontend;

import error.Error;
import error.ErrorList;
import token.Token;
import token.TokenList;

public class Lexer {
    public static String rawCode;
    public static int pos;

    public static String getWord() {
        String value = "";
        char ch = rawCode.charAt(pos);
        value += ch;
        while (true) {
            pos++;
            ch = rawCode.charAt(pos);
            if (Character.isLetter(ch) || ch == '_' || Character.isDigit(ch)) {
                value += ch;
            } else {
                pos--;
                break;
            }
        }
        return value;
    }

    public static String getNum() {
        String value = "";
        char ch = rawCode.charAt(pos);
        value += ch;
        while (true) {
            pos++;
            ch = rawCode.charAt(pos);
            if (Character.isDigit(ch)) {
                value += ch;
            } else {
                pos--;
                break;
            }
        }
        return value;
    }

    public static String getStr() {
        StringBuilder value = new StringBuilder();
        char ch = rawCode.charAt(pos);
        value.append(ch);
        boolean flag = false;
        while (true) {
            pos++;
            ch = rawCode.charAt(pos);
            if (ch == '\\' && !flag) {
                flag = true;
                value.append(ch);
            } else if (flag) {
                flag = false;
                value.append(ch);
            } else if (ch == '\"') {
                value.append(ch);
                break;
            } else {
                value.append(ch);
            }
        }
        return value.toString();
    }

    public static String getChr() {
        StringBuilder value = new StringBuilder();
        char ch = rawCode.charAt(pos);
        value.append(ch);
        boolean flag = false;
        while (true) {
            pos++;
            ch = rawCode.charAt(pos);
            if (ch == '\\' && !flag) {
                flag = true;
                value.append(ch);
            } else if (flag) {
                flag = false;
                value.append(ch);
            } else if (ch == '\'') {
                value.append(ch);
                break;
            } else {
                value.append(ch);
            }
        }
        return value.toString();
    }

    public static void lexer(TokenList tl, ErrorList er) {
        int line = 1;
        int length = rawCode.length();

        for (pos = 0; pos < length; pos++) {
            char ch = rawCode.charAt(pos);
            Token.TokenType type = null;
            String value = "";

            if (Character.isLetter(ch) || ch == '_') {
                // IDENFR *TK
                value = getWord();
                type = Token.TK.getOrDefault(value, Token.TokenType.IDENFR);
            } else if (Character.isDigit(ch)) {
                // INTCON
                value = getNum();
                type = Token.TokenType.INTCON;
            } else if (ch == '\"') {
                // STRCON
                value = getStr();
                type = Token.TokenType.STRCON;
            } else if (ch == '\'') {
                // CHRCON
                value = getChr();
                type = Token.TokenType.CHRCON;
            } else if (Token.SINGLE.containsKey(ch)) {
                value += ch;
                type = Token.SINGLE.get(ch);
            } else if (ch == '!') {
                // NOT NEQ
                value += ch;
                pos++;
                ch = rawCode.charAt(pos);
                if (ch == '=') {
                    value += ch;
                    type = Token.TokenType.NEQ;
                } else {
                    pos--;
                    type = Token.TokenType.NOT;
                }
            } else if (ch == '&') {
                value += ch;
                pos++;
                ch = rawCode.charAt(pos);
                if (ch == '&') {
                    value += ch;
                    type = Token.TokenType.AND;
                } else {
                    pos--;
                    er.addError(new Error(line, 'a'));
                }
            } else if (ch == '|') {
                value += ch;
                pos++;
                ch = rawCode.charAt(pos);
                if (ch == '|') {
                    value += ch;
                    type = Token.TokenType.OR;
                } else {
                    pos--;
                    er.addError(new Error(line, 'a'));
                }
            } else if (ch == '<') {
                value += ch;
                pos++;
                ch = rawCode.charAt(pos);
                if (ch == '=') {
                    value += ch;
                    type = Token.TokenType.LEQ;
                } else {
                    pos--;
                    type = Token.TokenType.LSS;
                }
            } else if (ch == '>') {
                value += ch;
                pos++;
                ch = rawCode.charAt(pos);
                if (ch == '=') {
                    value += ch;
                    type = Token.TokenType.GEQ;
                } else {
                    pos--;
                    type = Token.TokenType.GRE;
                }
            } else if (ch == '=') {
                value += ch;
                pos++;
                ch = rawCode.charAt(pos);
                if (ch == '=') {
                    value += ch;
                    type = Token.TokenType.EQL;
                } else {
                    pos--;
                    type = Token.TokenType.ASSIGN;
                }
            } else if (ch == '/') {
                value += ch;
                pos++;
                ch = rawCode.charAt(pos);
                if (ch == '/') {
                    pos++;
                    ch = rawCode.charAt(pos);
                    while (ch != '\n') {
                        pos++;
                        ch = rawCode.charAt(pos);
                    }
                    line++;
                } else if (ch == '*') {
                    pos++;
                    ch = rawCode.charAt(pos);
                    while (true) {
                        while (ch != '*') {
                            pos++;
                            ch = rawCode.charAt(pos);
                        }
                        pos++;
                        ch = rawCode.charAt(pos);
                        if (ch == '/') {
                            break;
                        }
                    }
                } else {
                    pos--;
                    type = Token.TokenType.DIV;
                }
            } else if (ch == '\n') {
                line++;
            }

            if (type != null) {
                tl.addToken(new Token(type, value, line));
            }
        }
    }
}

package token;

import config.Config;

import java.util.HashMap;
import java.util.Map;

public class Token {
    public enum TokenType {
        IDENFR,
        INTCON,
        STRCON,
        CHRCON,
        MAINTK, CONSTTK, INTTK, CHARTK, BREAKTK, CONTINUETK, RETURNTK, VOIDTK,
        IFTK, ELSETK,
        NOT, AND, OR,
        FORTK,
        GETINTTK, GETCHARTK, PRINTFTK,
        PLUS, MINU, MULT, DIV, MOD,
        LSS, LEQ, GRE, GEQ, EQL, NEQ, ASSIGN,
        SEMICN, COMMA,
        LPARENT, RPARENT,
        LBRACK, RBRACK,
        LBRACE, RBRACE
    }

    public static Map<String, TokenType> TK = new HashMap<>() {{
        put("main", TokenType.MAINTK);
        put("const", TokenType.CONSTTK);
        put("int", TokenType.INTTK);
        put("char", TokenType.CHARTK);
        put("break", TokenType.BREAKTK);
        put("continue", TokenType.CONTINUETK);
        put("return", TokenType.RETURNTK);
        put("void", TokenType.VOIDTK);
        put("if", TokenType.IFTK);
        put("else", TokenType.ELSETK);
        put("getint", TokenType.GETINTTK);
        put("getchar", TokenType.GETCHARTK);
        put("printf", TokenType.PRINTFTK);
    }};

    public static Map<Character, TokenType> SINGLE = new HashMap<>() {{
        put('+', TokenType.PLUS);
        put('-', TokenType.MINU);
        put('*', TokenType.MULT);
        put('%', TokenType.MOD);
        put(';', TokenType.SEMICN);
        put(',', TokenType.COMMA);
        put('(', TokenType.LPARENT);
        put(')', TokenType.RPARENT);
        put('[', TokenType.LBRACK);
        put(']', TokenType.RBRACK);
        put('{', TokenType.LBRACE);
        put('}', TokenType.RBRACE);
    }};

    TokenType type;
    String value;
    int line;

    public Token(TokenType type, String value, int line) {
        this.type = type;
        this.value = value;
        this.line = line;
    }

    @Override
    public String toString() {
        StringBuilder sb = new StringBuilder();
        sb.append(type);
        sb.append(" ");
        sb.append(value);
        if (Config.TOKEN_FULL_DESC){
            sb.append(" ");
            sb.append(line);
        }
        return sb.toString();
    }
}

package utils;

import java.io.*;
import java.nio.file.Files;
import java.nio.file.Path;
import java.util.Scanner;

import config.Config;

public class IOStream {
    public enum pathList{
        lexer,
        error
    }
    public static void init() throws IOException {
        for (pathList path: pathList.values()){
            if (Files.exists(Path.of(path + ".txt"))) {
                Files.delete(Path.of(path + ".txt"));
            }
        }
    }

    public static String getPath() {
        if (Config.DIY_TESTFILE == 0) {
            return "testfile.txt";
        } else {
            return String.format("../data/testfile%d.txt", Config.DIY_TESTFILE);
        }
    }

    public static String readFile() throws IOException {
        String path = getPath();
        StringBuilder sb = new StringBuilder();
        BufferedReader br = new BufferedReader(new FileReader(path));
        String line;

        while ((line = br.readLine()) != null) {
            sb.append(line + "\n");
        }
        return sb.toString();
    }

    public static void writeLexer(String tokens) throws IOException {
        try (BufferedWriter bufferedWriter = new BufferedWriter(new FileWriter("lexer.txt"))) {
            bufferedWriter.write(tokens);
        }
    }

    public static void writeError(String errors) throws IOException {
        try (BufferedWriter bufferedWriter = new BufferedWriter(new FileWriter("error.txt"))) {
            bufferedWriter.write(errors);
        }
    }
}
