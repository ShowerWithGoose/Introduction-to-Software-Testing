    private boolean skipComment() {
        if (getChar() == '/') {
            if (charNum + 1 >= getLine().length()) {
                return false;
            } else if (getLine().charAt(charNum + 1) == '/') {
                lineNum++;
                charNum = 0;
                return true;
            } else if (getLine().charAt(charNum + 1) == '*') {
                next();
                while (true) { // @@ Missing isEnd() exit condition: if the block comment is never closed, this loops infinitely and causes an out-of-bounds crash at end-of-file
                    next();
                    if (getChar() == '*') {
                        if (charNum + 1 < getLine().length() && getLine().charAt(charNum + 1) == '/') {
                            next();
                            next();
                            return true;
                        }
                    }
                }
            }
        }
        return false;
    }
