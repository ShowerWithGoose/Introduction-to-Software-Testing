char getChar(FILE *file) { // @@ `fgetc` returns an `int`, but assigning it to `char` may lose the `EOF` value (typically -1) if `char` is unsigned, causing incorrect EOF detection and potential infinite loops.
    char ch;
    if ((ch = fgetc(file)) != EOF) {
        return ch;
    } else {
        return EOF;
    }
}
