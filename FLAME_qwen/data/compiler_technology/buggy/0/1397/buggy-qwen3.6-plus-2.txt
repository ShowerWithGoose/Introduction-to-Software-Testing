    private Token parseIdent() throws IOException {
        StringBuilder buf = new StringBuilder();
        buf.append(ch);
        read();
        while (ch == '_' || Character.isLetter(ch) || Character.isDigit(ch)) {
            buf.append(ch);
            read();
        }
        unread();
        String ident = buf.toString();
        if (ident == "main") { // @@ In Java, '==' compares object references, not string content. Use 'ident.equals("main")' to correctly compare string values.
            return newToken(TokenType.MAINTK, ident);
        } else if (ident == "const") { // @@ Same error: '==' compares references. Use '.equals()' for string content comparison.
            return newToken(TokenType.CONSTTK, ident);
        } else if (ident == "int") { // @@ Same error: '==' compares references. Use '.equals()' for string content comparison.
            return newToken(TokenType.INTTK, ident);
        } else if (ident == "char") { // @@ Same error: '==' compares references. Use '.equals()' for string content comparison.
            return newToken(TokenType.CHARTK, ident);
        } else if (ident == "break") { // @@ Same error: '==' compares references. Use '.equals()' for string content comparison.
            return newToken(TokenType.BREAKTK, ident);
        } else if (ident == "continue") { // @@ Same error: '==' compares references. Use '.equals()' for string content comparison.
            return newToken(TokenType.CONTINUETK, ident);
        } else if (ident == "if") { // @@ Same error: '==' compares references. Use '.equals()' for string content comparison.
            return newToken(TokenType.IFTK, ident);
        } else if (ident == "else") { // @@ Same error: '==' compares references. Use '.equals()' for string content comparison.
            return newToken(TokenType.ELSETK, ident);
        } else if (ident == "for") { // @@ Same error: '==' compares references. Use '.equals()' for string content comparison.
            return newToken(TokenType.FORTK, ident);
        } else if (ident == "getint") { // @@ Same error: '==' compares references. Use '.equals()' for string content comparison.
            return newToken(TokenType.GETINTTK, ident);
        } else if (ident == "getchar") { // @@ Same error: '==' compares references. Use '.equals()' for string content comparison.
            return newToken(TokenType.GETCHARTK, ident);
        } else if (ident == "printf") { // @@ Same error: '==' compares references. Use '.equals()' for string content comparison.
            return newToken(TokenType.PRINTFTK, ident);
        } else if (ident == "return") { // @@ Same error: '==' compares references. Use '.equals()' for string content comparison.
            return newToken(TokenType.RETURNTK, ident);
        } else if (ident == "void") { // @@ Same error: '==' compares references. Use '.equals()' for string content comparison.
            return newToken(TokenType.VOIDTK, ident);
        } else {
            return newToken(TokenType.IDENFR, ident);
        }
    }
