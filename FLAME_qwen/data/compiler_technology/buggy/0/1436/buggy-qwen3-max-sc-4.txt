    class KeywordParser : public Parser
    {
    private:
        string target;
        TokenEnum key;

    public:
        KeywordParser(string target, TokenEnum key)
        {
            this->target = target;
            this->key = key;
        }
        ParseRes parse(string buf, char next) override
        {
            if (buf.size() == target.size())
            {
                return {true, false};
            }
            if (target[buf.size()] == next)
            {
                return {false, true};
            }
            return {false, false};
        }
        TokenEnum getKey() override
        {
            return key;
        }
    };
