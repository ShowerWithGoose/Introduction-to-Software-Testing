package Error;

public class error {
    char type;
    int lineno;
    public error(char type, int lineno){
        this.type = type;
        this.lineno = lineno;
    }
    public error(){}

    @Override
    public String toString(){
        StringBuffer sbf = new StringBuffer();
        sbf.append(this.lineno);
        sbf.append(' ');
        sbf.append(this.type);
        sbf.append('\n');
        return sbf.toString();
    }
}

package Token;

public class Token {
    TokenType type;
    String content;
    int lineno;
    public Token(TokenType tokentype, String content, int lineno){
        this.type = tokentype;
        this.content = content;
        this.lineno = lineno;
    }
    @Override
    public String toString(){
        return type.toString() + ' ' + content + '\n';
    }

}
