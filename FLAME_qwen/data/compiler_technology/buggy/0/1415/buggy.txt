import java.io.BufferedReader;

import java.io.IOException;

public class Compiler {
    public static void main(String[] args) {
        String filePath = "testfile.txt"; // 文件路径
        StringBuilder contentBuilder = new StringBuilder();

        try (BufferedReader br = new BufferedReader(new FileReader(filePath))) {
            String line;
            while ((line = br.readLine()) != null) {
                contentBuilder.append(line).append(System.lineSeparator());
            }
            String content = contentBuilder.toString();
            Lexical lexical = new Lexical(content);
            lexical.lexical_analysis();
            lexical.lexical_print();
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

import java.io.BufferedWriter;
import java.io.FileWriter;
import java.io.IOException;
import java.util.ArrayList;
import java.util.HashMap;

public class Lexical {
    private final String content;
    private final ArrayList<CategoryCode> categoryCodes;
    private final ArrayList<String> words;
    private char curChar;
    private int curCharIndex;
    private final StringBuilder token;
    private HashMap<String,CategoryCode> reserveCode;
    private int line;
    private Boolean error;
    private char errorText;
    private int errorLine;

    Lexical(String content) {
        this.content = content;
        this.categoryCodes = new ArrayList<>();
        this.words = new ArrayList<>();
        this.curCharIndex = 0;
        this.token = new StringBuilder();
        this.curChar = content.charAt(curCharIndex);
        init_reserveCode();
        this.line = 1;
        this.error = false;
    }

    private void init_reserveCode() {
        reserveCode = new HashMap<>();
        reserveCode.put("main",CategoryCode.MAINTK);
        reserveCode.put("const",CategoryCode.CONSTTK);
        reserveCode.put("int",CategoryCode.INTTK);
        reserveCode.put("char",CategoryCode.CHARTK);
        reserveCode.put("break",CategoryCode.BREAKTK);
        reserveCode.put("continue",CategoryCode.CONTINUETK);
        reserveCode.put("if",CategoryCode.IFTK);
        reserveCode.put("else",CategoryCode.ELSETK);
        reserveCode.put("!",CategoryCode.NOT);
        reserveCode.put("&&",CategoryCode.AND);
        reserveCode.put("||",CategoryCode.OR);
        reserveCode.put("for",CategoryCode.FORTK);
        reserveCode.put("getint",CategoryCode.GETINTTK);
        reserveCode.put("getchar",CategoryCode.GETCHARTK);
        reserveCode.put("printf",CategoryCode.PRINTFTK);
        reserveCode.put("return",CategoryCode.RETURNTK);
        reserveCode.put("+",CategoryCode.PLUS);
        reserveCode.put("-",CategoryCode.MINU);
        reserveCode.put("void",CategoryCode.VOIDTK);
        reserveCode.put("*",CategoryCode.MULT);
        reserveCode.put("/",CategoryCode.DIV);
        reserveCode.put("%",CategoryCode.MOD);
        reserveCode.put("<",CategoryCode.LSS);
        reserveCode.put("<=",CategoryCode.LEQ);
        reserveCode.put(">",CategoryCode.GRE);
        reserveCode.put(">=",CategoryCode.GEQ);
        reserveCode.put("==",CategoryCode.EQL);
        reserveCode.put("!=",CategoryCode.NEQ);
        reserveCode.put("=",CategoryCode.ASSIGN);
        reserveCode.put(";",CategoryCode.SEMICN);
        reserveCode.put(",",CategoryCode.COMMA);
        reserveCode.put("(",CategoryCode.LPARENT);
        reserveCode.put(")",CategoryCode.RPARENT);
        reserveCode.put("[",CategoryCode.LBRACK);
        reserveCode.put("]",CategoryCode.RBRACK);
        reserveCode.put("{",CategoryCode.LBRACE);
        reserveCode.put("}",CategoryCode.RBRACE);
    }

    public void lexical_analysis() {
        while (curCharIndex < content.length()) {
            token.setLength(0);
            getCharNbc();
            if (Character.isLowerCase(curChar)) {
                if (curChar != content.charAt(curCharIndex)) {
                    curCharIndex++;
                }
                while (Character.isLetter(curChar) || Character.isDigit(curChar)) {
                    cat();
                    getChar();
                }
                int cur = reserve();
                if (cur == 0) {
                    categoryCodes.add(CategoryCode.IDENFR);
                    words.add(token.toString());
                } else {
                    categoryCodes.add(reserveCode.get(token.toString()));
                    words.add(token.toString());
                }
            } else if (Character.isDigit(curChar)) {
                if (curChar != content.charAt(curCharIndex)) {
                    curCharIndex++;
                }
                while (Character.isDigit(curChar)) {
                    cat();
                    getChar();
                }
                categoryCodes.add(CategoryCode.INTCON);
                words.add(token.toString());
            } else if (curChar == '"') {
                cat();
                getChar();
                while ((curChar >= 32 && curChar <= 126)) {
                    if (curChar == '"') {
                        break;
                    }
                    cat();
                    getChar();
                }
                cat();
                getChar();
                categoryCodes.add(CategoryCode.STRCON);
                words.add(token.toString());
            } else if (curChar == '\'') {
                while (Character.isLetter(curChar) || curChar == '\'') {
                    cat();
                    getChar();
                }
                categoryCodes.add(CategoryCode.STRCON);
                words.add(token.toString());
            } else if (curChar == '!') {
                getChar();
                if (curChar == '=') {
                    categoryCodes.add(CategoryCode.NEQ);
                    words.add("!=");
                    getChar();
                } else {
                    categoryCodes.add(CategoryCode.NOT);
                    words.add("!");
                    unGetCh();
                }
            } else if (curChar == '&') {
                getChar();
                if (curChar == '&') {
                    categoryCodes.add(CategoryCode.AND);
                    words.add("&&");
                    getChar();
                } else {
                    error = true;
                    errorText = 'a';
                    errorLine = line;
                    categoryCodes.add(CategoryCode.AND);
                    words.add("&");
                }
            } else if (curChar == '|') {
                getChar();
                if (curChar == '|') {
                    categoryCodes.add(CategoryCode.OR);
                    words.add("||");
                    getChar();
                } else {
                    error = true;
                    errorText = 'a';
                    errorLine = line;
                    categoryCodes.add(CategoryCode.OR);
                    words.add("|");
                }
            } else if (curChar == '+') {
                categoryCodes.add(CategoryCode.PLUS);
                words.add("+");
                getChar();
            } else if (curChar == '-') {
                categoryCodes.add(CategoryCode.MINU);
                words.add("-");
                getChar();
            } else if (curChar == '*') {
                categoryCodes.add(CategoryCode.MULT);
                words.add("*");
                getChar();
            } else if (curChar == '/') {
                getChar();
                if (curChar == '/') {
                    while (curCharIndex < content.length() && curChar != '\n') {
                        getChar();
                    }
                    if (curChar == '\n') {
                        line++;
                    }
                }
                if (curChar == '*') {
                    while (curCharIndex < content.length()) {
                        getChar();
                        while (curChar != '*') {
                            if (curChar == '\n') {
                                line++;
                            }
                            getChar();
                        }
                        while (curChar == '*') {
                            getChar();
                        }
                        if (curChar == '/') {
                            getChar();
                            break;
                        }
                    }
                }
            } else if (curChar == '%') {
                categoryCodes.add(CategoryCode.MOD);
                words.add("%");
                getChar();
            } else if (curChar == '<') {
                getChar();
                if (curChar == '=') {
                    categoryCodes.add(CategoryCode.LEQ);
                    words.add("<=");
                    getChar();
                } else {
                    categoryCodes.add(CategoryCode.LSS);
                    words.add("<");
                    unGetCh();
                }
            } else if (curChar == '>') {
                getChar();
                if (curChar == '=') {
                    categoryCodes.add(CategoryCode.GEQ);
                    words.add(">=");
                    getChar();
                } else {
                    categoryCodes.add(CategoryCode.GRE);
                    words.add(">");
                    unGetCh();
                }
            } else if (curChar == '=') {
                getChar();
                if (curChar == '=') {
                    categoryCodes.add(CategoryCode.EQL);
                    words.add("==");
                    getChar();
                } else {
                    categoryCodes.add(CategoryCode.ASSIGN);
                    words.add("=");
                    unGetCh();
                }
            } else if (curChar == ';') {
                categoryCodes.add(CategoryCode.SEMICN);
                words.add(";");
                getChar();
            } else if (curChar == ',') {
                categoryCodes.add(CategoryCode.COMMA);
                words.add(",");
                getChar();
            } else if (curChar == '(') {
                categoryCodes.add(CategoryCode.LPARENT);
                words.add("(");
                getChar();
            } else if (curChar == ')') {
                categoryCodes.add(CategoryCode.RPARENT);
                words.add(")");
                getChar();
            } else if (curChar == '[') {
                categoryCodes.add(CategoryCode.LBRACK);
                words.add("[");
                getChar();
            } else if (curChar == ']') {
                categoryCodes.add(CategoryCode.RBRACK);
                words.add("]");
                getChar();
            } else if (curChar == '{') {
                categoryCodes.add(CategoryCode.LBRACE);
                words.add("{");
                getChar();
            } else if (curChar == '}') {
                categoryCodes.add(CategoryCode.RBRACE);
                words.add("}");
                getChar();
            }
        }
    }

    public void getChar() {
        curCharIndex++;
        if (curCharIndex < content.length()) {
            curChar = content.charAt(curCharIndex);
        }
    }

    private void getCharNbc() {
        while (Character.isWhitespace(curChar) && curCharIndex < content.length()) {
            if (curChar == '\n') {
                line++;
            }
            getChar();
        }
    }

    private void cat() {
        token.append(curChar);
    }

    private void unGetCh() {
        curCharIndex--;
    }

    private int reserve() {
        String curToken = token.toString();
        if (reserveCode.containsKey(curToken)) {
            return 1;
        } else {
            return 0;
        }
    }

    public void lexical_print() {
        if (!error) {
            String filePath = "lexer.txt";
            try (BufferedWriter writer = new BufferedWriter(new FileWriter(filePath, true))) {
                for (int i = 0;i < words.size();i++) {
                    writer.write(categoryCodes.get(i) + " " + words.get(i) + "\n");
                }
                writer.flush();
            } catch (IOException e) {
                e.printStackTrace();
                System.out.println("写入文件时发生错误。");
            }
        }
        else {
            String filePath = "error.txt";
            try (BufferedWriter writer = new BufferedWriter(new FileWriter(filePath, true))) {
                writer.write(errorLine + " " + errorText);
                writer.flush();
            } catch (IOException e) {
                e.printStackTrace();
                System.out.println("写入文件时发生错误。");
            }
        }
    }
}
