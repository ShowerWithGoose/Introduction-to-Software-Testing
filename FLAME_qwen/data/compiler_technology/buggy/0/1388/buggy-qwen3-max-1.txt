       try {
           this.curPos = br.read();
           if(this.curPos != '/' && this.curPos != '*') {
               this.tokens.add(new Token("/", LexType.DIV, this.curLine)); //除号 @@ [When encountering '/', the next character is read immediately. If it's neither '/' nor '*', the lexer incorrectly assumes the current token is a single '/' and adds a DIV token. However, the original '/' has already been consumed as curPos, and the next character (stored in curPos) is not pushed back or handled, causing it to be skipped in the main loop.]
               return 1;
           } else if(this.curPos == '/'){
               while ((this.curPos = br.read()) != -1){
                   if(this.curPos == '\n') {//行尾注释结束
                       this.curLine++;
                       break;
                   }
               }
           } else {
               while ((this.curPos = br.read()) != -1){
                   if(this.curPos == '*') {
                       this.curPos = br.read();
                       if(this.curPos == '/') break;//注释结束
                   }
                   else if(this.curPos == '\n') this.curLine++;
               }
           }
           return 0;
