    private boolean getOperator() {
        StringBuilder string = new StringBuilder();
        string.append(sourceCode.charAt(charIndex));
        if (charIndex + 1 < sourceCode.length()) { // 双字符运算符
            string.append(sourceCode.charAt(charIndex + 1));
            if (doubleCharOperators.containsKey(string.toString())) {
                tokens.add(new Token(doubleCharOperators.get(string.toString()), string.toString(), lineIndex));
                charIndex += 2;
                return true;
            } else {
                string.delete(1, 2);
            }
        }

        if (singleCharOperators.containsKey(string.toString())) {
            tokens.add(new Token(singleCharOperators.get(string.toString()), string.toString(), lineIndex));
            charIndex++;
            return true;
        } else if (sourceCode.charAt(charIndex) == '&' || sourceCode.charAt(charIndex) == '|') {
            if (sourceCode.charAt(charIndex) == '&') {
                tokens.add(new Token("AND", string.toString(), lineIndex)); // @@ [Error: When encountering a single '&' or '|', the token type should not be "AND" or "OR"; it should be an error token, but the code incorrectly adds a token with the wrong type before reporting the error.             } else {
