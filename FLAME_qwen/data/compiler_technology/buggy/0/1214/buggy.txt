import frontend.Lexer;
import java.io.*;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;

public class Compiler {
    public static void main(String[] args) throws FileNotFoundException {
        boolean lexerPrint = true;
        Path path = Paths.get("testfile.txt");
        String source;
        try {
            source = Files.readString(path);
        } catch (IOException e) {
            throw new RuntimeException(e);
        }
        Lexer lexer = new Lexer(source);
        lexer.next();
        if (lexerPrint) {
            printLexerMessage(lexer);
        }
    }

    private static void printLexerMessage(Lexer lexer) {
        if (lexer.isError()) {
            String outPath = "error.txt";
            try (BufferedWriter writer = new BufferedWriter(new FileWriter(outPath))) {
                for (Lexer.Error error : lexer.errors) {
                    writer.write(error.line + " " + error.errorType);
                    writer.newLine();
                }
            } catch (IOException e) {
                throw new RuntimeException(e);
            }
        } else {
            String outPath = "lexer.txt";
            try (BufferedWriter writer = new BufferedWriter(new FileWriter(outPath))) {
                for (Lexer.Token token : lexer.tokens) {
                    writer.write(token.token + " " + token.tokenType);
                    writer.newLine();
                }
            } catch (IOException e) {
                throw new RuntimeException(e);
            }
        }
    }
}

package frontend;

import java.util.ArrayList;
import java.util.HashMap;

public class Lexer {
    private String source;
    private int curPos;
    private int line;
    private int number;
    private String curToken;
    private TokenType curTokenType;
    private boolean error;
    public class Token {
        public String token;
        public TokenType tokenType;

        public Token(String token, TokenType tokenType) {
            this.token = token;
            this.tokenType = tokenType;
        }
    }
    public class Error {
        public int line;
        public char errorType;

        public Error(int line, char errorType) {
            this.line = line;
            this.errorType = errorType;
        }
    }
    public ArrayList<Token> tokens;
    public ArrayList<Error> errors;
    public static final HashMap<String, TokenType> reservedWords = new HashMap<>() {{
        put("main", TokenType.MAINTK);
        put("const", TokenType.CONSTTK);
        put("int", TokenType.INTTK);
        put("char", TokenType.CHARTK);
        put("void", TokenType.VOIDTK);
        put("if", TokenType.IFTK);
        put("else", TokenType.ELSETK);
        put("for", TokenType.FORTK);
        put("break", TokenType.BREAKTK);
        put("continue", TokenType.CONTINUETK);
        put("return", TokenType.RETURNTK);
        put("getint", TokenType.GETINTTK);
        put("getchar", TokenType.GETCHARTK);
        put("printf", TokenType.PRINTFTK);
    }};

    public Lexer(String source) {
        this.source = source;
        this.curPos = 0;
        this.line = 1;
        this.number = 0;
        this.curToken = "";
        this.error = false;
        this.tokens = new ArrayList<>();
        this.errors = new ArrayList<>();
    }

    private boolean isDigit(char c) {
        return c >= '0' && c <= '9';
    }

    private boolean isIdentifierNoneDigit(char c) {
        return (c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') || c == '_';
    }

    private boolean isIdentifier(char c) {
        return isIdentifierNoneDigit(c) || isDigit(c);
    }

    private boolean isBlank(char c) {
        return c == ' ' || c == '\t' || c == '\n' || c == '\r';
    }

    private boolean isReservedWord(String token) {
        return reservedWords.containsKey(token);
    }

    public boolean isError() {
        return error;
    }

    public void next() {
        while (curPos < source.length() && isBlank(source.charAt(curPos))) {
            if (source.charAt(curPos) == '\n') {
                line++;
            }
            curPos++;
        } // 忽略空白字符
        curToken = "";
        if (curPos >= source.length()) {
            return;
        }
        while (curPos < source.length() && isIdentifier(source.charAt(curPos))) {
            curToken += source.charAt(curPos);
            curPos++;
        }
        if (!curToken.isEmpty()) {
            if (isReservedWord(curToken)) {
                curTokenType = reservedWords.get(curToken);
            } else if (isDigit(curToken.charAt(0))) {
                curTokenType = TokenType.INTCON;
            } else {
                curTokenType = TokenType.IDENFR;
            }
            tokens.add(new Token(curToken, curTokenType));
            next();
            return;
        } // 保留字、标识符、整数常量
        if (source.charAt(curPos) == '\"') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.STRCON;
            curPos++;
            while (curPos < source.length() && source.charAt(curPos) != '\"') {
                curToken += source.charAt(curPos);
                curPos++;
            }
            if (curPos < source.length()) {
                curToken += source.charAt(curPos);
                curPos++;
            }
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == '\'') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.CHARCON;
            curPos++;
            if (curPos < source.length() && source.charAt(curPos) == '\\') {
                curToken += source.charAt(curPos);
                curPos++;
            } // 转义字符
            if (curPos < source.length()) {
                curToken += source.charAt(curPos);
                curPos++;
            }
            if (curPos < source.length() && source.charAt(curPos) == '\'') {
                curToken += source.charAt(curPos);
                curPos++;
            }
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == '+') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.PLUS;
            curPos++;
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == '-') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.MINU;
            curPos++;
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == '*') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.MULT;
            curPos++;
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == '%') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.MOD;
            curPos++;
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == '<') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.LSS;
            curPos++;
            if (curPos < source.length() && source.charAt(curPos) == '=') {
                curToken += source.charAt(curPos);
                curTokenType = TokenType.LEQ;
                curPos++;
            }
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == '>') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.GRE;
            curPos++;
            if (curPos < source.length() && source.charAt(curPos) == '=') {
                curToken += source.charAt(curPos);
                curTokenType = TokenType.GEQ;
                curPos++;
            }
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == '=') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.ASSIGN;
            curPos++;
            if (curPos < source.length() && source.charAt(curPos) == '=') {
                curToken += source.charAt(curPos);
                curTokenType = TokenType.EQL;
                curPos++;
            }
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == '!') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.NOT;
            curPos++;
            if (curPos < source.length() && source.charAt(curPos) == '=') {
                curToken += source.charAt(curPos);
                curTokenType = TokenType.NEQ;
                curPos++;
            }
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == '&') {
            curToken += source.charAt(curPos);
            curPos++;
            if (curPos < source.length() && source.charAt(curPos) == '&') {
                curToken += source.charAt(curPos);
                curTokenType = TokenType.AND;
                curPos++;
            } else {
                error = true;
                errors.add(new Error(line, 'a'));
            }
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == '|') {
            curToken += source.charAt(curPos);
            curPos++;
            if (curPos < source.length() && source.charAt(curPos) == '|') {
                curToken += source.charAt(curPos);
                curTokenType = TokenType.OR;
                curPos++;
            } else {
                error = true;
                errors.add(new Error(line, 'a'));
            }
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == ';') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.SEMICN;
            curPos++;
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == ',') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.COMMA;
            curPos++;
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == '(') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.LPARENT;
            curPos++;
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == ')') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.RPARENT;
            curPos++;
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == '[') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.LBRACK;
            curPos++;
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == ']') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.RBRACK;
            curPos++;
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == '{') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.LBRACE;
            curPos++;
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == '}') {
            curToken += source.charAt(curPos);
            curTokenType = TokenType.RBRACE;
            curPos++;
            Token t = new Token(curToken, curTokenType);
            tokens.add(t);
            next();
        } else if (source.charAt(curPos) == '/') {
            curToken += source.charAt(curPos);
            curPos++;
            if (curPos < source.length() && source.charAt(curPos) == '/') {
                while (curPos < source.length() && source.charAt(curPos) != '\n') {
                    curPos++;
                }
                line++;
                next();
            } else if (curPos < source.length() && source.charAt(curPos) == '*') {
                curPos++;
                while (curPos < source.length()) {
                    while (curPos < source.length() && source.charAt(curPos) != '*') {
                        if (source.charAt(curPos) == '\n') {
                            line++;
                        }
                        curPos++;
                    }
                    while (curPos < source.length() && source.charAt(curPos) == '*') {
                        curPos++;
                    }
                    if (curPos < source.length() && source.charAt(curPos) == '/') {
                        curPos++;
                        next();
                    }
                }
            } else {
                curTokenType = TokenType.DIV;
                Token t = new Token(curToken, curTokenType);
                tokens.add(t);
                next();
            }
        }
    }
}

    INTCON, // IntConst
    STRCON, // StringConst

    GETINTTK, // getint
    GETCHARTK, // getchar
    PRINTFTK, // printf
    RETURNTK, // return
    PLUS, // +
    MINU, // -
    VOIDTK, // void
    MULT, // *
    DIV, // /
    MOD, // %
    LSS, // <
    LEQ, // <=
    GRE, // >
    GEQ, // >=
    EQL, // ==
    NEQ, // !=
    ASSIGN, // =
    SEMICN, // ;
    COMMA, // ,
    LPARENT, // (
    RPARENT, // )
    LBRACK, // [
    RBRACK, // ]
    LBRACE, // {
    RBRACE, // }
}
