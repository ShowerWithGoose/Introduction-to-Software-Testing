    private void scanSymbol() throws IOException {
        int chr = reader.read();
        Pair<Integer, Integer> start = reader.location();
        switch (chr) {
            case '|' -> {
                chr = reader.read();
                if (chr != '|') {
                    errors.add(ErrorType.InvalidToken, location());
                    reader.unread(chr);
                }
                addToken(TokenType.Or, "||", start); // @@ [When a single '|' is encountered, it should not add an OR token "||", but the code always adds it regardless of whether the second '|' was found]
