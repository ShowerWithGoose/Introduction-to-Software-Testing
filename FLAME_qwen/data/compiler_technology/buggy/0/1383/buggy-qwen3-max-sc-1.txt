<Compiler.java>
import error.ErrorChecker;
import frontend.preprocess.Extractor;
import frontend.lexer.Lexer;

import java.io.IOException;

/**
 * @Description Compiler </br>
 * 前端：词法分析（源程序提取器、语法分析器、保留字检查器）</br>
 * 错误处理：（错误检查器）</br>
 */

public class Compiler {
    public static void main(String[] args) throws IOException {
        // 读取源程序文本文件
        String content = Extractor.extractSourceCode();

        // 词法分析
        Lexer.lexicalAnalysis(content);
        // Lexer.printResult();
        Lexer.printResultToFile("lexer.txt"); // @@ [The program should only output to lexer.txt for correct programs, but this always outputs lexer.txt even when errors exist]
        Lexer.clearResult();
        // ErrorChecker.printError();
        ErrorChecker.printErrorToFile("error.txt"); // @@ [The program should only output to error.txt for erroneous programs, but this always outputs error.txt even when no errors exist]
        ErrorChecker.clearError();

    }
}
