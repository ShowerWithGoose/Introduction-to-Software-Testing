    private void identLogical() throws IOException {
        int line = lineNumber;
        int column = columnNumber;
        if (currentChar == '&') {
            advance();
            if (currentChar != '&') {
                errors.add(new Error(line, column, 'a')); // @@ Missing advance() call after detecting a single '&' or '|', which leaves the lexer pointing to the next character and can cause incorrect tokenization or state misalignment.
            } else {
                words.add(new Word(line, column, "&&", Type.AND));
                advance();
            }
        } else if (currentChar == '|') {
            advance();
            if (currentChar != '|') {
                errors.add(new Error(line, column, 'a')); // @@ Missing advance() call after detecting a single '&' or '|', which leaves the lexer pointing to the next character and can cause incorrect tokenization or state misalignment.
            } else {
                words.add(new Word(line, column, "||", Type.OR));
                advance();
            }
        } else {
            System.out.println("Error : not Logical Char");
        }
    }
