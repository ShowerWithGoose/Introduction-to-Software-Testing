        if (currentChar == '&') {
            advance();
            if (currentChar != '&') {
                errors.add(new Error(line, column, 'a'));
                advance(); // @@ [After detecting a single '&', the lexer must consume the next character to continue scanning; otherwise, it gets stuck in an infinite loop or misreads the following token. The current code only adds the error but does not advance past the invalid single '&'.]
            } else {
                words.add(new Word(line, column, "&&", Type.AND));
                advance();
            }
        } else if (currentChar == '|') {
            advance();
            if (currentChar != '|') {
