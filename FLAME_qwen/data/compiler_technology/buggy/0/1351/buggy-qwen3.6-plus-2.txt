            readCh();
            while (curChar != '\n' && curChar != -1) {
                readCh();
            }
            if (curChar == '\n') {
                lineNo++;
            }
            readCh();//anon: the next token char to deal withs
        } else if (curChar == '*') {
            readCh();

                if (curChar == '*') {
                    readCh();
                    if (curChar == '/') {
                        readCh();
                        break;

                        continue;
