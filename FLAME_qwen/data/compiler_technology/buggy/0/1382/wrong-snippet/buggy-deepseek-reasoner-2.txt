case Start:  {
    if (Character.isDigit(current)) {   // current could be null
        state = State.Digit;
        token.append(current);
    }
    else if (Character.isAlphabetic(current) || current.equals('_')){
        state = State.NonDigit;
        token.append(current);
        peekChar();
    }
    else if (current.equals(' ')
            || current.equals('\r')
            || current.equals('\t')
            || current.equals('\n')) {
        peekChar();
        break;
    }
    else {
        state = State.Operation;
    }
    break;
}
