void Lexer::getSym() {
    string str;
    str += cur_line[index++];
    // str = "&"
    if (index < cur_line.length()) {
        string rts = str + cur_line[index];
        // rts = "&&"
        if (symbol.count(rts)) {
            this->cur_token = rts;
            this->cur_type = symbol.at(rts);
            index++;
        } else {
            this->cur_token = str;
            this->cur_type = symbol.at(str);
            if (str[0] == '&' || str[0] == '|') printOut_for_error();
            if (str[0] == '&') this->cur_type = AND; // @@ [Error: When a single '&' or '|' is encountered, the error is reported but then the token type is incorrectly overwritten to AND/OR, and then printOut() is called, which outputs the token. According to the problem, for erroneous programs, only the error should be output to error.txt, not the token to lexer.txt. The program should stop processing tokens after detecting the error, or at least not output the token.]
            if (str[0] == '|') this->cur_type = OR; // @@ [Same error as above]
        }
    } else {
        this->cur_token = str;
        this->cur_type = symbol.at(str);
        if (str[0] == '&' || str[0] == '|') printOut_for_error();
        if (str[0] == '&') this->cur_type = AND; // @@ [Same error as above]
        if (str[0] == '|') this->cur_type = OR; // @@ [Same error as above]
    }

    printOut(); // @@ [This line is called unconditionally, causing tokens to be output even when an error has been detected. For erroneous programs, only the error should be output, not the token.]
}
