void Lexer::getSym() {
    string str;
    str += cur_line[index++];
    // str = "&"
    if (index < cur_line.length()) {

        // rts = "&&"

            this->cur_token = rts;
            this->cur_type = symbol.at(rts);
            index++;
        } else {
            this->cur_token = str;
            this->cur_type = symbol.at(str); // @@ [if str is "&" or "|", symbol.at(str) returns ANDOP or OROP, but then the code overrides cur_type to AND/OR and still calls printOut(), which outputs a valid token instead of only reporting an error]
