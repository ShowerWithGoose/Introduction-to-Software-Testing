void Lexer::getSym() {
    string str;
    str += cur_line[index++];
    // str = "&"
    if (index < cur_line.length()) {
        string rts = str + cur_line[index];
        // rts = "&&"
        if (symbol.count(rts)) {
            this->cur_token = rts;
            this->cur_type = symbol.at(rts);
            index++;
        } else {
            this->cur_token = str;
            this->cur_type = symbol.at(str);
            if (str[0] == '&' || str[0] == '|') printOut_for_error();
            if (str[0] == '&') this->cur_type = AND; // @@ [Error: When a single '&' or '|' is encountered, the code incorrectly sets the token type to AND or OR instead of ANDOP or OROP, and then calls printOut() which outputs the wrong token type. Additionally, printOut_for_error() is called before setting the type, but the error output should be the only output for erroneous programs, not followed by a normal token output.]
            if (str[0] == '|') this->cur_type = OR; // @@ [Same error as above: should set to OROP, not OR]
        }
    } else {
        this->cur_token = str;
        this->cur_type = symbol.at(str);
        if (str[0] == '&' || str[0] == '|') printOut_for_error();
        if (str[0] == '&') this->cur_type = AND; // @@ [Same error as above]
        if (str[0] == '|') this->cur_type = OR; // @@ [Same error as above]
    }

    printOut(); // @@ [Error: For erroneous programs with a single '&' or '|', printOut() should not be called after printOut_for_error(), as it will output an incorrect token to lexer.txt when the program should only output to error.txt]
}
