    private Token getSymbol() {
        char ch = sourceCode.charAt(pos);
        StringBuilder sb = new StringBuilder().append(ch);
        switch (ch) {
            case '&':
            case '|':
                if (pos + 1 < sourceCode.length() && sourceCode.charAt(pos + 1) == ch) {
                    sb.append(ch);
                    pos++;
                }
                // @@ [After processing, if the symbol is just "&" or "|", it's an invalid token (error type 'a'), but the current code does not detect or report this error. Instead, it proceeds to create a token with type INVALID and does not trigger error handling.]
                break;
            case '<':
            case '>':
            case '=':
            case '!':
                if (pos + 1 < sourceCode.length() && sourceCode.charAt(pos + 1) == '=') {
                    sb.append('=');
                    pos++;
                }
                break;
            default:
        }
        pos++;
        String symbol = sb.toString();
        return new Token(symbol, Token.symbolToTokenType(symbol), line); // @@ [If symbol is "&" or "|", symbolToTokenType returns TokenType.INVALID, but the lexer does not check for this and add an error via ErrorHandler. Thus, lexical errors of type 'a' are not reported as required.]
    }
