    public String getLetter(){
        //用来获取字母
        StringBuilder sb = new StringBuilder();
        while (positionNum < inputLine.length() && Character.isLetter(inputLine.charAt(positionNum))){ // @@ Only checks Character.isLetter(), so identifiers containing digits after the first character (e.g., var1) are not fully consumed here; underscores are also never handled, making identifiers like _var or my_var unrecognized
            sb.append(inputLine.charAt(positionNum));
            positionNum++;
        }
        return sb.toString();
    }
