void Lexer::run() {
    std::ofstream out("lexer.txt", std::ofstream::out);
    while (pos < source.length()) {
        next();
    }
    for (int i = 0; i < tokens.size(); i++) {
        if (tokens[i].get_type() == Token::TokenType::Null) continue;
        out << tokens[i].to_string() << std::endl;
    }
    out.close();

    if (!errors.empty()) {
        std::ofstream error_out("error.txt", std::ofstream::out);
        error_out << errors[0].to_string() << std::endl; // @@ [Only outputs first error, but problem states each erroneous example has exactly one error, so this is acceptable; however, the presence of dummy "HELL" tokens may cause incorrect behavior if lexer continues after error]
        error_out.close();
    }
}
