    public String matchRemainingLine(Pattern pattern){
        String remainingLine=getRemainingLine();
        if(remainingLine==null){
            return null;
        }
        if(remainingLine.startsWith("'")&&pattern.pattern().equals("^'(.*?)'")&&remainingLine.substring(1,2).equals("\\")){
            return remainingLine.substring(0,4); // @@ This hardcoded length 4 may be incorrect for escape sequences longer than 2 characters (e.g., '\n' is 4 characters but '\x41' would be 6). It also doesn't handle the case where the escape sequence is not exactly 2 characters after the opening quote.
        }
        if(remainingLine.startsWith("\"")&&pattern.pattern().equals("^\\\"[^\\\"]*\\\"")&&remainingLine.substring(1,2).equals("\\")){
            return remainingLine.substring(0,4); // @@ Same issue as above: hardcoded length 4 may be incorrect for string constants with escape sequences longer than 2 characters.
        }
        Matcher matcher=pattern.matcher(remainingLine);
        if (matcher.find()) {
            return matcher.group(0);
        } else {
            //        exception
            if(pattern.pattern().equals("^&&")&&remainingLine.startsWith("&")){
                return "&";
            }
            if(pattern.pattern().equals("^\\|\\|")&&remainingLine.startsWith("|")){
                return "|";
            }
            // not match
            return null;
        }
    }
