    public int getTokens() {
        int normal; // 正常值为0
        int line   = 1;
        while (codes.hasNextLine()) { // 读一行
            String codeLine = codes.nextLine();
            codeLine += " ";
            normal = getSymbol(codeLine);
            if (normal < 0) {
                handlError("a", line);
                return line;
            }
            line++;
        }
        return 0;
    }
