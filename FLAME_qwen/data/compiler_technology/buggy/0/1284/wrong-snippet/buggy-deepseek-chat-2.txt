    private String readText() {
        String line;
        StringBuilder textContent = new StringBuilder();
        while (true) {
            try {
                if ((line = bufferedReader.readLine()) == null) break;
                textContent.append(line).append("\n");
            } catch (IOException e) {
                throw new RuntimeException(e);
            }
        }
        return textContent.toString();
    }
