    private Token processOrErr() {
        Token token = null;
        if (getCurrentChar() == '|') {
            token = new Token(KindCode.OR, "||", currentLineNumber);
            move(1);
        } else {
            token = new Token(KindCode.OR, "|", currentLineNumber);
            ErrorHandling.addError("a", currentLineNumber);
        }
        return token;
    }

    private Token processAndErr() {
        Token token = null;
        if (getCurrentChar() == '&') {
            token = new Token(KindCode.AND, "&&", currentLineNumber);
            move(1);
        } else {
            token = new Token(KindCode.AND, "&", currentLineNumber);
            ErrorHandling.addError("a", currentLineNumber);
        }
        return token;
    }
