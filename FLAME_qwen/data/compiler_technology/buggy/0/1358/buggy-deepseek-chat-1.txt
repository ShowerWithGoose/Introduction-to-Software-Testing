import java.io.*;
import java.util.List;

public class Compiler {
    public static void main(String[] args) {
        Lexer myLexer = new Lexer("testfile.txt");
        List<Token> tokens = myLexer.lexer();

        try (FileWriter writer = new FileWriter("lexer.txt", true)) { // @@ [Error: The FileWriter is opened in append mode (true), which will cause output from multiple runs to accumulate in the file, potentially producing incorrect results. It should be false to overwrite the file each time.]
            for (Token token: tokens) {
                writer.write(token.toString());
                writer.write('\n');
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
