package frontend;

import java.util.regex.Pattern;

public enum Error {
    a("^&|\\|"); // @@ [The regex "^&|\\|" matches either a single '&' at the start of the string OR a single '|' anywhere due to operator precedence; it should be grouped as "^(?:&|\\|)" or "^(&|\\|)" to correctly match only standalone '&' or '|' at the beginning of a token.]

    private final Pattern pattern;
    Error(String pattern) {
        this.pattern = Pattern.compile(pattern);
    }

    public Pattern getPattern() {
        return pattern;
    }
}
