    private boolean isEnter() {
        if (posIsValid() && source.charAt(curPos) == '\n') {
            lineNum++; // @@ [isEnter() increments lineNum as a side effect every time it is called. Since it is used as a while-loop condition alongside isTab() and isSpace(), a newline character causes lineNum++ on every iteration of the loop until the character advances, potentially double-counting line numbers.]
            return true;
        }else if (curPos < source.length() - 1 && source.charAt(curPos) == '\r' &&
                source.charAt(curPos + 1) == '\n') {
            lineNum++;
            return true;
        }
        return false;
    }
