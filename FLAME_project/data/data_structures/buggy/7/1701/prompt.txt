Node*insert(Node* root, int data) {
    if (root == NULL) {
        root = createNode(data);
        return root;
    }

    // 递归插入新节点
    if (data < root->data) { // 如果新节点数据小于当前节点数据，则插入到左子树中

    } else if (data > root->data) { // 如果新节点数据大于当前节点数据，则插入到右子树中

    return root; // 返回更新后的根节点指针
}
