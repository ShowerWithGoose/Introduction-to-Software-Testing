void prem(int* a, int N, int* returnSize, int** returnColumnSizes,int** returnNums,int start)
{
    if(start == N)
    {
        //遍历到末尾了
        //申请returnNums
        returnNums[*returnSize] = (int *)malloc(sizeof(int ) * N);
        //拷贝内容到returnNums
        memcpy(returnNums[*returnSize],a,sizeof(int) * N );
        //记录当前拷贝内容的长度
        (*returnColumnSizes)[*returnSize] = N;
        *returnSize = *returnSize + 1;

    }
    else
    {

        //回溯算法的核心
        int i;
        for(i = start; i < N; i++)
        {
            swap(a,i,start);//i 和 offset 交换
            prem(a,N,returnSize,returnColumnSizes,returnNums,start+1);
            swap(a,i,start);//i 和 offset 交换
        }
    }
}
