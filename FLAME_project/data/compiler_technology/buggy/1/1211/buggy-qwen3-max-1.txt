    public Stmt parseStmt() {
        StmtEle stmtEle;

        switch (curToken.getType()) {
            case SEMICN -> stmtEle = parseNullStmt();
            case LBRACE -> stmtEle = parseBlock();
            case IFTK -> stmtEle = parseIfStmt();
            case FORTK -> stmtEle = parseForLoopStmt();
            case BREAKTK -> stmtEle = parseBreakStmt();
            case CONTINUETK -> stmtEle = parseContinueStmt();
            case RETURNTK -> stmtEle = parseReturnStmt();
            case PRINTFTK -> stmtEle = parsePrintfStmt();
            case IDENFR -> stmtEle = dealIdentCase();
            case LPARENT, INTCON, CHRCON, PLUS, MINU, NOT -> stmtEle = parseExpStmt();

            default -> stmtEle = new NullStmt(handleIError()); // error @@ [This default case incorrectly handles missing semicolons by creating a NullStmt with handleIError(), but the real issue is that statements like printf, return, and variable declarations that are missing semicolons should be detected in their specific parsing methods, not in this generic fallback. The handleIError() method may not capture the correct line number for the actual statement that's missing the semicolon.]
        }

        return new Stmt(stmtEle);
    }
