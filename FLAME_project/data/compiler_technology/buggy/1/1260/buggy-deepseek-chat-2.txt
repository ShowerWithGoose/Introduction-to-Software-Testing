    private CompUnitNode CompUnit() {
        List<DeclNode> declNodes = new ArrayList<>();
        List<FuncDefNode> funcDefNodes = new ArrayList<>();
        MainFuncDefNode mainFuncDefNode;
        while (lexer.tokenList.get(pos + 1).getType() != Token.LexType.MAINTK && lexer.tokenList.get(pos + 2).getType() != Token.LexType.LPARENT) { // @@ The condition uses `&&` but should use `||` to correctly check if the next two tokens are not "main" and "("; also, the logic for detecting Decl vs FuncDef is flawed, as it doesn't properly distinguish between them based on the grammar.
            DeclNode declNode = Decl();
            declNodes.add(declNode);
        }
        while (lexer.tokenList.get(pos + 1).getType() != Token.LexType.MAINTK) { // @@ This loop condition is incorrect; it should check for FuncDef before MainFuncDef, but the condition only checks for MAINTK, which may cause infinite loops or incorrect parsing when there are no function definitions.
            FuncDefNode funcDefNode = FuncDef();
            funcDefNodes.add(funcDefNode);
        }
        mainFuncDefNode = MainFuncDef();
        return new CompUnitNode(declNodes, funcDefNodes, mainFuncDefNode);
    }
