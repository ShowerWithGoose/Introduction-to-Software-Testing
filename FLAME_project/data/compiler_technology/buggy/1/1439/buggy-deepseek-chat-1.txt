    private Node parseVarDecl() {
        Node varDeclNode = new Node("<VarDecl>");
        varDeclNode.addChild(parseBType());
        varDeclNode.addChild(parseVarDef());
        while (currentToken().type != TokenType.SEMICN) { // @@ [This while loop condition does not check for null pointer, and when encountering a missing semicolon, it may cause an infinite loop or null pointer exception. Additionally, the loop logic incorrectly assumes that after parsing the first VarDef, the next token must be a comma or semicolon, but in the test case, after parsing "int a", the next token is "for", which is neither, causing the loop to continue indefinitely.]
            if(currentToken().type == TokenType.COMMA){
                Node comma = new Node("COMMA", currentToken().value);
                varDeclNode.addChild(comma);
                advance();
            }
            varDeclNode.addChild(parseVarDef());
        }
        if(currentToken().type == TokenType.SEMICN){
            Node semicn = new Node("SEMICN", currentToken().value);
            varDeclNode.addChild(semicn);
            advance();
        }
        return varDeclNode;
