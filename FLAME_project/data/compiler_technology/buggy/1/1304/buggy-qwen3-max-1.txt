    //<Decl>:           [const] int/char <Ident>
    //<FuncDef>:        void/int/char func()
    //<MainFuncDef>:    int main()
    private CompUnitNode parseCompUnit() {
        //Decl
        ArrayList<DeclNode> declNodes = new ArrayList<>();
        while (peek(2) != TokenType.LPARENT) { // @@ [This lookahead logic fails when there are syntax errors like missing parentheses, causing the parser to not properly detect function definitions and skip error handling]
            declNodes.add(parseDecl());
        }

        //FuncDef
        ArrayList<FuncDefNode> funcDefNodes = new ArrayList<>();
        while (peek(1) != TokenType.MAINTK) { // @@ [This lookahead logic similarly fails with malformed syntax, preventing proper parsing of MainFuncDef and missing error detection for missing parentheses]
            funcDefNodes.add(parseFuncDef());
        }

        //MainFuncDef
        MainFuncDefNode mainFuncDefNode = parseMainFuncDef();

        return new CompUnitNode(declNodes, funcDefNodes, mainFuncDefNode);
    }
