    //<CompUnit> ::= {<Decl>} {<FuncDef>} <MainFuncDef>
    //<Decl>:           [const] int/char <Ident>
    //<FuncDef>:        void/int/char func()
    //<MainFuncDef>:    int main()
    private CompUnitNode parseCompUnit() {
        //Decl
        ArrayList<DeclNode> declNodes = new ArrayList<>();
        while (peek(2) != TokenType.LPARENT) { // @@ The condition `peek(2) != TokenType.LPARENT` is incorrect for detecting Decl. It should check for 'const', 'int', or 'char' tokens to identify Decl, not peek(2) which looks two tokens ahead for LPARENT. This causes the parser to fail when encountering function definitions with missing parentheses, as it cannot correctly distinguish between Decl and FuncDef.
            declNodes.add(parseDecl());
        }

        //FuncDef
        ArrayList<FuncDefNode> funcDefNodes = new ArrayList<>();
        while (peek(1) != TokenType.MAINTK) { // @@ The condition `peek(1) != TokenType.MAINTK` is incorrect for detecting FuncDef. It should check for function type tokens (void, int, char) followed by an identifier, not peek(1) which looks one token ahead for MAINTK. This causes the parser to fail when encountering the main function definition with missing parentheses, as it cannot correctly identify FuncDef vs MainFuncDef.
            funcDefNodes.add(parseFuncDef());
        }

        //MainFuncDef
        MainFuncDefNode mainFuncDefNode = parseMainFuncDef();

        return new CompUnitNode(declNodes, funcDefNodes, mainFuncDefNode);
    }
