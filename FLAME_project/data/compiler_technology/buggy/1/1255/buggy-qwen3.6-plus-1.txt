    public CompUnit parseCompUnit() {
        CompUnit compUnit = new CompUnit();
        MainFuncDef mainFuncDef;
        getNextToken();
        while (true) {
            Token token1 = getNextToken();
            Token token2 = getNextToken();
            getBack(2);
            if (token1.getTokenCode() == TokenCode.MAINTK || token2.getTokenCode() == TokenCode.LPARENT) {
                break;
            }
            compUnit.addDecl(parseDecl());
        }
        while (true) {
            if (getNextToken().getTokenCode() == TokenCode.MAINTK) { // @@ The condition only checks if the current token is 'main'. For "int main()", the current token is 'int' (INTTK), so this check fails. The parser proceeds to parse "int main()" as a regular FuncDef, consuming the 'main' identifier. Consequently, parseMainFuncDef() fails to find 'int main' and likely returns null or crashes, leading to empty output or runtime error.
                getBack(1);
                break;
            }
            getBack(1);
            compUnit.addFuncDef(parseFuncDef());
        }
        mainFuncDef = parseMainFuncDef();
        compUnit.setMainFuncDef(mainFuncDef);
        return compUnit;
    }
