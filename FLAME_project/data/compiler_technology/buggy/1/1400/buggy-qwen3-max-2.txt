    public Node parseExpStmt() {
        ArrayList<Node> children = new ArrayList<>();
        if (curToken.getType() == TokenType.IDENFR || curToken.getType() == TokenType.PLUS
                || curToken.getType() == TokenType.MINU || curToken.getType() == TokenType.NOT
                || curToken.getType() == TokenType.LPARENT || curToken.getType() == TokenType.INTCON
                || curToken.getType() == TokenType.CHRCON) {
            children.add(parseExp());
        }
        if (curToken.getValue().equals(";")) {
            children.add(new Node(curToken));
            curToken = tokens.read();
        } else {
            Printer.addError("i", tokens.lastToken().getLine()); // @@ [Using lastToken().getLine() gives the line of the last parsed token, not the line where the statement began. For missing semicolon errors, the error should be reported on the line where the statement starts, not where it ends.]
        }
        return new Node(children, SyntaxType.Stmt);
    }

    public Node parseAssignStmt() {
        ArrayList<Node> children = new ArrayList<>();
        children.add(parseLVal());
        children.add(new Node(curToken));
        curToken = tokens.read();
        children.add(parseExp());
        if (curToken.getValue().equals(";")) {
            children.add(new Node(curToken));
            curToken = tokens.read();
        } else {
            Printer.addError("i", tokens.lastToken().getLine()); // @@ [Using lastToken().getLine() gives the line of the last parsed token (from parseExp()), which may be different from the line where the assignment statement started. The error should be reported on the line containing the LVal.]
        }
        return new Node(children, SyntaxType.Stmt);
    }

    public Node parseReturnStmt() {
        ArrayList<Node> children = new ArrayList<>();
        children.add(new Node(curToken)); //return
        curToken = tokens.read();
        //Exp的FIRST集：Ident,+-!,(,number,character
        if (curToken.getType() == TokenType.IDENFR || curToken.getType() == TokenType.PLUS
                || curToken.getType() == TokenType.MINU || curToken.getType() == TokenType.NOT
                || curToken.getType() == TokenType.LPARENT || curToken.getType() == TokenType.INTCON
                || curToken.getType() == TokenType.CHRCON) {
            children.add(parseExp());
        }
        if (curToken.getType() == TokenType.SEMICN) {
            children.add(new Node(curToken));
            curToken = tokens.read();
        } else {
            Printer.addError("i", tokens.lastToken().getLine()); // @@ [Using lastToken().getLine() gives the line of the last token in the expression or the 'return' keyword, but for a missing semicolon after 'return 0', the error should be reported on the line where 'return' appears, not where the expression ends.]
        }
        return new Node(children, SyntaxType.Stmt);
    }
