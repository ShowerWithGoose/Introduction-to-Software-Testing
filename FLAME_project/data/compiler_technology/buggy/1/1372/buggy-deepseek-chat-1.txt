    public void handleTerminal(Node node, LexType correctType) {
        Token token = tokens.getThis();
        lineNum = token.getLineNum();
        Node termiNode = new Node(token);
        //System.out.println(termiNode.getToken().getLexType().toString());

        if (correctType != termiNode.getToken().getLexType()) {
            if (correctType == LexType.SEMICN) {
                errors.getErrors().add(new Error(tokens.getErrLine(), "i")); // @@ The error line number is incorrectly retrieved using tokens.getErrLine(), which returns the line number of the previous token instead of the current token's line number. This causes the error line number to be off by one or more lines.
            } else if (correctType == LexType.RPARENT) {
                errors.getErrors().add(new Error(tokens.getErrLine(), "j")); // @@ Same issue: tokens.getErrLine() returns the line number of the previous token, not the current token where the missing ')' is expected.
            } else if (correctType == LexType.RBRACK) {
                errors.getErrors().add(new Error(tokens.getErrLine(), "k")); // @@ Same issue: tokens.getErrLine() returns the line number of the previous token, not the current token where the missing ']' is expected.
            }
            return;
        }
        tokens.goNext();
        node.add(termiNode);
    }
