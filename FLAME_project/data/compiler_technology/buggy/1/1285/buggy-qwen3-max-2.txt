Node *CompUnit(int depth){
    Node *root = new Node;
    strcpy(root->GrammarItem, "CompUnit");
    root->parent = NULL;
    root->depth = depth;
    Node *child = NULL;
    while(strcmp(tokens[tokenPos + 2]->symbol, "LPARENT") != 0){ // @@ [This condition is unsafe: it assumes at least 3 tokens exist ahead and that Decl ends before any '('; it may access out-of-bounds tokens or misclassify FuncDef as Decl]
        child = Decl(depth + 1);
        root->children.push_back(child);
        child->parent = root;
    }
    while(strcmp(tokens[tokenPos + 1]->symbol, "MAINTK") != 0){ // @@ [This condition is unsafe: it assumes at least 2 tokens exist ahead and that MainFuncDef is always present; if input lacks main, it will access invalid memory or loop incorrectly]
        child = FuncDef(depth + 1);
        root->children.push_back(child);
        child->parent = root;
    }
    child = MainFuncDef(depth + 1);
    root->children.push_back(child);
    child->parent = root;
    return root;
}
