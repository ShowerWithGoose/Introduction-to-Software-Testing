Node *CompUnit(int depth){
    Node *root = new Node;
    strcpy(root->GrammarItem, "CompUnit");
    root->parent = NULL;
    root->depth = depth;
    Node *child = NULL;
    while(strcmp(tokens[tokenPos + 2]->symbol, "LPARENT") != 0){ // @@ The lookahead logic `tokens[tokenPos + 2]` is unsafe and incorrect for distinguishing Decl from FuncDef. It assumes a fixed structure that may not exist (e.g., at end of file or with different token sequences), leading to out-of-bounds access or infinite loops. It should check the current token type (e.g., CONSTTK, INTTK, CHARTK) to decide whether to parse a Decl or stop.
        child = Decl(depth + 1);
        root->children.push_back(child);
        child->parent = root;
    }
    while(strcmp(tokens[tokenPos + 1]->symbol, "MAINTK") != 0){ // @@ Similar lookahead issue. Checking `tokens[tokenPos + 1]` for MAINTK is fragile. It should verify if the current token sequence corresponds to a FuncDef (FuncType Ident ...) or if it has reached the MainFuncDef.         child = FuncDef(depth + 1);
        root->children.push_back(child);
        child->parent = root;
    }
    child = MainFuncDef(depth + 1);
    root->children.push_back(child);
    child->parent = root;
    return root;
}
