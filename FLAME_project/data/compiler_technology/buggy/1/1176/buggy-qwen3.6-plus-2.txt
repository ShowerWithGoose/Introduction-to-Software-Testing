    private void CompUnit(){
        root = new NNode("CompUnit");
        //curNode = root;
        while((isConst() || isFuncType()) && !end()){
            if(isConst()){
                Decl(root);
            } else if (isFuncType()){
                if(lexList.get(curTokenIndex + 1).lexType.equals(Lexer.LexType.IDENFR)){
                    if(lexList.get(curTokenIndex + 2).token.equals("(")){
                        FuncDef(root); // @@ The code incorrectly parses 'main' as a generic FuncDef. It should check if the identifier is 'main' and call MainFuncDef instead, otherwise it outputs <FuncDef> instead of the required <MainFuncDef>.
                    } else if (!curLex.lexType.equals(Lexer.LexType.VOIDTK)){
                        Decl(root);
                    }
                }
            }
            if(curLex.lexType.equals(Lexer.LexType.INTTK) && lexList.get(curTokenIndex + 1).lexType.equals(Lexer.LexType.MAINTK)) { // @@ This check is unreachable for 'main' because the block above consumes the tokens for 'main' as a FuncDef. Even if it were reachable, it's placed inside the loop that might have already advanced past main or failed to distinguish it.
                MainFuncDef(root);
            }
        }
    }
