    private Stmt parseAssignmentOrExpStmt(LexType ident) {
        int step = 0;
        ArrayList<LexType> symbols = new ArrayList<>();
        LexType token = tokens.getAndMove();
        step++;

        while (true) {
            token = tokens.getAndMove();
            step++;

            if (token.getType() == LexType.TokenType.ASSIGN) {
                tokens.retreat(step);
                LVal lVal = new ExpParser(tokens, errorLog).parserLVal();
                LexType assign = tokens.getAndMove();
                symbols.add(assign);

                token = tokens.getToken();
                if (token.getType() == LexType.TokenType.GETINTTK ||
                        token.getType() == LexType.TokenType.GETCHARTK) {
                    return parseGetStmt(lVal, symbols);
                } else {
                    return parseAssignStmt(lVal, assign);
                }
            } else if (token.getType() == LexType.TokenType.SEMICN) {
                tokens.retreat(step);
                ExpStmt expStmt = new ExpStmt(new ExpParser(tokens, errorLog).parserExp());
                return new Stmt(expStmt, tokens);
            } else if (!Objects.equals(token.getLine(), ident.getLine())) {
                tokens.retreat(step);
                ExpStmt expStmt = new ExpStmt(new ExpParser(tokens, errorLog).parserExp());
                errorLog.add(tokens.getlastToken(1).getLine(), 'i'); // @@ [The error line number is incorrectly determined. When a semicolon is missing, the error should be reported at the line of the last non-terminal before the semicolon, but here it uses tokens.getlastToken(1).getLine() which may point to the wrong line, especially when the statement spans multiple lines. In the test case, the missing semicolon on line 9 (after "int a") is not detected because the condition `!Objects.equals(token.getLine(), ident.getLine())` fails to trigger when the next token is on the same line as the identifier, causing the error to be missed.]
                return new Stmt(expStmt, tokens);
            }
        }
    }
