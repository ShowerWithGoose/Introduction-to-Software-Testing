package frontend.Parser.statement.stmt;

import frontend.Error;
import frontend.Lexer.Token;
import frontend.Lexer.TokenListManager;
import frontend.Lexer.TokenType;
import frontend.MyErrorType;

import java.util.ArrayList;

public class ParserOfBreakStmt {
    private TokenListManager tokenListManager;
    private Token breakTk;
    private Token semicn;
    private ArrayList<Error> errors;

    public ParserOfBreakStmt(TokenListManager tokenListManager) {
        this.tokenListManager = tokenListManager;
    }

    public BreakStmt parseBreakStmt() {
        this.errors = new ArrayList<>();
        Token fir = tokenListManager.getTokenAndMove();
        Token sec = tokenListManager.getTokenAndMove();
        if (fir.getTokenType().equals(TokenType.BREAKTK) &&
                sec.getTokenType().equals(TokenType.SEMICN)) {
            this.breakTk = fir;
            this.semicn = sec;
        } else if (fir.getTokenType().equals(TokenType.BREAKTK) &&
                !sec.getTokenType().equals(TokenType.SEMICN)) {
            this.breakTk = fir;
            tokenListManager.lookBackToken(2);
            Token token = tokenListManager.getTokenAndMove();
            Error error = new Error(token.getLineNo(), MyErrorType.i);
            this.errors.add(error);
            System.out.println("Error in parsing break statement");
        }
        return new BreakStmt(this.breakTk, this.semicn); // @@ [When the semicolon is missing, this.semicn remains null, but the BreakStmt is still returned with a null semicn. The error line number is incorrectly set to the line of the break token (via lookBackToken(2) and getTokenAndMove()), but the problem requires the error line number to be the line of the last non-terminal before the semicolon, which should be the break token itself. However, the actual bug is that the error is not reported for the second missing semicolon (line 9 in the test case) because the parser incorrectly handles the empty statement after the if block, causing the missing semicolon error to be missed or reported on the wrong line.]
    }
    public ArrayList<Error> getErrors() {
        return this.errors;
    }
}
