    private void analyseUnaryExp() {
        //进入为第一位
        if (thisToken.getType().equals("PLUS") || thisToken.getType().equals("MINU") || thisToken.getType().equals("NOT"))

            // Ident '(' [FuncRParams] ')'
        {
            nextToken();//(

    {
        String target = peekToken(0).getType();
        if (target.equals("PLUS") || target.equals("MINU") || target.equals("NOT") ||
                target.equals("IDENFR") || target.equals("LPARENT") || target.equals("INTCON") || target.equals("CHRCON"))
        {
            for (int i = 1;;i++)
            {
                if (peekToken(i).getType().equals("SEMICN"))
                {
                    return true;
                }
                if (peekToken(i).getType().equals("ASSIGN") || peekToken(i).getType().equals("RBRACE")) // @@ This lookahead logic is flawed for parsing function arguments within variable declarations. It assumes that encountering an ASSIGN token means the current expression is invalid or complete, but in cases like "int b = min(r, scan()), c = 58;", the ASSIGN belongs to the next variable 'c', not the current expression.
