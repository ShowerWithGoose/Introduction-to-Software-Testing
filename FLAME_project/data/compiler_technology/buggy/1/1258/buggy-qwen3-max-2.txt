    private void analyseUnaryExp() {
        //进入为第一位
        if (thisToken.getType().equals("PLUS") || thisToken.getType().equals("MINU") || thisToken.getType().equals("NOT"))
        {
            analyseUnaryOp();
            nextToken();
            analyseUnaryExp();
            grammar.add("<UnaryExp>");
        }
        //这里需要分辨是函数还是左值
        else if (thisToken.getType().equals("IDENFR") && peekToken(0).getType().equals("LPARENT"))
            // Ident '(' [FuncRParams] ')'
        {
            nextToken();//(
            if (peekToken(0).getType().equals("RPARENT"))
            {
                nextToken();//)
                grammar.add("<UnaryExp>");
                return;
            }
            else
            {
                if (isExp()) // @@ [The isExp() method checks peekToken(0) which looks at the token after the current argument, not the current argument itself. For function calls like min(r, scan()), after consuming '(', the current token is 'r' (IDENFR), but isExp() checks the next token ',' which is not a valid expression starter, causing isExp() to return false and skip parameter parsing]
                {
                    nextToken();//第一位
                    analyseFuncRParams();
                    if (peekToken(0).getType().equals("RPARENT"))
                    {
                        nextToken();//)
                        grammar.add("<UnaryExp>");
                        return;
                    }
                    else
                    {
                        errors.put(thisToken.getLineNumber(),"j");
                        grammar.add("<UnaryExp>");
                        return;
                    }
                }
                else
                {
                    errors.put(thisToken.getLineNumber(),"j");
                    grammar.add("<UnaryExp>");
                    return;
                }
            }
        }
        else
        {
            analysePrimaryExp();
            grammar.add("<UnaryExp>");
        }
    }
