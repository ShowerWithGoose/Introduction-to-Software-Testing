    public CompUnit handleCompUnit() {
        // CompUnit -> {Decl} {FuncDef} MainFuncDef
        List<Decl> declList = new ArrayList<>();
        List<FuncDef> funcDefList = new ArrayList<>();
        MainFuncDef mainFuncDef = null;
        // {Decl}
        while (getSymType(1).notEquals(LexType.MAINTK) && getSymType(2).notEquals(LexType.LPARENT)) { // @@ The condition uses peek offsets (1 and 2) to decide if the current token starts a declaration, but this is not a reliable method. For example, if a function definition has a name that is not 'main' and has an empty parameter list, offset2 will be '(' causing the condition to fail and skip declarations incorrectly. More critically, if the first token is the beginning of a function definition (e.g., 'int f()'), the condition fails and the code correctly exits the Decl loop, but this approach is fragile and may fail for other valid constructs. The correct approach is to check if the current token can start a declaration (CONSTTK, INTTK, CHARTK) and whether the following tokens do not form a function definition.
            declList.add(handleDecl());
        }
        // {FuncDef}
        while (getSymType(1).notEquals(LexType.MAINTK)) {
            funcDefList.add(handleFuncDef());
        }
        // MainFuncDef
        mainFuncDef = handleMainFuncDef();
        // todo 错误处理！！！
        return new CompUnit(declList, funcDefList, mainFuncDef);
    }
