    private void CompUnit() {
        // CompUnit → {Decl} {FuncDef} MainFuncDef
        if(isEOF()) { return; }
        // {Decl}
        while(true) {
            if (peekSymbol(1).getType() == SymbolType.CONSTTK) {
                Decl();
            } else if(peekSymbol(3).getType() != SymbolType.LPARENT && !isEOF()) { // @@ The condition for identifying VarDecl is incorrect. It checks if the third symbol is not '(', but for a VarDecl like "int a = 0;", the third symbol is 'a' (IDENFR), which is not '(', so it enters Decl(). However, this condition also incorrectly triggers for other cases, such as when the first symbol is 'int' and the third is '(' (function definition), it should not enter Decl(). The logic should check if the first symbol is INTTK or CHARTK (BType) and the second is IDENFR to distinguish VarDecl from FuncDef.
                Decl();
            } else {
                break;
            }
        }
        // {FuncDef}
        while(true) {
            if(peekSymbol(2).getType() != SymbolType.MAINTK && !isEOF()) { // @@ The condition for identifying FuncDef is incorrect. It checks if the second symbol is not 'main', but for a VarDecl like "int a = 0;", the second symbol is 'a' (IDENFR), which is not 'main', so it would incorrectly enter FuncDef(). The logic should check if the first symbol is a FuncType (VOIDTK, INTTK, CHARTK) and the second is IDENFR, and the third is LPARENT to distinguish FuncDef from VarDecl.
                FuncDef();
            } else {
                break;
            }
        }
        //MainFuncDef
        MainFuncDef();
        if(shouldPrint) System.out.println("<CompUnit>");
    }
