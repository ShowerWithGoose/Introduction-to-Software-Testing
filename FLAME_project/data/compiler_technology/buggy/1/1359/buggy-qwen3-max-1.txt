    private void handleLogicalOperator(char expected, Category category) {
        reader.catToken();
        reader.nextChar();
        char c = reader.getChar();
        if (c == expected) {
            reader.catToken();
            tokens.add(new Token(category, reader.getToken().toString(), reader.getCurLine()));
        } else {
            error.lexError(reader.getCurLine(), "a");
        }
    }
