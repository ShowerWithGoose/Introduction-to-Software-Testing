    private void parseLOrExp() {
        parseLAndExp();
        outputList.add("<LOrExp>\n");
        while (curToken.getCategory() == Category.OR) {
            skipTokens(1);
            parseLAndExp();
            outputList.add("<LOrExp>\n");
        }
    }

    private void parseLAndExp() {
        parseEqExp();
        outputList.add("<LAndExp>\n");
        while (curToken.getCategory() == Category.AND) {
            skipTokens(1);
            parseEqExp();
            outputList.add("<LAndExp>\n");
        }
    }
