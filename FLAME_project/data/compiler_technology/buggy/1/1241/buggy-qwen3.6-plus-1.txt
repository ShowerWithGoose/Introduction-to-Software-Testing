    private Node Block(){
        GComponent  currentComponent = GComponent.Block;
        NLNode result = new NLNode(currentComponent);
        Node node1 = matchToken(new TokenType[]{TokenType.LBRACE});
        result.addChild(node1);
        while(!getCurToken().getType().equals(TokenType.RBRACE)){
            int currentIndex = index;
            Node node2 = BlockItem();
            if(currentIndex == tokenList.size()-1)
            {
                break;
            }
            // @@ The loop does not check if `index` has advanced after calling `BlockItem()`. If `BlockItem()` (specifically `Stmt()` or `Decl()`) encounters an unexpected token and fails to consume it (due to error recovery in `matchToken` not advancing the index), `index` remains unchanged. If the current token is not `RBRACE` and not the last token, this results in an infinite loop. A check like `if (currentIndex == index) break;` or `next();` is needed to prevent TLE on syntax errors.
            result.addChild(node2);
        }
        Node node3 = matchToken(new TokenType[]{TokenType.RBRACE});
        result.addChild(node3);
        return result;
    }
