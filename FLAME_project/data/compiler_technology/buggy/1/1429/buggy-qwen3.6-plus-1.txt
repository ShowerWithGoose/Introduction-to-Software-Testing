    private void parseStmt() {
        if (nextmatch(nextIndex, TokenType.LBRACE)) {
            // Block
            parseBlock();
        } else if (nextmatch(nextIndex, TokenType.IFTK)) {
...
        } else {
            // 解析 LVal '=' Exp ';' 或 [Exp] ';'
            if (isLVal(nextIndex)) {
                parseLVal();  // 左值表达式
                if (nextmatch(nextIndex, TokenType.ASSIGN)) {
                    writeOutputWord();
                    nextIndex++;
                    if (nextmatch(nextIndex, TokenType.GETINTTK)) {
...
                    } else if (nextmatch(nextIndex, TokenType.GETCHARTK)) {
...
                    } else {
                        parseExp();  // 普通赋值语句
                        if (nextmatch(nextIndex, TokenType.SEMICN)) {
                            writeOutputWord();
                            nextIndex++;
                        } else {
                            addError(tokens.get(nextIndex - 1).lineNum, "i");//缺少分号 @@ If semicolon is missing, nextIndex is not advanced. The caller (parseBlock) will loop infinitely on the same token because the error recovery does not skip to the next statement or synchronize.
                        }
                    }
                } else {
                    addError(currentToken.lineNum, "i");  // 缺少等号 @@ If '=' is missing, nextIndex is not advanced. This leads to an infinite loop in the caller.
                }
            } else {
                if (!nextmatch(nextIndex, TokenType.SEMICN)) {
                    parseExp();  // 可选表达式
                }
                if (nextmatch(nextIndex, TokenType.SEMICN)) {
                    writeOutputWord();
                    nextIndex++;
                } else {
                    addError(tokens.get(nextIndex - 1).lineNum, "i");//缺少分号 @@ If semicolon is missing, nextIndex is not advanced. This leads to an infinite loop in the caller (parseBlock).
                }
            }
        }
        writeOutput("Stmt");
    }
