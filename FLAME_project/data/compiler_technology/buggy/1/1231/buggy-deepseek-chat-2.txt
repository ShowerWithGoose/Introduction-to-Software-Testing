package frontend.parser.stmt;

import error.ErrorList;
import error.MyErrorType;
import frontend.TokenIterator;
import nodes.Token;
import nodes.TokenType;

public class MyNopStmt implements StmtNode {
    private TokenIterator tokens;

    public MyNopStmt(TokenIterator tokens) {
        this.tokens = tokens;
    }

    private Token token;

    @Override
    public String syntaxPrinter() {
        return token.syntaxPrinter();
    }

    @Override
    public boolean parse() {
        token = tokens.next();
        if (token.getType().equals(TokenType.SEMICN)) {
            return true;
        } else {
            tokens.stepBack(2); // @@ [When the current token is not a semicolon, stepping back 2 may go past the beginning of the token list, causing an exception or incorrect behavior. Additionally, the error line number should be the line of the last non-terminal before the missing semicolon, but here it uses tokenBefore which may not be correct.]
            Token tokenBefore = tokens.next();
            ErrorList.addError(MyErrorType.I_MISSING_SEMICOLON, tokenBefore.getLineNum());
            token = new Token(TokenType.SEMICN, ";", tokenBefore.getLineNum(), 0);
            return false;
        }
    }

}
