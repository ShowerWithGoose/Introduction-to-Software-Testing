    // match IDDNFR; distinguish between assign, getchar|getint and exp
    // solve: check if there is an '=' before hit ';'
    private void assignOrExp() {
        int count = 0;
        while (tokens.hasNext()) {
            Token token = tokens.next();
            count++;
            if (token.getType().equals(TokenType.ASSIGN)) {
                Token tokenAfterAssign = tokens.next();
                tokens.stepBack(count + 1);
                switch (tokenAfterAssign.getType()) {
                    case GETINTTK:
                        stmtNode = new MyGetintStmt(tokens);
                        break;
                    case GETCHARTK:
                        stmtNode = new MyGetcharStmt(tokens);
                        break;
                    default:
                        stmtNode = new MyAssignStmt(tokens);
                        break;
                }
                break;
            }
            if (token.getType().equals(TokenType.SEMICN)) {
                tokens.stepBack(count);
                stmtNode = new MyExpStmt(tokens);
                break;
            }
        }
        // @@ If the loop finishes without finding ASSIGN or SEMICN (e.g., due to a syntax error like missing semicolon and next token is not part of expression), stmtNode remains null.
        // @@ When parse() calls stmtNode.parse(), it will throw a NullPointerException, causing the program to crash and produce no output.
        // @@ This happens in the test case for "printf(...)" if the lookahead in assignOrExp consumes tokens incorrectly or if this method is reached unexpectedly, 
        // @@ but more likely, this specific crash path is triggered if an identifier is followed by something that isn't = or ; and the loop exhausts tokens or breaks without setting stmtNode.
        // @@ Actually, looking at the test case: printf is handled by PRINTFTK case. 
        // @@ However, if there is an identifier at the start of a statement that is NOT followed by = or ;, and the loop ends (hasNext becomes false or logic falls through), stmtNode is null.
        // @@ In the provided test case, the errors are missing semicolons. 
        // @@ Let's look at `int a // error 2`. Next token is `for`.

        // @@ Wait, if `assignOrExp` doesn't set `stmtNode`, it remains null.
        // @@ Does `assignOrExp` handle the case where the loop terminates without hitting break?
        // @@ If `tokens.hasNext()` becomes false, the loop ends. `stmtNode` is null.
        // @@ Then `stmtNode.parse()` crashes.
    }
