public class MyNopStmt implements StmtNode {
    private TokenIterator tokens;

    public MyNopStmt(TokenIterator tokens) {
        this.tokens = tokens;
    }

    private Token token;

    @Override
    public String syntaxPrinter() {
        return token.syntaxPrinter();
    }

    @Override
    public boolean parse() {
        token = tokens.next();
        if (token.getType().equals(TokenType.SEMICN)) {
            return true;
        } else {
            tokens.stepBack(2); // @@ [stepping back 2 positions is incorrect when only 1 token was consumed; should step back 1]
            Token tokenBefore = tokens.next(); // @@ [this gets the same token again, not the previous one; logic error in error line number determination]
            ErrorList.addError(MyErrorType.I_MISSING_SEMICOLON, tokenBefore.getLineNum()); // @@ [reports wrong line number because tokenBefore is not the actual last valid token before missing semicolon]
            token = new Token(TokenType.SEMICN, ";", tokenBefore.getLineNum(), 0);
            return false; // @@ [returning false causes parsing to fail completely instead of continuing with error recovery like other statement handlers]
        }
    }

}
