    public token consumeToken(token.Type type){
        //提供所需要的类型消耗匹配，此时currentToken为下一个token，返回消耗的token
        if(currentToken.type.equals(type)){
            token temoToken = currentToken;
            if(i<this.tokens.size()-1){
                i=i+1;
                currentToken=this.tokens.get(i);
            }
            return temoToken;
        }else if(type== token.Type.SEMICN){
            //缺少分号
            errorList.getList().addError(new error(error.wrongType.i,this.tokens.get(i-1).lineno)); // @@ [When i=0, tokens.get(i-1) will cause IndexOutOfBoundsException, and more importantly, when at end of input, this doesn't handle the case where currentToken is null or we've reached EOF]
            return new token(token.Type.SEMICN,";",this.tokens.get(i-1).lineno);
        }else if(type== token.Type.RPARENT){
            //缺少右括号
            errorList.getList().addError(new error(error.wrongType.j,this.tokens.get(i-1).lineno)); // @@ [Same issue: tokens.get(i-1) may be invalid when i=0 or at EOF]
            return new token(token.Type.RPARENT,"）",this.tokens.get(i-1).lineno);
        }else if(type== token.Type.RBRACK){
            //缺少右括号
            errorList.getList().addError(new error(error.wrongType.k,this.tokens.get(i-1).lineno)); // @@ [Same issue: tokens.get(i-1) may be invalid when i=0 or at EOF]
            return new token(token.Type.RBRACK,"）",this.tokens.get(i-1).lineno);
        }else {
            throw new RuntimeException("wrong at"+this.tokens.get(i-1).lineno+" "+currentToken); // @@ [This will crash the program instead of gracefully handling syntax errors, and tokens.get(i-1) may be invalid]
        }
    }
