void getc() {
    if (*now != '\0' && *now!='\r') {
        *now++; // 返回当前字符并移动指针 @@ [错误：`*now++` 递增的是指针 `now` 所指向的字符值，而不是指针本身。应该使用 `now++` 来移动指针。]
    }
    //
    while(*now == '\0' || *now=='\r' || *now=='\n'){
        isNewLine=1;
        if(readNextLine()==-1){
            endFlag=1;
            break;
        }
    }
}
