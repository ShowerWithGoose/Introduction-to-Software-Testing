void CompUnit(int father)
{
    ++node_cnt;
    node[node_cnt].endToken.sym = "<CompUnit>";
    int rt = node_cnt;
    node[father].son.push_back(rt);

    while (token_ans[token_cnt + 1].sym != "MAINTK") // @@ Potential out-of-bounds access if MAINTK is not found or token_cnt exceeds array bounds, leading to undefined behavior or crash.
    {
        if (token_ans[token_cnt].sym == "CONSTTK")
            Decl(rt);
        else
        {
            if (token_ans[token_cnt + 2].sym == "LPARENT") // @@ Potential out-of-bounds access if token_cnt is near the end of the token array.
                FuncDef(rt);
            else
                Decl(rt);
        }
    }

    MainFuncDef(rt);
}
