Based on the problem description and code review, the issue appears to be in the `parseCompUnit` function in parser.cpp. The function incorrectly assumes that any sequence with a '(' at position 2 must be a function definition, but it doesn't properly handle the case where we have a main function versus regular functions. The logic for detecting MainFuncDef is flawed because it checks `getType(2)==LPARENT` and then looks at `words[index+1].content`, but this doesn't account for the proper structure of "int main()" which should be handled as MainFuncDef.

static void parseCompUnit(CompUnit &it){
    bool fl=false;
    while(index<size){
        if(getType(2)==LPARENT){
            if(isWordType(words[index+1].content,MAINTK)){
                fl=true;
                parseFuncDef(it.mainFuncDef);
            }
            else{
                it.funcDefs.emplace_back();
                parseFuncDef(it.funcDefs.back());
            }
        }
        else{
            it.decls.emplace_back();
            parseBlockItem(it.decls.back());
        }
    }
    ast(fl);
}
