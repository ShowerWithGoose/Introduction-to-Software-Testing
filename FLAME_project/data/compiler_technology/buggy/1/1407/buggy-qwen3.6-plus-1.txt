    public VarDeclBlock VarDecl() {
        BTypeBlock bTypeBlock;
        List<VarDefBlock> varDefBlockList = new ArrayList<>();
        List<Token> commaList = new ArrayList<>();
        Token semicn;

        bTypeBlock = BType();
        varDefBlockList.add(VarDef());
        while(nowToken.type!=TokenType.SEMICN){
            commaList.add(getToken(TokenType.COMMA)); // @@ The parser expects a comma if the next token is not a semicolon. If a semicolon is missing (error 'i'), the next token might be a keyword or identifier from the next statement. getToken(COMMA) will fail and throw a RuntimeException because COMMA is not handled in the error recovery switch case, causing the program to crash before writing errors to file.
            varDefBlockList.add(VarDef());
        }
        semicn = getToken(TokenType.SEMICN);
        return new VarDeclBlock(bTypeBlock,varDefBlockList,commaList,semicn);
    }
