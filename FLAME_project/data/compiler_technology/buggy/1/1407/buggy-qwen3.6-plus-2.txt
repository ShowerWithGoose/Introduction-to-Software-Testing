    public ConstDeclBlock ConstDecl() {
        Token constToken = getToken(TokenType.CONSTTK);
        BTypeBlock bTypeBlock = BType();
        List<ConstDefBlock> constDefBlockList = new ArrayList<>();
        List<Token> commaList = new ArrayList<>();
        constDefBlockList.add(ConstDef());
        while(nowToken.type != TokenType.SEMICN){
            commaList.add(getToken(TokenType.COMMA)); // @@ The parser expects a comma if the next token is not a semicolon. If a semicolon is missing (error i), the next token will likely be a keyword or identifier (e.g., 'for', 'int'), causing getToken(COMMA) to throw a RuntimeException instead of reporting the missing semicolon error.
            constDefBlockList.add(ConstDef());
        }
        Token semicnToken = getToken(TokenType.SEMICN);
        return new ConstDeclBlock(constToken,bTypeBlock,constDefBlockList,commaList,semicnToken);
    }

    public VarDeclBlock VarDecl() {
        BTypeBlock bTypeBlock;
        List<VarDefBlock> varDefBlockList = new ArrayList<>();
        List<Token> commaList = new ArrayList<>();
        Token semicn;

        bTypeBlock = BType();
        varDefBlockList.add(VarDef());
        while(nowToken.type!=TokenType.SEMICN){
            commaList.add(getToken(TokenType.COMMA)); // @@ Similar to ConstDecl, if a semicolon is missing (error i), the next token is not a comma. This causes getToken(COMMA) to crash with a RuntimeException instead of detecting the missing semicolon and allowing the parser to continue or report the error correctly.
            varDefBlockList.add(VarDef());
        }
        semicn = getToken(TokenType.SEMICN);
        return new VarDeclBlock(bTypeBlock,varDefBlockList,commaList,semicn);
    }
