    private ParseResult parseCompUnit(Token currToken) throws LexerException, ParserException {
        CompUnitTree subTree = new CompUnitTree();
        ParseResult result;
        Token preRead, prePreRead;

        preRead = getNextToken();
        prePreRead = getNext_NextToken();
        while (isMatch(currToken, LexType.CONSTTK)
                || (isMatch(currToken, LexType.INTTK)
                && isMatch(preRead, LexType.IDENFR)
                && isNotMatch(prePreRead, LexType.LPARENT)
        )|| isMatch(currToken, LexType.CHARTK) // @@ [The CHARTK case doesn't check if it's followed by an identifier and then LPARENT, so char variable declarations like "char c;" will be incorrectly included in the declaration loop, but more critically, char function definitions like "char func()" might be misclassified because this condition accepts any CHARTK regardless of what follows]
        ) {
            //为什么要倒一下
            result = parseDecl(currToken);
            currToken = result.getNextToken();
            subTree.declares.add((DeclTree) result.getSubtree());

            preRead = getNextToken();
            prePreRead = getNext_NextToken();
        }

        preRead = getNextToken();
        while (isMatch(currToken, LexType.VOIDTK)
                || (isMatch(currToken, LexType.INTTK)
                && isMatch(preRead, LexType.IDENFR)
                || isMatch(currToken, LexType.CHARTK)) // @@ [The CHARTK case here doesn't verify that it's followed by an identifier and LPARENT, so it may incorrectly try to parse char variable declarations as function definitions, and the INTTK case also lacks verification that prePreRead is LPARENT]
        ) {
            result = parseFuncDef(currToken);
            currToken = result.getNextToken();
            subTree.funcs.add((FuncDefTree) result.getSubtree());

            preRead = getNextToken();
        }
