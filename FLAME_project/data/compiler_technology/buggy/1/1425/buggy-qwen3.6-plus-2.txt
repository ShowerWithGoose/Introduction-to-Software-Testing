public class CompileException{
    private int lineNumber;
    private ErrorType errorType;

    public CompileException(int lineNumber,ErrorType errorType){
        this.lineNumber=lineNumber;
        this.errorType=errorType;
    }
    public void print(){ // @@ The error is printed immediately to the file. Since lexical analysis runs before syntax analysis, lexical errors (like line 18) are written to the file before syntax errors (like line 16). This results in an unsorted output. Errors should be collected in a list and sorted by line number before being written to the file.
        IO.lexer_error(lineNumber + " " + errorType+'\n');
    }
}
