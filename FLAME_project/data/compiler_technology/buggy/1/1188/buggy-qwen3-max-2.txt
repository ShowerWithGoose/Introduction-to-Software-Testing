    private static void error(char c) {
        switch (c){
            case 'i':
                fileOut2.printf("%d %c\n", Lexer.getLine(), 'i'); // @@ [Lexer.getLine() returns the current global line counter, which may have advanced past the actual line containing the incomplete statement, leading to wrong error line numbers]
                break;
            case 'j':
                fileOut2.printf("%d %c\n", Lexer.getLine(), 'j'); // @@ [Same issue: line number corresponds to current scan position, not the location of the missing parenthesis]
                break;
            case 'k':
                fileOut2.printf("%d %c\n", Lexer.getLine(), 'k'); // @@ [Same issue: line number is not tied to the token before the missing bracket]
                break;
            default:
                break;
        }
    }
