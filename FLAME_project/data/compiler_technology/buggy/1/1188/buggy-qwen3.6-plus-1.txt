    private static void error(char c) {
        switch (c){
            case 'i':
                fileOut2.printf("%d %c\n", Lexer.getLine(), 'i'); // @@ The error line number should correspond to the line where the last non-terminal before the semicolon is located. Lexer.getLine() returns the current line of the lookahead token (which has skipped whitespace/newlines), causing it to report the line of the next statement or EOF instead of the line where the missing semicolon was expected.
                break;
            case 'j':
                fileOut2.printf("%d %c\n", Lexer.getLine(), 'j'); // @@ Similarly, for missing right parenthesis, the line number should reflect the location of the construct expecting the parenthesis, not the current lookahead position which may have advanced past newlines.
                break;
            case 'k':
                fileOut2.printf("%d %c\n", Lexer.getLine(), 'k'); // @@ Similarly, for missing right square bracket, the line number should reflect the location of the array access/definition, not the current lookahead position.
                break;
            default:
                break;
        }
    }
