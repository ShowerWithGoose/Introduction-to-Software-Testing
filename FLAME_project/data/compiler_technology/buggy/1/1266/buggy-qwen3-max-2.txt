    private AstCompUnit parseCompUnit() throws IOException {
        AstCompUnit compUnit = new AstCompUnit();

        // { Decl }
        while (lexer.lookAhead(1).getTokenId() != MAINTK
                && lexer.lookAhead(2).getTokenId() != LPARENT) { // @@ [The lookahead logic is flawed: it assumes that after a declaration there must be either 'main' or a function definition starting with 'main(', but in erroneous input like "void f(int a {}" the parser cannot properly detect where declarations end and function definitions begin, causing it to skip error detection in parseFuncDef]
            compUnit.addDecl(parseDecl());
        }
        // { FuncDef }
        while (lexer.lookAhead(1).getTokenId() != MAINTK) { // @@ [This condition also uses incorrect lookahead; it should check the current token (lookAhead(0)) for function type keywords like VOIDTK, INTTK, CHARTK rather than looking for MAINTK in lookAhead(1), which fails to properly parse malformed function definitions and prevents error 'j' from being reported at line 1]
            compUnit.addFuncDef(parseFuncDef());
        }
        // MainFuncDef
        compUnit.setMainFuncDef(parseMainFuncDef());

        loggerOut.println(compUnit.output());
        return compUnit;
    }
