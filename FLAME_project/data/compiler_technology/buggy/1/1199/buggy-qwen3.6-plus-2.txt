    private void error(TokenType tokenType) {
        if (tokenType == TokenType.SEMICN) {
            lexer.back();
            lexer.getError().get(lexer.peek().getLineNumber()).add('i');
            lexer.next();
        } else if (tokenType == TokenType.RPARENT) {
            lexer.back();
            lexer.getError().get(lexer.peek().getLineNumber()).add('i'); // @@ The error code for missing right parenthesis ')' should be 'j', not 'i'.
            lexer.next();
        } else if (tokenType == TokenType.RBRACK) {
            lexer.back();
            lexer.getError().get(lexer.peek().getLineNumber()).add('i'); // @@ The error code for missing right square bracket ']' should be 'k', not 'i'.
            lexer.next();
        }
    }
