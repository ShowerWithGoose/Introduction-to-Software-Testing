    private void error(TokenType tokenType) {
        if (tokenType == TokenType.SEMICN) {
            lexer.back();
            lexer.getError().get(lexer.peek().getLineNumber()).add('i'); // @@ The error category code should be 'j' for missing right parenthesis ')', not 'i' for missing semicolon.
            lexer.next();
        } else if (tokenType == TokenType.RPARENT) {
            lexer.back();
            lexer.getError().get(lexer.peek().getLineNumber()).add('i'); // @@ The error category code should be 'j' for missing right parenthesis ')', not 'i' for missing semicolon.
            lexer.next();
        } else if (tokenType == TokenType.RBRACK) {
            lexer.back();
            lexer.getError().get(lexer.peek().getLineNumber()).add('i'); // @@ The error category code should be 'k' for missing right square bracket ']', not 'i' for missing semicolon.
            lexer.next();
        }
    }
