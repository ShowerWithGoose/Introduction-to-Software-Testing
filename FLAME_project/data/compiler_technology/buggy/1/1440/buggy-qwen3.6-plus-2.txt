    public CompUnit parseCompUnit() {
        CompUnit compUnit = new CompUnit();
        // 循环解析 Decl 部分
        while (token.getType() == tokenType.CONSTTK || ((token.getType() == tokenType.INTTK || token.getType() == tokenType.CHARTK) && index2Token(2).getType() != tokenType.LPARENT)) { // @@ The lookahead logic is incorrect. For "int main()", index2Token(2) returns "main" (MAINTK), which is not LPARENT, causing the parser to incorrectly treat the main function definition as a variable declaration (Decl). This leads to parsing errors and potentially no output or incorrect error reporting. It should check if the identifier is followed by '(' to distinguish FuncDef/MainFuncDef from VarDecl.
            compUnit.decls.add(parseDecl());

        }// 循环解析 FuncDef 部分
        //int ident ()
        while (token.getType() == tokenType.VOIDTK || token.getType() == tokenType.CHARTK || (token.getType() == tokenType.INTTK && index2Token(1).getType() != tokenType.MAINTK)) {
            compUnit.funcDefs.add(parseFuncDef());
        }
        // 解析主函数定义 MainFuncDef
        compUnit.MainFunctionDef = parseMainFuncDef();
        return compUnit;
    }
