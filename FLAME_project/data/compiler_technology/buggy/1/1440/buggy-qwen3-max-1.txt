    private int StmtType(){
        // 检查当前 token 是否为标识符
        if (token.getType() != tokenType.IDENFR) {
            return 0;
        }
        tokenType nextType = index2Token(1).getType();
        // 处理普通变量赋值
        if (nextType == tokenType.ASSIGN) {
            if (index2Token(2).getType() == tokenType.GETINTTK) {
                return 1;
            } else if (index2Token(2).getType() == tokenType.GETCHARTK) {
                return 2;
            } else {
                return 3;
            }
        }
        // 处理数组元素赋值
        if (nextType == tokenType.LBRACK) {
            int currentIndex = this.index;  // 保存当前索引
            token mytoken=this.token;
            match();   //匹配ident;
            match();  // 匹配 LBRACK
            parseExp();  // 解析括号内表达式 @@ [This call consumes arbitrary tokens via recursive descent, but the parser state (index and token) is only partially restored afterward. The internal token consumption during parseExp() cannot be undone by simply resetting index and token because sub-parsers may have advanced the index far beyond what is captured here, leading to incorrect lookahead and token stream desynchronization.]
            if (token.getType() == tokenType.ASSIGN || index2Token(1).getType() == tokenType.ASSIGN) {
                if (index2Token(2).getType() == tokenType.GETINTTK) {
                    this.index = currentIndex;
                    this.token= mytoken;
                    return 1;
                } else if (index2Token(2).getType() == tokenType.GETCHARTK) {
                    this.index = currentIndex;
                    this.token= mytoken;
                    return 2;
                } else {
                    this.index = currentIndex;
                    this.token= mytoken;
                    return 3;
                }
            } else {
                this.index=currentIndex;
                this.token=mytoken;
                return 0;
            }
        }
        return 0;
    }
