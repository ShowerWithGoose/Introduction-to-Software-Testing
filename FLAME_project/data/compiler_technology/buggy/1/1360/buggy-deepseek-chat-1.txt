//编译单元 CompUnit → {Decl} {FuncDef} MainFuncDef // 1.是否存在Decl 2.是否存在FuncDef
std::shared_ptr<Node> parser::handle_CompUnit(int d) {
    Grammar compUnit = {CompUnit, "CompUnit"};
    std::shared_ptr<Node> parent = std::make_shared<Node>(nullptr, d, compUnit);
    std::shared_ptr<Node> child = nullptr;
    //Decl
    while (this->tokens[this->now + 2].type != LPARENT) { // @@ [Error: This condition incorrectly assumes that a function definition always has '(' at position now+2, but for functions like 'int func()', the '(' is at now+2, while for 'void func()' it's also at now+2. However, this logic fails when there are no declarations (Decl) before function definitions, or when the token at now+2 is not '(' (e.g., for array declarations). Additionally, the condition does not properly handle the case where there are no Decl items, causing an infinite loop or out-of-bounds access.]
        child = this->handle_Decl(d + 1);
        parent->addChildren(child);
        child->addParent(parent);
    }
    //FuncDef
    while (this->tokens[this->now + 1].type != MAINTK) { // @@ [Error: This condition assumes that 'main' is always at position now+1 after processing Decl, but this is not guaranteed. If there are no FuncDef items, the loop may skip the main function or cause an out-of-bounds access. Also, the condition does not check for the end of tokens, leading to potential crashes.]
        child = this->handle_FuncDef(d + 1);
        parent->addChildren(child);
        child->addParent(parent);
    }
    //MainFuncDef
    child = this->handle_MainFuncDef(d + 1);
    parent->addChildren(child);
    child->addParent(parent);
    return parent;
}
