    //MainFuncDef → 'int' 'main' '(' ')' Block // j
    private MainFuncDef parseMainFuncDef() {
        tokens.consumeExpect(TokenType.INTTK);
        tokens.consumeExpect(TokenType.MAINTK);
        tokens.consumeExpect(TokenType.LPARENT);
        tokens.consumeExpect(TokenType.RPARENT); // @@ The return value of consumeExpect is ignored, so if the right parenthesis is missing, no error is logged. It should be checked using Logger.logger.check() similar to other parsing methods.
        Block body = parseBlock();
        return new MainFuncDef(body);
    }
