    public CompUnit parseCompUnit() {
        CompUnit compUnit = new CompUnit();
        while (words.get(pos).getStr().equals("const") ||
            !words.get(pos + 2).getStr().equals("(")) { // @@ The condition `!words.get(pos + 2).getStr().equals("(")` is incorrect for identifying Decl vs FuncDef. It assumes that a function definition always has '(' at position pos+2, but for a variable declaration like `int a = 0;`, pos+2 may be out of bounds or not '(' causing the loop to incorrectly treat it as a Decl or fail. This leads to parsing errors and no output.
            compUnit.addDecl(parseDecl());
        }
        while (!words.get(pos + 1).getStr().equals("main")) { // @@ This condition assumes that the next token after a function name is always "main" for MainFuncDef, but it does not correctly handle the case where there are no function definitions before main. It may cause an IndexOutOfBoundsException or incorrect parsing.
            compUnit.addFuncDef(parseFuncDef());
        }
        compUnit.setMainFuncDef(parseMainFuncDef());
        return compUnit;
    }
