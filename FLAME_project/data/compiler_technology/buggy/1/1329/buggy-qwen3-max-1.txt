    public CompUnit parseCompUnit() {
        CompUnit compUnit = new CompUnit();
        while (pos < words.size() && (words.get(pos).getStr().equals("const") ||
            (pos + 2 < words.size() && !words.get(pos + 2).getStr().equals("(")))) { // @@ [Original condition can cause IndexOutOfBoundsException when pos+2 exceeds words list size, preventing proper parsing completion]
            compUnit.addDecl(parseDecl());
        }
        while (pos + 1 < words.size() && !words.get(pos + 1).getStr().equals("main")) { // @@ [Original condition can cause IndexOutOfBoundsException when pos+1 exceeds words list size]
            compUnit.addFuncDef(parseFuncDef());
        }
        compUnit.setMainFuncDef(parseMainFuncDef());
        return compUnit;
    }
