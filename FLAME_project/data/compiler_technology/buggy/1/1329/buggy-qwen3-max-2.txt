    public CompUnit parseCompUnit() {
        CompUnit compUnit = new CompUnit();
        while (words.get(pos).getStr().equals("const") ||
            !words.get(pos + 2).getStr().equals("(")) { // @@ [This condition can cause ArrayIndexOutOfBoundsException when pos+2 exceeds the words list size, and the logic is incorrect for distinguishing declarations from function definitions. For variable declarations like "int a = 0;", pos+2 might not exist or might not be "(", causing premature termination of declaration parsing.]
            compUnit.addDecl(parseDecl());
        }
        while (!words.get(pos + 1).getStr().equals("main")) { // @@ [This condition also risks ArrayIndexOutOfBoundsException when pos+1 exceeds bounds, and fails to properly handle cases where there are no function definitions before main.]
            compUnit.addFuncDef(parseFuncDef());
        }
        compUnit.setMainFuncDef(parseMainFuncDef());
        return compUnit;
    }
