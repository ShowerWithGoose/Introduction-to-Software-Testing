    public Token getLogicalOperator() {

        if (end() || input.charAt(pos) != c) {
            ErrorLog.getInstance().addError(new Error(row,"a"));
        } else {
            stringBuilder.append(c);
        }
        if (!end()) {
            move(); // @@ [This move() is incorrect when the second character doesn't match; it consumes an extra character even when only a single '&' or '|' is present, and the method returns an AND/OR token instead of treating the single character as invalid]
        }
        if (c == '&') {
            return new Token(TokenType.AND, stringBuilder.toString(), row);

            return new Token(TokenType.OR, stringBuilder.toString(), row);
        }
