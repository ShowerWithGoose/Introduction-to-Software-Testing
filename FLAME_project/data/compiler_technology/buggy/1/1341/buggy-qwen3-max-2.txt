    public FuncDef parseFuncDef() {
        FuncDef funcDef = new FuncDef(tokens.next(), tokens.next());
        funcDef.setLparent(tokens.next());//(
        if (!tokens.peek().is(TokenType.RPARENT)) {
            if (tokens.peek().is(TokenType.LBRACE)) {
                //error j
                addError("j"); @@ [This adds error 'j' but doesn't consume the LBRACE token, causing the parser to continue with incorrect token positioning]
                funcDef.setBlock(parseBlock());
                return funcDef;
            } else {
                funcDef.setFuncFParams(parseFuncFParams());
            }
        }
        //error j
        if (!tokens.peek().is(TokenType.RPARENT)) {
            addError("j"); @@ [This condition is reached even when we already encountered LBRACE, but tokens.peek() is now LBRACE, not RPARENT, so it adds another error incorrectly]
        } else {
            funcDef.setRparent(tokens.next());
        }
        funcDef.setBlock(parseBlock());
        return funcDef;

    public UnaryExp parseUnaryExp() {
        UnaryExp unaryExp = new UnaryExp();
        while (tokens.peek().is(TokenType.NOT) || tokens.peek().is(TokenType.PLUS) || tokens.peek().is(TokenType.MINU)) {
            unaryExp.addUnaryOp(new UnaryOp(tokens.next()));
        }
        if (tokens.peek().is(TokenType.IDENFR) && tokens.peek(1).is(TokenType.LPARENT)) {
            unaryExp.setIdent(tokens.next());
            unaryExp.setLparent(tokens.next());//(
            if (!tokens.peek().is(TokenType.RPARENT)) {
                FuncRParams funcRParams = tryFuncRParams();
                if (funcRParams != null) {
                    unaryExp.setFuncRParams(funcRParams);
                } else {
                    //error j
                    addError("j"); @@ [When tryFuncRParams fails due to missing RPARENT, this adds error 'j' but then the following check also adds another error 'j', causing duplicate errors]
                    return unaryExp;
                }
            }
            //error j
            if (!tokens.peek().is(TokenType.RPARENT)) {
                addError("j"); @@ [This adds a second error 'j' for the same missing parenthesis, and doesn't handle the case where we should stop parsing due to the error]
            } else {
                unaryExp.setRparent(tokens.next());
            }
        } else {
            unaryExp.setPrimaryExp(parsePrimaryExp());
        }
        return unaryExp;
    }
