    private Node parseBlock() { //  Block → '{' { BlockItem } '}'
        Node block = new Node(ParType.Block);
        LeafNode lBrace = new LeafNode(now.getLineNum(),now.getName(),0,now.getLexType(),block); // @@ [Left brace leaf created from current token 'now']
        next(); // @@ [Error: next() called before adding the left brace leaf to block, causing the left brace token to be lost]
        block.addChildNode(lBrace); // @@ [Left brace added after next(), but 'now' already points to the token after '{']
        while (now.getLexType() != LexType.RBRACE) {
            Node blockItem = parseBlockItem();
            blockItem.setParentNode(block);
            block.addChildNode(blockItem);
        }
        LeafNode rBrace = new LeafNode(now.getLineNum(),now.getName(),0,now.getLexType(),block);
        block.addChildNode(rBrace);
        if (iter.hasNext()) {
            next();
        }
        return block;
    }
