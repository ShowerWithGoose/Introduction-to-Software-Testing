tokenItem Token::getNextToken() {
    tokenIndex++;
    if (tokenIndex < tokenList.size()) {
        if (tokenIndex > 0) { // @@ Incorrect: This condition prevents printing the first token (index 0) and prints the previous token instead of the current one.
            if (tokenList[tokenIndex - 1].type == "CHRCON") {
                tokenList[tokenIndex - 1].name = "\'" + tokenList[tokenIndex - 1].name + "\'";
            }
            Util::printParserInfo((tokenList[tokenIndex - 1].type + " " + tokenList[tokenIndex - 1].name).c_str()); // @@ Prints the previous token's info, not the current one being returned.
        }
        return tokenList[tokenIndex];
    }
    if (tokenIndex == tokenList.size()) {
        Util::printParserInfo((tokenList[tokenIndex - 1].type + " " + tokenList[tokenIndex - 1].name).c_str()); // @@ Prints the last token again when reaching end, causing duplicate output.
    }
    return {"", "", -1};
}
