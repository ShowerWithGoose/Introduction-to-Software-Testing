void Util::printErrorInfo(int lineNum, const char *error) {
    std::ofstream errorFile("error.txt", std::ios::app);
    errorFile << Token::tokenList[Token::tokenIndex - 1].lineNum << " " << error << std::endl; // @@ The function parameter `lineNum` is ignored; instead, it uses `Token::tokenList[Token::tokenIndex - 1].lineNum`, which may be incorrect when the error is detected at the current token (not the previous one). This can cause the wrong line number to be output for error types like 'i', 'j', 'k'.
    errorFile.close();
}
