void Util::printErrorInfo(int lineNum, const char *error) {
    std::ofstream errorFile("error.txt", std::ios::app);
    errorFile << Token::tokenList[Token::tokenIndex - 1].lineNum << " " << error << std::endl; // @@ [The error line number should be the passed parameter `lineNum`, not `Token::tokenList[Token::tokenIndex - 1].lineNum`. Using the token's line number may be incorrect when the error occurs at a different line, such as when a semicolon is missing and the last non-terminal is on a different line than the token being checked.]
    errorFile.close();
}
