    public void parseDecl(){
        //如果预读的第二个是(结束
        while (tokenList.hasNext()){
            if (tokenList.preRead(2).getType().equals(TokenType.LPARENT)){ // @@ IndexOutOfBoundsException may occur if there are fewer than 2 tokens remaining in the list. preRead does not check bounds.
                //有(，说明是函数了
                return;
            }
            Decl decl = new Decl(tokenList);
            decl.parseDecl();
            declArrayList.add(decl);
        }
    }
    public void parseFuncDef(){
        //如果预读的第一个是main结束
        while (tokenList.hasNext()){
            if (tokenList.preRead(1).getType().equals(TokenType.MAINTK)){ // @@ IndexOutOfBoundsException may occur if there is only 1 token remaining (the last one) and preRead tries to access index size (out of bounds). Although hasNext ensures at least 1, preRead(1) needs index position+1 which might be equal to size.
                //有main，说明是主函数了
                return;
            }
            FuncDef funcDef = new FuncDef(tokenList);
            funcDef.parseFuncDef();
            funcDefArrayList.add(funcDef);
        }
    }
