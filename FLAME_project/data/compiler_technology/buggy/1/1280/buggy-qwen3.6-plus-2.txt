    private void UnaryExp() {
        // UnaryExp → PrimaryExp | Ident '(' [FuncRParams] ')' | UnaryOp UnaryExp
        updateNumber();
        Type type = words.get(wordPoint).getType();
        if (type == Type.PLUS
                || type == Type.MINU
                || type == Type.NOT) {
            UnaryOp();
            UnaryExp();
        } else if (type == Type.IDENFR
                && words.get(wordPoint + 1).getType() == Type.LPARENT) { // @@ Accessing wordPoint + 1 without checking bounds can cause IndexOutOfBoundsException if IDENFR is the last token.
            //Ident
            addOut();
            //'('
            addOut();
            while (words.get(wordPoint).getType() != Type.RPARENT) { // @@ If RPARENT is missing, this loop may run indefinitely or cause IndexOutOfBoundsException when wordPoint exceeds list size, as FuncRParams does not guarantee consuming tokens that lead to RPARENT or EOF in error cases.
                FuncRParams();
            }
            //')'
            if (words.get(wordPoint).getType() == Type.RPARENT) {
                addOut();
            } else {
                addError('j');
            }
        } else {
            PrimaryExp();
        }

        outStr.add("<UnaryExp>");
    }
