    private static void dump(ArrayList<Object> nodes) throws IOException {
        for (Object node : nodes) {
            if (node != null) {
                if (node instanceof ASTNode branchNode) {
                    dump(branchNode.explore());
                    out.write("<" + branchNode.getClass().getSimpleName() + ">" + "\n"); // @@ The output order is incorrect: the syntax component name should be output before analyzing its children, but here it is output after the children have been analyzed.
                } else if (node instanceof ASTNodeWithOption<?> branchNodeWithOption) {
                    dump(branchNodeWithOption.extract().explore());
                    out.write("<" + branchNodeWithOption.getClass().getSimpleName() + ">" + "\n"); // @@ Same issue: the syntax component name is output after the children, but the problem requires outputting the name before analyzing the children.
                } else if (node instanceof Token leafNode) {
                    out.write(leafNode.type().toString() + " " + leafNode.strVal() + "\n");
                }
            }
        }
    }
