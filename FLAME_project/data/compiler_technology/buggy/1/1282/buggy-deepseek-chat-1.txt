    private static void dump(ArrayList<Object> nodes) throws IOException {
        for (Object node : nodes) {
            if (node != null) {
                if (node instanceof ASTNode branchNode) {
                    dump(branchNode.explore());
                    out.write("<" + branchNode.getClass().getSimpleName() + ">" + "\n"); // @@ The syntax component name is output after exploring its children, but the problem requires outputting the name before analyzing the component. This causes the output order to be reversed for nested components.
                } else if (node instanceof ASTNodeWithOption<?> branchNodeWithOption) {
                    dump(branchNodeWithOption.extract().explore());
                    out.write("<" + branchNodeWithOption.getClass().getSimpleName() + ">" + "\n"); // @@ Same issue: the name is output after the children are explored, but it should be output before.
                } else if (node instanceof Token leafNode) {
                    out.write(leafNode.type().toString() + " " + leafNode.strVal() + "\n");
                }
            }
        }
    }
