package symbol;

import java.util.HashMap;
import java.util.Map;

public class FuncSym extends Symbol {
    private Map<String, Symbol> args = new HashMap<>(); // @@ Using HashMap does not preserve the insertion order of function parameters, which causes the argument types to be retrieved in an undefined order during semantic analysis, leading to incorrect parameter type mismatch errors (Error e). It should be changed to LinkedHashMap.

    public FuncSym(String name, SymbolType type, int scope) {
        super(name, type, scope);
    }

    public Map<String, Symbol> getArgs() {
        return args;
    }

    public boolean isExist(String name) {
        return args.containsKey(name);
    }

    public void addArg(Symbol arg) {
        args.put(arg.getName(), arg);
    }
}
