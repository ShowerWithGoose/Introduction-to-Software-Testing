void judge()
{
    int j = i;
    if (tks[j + 1].type == MAINTK) // @@ [Potential out-of-bounds access if j+1 >= tks.size(); also after processing one unit, i may not be updated correctly?]
    {
        opmain();
    }
    else if (tks[j + 2].type == LPARENT) // @@ [Same out-of-bounds risk; also typo in identifier: should be "tks" not "tks"? Actually it's fine but variable name is tks]
    {
        opfuncdef();
    }
    else
    {
        opdecl();
    }
}

void opfuncdef()
{
    opfunctype();
    pttk();
    pttk();
    if (tks[i].type == LBRACE) // @@ [After consuming '(' and identifier? Actually after functype and ident and '(', next token should be either ')' or first param. LBRACE indicates missing ')' – but error handling calls jerr then proceeds to block, which is inconsistent grammar.]
    {
        jerr(i);
        opblock();
    }
    else if (tks[i].type == RPARENT)
    {
        pttk();
        opblock();
    }
    else
    {
        opfuncfparams();
        if (tks[i].type == LBRACE) // @@ [Duplicate LBRACE check: after params, we expect ')', not LBRACE. This case should not occur; should be an error for missing ')' instead.]
        {
            jerr(i);
            opblock();
        }
        else if (tks[i].type == RPARENT)
        {
            pttk();
            opblock();
        }
    }
    wenfafile << "<FuncDef>" << std::endl;
}
