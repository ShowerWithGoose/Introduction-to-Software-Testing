    private Stmt stmtAssOrExpParser() {
        iterator.addRecord();
        LVal lVal = lValParser(); // @@ Errors added here (e.g., for missing ']') are not removed on backtrack
        if (iterator.nextToken().isType(TokenType.ASSIGN)) {
            Token assign = iterator.readNextToken();
            Exp exp = expParser();
            Token semicn = null;
            if (iterator.nextToken().isType(TokenType.SEMICN)) {
                semicn = iterator.readNextToken();
            } else {
                errors.add(new Error(Error.ErrorType.i, iterator.lastToken().getLineNum()));
            }
            return new StmtAssign(lVal, assign, exp, semicn);
        } else {
            iterator.returnToRecord(); // @@ Backtrack, but previously added errors are not undone
            Exp exp = expParser();
            Token semicn = null;
            if (iterator.nextToken().isType(TokenType.SEMICN)) {
                semicn = iterator.readNextToken();
            } else {
                errors.add(new Error(Error.ErrorType.i, iterator.lastToken().getLineNum()));
            }
            return new StmtExp(exp, semicn);
        }
    }
