    public ArrayList<SynNode> parseHead() {
        ArrayList<SynNode> children = new ArrayList<>();
        ExpParser expParser = new ExpParser(lexer);
        SynNode root = expParser.parse();
        updateError(expParser.getErrorList());
        root = analyseExpLVal(root);
        if (root.getType() != SynType.EMPTY) {
            children.add(root);
        }
        if (root.getType() == SynType.LVal) {
            children.add(new SynLeaf(lexer.getAndNext())); // =
            Token t = lexer.getAndNext();
            if (t.getType() == TokenType.GETCHARTK || t.getType() == TokenType.GETINTTK) {
                children.add(new SynLeaf(t));
                children.add(new SynLeaf(lexer.getAndNext())); // (
                t = lexer.getAndNext();
                if (t.getType() == TokenType.RPARENT) { // )
                    children.add(new SynLeaf(t));
                } else {
                    lexer.move(-1);
                    addError(new ErrorSyntax(children.get(1).getLine(), ErrorType.j));
                }
                t = lexer.getAndNext();
                if (t.getType() == TokenType.SEMICN) {
                    children.add(new SynLeaf(t));
                } else {
                    lexer.move(-1);
                    addError(new ErrorSyntax(children.get(1).getLine(), ErrorType.i));
                }
            } else { // Exp; // i
                lexer.move(-1);
                ExpParser expParser2 = new ExpParser(lexer);
                children.add(expParser2.parse());
                updateError(expParser2.getErrorList());
                t = lexer.getAndNext();
                if (t.getType() == TokenType.SEMICN) {
                    children.add(new SynLeaf(t));
                } else {
                    lexer.move(-1);
                    addError(new ErrorSyntax(children.get(1).getLine(), ErrorType.i));
                }
            }
        } else { // Exp(done) ; // i
            Token t = lexer.getAndNext();
            if (t.getType() == TokenType.SEMICN) {
                children.add(new SynLeaf(t));
            } else {
                lexer.move(-2); // @@ If getAndNext() consumed the last token, curIdx is tokens.size(). move(-2) makes it tokens.size()-2. If there was only 1 token left, this becomes -1, causing IndexOutOfBoundsException in the next line.
                addError(new ErrorSyntax(lexer.get().getLine(), ErrorType.i)); // @@ Accessing lexer.get() with an invalid index (potentially -1 or out of bounds) causes Runtime Error.
                lexer.move(1);
            }
        }
        return children;
    }
