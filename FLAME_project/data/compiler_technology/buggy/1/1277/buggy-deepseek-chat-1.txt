    public ArrayList<SynNode> parseHead() {
        ArrayList<SynNode> children = new ArrayList<>();
        ExpParser expParser = new ExpParser(lexer);
        SynNode root = expParser.parse();
        updateError(expParser.getErrorList());
        root = analyseExpLVal(root);
        if (root.getType() != SynType.EMPTY) {
            children.add(root);
        }
        if (root.getType() == SynType.LVal) {
            children.add(new SynLeaf(lexer.getAndNext())); // =
            Token t = lexer.getAndNext();
            if (t.getType() == TokenType.GETCHARTK || t.getType() == TokenType.GETINTTK) {
                children.add(new SynLeaf(t));
                children.add(new SynLeaf(lexer.getAndNext())); // (
                t = lexer.getAndNext();
                if (t.getType() == TokenType.RPARENT) { // )
                    children.add(new SynLeaf(t));
                } else {
                    lexer.move(-1);
                    addError(new ErrorSyntax(children.get(1).getLeaf().getLine(), ErrorType.j)); // @@ The error line for missing ')' should be the line of the last non-terminal before the ')', which is the line of the '(' or the function name. Here it uses children.get(1).getLeaf().getLine() which is the line of '=', but the expected line is 16 (the line of 'getchar' or '('). The correct line should be the line of the last token before the missing ')', which is the line of '(' (or the function name). In the test case, the missing ')' is after 'getchar(', so the line should be 16.
                }
                t = lexer.getAndNext();
                if (t.getType() == TokenType.SEMICN) {
                    children.add(new SynLeaf(t));
                } else {
                    lexer.move(-1);
                    addError(new ErrorSyntax(children.get(1).getLeaf().getLine(), ErrorType.i)); // @@ Similar: the error line for missing ';' should be the line of the last non-terminal before the ';', which is the line of the last token in the statement (e.g., the ')' or the function call). Here it uses the line of '=', which is wrong. The expected line is 16.
                }
            } else { // Exp; // i
                lexer.move(-1);
                ExpParser expParser2 = new ExpParser(lexer);
                children.add(expParser2.parse());
                updateError(expParser2.getErrorList());
                t = lexer.getAndNext();
                if (t.getType() == TokenType.SEMICN) {
                    children.add(new SynLeaf(t));
                } else {
                    lexer.move(-1);
                    addError(new ErrorSyntax(children.get(1).getLeaf().getLine(), ErrorType.i));
                }
            }
        } else { // Exp(done) ; // i
            Token t = lexer.getAndNext();
            if (t.getType() == TokenType.SEMICN) {
                children.add(new SynLeaf(t));
            } else {
                lexer.move(-2);
                addError(new ErrorSyntax(lexer.get().getLine(), ErrorType.i));
                lexer.move(1);
            }
        }
        return children;
    }

    public ArrayList<SynNode> parseIfStmt() {

        children.add(new SynLeaf(lexer.getAndNext())); // if
        children.add(new SynLeaf(lexer.getAndNext())); // (
        children.add(parseCond()); // Cond
        Token t = lexer.getAndNext();
        if (t.getType() == TokenType.RPARENT) {
            children.add(new SynLeaf(t));
        } else {
            lexer.move(-1);
            addError(new ErrorSyntax(children.get(0).getLeaf().getLine(), ErrorType.j)); // @@ The error line for missing ')' should be the line of the last non-terminal before the ')', which is the line of the last token in the Cond (or the '('). Here it uses children.get(0).getLeaf().getLine() which is the line of 'if', but the expected line is 18 (the line of the last token before the missing ')', which is the line of the last token in the condition, e.g., 'f1()' or '&'). The correct line should be the line of the last token before the missing ')', which is the line of the last token in the condition. In the test case, the missing ')' is after 'f2() & f1()', so the line should be 18.
        }
        children.add(parseStmt());
        t = lexer.getAndNext();
        if (t.getType() == TokenType.ELSETK) {
            children.add(new SynLeaf(t));
            children.add(parseStmt());
        } else {
            lexer.move(-1);
        }
        return children;
    }
