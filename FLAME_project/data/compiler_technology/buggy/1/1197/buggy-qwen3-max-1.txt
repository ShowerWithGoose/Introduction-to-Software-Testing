public static Node parse() {
    BType bType = new BType();
    if (Lexer.INSTANCE.getLexType() == LexType.INTTK ||
        Lexer.INSTANCE.getLexType() == LexType.CHARTK) {
        bType.parseAndAdd(NodeRegistry.TERMINAL);
    } else {
        processError(ErrorType.NONE); // @@ [ErrorType.NONE should not be reported as an actual error; this causes the unwanted "1 NONE" output in the error file. The BType parser should not generate errors when called in contexts where type tokens are expected but missing, as higher-level parsers should handle the error reporting with appropriate error codes like 'j' for missing parentheses.]
    }
    return bType;
}
