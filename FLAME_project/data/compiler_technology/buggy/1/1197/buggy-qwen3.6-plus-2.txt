package frontend.nodes;

import frontend.ErrorType;
import frontend.LexType;
import frontend.NodeRegistry;
import frontend.NodeType;

public class MainFuncDef extends Node {
    public MainFuncDef() {
        super(NodeType.MAIN_FUNC_DEF);
    }

    public static Node parse() {
        MainFuncDef mainFuncDef = new MainFuncDef();
        mainFuncDef.terminalCheck(LexType.INTTK); // @@ The default error type for this terminalCheck is ErrorType.NONE. When the parser incorrectly reaches here (e.g., after failing to parse a preceding FuncDef due to syntax errors), it logs a spurious "NONE" error instead of handling the mismatch or synchronizing. This causes the extra "1 NONE" in the output.
        mainFuncDef.terminalCheck(LexType.MAINTK);
        mainFuncDef.terminalCheck(LexType.LPARENT);
        mainFuncDef.terminalCheck(LexType.RPARENT, ErrorType.j);
        mainFuncDef.parseAndAdd(NodeRegistry.BLOCK);
        return mainFuncDef;
    }
}
