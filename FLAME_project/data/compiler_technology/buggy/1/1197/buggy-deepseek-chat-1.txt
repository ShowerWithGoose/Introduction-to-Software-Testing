package frontend.nodes;

import frontend.*;

public class FuncDef extends Node {
    public FuncDef() {
        super(NodeType.FUNC_DEF);
    }

    public static Node parse() {
        FuncDef funcDef = new FuncDef();
        funcDef.parseAndAdd(NodeRegistry.FUNC_TYPE);
        funcDef.terminalCheck(LexType.IDENFR);
        funcDef.terminalCheck(LexType.LPARENT);
        switch (Lexer.INSTANCE.getLexType()) {
            case INTTK, CHARTK -> {
                funcDef.parseAndAdd(NodeRegistry.TERMINAL);
                funcDef.parseAndAdd(NodeRegistry.FUNC_F_PARAMS);
                funcDef.terminalCheck(LexType.RPARENT, ErrorType.j);
            }
            case RPARENT -> funcDef.parseAndAdd(NodeRegistry.TERMINAL);
            default -> processError(ErrorType.j); // @@ [When the next token is not INTTK, CHARTK, or RPARENT (e.g., LBRACE from a missing ')'), this reports error j but does not consume the token, causing the subsequent Block parse to fail or produce incorrect error line numbers.]
        }
        funcDef.parseAndAdd(NodeRegistry.BLOCK);
        return funcDef;
    }
}
