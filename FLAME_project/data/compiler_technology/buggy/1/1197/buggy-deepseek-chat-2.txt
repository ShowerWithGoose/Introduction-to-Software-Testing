    public static Node parse() {
        FuncDef funcDef = new FuncDef();
        funcDef.parseAndAdd(NodeRegistry.FUNC_TYPE);
        funcDef.terminalCheck(LexType.IDENFR);
        funcDef.terminalCheck(LexType.LPARENT);
        switch (Lexer.INSTANCE.getLexType()) {
            case INTTK, CHARTK -> {
                funcDef.parseAndAdd(NodeRegistry.TERMINAL); // @@ This line incorrectly consumes the BType token (int/char) as a terminal, but it should be parsed as part of FuncFParam via BType. This causes the subsequent FuncFParams parsing to miss the BType token.
                funcDef.parseAndAdd(NodeRegistry.FUNC_F_PARAMS);
                funcDef.terminalCheck(LexType.RPARENT, ErrorType.j);
            }
            case RPARENT -> funcDef.parseAndAdd(NodeRegistry.TERMINAL);
            default -> processError(ErrorType.j); // @@ This default case is triggered when the next token is not INTTK, CHARTK, or RPARENT. In the test case "void f(int a {}", after parsing the parameter "int a", the next token is LBRACE, not RPARENT. This default case reports error j on the wrong line (line 1) instead of the line where the missing ')' should be (line 1 is correct for this case, but the error is reported before the block is parsed, causing the block to be parsed incorrectly and leading to additional errors on lines 3 and 5).
        }
        funcDef.parseAndAdd(NodeRegistry.BLOCK);
        return funcDef;
    }
