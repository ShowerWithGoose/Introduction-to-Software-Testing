    public void next() throws IOException {
        if (curType != null && curType != LexType.ERROR) {
            out.write(curWord.getType() + " " + curWord.getToken() + "\n");
        }
        lastWord = curWord;
        curPos++;
        if (curPos >= words.size()) {
            return;
        }
        curWord = words.get(curPos);
        curType = curWord.getType();
        curToken = curWord.getToken();
        curLineNum = curWord.getLineNum();
        if (curType == LexType.ERROR) {
            errors.add(new Error(curLineNum, 'a')); // @@ [This line duplicates error 'a' reporting because the Lexer already added the error to the error file when encountering single '&' or '|'. The Parser should not add another error for the same issue.]
            curType = LexType.AND;
            curToken = "&&";
        }

    }
