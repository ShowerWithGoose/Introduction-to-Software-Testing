    int parseInitVal() {
        tr[++nodecnt]=Parser_unit(0, "InitVal");
        int cur = nodecnt;
        if (peek().tk == "LBRACE") {
            tr[cur].push(parseEnd());
            tr[cur].push(parseExp()); // @@ The grammar allows empty braces '{}', but this code unconditionally parses an Exp, which will fail or produce incorrect results if the next token is '}'. It should check if an expression starts here.
            while (peek().tk == "COMMA") {
                tr[cur].push(parseEnd());
                tr[cur].push(parseExp());
            }
            tr[cur].push(parseEnd());
        } else if (peek().tk == "STRCON") {
            tr[cur].push(parseEnd());
        } else
            tr[cur].push(parseExp());
        return cur;
    }
