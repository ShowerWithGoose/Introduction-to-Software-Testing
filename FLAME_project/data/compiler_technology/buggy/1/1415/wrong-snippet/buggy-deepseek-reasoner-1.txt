    public void lexical_print() {
        if (!errorInfo.isEmpty()) {   // Should be !errorInfo.isEmpty()? Actually, lexical output should be written when there are no errors.
            String filePath = "lexer.txt";
            try (BufferedWriter writer = new BufferedWriter(new FileWriter(filePath, true))) {
                for (Token word : words) {
                    writer.write(word + "\n");
                }
                writer.flush();
            } catch (IOException e) {
                e.printStackTrace();
                System.out.println("写入文件时发生错误。");
            }
        } else {
            String filePath = "error.txt";   // This writes an empty file when there are no errors, but should not be written at all.
            try (BufferedWriter writer = new BufferedWriter(new FileWriter(filePath, true))) {
                writer.flush();
            } catch (IOException e) {
                e.printStackTrace();
                System.out.println("写入文件时发生错误。");
            }
        }
    }
