    public static Node VarDecl() {
        Node node = new Node("<VarDecl>");
        node.addChild(BType());
        node.addChild(VarDef());
        while (now.getName().equals(",")) {
            node.addChild(new Node(/*","*/now.getType() + " " + now.getName()));
            now = Lexer.outs.get(++i);
            node.addChild(VarDef());
        }
        if (now.getName().equals(";")) {
            node.addChild(new Node(/*";"*/now.getType() + " " + now.getName()));
            if (i < Lexer.outs.size() - 1) {
                now = Lexer.outs.get(++i);
            }
        } else {
            //错误处理
            w.add(cnt, new WrongExample(Lexer.outs.get(i - 1).getLine_num(), "i")); // @@ [Using i-1 to get line number is incorrect because after processing VarDef(), the current token 'now' points to the next token after the variable declaration, so i-1 may not correspond to the actual line where the missing semicolon should be. The correct approach should track the line number of the last token in the declaration itself, not rely on i-1 which can be off due to token consumption patterns.]
            cnt++;
            //System.out.println(Lexer.outs.get(i - 1).getLine_num() + " " + "i");
        }
        return node;
    }
