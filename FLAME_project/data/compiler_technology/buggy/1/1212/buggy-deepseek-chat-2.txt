    private boolean isLVal() {
        for (int i = 1; !Objects.equals(Objects.requireNonNull(NextToken(i)).getToken(), ";"); i++) { // @@ The loop condition may cause an infinite loop or NullPointerException if there is no ';' in the remaining tokens, and it does not check for end of token list.
            if (Objects.equals(Objects.requireNonNull(NextToken(i)).getToken(), "=")) {
                return true;
            }
            if (BeginExp(i)) {
                if (LToken1(i).getType().equals("LBRACK")) { // @@ LToken1(i) returns token at index (currentTokenIndex - 1 + i), which is incorrect; it should be NextToken(i) to look ahead.
                    continue;
                }
                else {
                    return false;
                }
            }
        }
        return false;
    }
