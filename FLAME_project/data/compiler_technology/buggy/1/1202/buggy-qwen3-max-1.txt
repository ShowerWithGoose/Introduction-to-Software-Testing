    public FuncDef parseFuncDef() {
        ArrayList<Syntax> syntaxes = new ArrayList<>();
        syntaxes.add(parseFuncType());
        syntaxes.add(getToken());//Ident
        next();
        syntaxes.add(getToken());//(
        next();
        if (getToken().getType().equals(Token.TokenType.CHARTK.toString())
                || getToken().getType().equals(Token.TokenType.INTTK.toString())) {
            syntaxes.add(parseFuncFParams());
        }
        if (getToken().getType().equals(Token.TokenType.RPARENT.toString())) {
            syntaxes.add(getToken());
            next();
        } else {//没有小括号
            back();
            ErrorLog errorLog = new ErrorLog(getToken().getLineNum(), 'j', pos); // @@ [This error handling is incorrect because after detecting a missing ')', it calls next() which advances the position, but doesn't properly recover from the error state. This causes subsequent parsing attempts to fail and generate additional spurious error messages for the same missing parenthesis.]
            errorLogs.add(errorLog);
            next();
        }
        syntaxes.add(parseBlock());
        return new FuncDef(syntaxes);
    }
