void Lexer::getSym() {
    string str;
    str += cur_line[index++];
    // str = "&"
    if (index < cur_line.length()) {
        string rts = str + cur_line[index];
        // rts = "&&"
        if (symbol.count(rts)) {
            this->cur_token = rts;
            this->cur_type = symbol.at(rts);
            index++;
        } else {
            this->cur_token = str;
            this->cur_type = symbol.at(str); // @@ [if str is "&" or "|", symbol.at(str) will return ANDOP or OROP, but then the code overrides cur_type to AND/OR and prints error; however, it still calls printOut(), which outputs a token instead of only reporting an error]
            if (str[0] == '&' || str[0] == '|') printOut_for_error();
            if (str[0] == '&') this->cur_type = AND;
            if (str[0] == '|') this->cur_type = OR;
        }
    } else {
        this->cur_token = str;
        this->cur_type = symbol.at(str); // @@ [same issue: for standalone "&" or "|", it logs an error but still proceeds to print a token via printOut(), violating the requirement that erroneous programs should only output to error.txt]
        if (str[0] == '&' || str[0] == '|') printOut_for_error();
        if (str[0] == '&') this->cur_type = AND;
        if (str[0] == '|') this->cur_type = OR;
    }

    printOut(); // @@ [this line should not be executed when an error (like single '&') is detected; currently, both error.txt and lexer.txt get output, which is incorrect]
}
