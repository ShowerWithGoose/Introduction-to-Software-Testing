    private void scanString() throws IOException {
        int chr = reader.read();
        assert chr == '"' : "Invariant broken";
        Pair<Integer, Integer> start = reader.location();
        StringBuilder str = new StringBuilder();
        while ((chr = reader.read()) != '"') {
            if (chr == '%') {
                chr = reader.read();
                if (chr != 'd') {
                    errors.add(ErrorType.InvalidFormatString, location());
                    continue;
                }
                str.append('%').append('d');
            } else if (chr == '\\') {
                chr = reader.read();
                if (chr != 'n') {
                    reader.unread(chr);
                    errors.add(ErrorType.InvalidFormatString, location());
                    continue;
                }
                str.append('\n');
            } else if (chr != (char) 32 && chr != (char) 33 && (chr < (char) 40 || chr > (char) 126)) {
                errors.add(ErrorType.InvalidFormatString, location());
            } else {
                str.append((char) chr);
            }
        }
        addToken(TokenType.FString, '"' + str.toString() + '"', start); // @@ The token type should be STRCON, not FString. The token type FString is not defined in the problem description; the correct token type for string constants is STRCON.
    }
