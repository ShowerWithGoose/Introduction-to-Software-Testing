        int chr;
        while ((chr = reader.read()) != -1) {
            if (Character.isWhitespace(chr)) {
                continue;
            }
            reader.unread(chr);

            if (Character.isDigit(chr)) {
                scanNumber();
            } else if (Character.isLetter(chr) || chr == '_') {
                scanIdentifier();
            } else if (chr == '"') {
                scanString();
            } else if (chr == '/') {
                scanSlash();
            } else {
                scanSymbol();
            }
        }
        return this;
