    public static TokenType getBySymbol(String symbol) {
        for (TokenType tokenType : TokenType.values()) {
            if (tokenType.getSymbol().equals(symbol)) { // @@ NullPointerException occurs here because tokenType.getSymbol() returns null for enums like IDENFR, INTCON, etc., which do not have a symbol assigned in their constructor. Calling .equals() on a null reference causes the crash.
                return tokenType;
            }
        }
        return null;
    }
