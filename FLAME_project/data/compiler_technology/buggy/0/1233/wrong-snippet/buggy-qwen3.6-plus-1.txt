    public String getLetter(){
        //用来获取字母
        StringBuilder sb = new StringBuilder();
        while (positionNum < inputLine.length() && (Character.isLetter(inputLine.charAt(positionNum)) || Character.isDigit(inputLine.charAt(positionNum)) || inputLine.charAt(positionNum) == '_')){
            sb.append(inputLine.charAt(positionNum));
            positionNum++;
        }
        return sb.toString();
    }
