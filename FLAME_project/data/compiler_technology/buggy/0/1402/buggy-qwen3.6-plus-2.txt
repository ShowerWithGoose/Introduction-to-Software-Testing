    public static void writeError(String filePath, int lineNumber, ErrorType type) throws IOException {
        try (BufferedWriter bufferedWriter = new BufferedWriter(new FileWriter(filePath, true))) { // @@ [Performance Issue: Opening and closing the file for every single error write is extremely inefficient and can cause Time Limit Exceeded. The writer should be initialized once and reused, or output should be buffered in memory and written at the end.]
            bufferedWriter.write(String.format("%d %s\n", lineNumber, type));
        }
    }

    public static void writeRight(String filePath, TokenType type, String content) throws IOException {
        try (BufferedWriter bufferedWriter = new BufferedWriter(new FileWriter(filePath, true))) { // @@ [Performance Issue: Opening and closing the file for every single token write is extremely inefficient and is the primary cause of Time Limit Exceeded. The writer should be initialized once and reused, or output should be buffered in memory and written at the end.]
            bufferedWriter.write(String.format("%s %s\n", type, content));
        }
    }
