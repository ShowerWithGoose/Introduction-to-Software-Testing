    public static void writeError(String filePath, int lineNumber, ErrorType type) throws IOException {
        try (BufferedWriter bufferedWriter = new BufferedWriter(new FileWriter(filePath, true))) { // @@ [Performance Error: Opening and closing a file for every single error write is extremely inefficient and causes Time Limit Exceeded. The writer should be opened once and reused.]
            bufferedWriter.write(String.format("%d %s\n", lineNumber, type));
        }
    }

        }
