import frontend.Config;
import frontend.Lexer;
import java.nio.file.Files;
import java.nio.file.Paths;

public class Compiler {
    public static void main(String[] args) throws Exception {
        Lexer instance = Lexer.getInstace();
        Config.init();
        String source;
        try {
            source = Files.readString(Paths.get(Config.fileInPath));
        } catch (Exception e) {
            source = null;
            e.printStackTrace();
        }
        boolean success = !instance.parse(source + " ");
        if(success) {
            instance.printTokens();
        }
        else {
            Config.error();
            instance.printErrors();
        }
    }
}

package frontend;

public class Error {
    private String stage;
    private int position;
    private char errorCode;
    Error(String stage, int position, char errorCode) {
        this.stage = stage;
        this.position = position;
        this.errorCode = errorCode;
    }
    public String toString() {
        return position + " " + Character.toString(errorCode);
    }
}
