<lexer/lexer.cpp>
#include "lexer.h"
#include <cctype>
#include <fstream>
#include <unordered_map>
#include <iostream>

Lexer::Lexer(const std::string &inputFile, const std::string &lexerFile, const std::string &errorFile)
        : lineno(1), hasError(false) {
    input.open(inputFile);
    lexerOutput.open(lexerFile);
    errorOutput.open(errorFile);
}

Lexer::~Lexer() {
    input.close();
    lexerOutput.close();
    errorOutput.close();
}

void Lexer::analyze() {
    char ch;
    while (input.get(ch)) {
        if (std::isalpha(ch) || ch == '_') {
            std::string content;
            content.push_back(ch);
            while (input.get(ch) && (std::isalnum(ch) || ch == '_')) {
                content.push_back(ch);
            }
            input.unget();
            if (keywords.find(content) != keywords.end()) {
                writeToken(keywords[content], content);
            } else {
                writeToken(TokenType::IDENFR, content);
            }
        } else if (std::isdigit(ch)) {
            std::string content;
            content.push_back(ch);
            while (input.get(ch) && std::isdigit(ch)) {
                content.push_back(ch);
            }
            input.unget();
            writeToken(TokenType::INTCON, content);
        } else if (ch == '"') {
            handleStringOrChar('"');
        } else if (ch == '\'') {
            handleStringOrChar('\'');
        } else if (std::isspace(ch)) {
            if (ch == '\n') lineno++;
        } else {
            switch (ch) {
                case '+': writeToken(TokenType::PLUS, "+"); break;
                case '-': writeToken(TokenType::MINU, "-"); break;
                case '*': writeToken(TokenType::MULT, "*"); break;
                case '/': handleSlash(); break;
                case '%': writeToken(TokenType::MOD, "%"); break;
                case '<': handleLessThan(); break;
                case '>': handleGreaterThan(); break;
                case '=': handleEqualSign(); break;
                case '!': handleExclamation(); break;
                case '&': handleAndOperator(); break;
                case '|': handleOrOperator(); break;
                case '(': writeToken(TokenType::LPARENT, "("); break;
                case ')': writeToken(TokenType::RPARENT, ")"); break;
                case '[': writeToken(TokenType::LBRACK, "["); break;
                case ']': writeToken(TokenType::RBRACK, "]"); break;
                case '{': writeToken(TokenType::LBRACE, "{"); break;
                case '}': writeToken(TokenType::RBRACE, "}"); break;
                case ';': writeToken(TokenType::SEMICN, ";"); break;
                case ',': writeToken(TokenType::COMMA, ","); break;
                default: handleError(ch); break;
            }
        }
    }

    if (!hasError) {
        errorOutput.close();
    }
}

void Lexer::handleLessThan() {
    char ch;
    if (input.get(ch) && ch == '=') {
        writeToken(TokenType::LEQ, "<=");
    } else {
        input.unget();
        writeToken(TokenType::LSS, "<");
    }
}

void Lexer::handleGreaterThan() {
    char ch;
    if (input.get(ch) && ch == '=') {
        writeToken(TokenType::GEQ, ">=");
    } else {
        input.unget();
        writeToken(TokenType::GRE, ">");
    }
}

void Lexer::handleEqualSign() {
    char ch;
    if (input.get(ch) && ch == '=') {
        writeToken(TokenType::EQL, "==");
    } else {
        input.unget();
        writeToken(TokenType::ASSIGN, "=");
    }
}

void Lexer::handleExclamation() {
    char ch;
    if (input.get(ch) && ch == '=') {
        writeToken(TokenType::NEQ, "!=");
    } else {
        input.unget();
        writeToken(TokenType::NOT, "!");
    }
}

void Lexer::handleAndOperator() {
    char ch;
    if (input.get(ch) && ch == '&') {
        writeToken(TokenType::AND, "&&");
    } else {
        handleError('&');
    }
}

void Lexer::handleOrOperator() {
    char ch;
    if (input.get(ch) && ch == '|') {
        writeToken(TokenType::OR, "||");
    } else {
        handleError('|');
    }
}

void Lexer::handleError(char ch) {
    errorOutput << lineno << " " << static_cast<char>(ErrorType::ILLEGAL_SYMBOL) << std::endl;
    hasError = true;
}

void Lexer::writeToken(TokenType type, const std::string &value) {
    lexerOutput << static_cast<int>(type) << " " << value << std::endl;
}

void Lexer::handleStringOrChar(char boundary) {
    std::string content(1, boundary);
    char ch;
    while (input.get(ch) && (ch != boundary || content.back() == '\\')) {
        content.push_back(ch);
    }
    content.push_back(boundary);
    if (boundary == '\'') {
        writeToken(TokenType::CHRCON, content);
    } else {
        writeToken(TokenType::STRCON, content);
    }
}

void Lexer::handleSlash() {
    char ch;
    if (input.get(ch)) {
        if (ch == '/') {
            while (input.get(ch) && ch != '\n') {}
            lineno++;
        } else if (ch == '*') {
            while (input.get(ch)) {
                if (ch == '*') {
                    if (input.get(ch) && ch == '/') {
                        break;
                    }
                }
                if (ch == '\n') lineno++;
            }
        } else {
            input.unget();
            writeToken(TokenType::DIV, "/");
        }
    }
}

int main() {
    std::string inputFile = "testfile.txt";
    std::string lexerOutputFile = "lexer.txt";
    std::string errorOutputFile = "error.txt";

    Lexer lexer(inputFile, lexerOutputFile, errorOutputFile);

    lexer.analyze();

    return 0;
}
<lexer/lexer.h>
#ifndef LEXER_H
#define LEXER_H

#include <string>
#include <fstream>
#include <unordered_map>

enum class TokenType {
    IDENFR, INTCON, STRCON, CHRCON, PLUS, MINU, MULT, DIV, MOD,
    LSS, LEQ, GRE, GEQ, EQL, NEQ, ASSIGN, AND, OR, NOT,
    LPARENT, RPARENT, LBRACK, RBRACK, LBRACE, RBRACE, SEMICN, COMMA,
    UNKNOWN
};

enum class ErrorType {
    ILLEGAL_SYMBOL = 'a'
};

class Lexer {
private:
    std::ifstream input;
    std::ofstream lexerOutput;
    std::ofstream errorOutput;
    int lineno;
    bool hasError;  // 添加 hasError 成员变量
    std::unordered_map<std::string, TokenType> keywords = {
            {"int", TokenType::IDENFR}, {"char", TokenType::IDENFR}, {"const", TokenType::IDENFR},
            {"if", TokenType::IDENFR}, {"else", TokenType::IDENFR}, {"for", TokenType::IDENFR},
            {"break", TokenType::IDENFR}, {"continue", TokenType::IDENFR}, {"return", TokenType::IDENFR},
            {"void", TokenType::IDENFR}, {"getint", TokenType::IDENFR}, {"getchar", TokenType::IDENFR},
            {"printf", TokenType::IDENFR}, {"main", TokenType::IDENFR}
    };

    void handleLessThan();
    void handleGreaterThan();
    void handleEqualSign();
    void handleExclamation();
    void handleAndOperator();
    void handleOrOperator();
    void handleError(char ch);
    void writeToken(TokenType type, const std::string &value);
    void handleStringOrChar(char boundary);
    void handleSlash();

public:
    Lexer(const std::string &inputFile, const std::string &lexerFile, const std::string &errorFile);
    ~Lexer();
    void analyze();
};

#endif
