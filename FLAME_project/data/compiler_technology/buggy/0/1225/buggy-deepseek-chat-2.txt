    public static ArrayList<Token> getTokenToPrint() {
        if (errTokenList.isEmpty()) {
            return tokenList;
        } else {
            int prevLineNo = 0;
            Iterator<Token> it = errTokenList.iterator();
            while (it.hasNext()) {
                Token item = it.next();
                int curLineNo = item.getLineNo();
                if (prevLineNo == curLineNo) {
                    it.remove();
                } else {
                    prevLineNo = curLineNo;
                }
            }
            return errTokenList;
        }
    }
