    private boolean getCharConst() {
        if (sourceCode.charAt(charIndex) == '\'') {
            charIndex++;
            if (charIndex < sourceCode.length()) {
                char c = sourceCode.charAt(charIndex);
                if (c == '\\') { // 转义字符
                    charIndex++;
                    c = sourceCode.charAt(charIndex);
                    if (charIndex + 1 < sourceCode.length()
                            && escapeChars.contains(String.valueOf(c))
                            && sourceCode.charAt(charIndex + 1) == '\'') {
                        tokens.add(new Token("CHARCON", "\'\\" + String.valueOf(c) + "\'", lineIndex)); // @@ The token value for char constants is incorrectly formatted. The expected output uses "CHRCON" as the token type and the value should be the character literal without extra backslashes, e.g., "'a'" not "'\\a'". The type should be "CHRCON" instead of "CHARCON".
                        charIndex += 2;
                    } else {
                        System.out.println("Missing ' at line " + lineIndex);
                        return false;
                    }
                } else if (asciiNotEscapeChars.contains(String.valueOf(c))) { // 非转义字符
                    charIndex++;
                    if (sourceCode.charAt(charIndex) == '\'') {
                        tokens.add(new Token("CHARCON", "\'" + String.valueOf(c) + '\'', lineIndex)); // @@ The token type should be "CHRCON" instead of "CHARCON" to match the expected output.
                        charIndex++;
                    } else {
                        System.out.println("Missing ' at line " + lineIndex);
                        return false;
                    }
                } else {
                    System.out.println("Invalid character");
                    return false;
                }
            } else {
                System.out.println("Missing ' at line " + lineIndex);
                return false;
            }
            return true;
        }
        return false;
    }
