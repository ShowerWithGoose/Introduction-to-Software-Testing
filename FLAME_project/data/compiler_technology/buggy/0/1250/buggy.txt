package frontend;

import java.util.HashMap;
import java.io.PrintWriter;
import java.io.FileWriter;
import java.io.IOException;
import java.util.ArrayList;

public class Lexer {
    private String sourceCode;
    private static HashMap<String, String> reservedWords = new HashMap<>();
    private static HashMap<String, String> singleCharOperators = new HashMap<>();
    private static HashMap<String, String> doubleCharOperators = new HashMap<>();
    private static HashMap<String, String> separators = new HashMap<>();
    // 存储tokens，<type, value, lineIndex>
    private static ArrayList<Token> tokens = new ArrayList<>();

    private static String spaceChars = " \t\n\r";
    private static String asciiNotEscapeChars = " !#$%&()*+,-./0123456789:;<=>?@ABCDEFGHIJKLMNOPQRSTUVWXYZ[]^_`abcdefghijklmnopqrstuvwxyz{|}~";
    private static String escapeChars = "abtnvf\"\'\\0";

    private int lineIndex;
    private int charIndex;

    public Lexer(String sourceCode) {
        this.sourceCode = sourceCode;
        reservedWords.put("main", "MAINTK");
        reservedWords.put("const", "CONSTTK");
        reservedWords.put("int", "INTTK");
        reservedWords.put("char", "CHARTK");
        reservedWords.put("break", "BREAKTK");
        reservedWords.put("continue", "CONTINUETK");
        reservedWords.put("if", "IFTK");
        reservedWords.put("else", "ELSETK");
        reservedWords.put("for", "FORTK");
        reservedWords.put("getint", "GETINTTK");
        reservedWords.put("getchar", "GETCHARTK");
        reservedWords.put("printf", "PRINTFTK");
        reservedWords.put("return", "RETURNTK");
        reservedWords.put("void", "VOIDTK");
        singleCharOperators.put("+", "PLUS");
        singleCharOperators.put("-", "MINU");
        singleCharOperators.put("*", "MULT");
        singleCharOperators.put("/", "DIV");
        singleCharOperators.put("%", "MOD");
        singleCharOperators.put("<", "LSS");
        singleCharOperators.put(">", "GRE");
        singleCharOperators.put("=", "ASSIGN");
        singleCharOperators.put("!", "NOT");
        doubleCharOperators.put("==", "EQL");
        doubleCharOperators.put("!=", "NEQ");
        doubleCharOperators.put("&&", "AND");
        doubleCharOperators.put("||", "OR");
        doubleCharOperators.put(">=", "GEQ");
        doubleCharOperators.put("<=", "LEQ");
        separators.put("(", "LPARENT");
        separators.put(")", "RPARENT");
        separators.put("[", "LBRACK");
        separators.put("]", "RBRACK");
        separators.put("{", "LBRACE");
        separators.put("}", "RBRACE");
        separators.put(";", "SEMICN");
        separators.put(",", "COMMA");
        lineIndex = 1;
        charIndex = 0;
    }

    public ArrayList<Token> getTokens() {
        return tokens;
    }

    public void toTokens() {

        while (charIndex < sourceCode.length()) {
            if (skipSpace()) {
                continue;
            }
            if (skipComment()) {
                continue;
            }
            if (getIdentifierOrReservedWord()) {
                continue;
            }
            if (getIntConst()) {
                continue;
            }
            if (getCharConst()) {
                continue;
            }
            if (getStringConst()) {
                continue;
            }
            if (getOperator()) {
                continue;
            }
            if (getSeparator()) {
                continue;
            }
        }
    }

    public void printTokens(String outputFile) {
        try {
            PrintWriter writer = new PrintWriter(outputFile);
            for (Token token : tokens) {
                writer.println(token.getType() + " " + token.getValue());
            }
            writer.close();
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    private boolean skipSpace() {
        boolean spaceFlag = false;
        while (charIndex < sourceCode.length() &&
                spaceChars.contains(String.valueOf(sourceCode.charAt(charIndex)))) {
            spaceFlag = true;
            if (sourceCode.charAt(charIndex) == '\n') {
                lineIndex++;
            }
            charIndex++;
        }
        return spaceFlag;
    }

    private boolean skipComment() {
        char c = sourceCode.charAt(charIndex);
        if (c == '/') {
            if (charIndex + 1 < sourceCode.length() && sourceCode.charAt(charIndex + 1) == '/') {
                // 处理单行注释
                charIndex += 2;
                while (charIndex < sourceCode.length() && sourceCode.charAt(charIndex) != '\n') {
                    charIndex++;
                }
                return true;
            } else if (charIndex + 1 < sourceCode.length() && sourceCode.charAt(charIndex + 1) == '*') {
                // 处理多行注释
                charIndex += 2;
                while (charIndex < sourceCode.length()
                        && !(sourceCode.charAt(charIndex) == '*' && sourceCode.charAt(charIndex + 1) == '/')) {
                    if (sourceCode.charAt(charIndex) == '\n') {
                        lineIndex++;
                    }
                    charIndex++;
                }
                if (charIndex < sourceCode.length()) {
                    charIndex += 2; // 跳过注释的结束符号
                } else {
                    System.out.println("Missing */");
                }
                return true;
            }
        }
        return false;
    }

    private boolean getIdentifierOrReservedWord() {
        if (Character.isLetter(sourceCode.charAt(charIndex)) || sourceCode.charAt(charIndex) == '_') {
            StringBuilder identifier = new StringBuilder(); // 非数字开头
            identifier.append(sourceCode.charAt(charIndex));
            charIndex++;
            while (charIndex < sourceCode.length() &&
                    (Character.isLetterOrDigit(sourceCode.charAt(charIndex)) ||
                            sourceCode.charAt(charIndex) == '_')) {
                identifier.append(sourceCode.charAt(charIndex));
                charIndex++;
            }
            String identifierStr = identifier.toString();
            if (reservedWords.containsKey(identifierStr)) {
                tokens.add(new Token(reservedWords.get(identifierStr), identifierStr, lineIndex));
            } else {
                tokens.add(new Token("IDENFR", identifierStr, lineIndex));
            }
            return true;
        }
        return false;
    }

    private boolean getIntConst() {
        if (Character.isDigit(sourceCode.charAt(charIndex))) {
            StringBuilder number = new StringBuilder();
            number.append(sourceCode.charAt(charIndex));
            charIndex++;
            if (sourceCode.charAt(charIndex - 1) != '0') { // 无前导零
                while (charIndex < sourceCode.length() && Character.isDigit(sourceCode.charAt(charIndex))) {
                    number.append(sourceCode.charAt(charIndex));
                    charIndex++;
                }
            }
            tokens.add(new Token("INTCON", number.toString(), lineIndex));
            return true;
        }
        return false;
    }

    private boolean getCharConst() {
        if (sourceCode.charAt(charIndex) == '\'') {
            charIndex++;
            if (charIndex < sourceCode.length()) {
                char c = sourceCode.charAt(charIndex);
                if (c == '\\') { // 转义字符
                    charIndex++;
                    c = sourceCode.charAt(charIndex);
                    if (charIndex + 1 < sourceCode.length()
                            && escapeChars.contains(String.valueOf(c))
                            && sourceCode.charAt(charIndex + 1) == '\'') {
                        tokens.add(new Token("CHARCON", "\'\\" + String.valueOf(c) + "\'", lineIndex));
                        charIndex += 2;
                    } else {
                        System.out.println("Missing ' at line " + lineIndex);
                        return false;
                    }
                } else if (asciiNotEscapeChars.contains(String.valueOf(c))) { // 非转义字符
                    charIndex++;
                    if (sourceCode.charAt(charIndex) == '\'') {
                        tokens.add(new Token("CHARCON", "\'" + String.valueOf(c) + '\'', lineIndex));
                        charIndex++;
                    } else {
                        System.out.println("Missing ' at line " + lineIndex);
                        return false;
                    }
                } else {
                    System.out.println("Invalid character");
                    return false;
                }
            } else {
                System.out.println("Missing ' at line " + lineIndex);
                return false;
            }
            return true;
        }
        return false;
    }

    private boolean getStringConst() {
        if (sourceCode.charAt(charIndex) == '\"') {
            StringBuilder string = new StringBuilder();
            charIndex++;
            while (charIndex < sourceCode.length() && sourceCode.charAt(charIndex) != '\"') {
                if (sourceCode.charAt(charIndex) == '\n') {
                    System.out.println("Missing \" at line " + lineIndex);
                    return false;
                }
                string.append(sourceCode.charAt(charIndex));
                charIndex++;
            }
            if (sourceCode.charAt(charIndex) == '\"') {
                tokens.add(new Token("STRCON", "\"" + string.toString() + "\"", lineIndex));
                charIndex++;
            } else {
                System.out.println("Missing \" at line " + lineIndex);
                return false;
            }
            return true;
        }
        return false;
    }

    private boolean getSeparator() {
        String s = String.valueOf(sourceCode.charAt(charIndex));
        if (separators.containsKey(s)) {
            tokens.add(new Token(separators.get(s), s, lineIndex));
            charIndex++;
            return true;
        }
        return false;
    }

    private boolean getOperator() {
        StringBuilder string = new StringBuilder();
        string.append(sourceCode.charAt(charIndex));
        if (charIndex + 1 < sourceCode.length()) { // 双字符运算符
            string.append(sourceCode.charAt(charIndex + 1));
            if (doubleCharOperators.containsKey(string.toString())) {
                tokens.add(new Token(doubleCharOperators.get(string.toString()), string.toString(), lineIndex));
                charIndex += 2;
                return true;
            } else {
                string.delete(1, 2);
            }
        }

        if (singleCharOperators.containsKey(string.toString())) {
            tokens.add(new Token(singleCharOperators.get(string.toString()), string.toString(), lineIndex));
            charIndex++;
            return true;
        } else if (sourceCode.charAt(charIndex) == '&' || sourceCode.charAt(charIndex) == '|') {
            if (sourceCode.charAt(charIndex) == '&') {
                tokens.add(new Token("AND", string.toString(), lineIndex));
            } else {
                tokens.add(new Token("OR", string.toString(), lineIndex));
            }
            try {
                PrintWriter writer = new PrintWriter(new FileWriter("error.txt", true)); // 追加写
                writer.println(lineIndex + " a");
                writer.close();
            } catch (IOException e) {
                System.out.println("写入错误文件时发生异常：" + e.getMessage());
            }
            charIndex++;
            return true;
        }
        return false;
    }

}

package frontend;

class Token {

    private String type;
    private String value;
    private int lineIndex;

    public Token(String type, String value, int lineIndex) {
        this.type = type;
        this.value = value;
        this.lineIndex = lineIndex;
    }

    public String getType() {
        return type;
    }

    public String getValue() {
        return value;
    }

    public int getLineIndex() {
        return lineIndex;
    }

}
