        StringBuilder sb = new StringBuilder();

        readCh();
        while (curChar != '\"') {
            if (curChar == -1) {
                throw new RuntimeException("no \" to match");
            }
            //"\n" => "\\n"
            /*if (curChar == '\\') {
                sb.append((char) "\\\\");
            } else {

            }*/

            readCh();
        }
        Token token = new Token(STRCON, sb.toString(), lineNo); // @@ The closing double quote '"' is currently held in curChar after the loop exits, but it is not appended to sb before creating the token. This results in the output string missing the closing quote.

    }
