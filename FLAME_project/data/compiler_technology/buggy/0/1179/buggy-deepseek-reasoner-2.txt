void checkTypeA(const char* line, int lineNumber) {
    int i = 0;
    FILE* errorFile = fopen(ERROR_FILENAME, "a");
    if (!errorFile) {
        perror("无法打开错误文件");
        return;
    }

    while (line[i] != '\0' && line[i] != '"') {  // 只处理到遇到双引号的部分 @@ [This condition does not skip line comments (//), possibly causing false positives for '&' or '|' inside comments]
        if (line[i] == '&') {
            if (line[i + 1] == '&') {
                i += 2;
            }
            else {
                fprintf(errorFile, "%d a\n", lineNumber);
                errorCount++;
                break;  // @@ [Breaks after first error; but spec says each erroneous example has exactly one error, so this is acceptable]
            }
        }
        else if (line[i] == '|') {
            if (line[i + 1] == '|') {
                i += 2;
            }
            else {
                fprintf(errorFile, "%d a\n", lineNumber);
                errorCount++;
                break;
            }
        }
        else {
            i++;
        }
    }

    fclose(errorFile);
}
