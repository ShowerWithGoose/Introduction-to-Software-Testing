    // 处理其他符号（如运算符，分隔符等）
    private void analyzeSymbol() {
        if (currentChar == '&' && peekChar() == '&') {
            tokens.add(new Token("AND", "&&", lineNumber));
            nextChar();  // 跳过第二个 &
            currentChar = nextChar();
        } else if (currentChar == '|' && peekChar() == '|') {
            tokens.add(new Token("OR", "||", lineNumber));
            nextChar();  // 跳过第二个 |
            currentChar = nextChar();
        } else {
            String tokenType = Keywords.getTokenType(Character.toString(currentChar));
            if (!tokenType.equals("IDENFR")) {
                tokens.add(new Token(tokenType, Character.toString(currentChar), lineNumber));
            } else {
                errors.add(lineNumber + " a");  // 非法字符 @@ [This incorrectly treats any unrecognized single character as error 'a', but in the test case, the issue is that "//" (comment) is not handled. The lexer treats each '/' as a DIV token, and then processes the rest of the line—including Chinese characters—as identifiers or invalid tokens. However, according to typical C-like language rules, "//" should start a line comment which must be skipped entirely. Since the lexer does not handle comments, it erroneously tokenizes the content after "//", leading to extra tokens like "DIV /", "DIV /", "IDENFR 输出", etc., instead of skipping the entire comment. This causes the output to include unexpected tokens rather than ignoring the comment line.]
            }
            currentChar = nextChar();  // 推进到下一个字符
        }
    }
