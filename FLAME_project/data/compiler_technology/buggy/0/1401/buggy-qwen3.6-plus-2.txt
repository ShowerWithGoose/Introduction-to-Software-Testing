package src;

public class Words {
    public String string;
    public Integer i;

    public Words(String string) {
        this.string=string;
        i=0;
    }
    public Character get(){
        if (i<string.length()) {
            Character a = string.charAt(i);
            i++;
            return a;
        }return '\n';
    }
    public void back(){
        i--;
    }
    public boolean max(){
        return i >= string.length()-1; // @@ The condition `i >= string.length()-1` causes the loop in State1 to terminate one character early. It should be `i >= string.length()` to ensure all characters, including the last one (like the final '}'), are processed.
    }
}
