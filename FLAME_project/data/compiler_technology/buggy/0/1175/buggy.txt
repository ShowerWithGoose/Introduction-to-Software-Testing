            fw.write("");
        } catch(IOException e){e.printStackTrace();}
        lexer = Lexer.getInstance();
        lexer.getSource();
        while(!lexer.isFinish()){
           // System.out.println(1);
            lexer.next();
        }
        if(lexer.isLegal()){
            try(FileWriter fw = new FileWriter("lexer.txt",true)){
                for(int i=0;i<lexer.wordDatas.size();i++){
                    fw.write(lexer.wordDatas.get(i).lexType + " " + lexer.wordDatas.get(i).token + "\n");
                }
            } catch(IOException e){
                e.printStackTrace();
            }
        }
        else{
            try(FileWriter fw = new FileWriter("error.txt",true)){
                for(int i=0;i<lexer.illegalWordDatas.size();i++){
                    fw.write(lexer.illegalWordDatas.get(i).lineNum + " " + lexer.illegalWordDatas.get(i).errorType + "\n");
                }
            } catch(IOException e){
                e.printStackTrace();
            }
        }
    }
}

package frontend;

import java.io.*;
import java.util.*;

public class Lexer {
    public static Lexer lexer;
    public static Lexer getInstance()
    {
        if (lexer == null) {
            lexer = new Lexer();
            lexer.Init();
        }
        return lexer;
    }

    public class WordData{
        public String token;
        public LexType lexType;
        public int lineNum;
        public boolean isLegal;
        public WordData (String token, LexType lexType, int lineNum,boolean isLegal){
            this.token = token;
            this.lexType = lexType;
            this.lineNum = lineNum;
        }
    }
    public List<WordData> wordDatas;
    private WordData wordData_tmp;

    public class IllegalWordData{
        public String token;
        public LexType lexType;
        public int lineNum;
        public char errorType;
        public IllegalWordData (String token, LexType lexType, int lineNum,char errorType){
            this.token = token;
            this.lexType = lexType;
            this.lineNum = lineNum;
            this.errorType = errorType;
        }
    }
    public List<IllegalWordData> illegalWordDatas;
    private IllegalWordData illegalWordData_tmp;

    private Map<String, LexType> map = new HashMap<String, LexType>();
    private String source;
    private int curPos;
    private String token;
    private LexType lexType;
    private int lineNum;
    private int number;
    private boolean isError;

    private char currentChar;

    public void getSource() throws IOException
    {
        FileReader fr = new FileReader("testfile.txt");
        source = "";
        int char_tmp;
        while((char_tmp = fr.read()) != -1){
            source += (char)char_tmp;
        }
        System.out.println(source.length());
    }
    public void next()
    {
        if(isFinish())return;
        token = "";

        getchar();
        //System.out.printf("%d %d\n",(int)currentChar,curPos);
        while(isSpace() || isNewLine() || isTab() || currentChar == '\r'){
            if(isFinish())return;
            if(isNewLine())lineNum++;
            getchar();
        }//读第一个非空字符

        //保留字or标识符
        if(isLetter() || isUnderLine()){
            while(isLetter() || isDigit() || isUnderLine()){
                token += currentChar;
                getchar();
            }
            retract();
            //保留字
            if(map.containsKey(token) && ( !token.equals("IntConst") && !token.equals("StringConst") && !token.equals("CharConst"))){
                lexType = map.get(token);
            }
            //标识符
            else lexType = map.get("Ident");
            wordData_tmp = new WordData(token,lexType,lineNum,true);
            wordDatas.add(wordData_tmp);
        }
        //数字
        else if(isDigit()){
            while(isDigit()){
                token += currentChar;
                getchar();
            }
            retract();
            number = Integer.parseInt(token);
            lexType = map.get("IntConst");
            wordData_tmp = new WordData(token,lexType,lineNum,true);
            wordDatas.add(wordData_tmp);
        }
        //字符串常量
        else if(isDuobleQuotation()){
            token += currentChar;
            getchar();
            while(!isDuobleQuotation()) {
                token += currentChar;
                getchar();

            }
            token += currentChar;
            lexType = map.get("StringConst");
            wordData_tmp = new WordData(token,lexType,lineNum,true);
            wordDatas.add(wordData_tmp);
        }
        //字符常量
        else if(isSingleQuotation()){
            token += currentChar;
            getchar();
            while(!isSingleQuotation()) {
                token += currentChar;
                getchar();

            }
            token += currentChar;
            lexType = map.get("CharConst");
            wordData_tmp = new WordData(token,lexType,lineNum,true);
            wordDatas.add(wordData_tmp);
        }
        //注释or除法
        else if(isDivide()){
            getchar();
            if(isStar()){
                do{
                    do{
                        getchar();
                    }while(!isStar());
                    do{
                        getchar();
                        if(isDivide())return;
                    }while(isStar());
                }while(!isStar());
            }
            else if(isDivide()){
                do{
                    getchar();
                }while(!isNewLine() || isFinish());
                retract();
            }
            else {
                retract();
                token += currentChar;
                lexType = map.get(token);
                wordData_tmp = new WordData(token,lexType,lineNum,true);
                wordDatas.add(wordData_tmp);
            }
        }
        //其他字符
        else{
            token += currentChar;
            if(currentChar == '<' || currentChar == '>' || currentChar == '!' || currentChar == '='){
                getchar();
                if(currentChar == '='){
                    token += currentChar;
                }
                else retract();
            }
            else if(currentChar == '|'){
                getchar();
                if(currentChar != '|'){
                    isError = true;
                    token += "|";
                    lexType = map.get(token);
                    wordData_tmp = new WordData(token,lexType,lineNum,false);
                    wordDatas.add(wordData_tmp);
                    illegalWordData_tmp = new IllegalWordData(token,lexType,lineNum,'a');
                    illegalWordDatas.add(illegalWordData_tmp);
                    return ;
                }
                token += currentChar;
            }
            else if(currentChar == '&'){
                getchar();
                if(currentChar != '&'){
                    isError = true;
                    token += "&";
                    lexType = map.get(token);
                    wordData_tmp = new WordData(token,lexType,lineNum,false);
                    wordDatas.add(wordData_tmp);
                    illegalWordData_tmp = new IllegalWordData(token,lexType,lineNum,'a');
                    illegalWordDatas.add(illegalWordData_tmp);
                    return ;
                }
                token += currentChar;
            }
            lexType = map.get(token);
            wordData_tmp = new WordData(token,lexType,lineNum,true);
            wordDatas.add(wordData_tmp);
        }


    }
    public String getToken() {return token;}
    public LexType getLexType() {return lexType;}
    public int getLineNum() {return lineNum;}
    public boolean isLegal(){return !isError;}
    public boolean isFinish(){
        if (curPos == source.length() - 1) return true;
        else return false;
    }
    public void Init(){
        illegalWordDatas = new ArrayList<IllegalWordData>();
        wordDatas = new ArrayList<WordData>();
        source = "";
        curPos = -1;
        token = "";
        lexType = null;
        lineNum = 1;
        number = 0;
        map.put("main", LexType.MAINTK);
        map.put("const", LexType.CONSTTK);
        map.put("int", LexType.INTTK);
        map.put("char", LexType.CHARTK);
        map.put("break", LexType.BREAKTK);
        map.put("continue", LexType.CONTINUETK);
        map.put("if", LexType.IFTK);
        map.put("else", LexType.ELSETK);
        map.put("!", LexType.NOT);
        map.put("&&", LexType.AND);
        map.put("||", LexType.OR);
        map.put("for", LexType.FORTK);
        map.put("getint", LexType.GETINTTK);
        map.put("getchar", LexType.GETCHARTK);
        map.put("printf", LexType.PRINTFTK);
        map.put("return", LexType.RETURNTK);
        map.put("+", LexType.PLUS);
        map.put("-", LexType.MINU);
        map.put("void", LexType.VOIDTK);
        map.put("*", LexType.MULT);
        map.put("/", LexType.DIV);
        map.put("%", LexType.MOD);
        map.put("<", LexType.LSS);
        map.put("<=", LexType.LEQ);
        map.put(">", LexType.GRE);
        map.put(">=", LexType.GEQ);
        map.put("==", LexType.EQL);
        map.put("!=", LexType.NEQ);
        map.put("=", LexType.ASSIGN);
        map.put(";", LexType.SEMICN);
        map.put(",", LexType.COMMA);
        map.put("(", LexType.LPARENT);
        map.put(")", LexType.RPARENT);
        map.put("[", LexType.LBRACK);
        map.put("]", LexType.RBRACK);
        map.put("{", LexType.LBRACE);
        map.put("}", LexType.RBRACE);
        map.put("IntConst", LexType.INTCON);
        map.put("Ident", LexType.IDENFR);
        map.put("StringConst", LexType.STRCON);
        map.put("CharConst", LexType.CHRCON);
    }

    private boolean isSpace(){return currentChar == ' ';}
    private boolean isDigit(){return Character.isDigit(currentChar);}
    private boolean isLetter(){return Character.isLetter(currentChar);}
    private boolean isNewLine(){return currentChar == '\n';}
    private boolean isTab(){return currentChar == '\t';}
    private boolean isUnderLine(){return currentChar == '_';}
    private boolean isDuobleQuotation(){return currentChar == '\"';}
    private boolean isSingleQuotation(){return currentChar == '\'';}
    private boolean isDivide(){return currentChar == '/';}
    private boolean isStar(){return currentChar == '*';}
    private void getchar(){
        if(curPos < source.length() - 1){
            currentChar = source.charAt(++curPos);
        }
        else currentChar = source.charAt(curPos);
    }
    private void retract() {
        if(curPos>0){
            currentChar = source.charAt(--curPos);
        }
        else currentChar = source.charAt(curPos);
    }




}
