    private void tokenizeCharConstant() {
        StringBuilder tokenBuilder = new StringBuilder();
        tokenBuilder.append('\''); // 添加起始引号
        curPos++;
        if (curPos < source.length() && curPos + 1 < source.length() && source.charAt(curPos + 1) == '\'') { // @@ [This condition fails to handle escape sequences like '\0', '\n', etc., which have more than one character between the quotes. For example, '\0' has two characters '\\' and '0', so source.charAt(curPos + 1) is '0', not '\'', causing the condition to be false and leading to an error being reported incorrectly.]
            tokenBuilder.append(source.charAt(curPos));
            tokenBuilder.append('\''); // 添加结束引号
            token = tokenBuilder.toString();
            lexType = LexType.CHRCON;
            curPos += 2;
        } else {
            addError("Unterminated char constant at line " + lineNum);
            lexType = LexType.ERROR;
            while (curPos < source.length() && source.charAt(curPos) != '\n') {
                curPos++;
            }
        }
    }
