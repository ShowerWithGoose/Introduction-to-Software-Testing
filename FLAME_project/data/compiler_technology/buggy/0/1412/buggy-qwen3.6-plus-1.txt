void Lexer::skip_multi_line_comment() {
    State state = State::S1; // start state
    while (pos < source.length()) {
        switch (state) {
            case State::S1:
                if (source[pos] == '*')
                    state = State::S2;
                break; // @@ Error: Does not increment line_number when encountering '\n' inside a multi-line comment, causing incorrect line numbers for subsequent tokens.
            case State::S2:
                if (source[pos] == '/')
                    state = State::S3;
                else if (source[pos] == '*')
                    state = State::S2;
                else
                    state = State::S1;
                break;    // @@ Error: Does not increment line_number when encountering '\n' inside a multi-line comment, causing incorrect line numbers for subsequent tokens.
            case State::S3:
                break;
            default:
                break;
        }
        pos++;
        if (state == State::S3) break;
    }
}

void Lexer::chrcon() {
    std::string chr = "";
    pos++;
    // 需要考虑转义字符
    if (source[pos] == '\\') {
        std::string chr1 = "";
        chr1.push_back(source[pos++]);
        chr1 = "\'" + chr1 + "\'";
        tokens.push_back(Token{chr1, Token::TokenType::CHRCON, line_number}); // @@ Error: Incorrectly generates a token for the escape character prefix, splitting one character constant into two tokens.
        chr.push_back(source[pos]);
        pos +=2;
        tokens.push_back(Token{chr, Token::TokenType::CHRCON, line_number}); // @@ Error: Incorrectly generates a second token for the rest of the escape sequence. Should consume the entire escape sequence and closing quote to produce a single CHRCON token.
    } else {
        chr.push_back(source[pos]);
        pos +=2;
        chr = "\'" + chr + "\'";
        tokens.push_back(Token{chr, Token::TokenType::CHRCON, line_number});
    }
}
