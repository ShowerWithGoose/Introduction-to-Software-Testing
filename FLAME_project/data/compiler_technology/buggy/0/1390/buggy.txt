package frontend;

import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

public class Lexer {

    private final ArrayList<Token> tokens = new ArrayList<>();

    private HashMap<String, Type> keywords = new HashMap<>();

    private ArrayList<String> errorLog = new ArrayList<>();

    private int column = 1;

    private int pos = 0;

    private String input;

    public Lexer(String input) {
        this.input = input;
        this.input = removeAnnotation();
        addKeyword();
        lexTokens();
    }

    private void skipBlank() {
        while (pos < input.length() && (input.charAt(pos) == ' ' || input.charAt(pos) == '\t' || input.charAt(pos) == '\n')) {
            if (input.charAt(pos) == '\n') {
                column++;
            }
            pos++;
        }
    }

    private String removeAnnotation() {
        // 正则表达式：匹配单行注释和多行注释
        String regex = "(//.*?$)|(/\\*[^*]*\\*+(?:[^/*][^*]*\\*+)*/)";

        // 使用 Pattern.DOTALL 使 '.' 匹配换行符，Pattern.MULTILINE 使 '^' 和 '$' 匹配每一行的开始和结束
        Pattern pattern = Pattern.compile(regex, Pattern.DOTALL | Pattern.MULTILINE);
        Matcher matcher = pattern.matcher(input);

        // 替换所有匹配到的注释，单行注释直接移除，多行注释替换为空行（保留换行符）
        StringBuffer result = new StringBuffer();
        while (matcher.find()) {
            if (matcher.group(1) != null) {
                // 这是单行注释，移除它但保留换行符
                matcher.appendReplacement(result, "");
            } else if (matcher.group(2) != null) {
                // 这是多行注释，保留注释中的换行符
                String multiLineComment = matcher.group(2);
                String newLines = multiLineComment.replaceAll("[^\n]", ""); // 只保留换行符
                matcher.appendReplacement(result, newLines);
            }
        }
        matcher.appendTail(result);
        return result.toString();
    }

    private void addKeyword() {
        /*try (BufferedReader br = new BufferedReader(new FileReader("keyword.txt"))) {
            String line;
            while ((line = br.readLine()) != null) {
                // 读取关键字
                String keyword = line.trim();

                // 读取枚举类型
                if ((line = br.readLine()) != null) {
                    String typeString = line.trim();
                    Type type = Type.valueOf(typeString);  // 将字符串转换为枚举类型

                    // 将关键字和对应的枚举值存入 HashMap
                    keywords.put(keyword, type);
                } else {
                    throw new IOException("Unexpected end of file: each keyword must have a corresponding frontend.Type.");
                }
            }
        } catch (IOException | IllegalArgumentException e) {
            e.printStackTrace();
        }*/
        keywords.put("main", Type.MAINTK);
        keywords.put("const", Type.CONSTTK);
        keywords.put("int", Type.INTTK);
        keywords.put("char", Type.CHARTK);
        keywords.put("break", Type.BREAKTK);
        keywords.put("continue", Type.CONTINUETK);
        keywords.put("if", Type.IFTK);
        keywords.put("else", Type.ELSETK);
        keywords.put("for", Type.FORTK);
        keywords.put("getint", Type.GETINTTK);
        keywords.put("getchar", Type.GETCHARTK);
        keywords.put("printf", Type.PRINTFTK);
        keywords.put("return", Type.RETURNTK);
        keywords.put("void", Type.VOIDTK);
        keywords.put("+", Type.PLUS);
        keywords.put("-", Type.MINU);
        keywords.put("*", Type.MULT);
        keywords.put("/", Type.DIV);
        keywords.put("%", Type.MOD);
        keywords.put(";", Type.SEMICN);
        keywords.put(",", Type.COMMA);
        keywords.put("(", Type.LPARENT);
        keywords.put(")", Type.RPARENT);
        keywords.put("{", Type.LBRACE);
        keywords.put("}", Type.RBRACE);
        keywords.put("[", Type.LBRACK);
        keywords.put("]", Type.RBRACK);
    }

    private Type checkKeyword(String keyword) {
        if (keywords.containsKey(keyword)) {
            return keywords.get(keyword);
        }
        return Type.IDENFR;
    }

    private void lexTokens() {
        while (pos < input.length()) {
            StringBuilder tmp = new StringBuilder();
            skipBlank();
            if (pos >= input.length()) {
                return;
            }
            if (Character.isLetter(input.charAt(pos)) || input.charAt(pos) == '_') {
                while (pos < input.length() && (Character.isLetter(input.charAt(pos)) || Character.isDigit(input.charAt(pos)) || input.charAt(pos) == '_')) {
                    tmp.append(input.charAt(pos));
                    pos++;
                }
                // 判断保留字 or 标识符
                tokens.add(new Token(checkKeyword(tmp.toString()), tmp.toString()));
            } else if (Character.isDigit(input.charAt(pos))) {
                while (pos < input.length() && Character.isDigit(input.charAt(pos))) {
                    tmp.append(input.charAt(pos));
                    pos++;
                }
                tokens.add(new Token(Type.INTCON, tmp.toString()));
            } else if (input.charAt(pos) == '\"') {
                tmp.append('\"');
                pos++;
                while (pos < input.length() && input.charAt(pos) != '\"') {
                    tmp.append(input.charAt(pos));
                    pos++;
                }
                tmp.append('\"');
                pos++;
                tokens.add(new Token(Type.STRCON, tmp.toString()));
            } else if (input.charAt(pos) == '\'') {
                tmp.append('\'');
                pos++;
                /*while (pos < input.length() && input.charAt(pos) != '\'') {
                    tmp.append(input.charAt(pos));
                    pos++;
                }*/
                if (input.charAt(pos) == '\\') {
                    pos += 2;
                    tmp.append('\\').append(input.charAt(pos - 1));
                } else {
                    tmp.append(input.charAt(pos));
                    pos++;
                }
                tmp.append('\'');
                pos++;
                tokens.add(new Token(Type.CHRCON, tmp.toString()));
            } else if (input.charAt(pos) == '!') {
                pos++;
                if (input.charAt(pos) == '=') {
                    pos++;
                    tmp.append("!=");
                    tokens.add(new Token(Type.NEQ, tmp.toString()));
                } else {
                    tmp.append('!');
                    tokens.add(new Token(Type.NOT, tmp.toString()));
                }
            } else if (input.charAt(pos) == '&') {
                pos++;
                if (input.charAt(pos) == '&') {
                    pos++;
                    tmp.append("&&");
                    tokens.add(new Token(Type.AND, tmp.toString()));
                } else {
                    errorLog.add(column + " a");
                    tmp.append('&');
                    tokens.add(new Token(Type.AND, tmp.toString()));
                }
            } else if (input.charAt(pos) == '|') {
                pos++;
                if (input.charAt(pos) == '|') {
                    pos++;
                    tmp.append("||");
                    tokens.add(new Token(Type.OR, tmp.toString()));
                } else {
                    errorLog.add(column + " a");
                    tmp.append('|');
                    tokens.add(new Token(Type.OR, tmp.toString()));
                }
            } else if (input.charAt(pos) == '<') {
                pos++;
                if (input.charAt(pos) == '=') {
                    pos++;
                    tmp.append("<=");
                    tokens.add(new Token(Type.LEQ, tmp.toString()));
                } else {
                    tmp.append('<');
                    tokens.add(new Token(Type.LSS, tmp.toString()));
                }
            } else if (input.charAt(pos) == '>') {
                pos++;
                if (input.charAt(pos) == '=') {
                    pos++;
                    tmp.append(">=");
                    tokens.add(new Token(Type.GEQ, tmp.toString()));
                } else {
                    tmp.append('>');
                    tokens.add(new Token(Type.GRE, tmp.toString()));
                }
            } else if (input.charAt(pos) == '=') {
                pos++;
                if (input.charAt(pos) == '=') {
                    pos++;
                    tmp.append("==");
                    tokens.add(new Token(Type.EQL, tmp.toString()));
                } else {
                    tmp.append('=');
                    tokens.add(new Token(Type.ASSIGN, tmp.toString()));
                }
            } else {
                tmp.append(input.charAt(pos));
                tokens.add(new Token(checkKeyword(tmp.toString()), tmp.toString()));
                pos++;
            }
        }
    }

    public String toString() {
        StringBuilder sb = new StringBuilder();
        if (errorLog.size() > 0) {
            for (String error : errorLog) {
                sb.append(error).append('\n');
            }
        } else {
            for (Token token : tokens) {
                sb.append(token.toString()).append('\n');
            }
        }
        return sb.toString();
    }
}

package frontend;

public class Token {
    private Type type;
    private String value;

    public Token(Type type, String value) {
        this.type = type;
        this.value = value;
    }

    public Type getType() {
        return type;
    }

    public String getValue() {
        return value;
    }

    public String toString() {
        return type.toString() + " " + value;
    }
}

package frontend;

public enum Type {
    IDENFR, INTCON, STRCON, CHRCON,
    MAINTK, CONSTTK, INTTK, CHARTK, BREAKTK, CONTINUETK, IFTK, ELSETK, FORTK, GETINTTK, GETCHARTK, PRINTFTK, RETURNTK, VOIDTK,
    NOT, AND, OR, PLUS, MINU, MULT, DIV, MOD, LSS, LEQ, GRE, GEQ, EQL, NEQ, ASSIGN, SEMICN, COMMA, LPARENT, RPARENT, LBRACK, RBRACK, LBRACE, RBRACE,

    // 字母开头
    // IDENFR, MAINTK, CONSTTK, INTTK, CHARTK, BREAKTK, CONTINUETK, IFTK, ELSETK, FORTK, GETINTTK, GETCHARTK, PRINTFTK, RETURNTK, VOIDTK,

    // 数字开头
    // INTCON

    // 符号开头（单字符识别）
    // STRCON("), CHRCON('), PLUS(+), MINU(-), MULT(*), DIV(/),
    // MOD(%), SEMICN(;), COMMA(,),
    // LPARENT('('), RPARENT(')'), LBRACK('['), RBRACK(']'),
    // LBRACE('{'), RBRACE('}')

    // 符号开头（多字符识别）
    // NOT(!), NEQ(!=)
    // AND(&&)
    // OR(||)
    // LSS(<), LEQ(<=)
    // GRE(>), GEQ(>=)
    // ASSIGN(=), EQL(==)
}
