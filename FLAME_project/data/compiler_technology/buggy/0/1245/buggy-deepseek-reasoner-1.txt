    public static int read() throws IOException {
        pos++; // @@ [Error: Incrementing pos before reading the first character of a line causes the first character to be skipped. This leads to incorrect token recognition and potential failure to detect errors like single '&' or '|'.]
        if(storage==null || pos >= storage.length()){
            readLine();
        }
        if(storage==null){
            //读到文件结束
            pos = 0;
            lineno = 0;
            return -1;
        }
        if(storage.isEmpty()){
            return read();
        }
        p = storage.charAt(pos);
        if(checkSign(p)){
            return 1;
        }
        if(checkInt(p)){
            return 1;
        }
        if(checkString(p)){
            return 1;
        }
        if(checkChar(p)){
            return 1;
        }
        if(checkIdent(p)){
            return 1;
        }
        return read();
    }
