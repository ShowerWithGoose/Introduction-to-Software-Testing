package mainSys;

public class Error {
    char type;
    int line;
    // toString
    public String toString() {
        return type + " " + line; // @@ The output format requires "line number error category code", but this returns "error category code line number". It should be `return line + " " + type;`.
    }

    public Error(char type, int line) {
        this.type = type;
        this.line = line;
    }
}
