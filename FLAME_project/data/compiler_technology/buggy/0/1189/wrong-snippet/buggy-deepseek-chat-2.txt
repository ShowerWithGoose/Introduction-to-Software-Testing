void Lexer::getChar() {
    if (curPos < fileText.size()) {
        this->LastChar = this->Char;
        this->Char = fileText[curPos];
        this->curPos++;
    } else {
        this->Char = '\0';
    }
}
...
void Lexer::retract() {
    this->curPos--;
    this->Char = LastChar;
}
