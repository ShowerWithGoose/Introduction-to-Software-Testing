    private void exeg() throws IOException {
        int line = lineNumber;
        int column = columnNumber;

        advance();
        if (currentChar == '/') {   //双行注释
            while (currentChar != '\n') {   //跳过当前行 @@ The loop condition does not check for EOF (-1). If the file ends with a line comment without a trailing newline, currentChar becomes -1, which is not equal to '\n', causing an infinite loop as advance() keeps returning -1.
                advance();
            }
            advance();
        } else if (currentChar == '*') {    //多行注释
            while (true) {
                if (currentChar == '*') {
                    advance();
                    if (currentChar == '/') {   //结束多行注释
                        break;
                    }
                } else {
                    advance();
                }
            }
            advance();
        } else {    //只是单独的除号
            words.add(new Word(line, column, "/", Type.DIV));
        }
    }
