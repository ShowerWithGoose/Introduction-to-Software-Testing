    private void identChars() throws IOException {
        StringBuilder stringBuilder = new StringBuilder();
        int line = lineNumber;
        int column = columnNumber;

        if (currentChar == '"') {   //字符串常量
            do {
                stringBuilder.append((char) currentChar);
                advance();
            } while (currentChar != '"'); // @@ If the string is unterminated and reaches EOF, currentChar becomes -1. Since -1 != '"', the loop continues indefinitely, causing Time Limit Exceeded.
            stringBuilder.append((char) currentChar);
            words.add(new Word(line, column, stringBuilder.toString(), Type.STRCON));
            advance();
        } else if (currentChar == '\'') {   //字符常量
            do {
                stringBuilder.append((char) currentChar);
                advance();
            } while (currentChar != '\''); // @@ If the char literal is unterminated and reaches EOF, currentChar becomes -1. Since -1 != ''', the loop continues indefinitely, causing Time Limit Exceeded.
            stringBuilder.append((char) currentChar);
            words.add(new Word(line, column, stringBuilder.toString(), Type.CHRCON));
            advance();
        } else {
            System.out.printf("Error : Not char or String in line %d column %d%n", line, column);
        }
    }
