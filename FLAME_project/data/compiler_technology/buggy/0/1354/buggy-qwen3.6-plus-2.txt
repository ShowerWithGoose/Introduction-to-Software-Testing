    private int dealWithDoubleOperator(BufferedInputStream bis, int c) throws IOException {
        StringBuilder str = new StringBuilder();
        str.append((char) c);

        // 读取下一个字符来判断是否是双字符操作符
        int nextChar = getChar(bis);
        String operator = str.toString();

        Token tk = null;

        switch (operator) {
            case "&":
                if (nextChar == '&') {
                    tk = new Token(AND, "&&", lineIndex);
                } else {
                    tk = new Token(AND, operator, lineIndex); // @@ The token type for a standalone '&' should not be AND (which represents '&&'). According to the problem, this is an error (type 'a'), so no valid token should be appended to the token list, or at least not with type AND. The current code appends a token with type AND and value "&", which causes the output "25 &" instead of detecting the error correctly as "25 a".
                    tokens.append(tk);
                    dealWithError('&');
                    return nextChar;
                }
                break;
            case "|":
                if (nextChar == '|') {
                    tk = new Token(OR, "||", lineIndex);
                } else {
                    tk = new Token(OR, operator, lineIndex); // @@ Similarly, the token type for a standalone '|' should not be OR (which represents '||'). This is an error (type 'a'). The code appends a token with type OR and value "|", contributing to the incorrect output.
                    dealWithError('|');
                    tokens.append(tk);
                    return nextChar;
                }
