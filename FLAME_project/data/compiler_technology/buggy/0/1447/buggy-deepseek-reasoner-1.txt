#include <exception>

class LexicalError : public std::exception {
private:
    int m_nLine;
    int m_nColumn;
    std::string* m_strLexme;

public:
    LexicalError(int line, int column, std::string* lexme) {
        this->m_nLine = line;
        this->m_nColumn = column;
        this->m_strLexme = lexme; // @@ Raw pointer assignment: if the exception object is ever copied, the pointer will be shared, leading to double deletion of the string.
    }

    int GetLine() {
        return m_nLine;
    }

    int GetColumn() {
        return m_nColumn;
    }

    std::string* GetLexme() {
        return m_strLexme;
    }
};
