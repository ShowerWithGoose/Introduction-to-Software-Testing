    public LexType nextChar() {
        StringBuilder sb = new StringBuilder();
        sb.append('\'');
        getChar();
        sb.append(curChar);
        if (curChar == '\\') { // Error: The lexer does not handle escape sequences in character constants correctly. It only reads one character after the opening quote.
            getChar();
            sb.append(curChar);
        }
        getChar();
        sb.append('\'');
        getChar();
        return new LexType(LexType.TokenType.CHRCON, sb.toString(), line);
    }
