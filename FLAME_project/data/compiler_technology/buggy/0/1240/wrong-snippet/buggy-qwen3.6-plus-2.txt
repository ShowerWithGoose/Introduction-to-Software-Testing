    public LexType nextChar() {
        StringBuilder sb = new StringBuilder();
        sb.append('\'');
        getChar();
        sb.append(curChar);
        if (curChar == '\\') {
            getChar();
            sb.append(curChar);
        }
        getChar();
        sb.append('\'');
        getChar();
        return new LexType(LexType.TokenType.CHRCON, sb.toString(), line);
    }
