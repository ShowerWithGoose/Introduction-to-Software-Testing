    private Token parseIdent() throws IOException {
        StringBuilder buf = new StringBuilder();
        buf.append(ch);
        read();
        while (ch == '_' || Character.isLetter(ch) || Character.isDigit(ch)) {
            buf.append(ch);
            read();
        }
        unread();
        String ident = buf.toString();
        if (ident == "main") { // @@ [Using '==' to compare strings instead of .equals() leads to incorrect keyword recognition; this causes keywords like 'int' and 'printf' to be treated as IDENFR]
            return newToken(TokenType.MAINTK, ident);
        } else if (ident == "const") { // @@ [Same issue: string comparison with '==' is unreliable for content equality in Java]
            return newToken(TokenType.CONSTTK, ident);
        } else if (ident == "int") { // @@ [This condition fails due to '==' comparison, so 'int' is incorrectly categorized as IDENFR]
            return newToken(TokenType.INTTK, ident);
        } else if (ident == "char") { // @@ [Same problem: should use .equals()]
            return newToken(TokenType.CHARTK, ident);
        } else if (ident == "break") { // @@ [Same problem]
            return newToken(TokenType.BREAKTK, ident);
        } else if (ident == "continue") { // @@ [Same problem]
            return newToken(TokenType.CONTINUETK, ident);
        } else if (ident == "if") { // @@ [Same problem]
            return newToken(TokenType.IFTK, ident);
        } else if (ident == "else") { // @@ [Same problem]
            return newToken(TokenType.ELSETK, ident);
        } else if (ident == "for") { // @@ [Same problem]
            return newToken(TokenType.FORTK, ident);
        } else if (ident == "getint") { // @@ [Same problem]
            return newToken(TokenType.GETINTTK, ident);
        } else if (ident == "getchar") { // @@ [Same problem]
            return newToken(TokenType.GETCHARTK, ident);
        } else if (ident == "printf") { // @@ [This causes 'printf' to be output as IDENFR instead of PRINTFTK]
            return newToken(TokenType.PRINTFTK, ident);
        } else if (ident == "return") { // @@ [Same problem: 'return' becomes IDENFR instead of RETURNTK]
            return newToken(TokenType.RETURNTK, ident);
        } else if (ident == "void") { // @@ [Same problem]
            return newToken(TokenType.VOIDTK, ident);
        } else {
            return newToken(TokenType.IDENFR, ident);
        }
    }
