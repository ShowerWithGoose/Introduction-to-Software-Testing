    public void analyze(String lexerOutputFile, String errorOutputFile) {
        try {
            PrintWriter lexerOutput = new PrintWriter(new FileWriter(lexerOutputFile));
            PrintWriter errorOutput = new PrintWriter(new FileWriter(errorOutputFile));

            for(int lineNum = 0; lineNum < lines.size(); lineNum++){ // @@ [The loop processes each line independently, but multi-line comments span multiple lines. When a multi-line comment starts on one line and ends on another, the content on subsequent lines is not skipped, causing it to be incorrectly tokenized.]
                String line = lines.get(lineNum);
                List<Token> tokens = Token.tokenize(line); // @@ [The tokenize method only handles multi-line comments within a single line. It does not track whether we are inside a multi-line comment across lines, so when the comment continues on the next line, the content is processed as normal tokens.]
                for(Token token: tokens) {
                    if(token.getType() != TokenType.ERROR){
                        lexerOutput.println(token.getType() + " " + token.getValue());
                    } else {
                        errorOutput.println((lineNum + 1) + " a");
                    }
                }
            }

            lexerOutput.close();
            errorOutput.close();
        } catch (IOException e) {
            throw new RuntimeException(e);
        }
    }
