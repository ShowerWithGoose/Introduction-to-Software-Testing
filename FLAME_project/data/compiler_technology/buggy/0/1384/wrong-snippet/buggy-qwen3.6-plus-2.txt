import frontend.Lexer.Token;
import frontend.Lexer.TokenLexer;
import frontend.SrcFileLexer;

import java.io.FileInputStream;

import java.io.FileOutputStream;
import java.io.IOException;
import java.io.InputStream;
import java.io.OutputStream;
import java.util.Map;

public class Compiler {
    public static void main(String args[]){
        try {
            InputStream inputStream=new FileInputStream("testfile.txt");
            SrcFileLexer srcFileLexer=new SrcFileLexer(inputStream);
            srcFileLexer.readLines();
            TokenLexer tokenLexer=new TokenLexer(srcFileLexer);
            tokenLexer.beginTokenLexer();
            //"lexer.txt";
            OutputStream outputStream=new FileOutputStream("lexer.txt");
            if(!tokenLexer.getErrorMap().isEmpty()){
                for (Map.Entry<Integer, String> entry : tokenLexer.getErrorMap().entrySet()) {
                    Integer key = entry.getKey();
                    String value = entry.getValue();
                    try {

                    } catch (IOException e){
                        throw new RuntimeException(e);
                    }
                }
            }
            else{
                try {
                    for (Token token: tokenLexer.getTokenList()) {
                        outputStream.write((token.getType()+" " +token.getStringContent()+"\n").getBytes());
                    }
                } catch (IOException e) {
                    throw new RuntimeException(e);
                }
            }
        } catch (FileNotFoundException e) {
