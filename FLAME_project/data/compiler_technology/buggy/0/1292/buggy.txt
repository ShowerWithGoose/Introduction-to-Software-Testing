            ArrayList<String> arr = new ArrayList<>();
            String line = null;
            InputStream inStr = new FileInputStream("testfile.txt");
            BufferedReader bufferedReader = new BufferedReader(new InputStreamReader(inStr));
            while (true) {
                line = bufferedReader.readLine();
                if (line == null) {
                    break;
                }
                arr.add(line);
            }
            TokenLexer tokenLexer = new TokenLexer(arr);
            tokenLexer.tokenScan();
        } catch (FileNotFoundException e) {
            throw new RuntimeException(e);
        } catch (IOException e) {
            throw new RuntimeException(e);
        }
    }
}

package frontend;

import frontend.LexType;

import java.io.FileWriter;
import java.io.IOException;

public class Token {
    private String token;
    private int lineNum;
    private LexType type;

    public Token(String token, int lineNum, LexType type) {
        this.token = token;
        this.lineNum = lineNum;
        this.type = type;
    }

    public void printOut() {
        try {
            FileWriter fileWriter = new FileWriter("lexer.txt", true);
            fileWriter.write(type + " " + token + "\n");
            fileWriter.close();
        } catch (IOException e) {
            throw new RuntimeException(e);
        }
    }
}

package frontend;

import frontend.LexType;
import frontend.Token;

import java.io.FileWriter;
import java.io.IOException;
import java.util.ArrayList;

public class TokenLexer {
    private ArrayList<String> arr = new ArrayList<>();         //储存输入的字符串组
    private int lineNum;
    private State state;
    private int curPos;
    private char cur;
    private StringBuilder sb = null;

    public TokenLexer(ArrayList<String> arr) {
        this.arr = arr;
        this.lineNum = 0;
        this.curPos = 0;
        this.state = State.START;
    }

    enum State {
        START, INTCON, STRCON, CHRCON, OPERATOR, NOTE, NOTES, SPACE, END,
    }

    public void peek() {
        while (lineNum < arr.size() && curPos >= arr.get(lineNum).length()) {
            curPos = 0;
            lineNum += 1;
        }
        if (lineNum >= arr.size()) {
            state = State.END;
            return;
        }
        //System.out.println(curPos);
        cur = arr.get(lineNum).charAt(curPos);
        curPos += 1;
    }

    public void tokenScan() {
        while (lineNum < arr.size()) {
            if (state == State.END) {
                break;
            }
            parse();
        }
    }

    public void parse() {
        switch (state) {
            case START : {
                sb = new StringBuilder();
                peek();
                if (lineNum >= arr.size()) {
                    break;
                } else if (Character.isWhitespace(cur)) {
                    state = State.SPACE;
                    break;
                } else if (Character.isAlphabetic(cur) || cur == '_') {
                    sb.append(cur);
                    while (curPos < arr.get(lineNum).length() &&
                            (Character.isAlphabetic(arr.get(lineNum).charAt(curPos))
                                    || Character.isDigit(arr.get(lineNum).charAt(curPos))
                                    || arr.get(lineNum).charAt(curPos) == '_')) {
                        peek();
                        sb.append(cur);
                    }
                    reserve();
                    break;
                } else if (Character.isDigit(cur)) {
                    sb.append(cur);
                    while (curPos < arr.get(lineNum).length() && Character.isDigit(arr.get(lineNum).charAt(curPos))) {
                        peek();
                        sb.append(cur);
                    }
                    state = State.INTCON;
                    break;
                } else if (cur == '"') {
                    sb.append(cur);
                    while (curPos < arr.get(lineNum).length() && arr.get(lineNum).charAt(curPos) != '"') {
                        peek();
                        sb.append(cur);
                    }
                    peek();
                    sb.append(cur);
                    state = State.STRCON;
                    break;
                } else if (cur == '\'') {
                    sb.append(cur);
                    while (curPos < arr.get(lineNum).length() && arr.get(lineNum).charAt(curPos) != '\'') {
                        peek();
                        sb.append(cur);
                    }
                    peek();
                    sb.append(cur);
                    state = State.CHRCON;
                    break;
                } else if (cur == '/') {
                    if (curPos < arr.get(lineNum).length() && arr.get(lineNum).charAt(curPos) == '/') {
                        peek();
                        state = State.NOTE;
                        break;
                    } else if (curPos < arr.get(lineNum).length() && arr.get(lineNum).charAt(curPos) == '*') {
                        peek();
                        state = State.NOTES;
                        break;
                    } else {
                        Token token = new Token("/", lineNum, LexType.DIV);
                        token.printOut();
                        state = State.START;
                        break;
                    }
                } else {
                    sb.append(cur);
                    if (curPos < arr.get(lineNum).length() && !Character.isDigit(arr.get(lineNum).charAt(curPos))
                            && !Character.isAlphabetic(arr.get(lineNum).charAt(curPos))
                            && arr.get(lineNum).charAt(curPos) != '_' && cur != '(' && cur != '[' && cur != '{'
                            && cur != ')' && cur != '}' && cur != ']' && cur != ';' && cur != ','
                            && arr.get(lineNum).charAt(curPos) != '(' && arr.get(lineNum).charAt(curPos) != '['
                            && arr.get(lineNum).charAt(curPos) != '+' && arr.get(lineNum).charAt(curPos) != '-'
                            && arr.get(lineNum).charAt(curPos) != '{' && !Character.isWhitespace(arr.get(lineNum).charAt(curPos))) {
                        peek();
                        sb.append(cur);
                    }
                    state = State.OPERATOR;
                    //System.out.println(sb.toString());
                    break;
                }
            }
            case INTCON: {
                Token token = new Token(sb.toString(), lineNum, LexType.INTCON);
                token.printOut();
                state = State.START;
                break;
            }
            case STRCON: {
                Token token = new Token(sb.toString(), lineNum, LexType.STRCON);
                token.printOut();
                state = State.START;
                break;
            }
            case CHRCON: {
                Token token = new Token(sb.toString(), lineNum, LexType.CHRCON);
                token.printOut();
                state = State.START;
                break;
            }
            case OPERATOR: {
                operate();
                break;
            }
            case NOTE: {
                lineNum += 1;
                curPos = 0;
                state = State.START;
                break;
            }
            case NOTES: {
                while (true) {
                    if (curPos < arr.get(lineNum).length() && cur == '*' && arr.get(lineNum).charAt(curPos) == '/') {
                        state = State.START;
                        peek();
                        break;
                    }
                    peek();
                }
                break;
            }
            case SPACE: {
                while (curPos < arr.get(lineNum).length() && Character.isWhitespace(arr.get(lineNum).charAt(curPos))) {
                    peek();
                }
                state = State.START;
                break;
            }
            case END: break;
        }
    }

    public void reserve() {
        String str = sb.toString();
        Token token = null;
        switch (str) {
            case "main": {
                token = new Token("main", lineNum, LexType.MAINTK);
                break;
            }
            case "const": {
                token = new Token("const", lineNum, LexType.CONSTTK);
                break;
            }
            case "int": {
                token = new Token("int", lineNum, LexType.INTTK);
                break;
            }
            case "char": {
                token = new Token("char", lineNum, LexType.CHARTK);
                break;
            }
            case "break": {
                token = new Token("break", lineNum, LexType.BREAKTK);
                break;
            }
            case "continue": {
                token = new Token("continue", lineNum, LexType.CONTINUETK);
                break;
            }
            case "if": {
                token = new Token("if", lineNum, LexType.IFTK);
                break;
            }
            case "else": {
                token = new Token("else", lineNum, LexType.ELSETK);
                break;
            }
            case "for": {
                token = new Token("for", lineNum, LexType.FORTK);
                break;
            }
            case "getint": {
                token = new Token("getint", lineNum, LexType.GETINTTK);
                break;
            }
            case "getchar": {
                token = new Token("getchar", lineNum, LexType.GETCHARTK);
                break;
            }
            case "printf": {
                token = new Token("printf", lineNum, LexType.PRINTFTK);
                break;
            }
            case "return": {
                token = new Token("return", lineNum, LexType.RETURNTK);
                break;
            }
            case "void": {
                token = new Token("void", lineNum, LexType.VOIDTK);
                break;
            }
            default: {
                token = new Token(str, lineNum, LexType.IDENFR);
                break;
            }
        }
        token.printOut();
        state = State.START;
    }

    public void operate() {
        String str = sb.toString();
        Token token = null;
        switch (str) {
            case "!" : {
                token = new Token(str, lineNum, LexType.NOT);
                break;
            }
            case "&&" : {
                token = new Token(str, lineNum, LexType.AND);
                break;
            }
            case "||" : {
                token = new Token(str, lineNum, LexType.OR);
                break;
            }
            case "+" : {
                token = new Token(str, lineNum, LexType.PLUS);
                break;
            }
            case "-" : {
                token = new Token(str, lineNum, LexType.MINU);
                break;
            }
            case "*" : {
                token = new Token(str, lineNum, LexType.MULT);
                break;
            }
            case "%" : {
                token = new Token(str, lineNum, LexType.MOD);
                break;
            }
            case "<=" : {
                token = new Token(str, lineNum, LexType.LEQ);
                break;
            }
            case "<" : {
                token = new Token(str, lineNum, LexType.LSS);
                break;
            }
            case ">=" : {
                token = new Token(str, lineNum, LexType.GEQ);
                break;
            }
            case ">" : {
                token = new Token(str, lineNum, LexType.GRE);
                break;
            }
            case "==" : {
                token = new Token(str, lineNum, LexType.EQL);
                break;
            }
            case "!=" : {
                token = new Token(str, lineNum, LexType.NEQ);
                break;
            }
            case "=" : {
                token = new Token(str, lineNum, LexType.ASSIGN);
                break;
            }
            case ";" : {
                token = new Token(str, lineNum, LexType.SEMICN);
                break;
            }
            case "," : {
                token = new Token(str, lineNum, LexType.COMMA);
                break;
            }
            case "(" : {
                token = new Token(str, lineNum, LexType.LPARENT);
                break;
            }
            case ")" : {
                token = new Token(str, lineNum, LexType.RPARENT);
                break;
            }
            case "[" : {
                token = new Token(str, lineNum, LexType.LBRACK);
                break;
            }
            case "]" : {
                token = new Token(str, lineNum, LexType.RBRACK);
                break;
            }
            case "{" : {
                token = new Token(str, lineNum, LexType.LBRACE);
                break;
            }
            case "}" : {
                token = new Token(str, lineNum, LexType.RBRACE);
                break;
            }
            case "&" : {
                try {
                    FileWriter fileWriter = new FileWriter("error.txt", true);
                    fileWriter.write((lineNum + 1) + " a\n");
                    fileWriter.close();
                } catch (IOException e) {
                    throw new RuntimeException(e);
                }
                break;
            }
            case "|" : {
                try {
                    FileWriter fileWriter = new FileWriter("error.txt", true);
                    fileWriter.write((lineNum + 1) + " a\n");
                    fileWriter.close();
                } catch (IOException e) {
                    throw new RuntimeException(e);
                }
                break;
            }
        }
        if (!str.equals("&") && !str.equals("|")) {
            token.printOut();
        }
        state = State.START;
    }
}
