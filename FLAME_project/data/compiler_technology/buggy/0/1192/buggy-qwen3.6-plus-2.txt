bool isDigit(const char *p) { return *p >= '0' && *p <= '9'; }

bool isNonDigit(const char *p) { return *p == '_' || (*p>='A' && *p<='Z') || (*p>='a' && *p<='z'); }

bool isSpace(const char *p) { return *p == ' '; }

bool isTab(const char *p) { return *p == '\t'; }

bool isNewline(const char *p) {return *p == '\n' || (*p == '\r' && *(p+1) == '\n'); } // @@ Potential out-of-bounds access if p points to the last character '\r' of the buffer. Should check for null terminator.

static char token_buf[global_lexerBufferMaxSize], *token;
static CODE_TOKEN token_code;
static int token_num;
static unsigned lineNumber;

void catToken(char **p) {
    *token = **p;
    *(token + 1) = 0;
    ++token;
    ++(*p);
}

void clearToken() {
    token = token_buf;
    *token = 0;
}

bool bufEqual(const char comp[]) {
    const char *p = comp, *q = token_buf;
    while (*p && *p == *q) ++p,++q;
    return *p == *q;
}

int buf2Num() {
    int ret = 0;
    for (const char *p = token_buf; p != token; ++p)
        ret = ret * 10 + (*p - '0');
    return ret;
}

CODE_TOKEN reserve() {
    if (bufEqual("main")) return TOKEN_MAINTK;
    else if (bufEqual("const")) return TOKEN_CONSTTK;
    else if (bufEqual("int")) return TOKEN_INTTK;
    else if (bufEqual("char")) return TOKEN_CHARTK;
    else if (bufEqual("break")) return TOKEN_BREAKTK;
    else if (bufEqual("continue")) return TOKEN_CONTINUETK;
    else if (bufEqual("if")) return TOKEN_IFTK;
    else if (bufEqual("else")) return TOKEN_ELSETK;
    else if (bufEqual("for")) return TOKEN_FORTK;
    else if (bufEqual("getint")) return TOKEN_GETINTTK;
    else if (bufEqual("getchar")) return TOKEN_GETCHARTK;
    else if (bufEqual("printf")) return TOKEN_PRINTFTK;
    else if (bufEqual("return")) return TOKEN_RETURNTK;
    else if (bufEqual("void")) return TOKEN_VOIDTK;
    else return TOKEN_UNIDENTIFIED;
}

int getToken(char *p, char **begin, char **end) {
    clearToken();
    while (isSpace(p) || isTab(p) || isNewline(p)) {
        if (isNewline(p)) ++lineNumber;
        ++p;
    }
    *begin = *end = p;
    if (!(*p) || *p == EOF) return LEXER_EOF;
    if (isNonDigit(p)) {
        while (isNonDigit(p) || isDigit(p)) catToken(&p);
        CODE_TOKEN resultValue = reserve();
        if (resultValue == TOKEN_UNIDENTIFIED)
            token_code = TOKEN_IDENFR;
        else
            token_code = resultValue;
    }
    else if (isDigit(p)) {
        while (isDigit(p)) catToken(&p);
        token_num = buf2Num();
        token_code = TOKEN_INTCON;
    }
    else {
        char cur = *p;
        catToken(&p);
        switch (cur) {
            case '"' :
                while (*p != '"') {
                    if (*p == '\\') catToken(&p);
                    catToken(&p);
                }
                catToken(&p);
                token_code = TOKEN_STRCON; break;
            case '\'' :
                if (*p == '\\') catToken(&p);
                catToken(&p); //content character
                if (*p != '\'') {
                    *end = p;
                    return -1; // Error code : not single character
                }
                catToken(&p);
                token_code = TOKEN_CHRCON; break;
            case '!' :
                if (*p == '=') {
                    catToken(&p);
                    token_code = TOKEN_NEQ;
                } else token_code = TOKEN_NOT;
                break;
            case '&' :
                if (*p != '&') {
                    *end = p;
                    return LEXER_ERROR_A;
                }
                catToken(&p);
                token_code = TOKEN_AND; break;
            case '|' :
                if (*p != '|') {
                    *end = p;
                    return LEXER_ERROR_A;
                }
                catToken(&p);
                token_code = TOKEN_OR; break;
            case '+' : token_code = TOKEN_PLUS; break;
            case '-' : token_code = TOKEN_MINU; break;
            case '*' : token_code = TOKEN_MULT; break;
            case '/' :
                if (*p == '/') {
                    while (!isNewline(p)) ++p;
                    ++lineNumber;
                    return getToken(p, begin, end);
                } else if (*p == '*') {
                    ++p;
                    while (*p != '*' || *(p+1) != '/') ++p; // @@ Potential out-of-bounds access if comment is unterminated.
                    p += 2;
                    return getToken(p, begin, end);
                }
                token_code = TOKEN_DIV;
                break;
            case '%' : token_code = TOKEN_MOD; break;
            case '<' :
                if (*p == '=') {
                    catToken(&p);
                    token_code = TOKEN_LEQ;
                } else token_code = TOKEN_LSS;
                break;
            case '>' :
                if (*p == '=') {
                    catToken(&p);
                    token_code = TOKEN_GEQ;
                } else token_code = TOKEN_GRE;
                break;
            case '=' :
                if (*p == '=') {
                    catToken(&p);
                    token_code = TOKEN_EQL;
                } else token_code = TOKEN_ASSIGN;
                break;
            case ';' : token_code = TOKEN_SEMICN; break;
            case ',' : token_code = TOKEN_COMMA; break;
            case '(' : token_code = TOKEN_LPARENT; break;
            case ')' : token_code = TOKEN_RPARENT; break;
            case '[' : token_code = TOKEN_LBRACK; break;
            case ']' : token_code = TOKEN_RBRACK; break;
            case '{' : token_code = TOKEN_LBRACE; break;
            case '}' : token_code = TOKEN_RBRACE; break;
            default :
                token_code = TOKEN_UNIDENTIFIED;
        }
    }
    *end = p;
    return 0;
}

struct lexerTable {
    CODE_TOKEN token_code;
    char *token;
    unsigned tokenLength;
    lexerTable() {
        this->token_code = 0;
        this->token = nullptr;
        this->tokenLength = 0;
    }
    lexerTable(CODE_TOKEN token_code, char *token, unsigned len) {
        this->token_code = token_code;
        this->token = token;
        tokenLength = len;
    }
} lexer[global_lexerTableMaxSize];
static int lexerSize;
struct lexerErrorTable {
    unsigned lineNumber;
    int type;
    lexerErrorTable() {
        this->lineNumber = 0;
        this->type = 0;
    }
    lexerErrorTable(unsigned lineNumber, int type) {
        this->lineNumber = lineNumber;
        this->type = type;
    }
} lexerError[global_lexerTableMaxSize];
static unsigned lexerErrorSize;

int initLexer(char *src_code, unsigned src_code_size) {
    lexerSize = 0;
    lexerErrorSize = 0;
    lineNumber = 1;
    int return_value = 0;
    char *current = src_code, *begin, *next;
    while (current - src_code < src_code_size && *current != EOF) {
        int return_code = getToken(current, &begin, &next);
        if (return_code > 0) { // error type
            return_value = -1;
            lexerError[lexerErrorSize++] = lexerErrorTable(lineNumber, return_code);
        } else if (return_code == 0) {
            lexer[lexerSize] = lexerTable(token_code, begin, next - begin);
            ++lexerSize;
        }
        current = next;
    }
    return return_value;
}

            case TOKEN_OR :     fprintf(file, "OR "); break;
            case TOKEN_FORTK :  fprintf(file, "FORTK "); break;
            case TOKEN_GETINTTK :fprintf(file, "GETINTTK "); break;
            case TOKEN_GETCHARTK :fprintf(file, "GETCHARTK "); break;
            case TOKEN_PRINTFTK :fprintf(file, "PRINTFTK "); break;
            case TOKEN_RETURNTK :fprintf(file, "RETURNTK "); break;
            case TOKEN_PLUS :   fprintf(file, "PLUS "); break;
            case TOKEN_MINU :   fprintf(file, "MINU "); break;
            case TOKEN_VOIDTK : fprintf(file, "VOIDTK "); break;
            case TOKEN_MULT :   fprintf(file, "MULT "); break;
            case TOKEN_DIV :    fprintf(file, "DIV "); break;

        p[len] = tmp;
    }
    fclose(file);
}
void errorLexer() {
    FILE *file = fopen("error.txt", "w");
    for (int i=0; i<lexerErrorSize; ++i)
        fprintf(file, "%u %c\n", lexerError[i].lineNumber, (char) lexerError[i].type);
    fclose(file);
}
