    private Token getSymbol() {
        char ch = sourceCode.charAt(pos);
        StringBuilder sb = new StringBuilder().append(ch);
        switch (ch) {
            case '&':
            case '|':
                if (pos + 1 < sourceCode.length() && sourceCode.charAt(pos + 1) == ch) {
                    sb.append(ch);
                    pos++;
                }
                break;
            case '<':
            case '>':
            case '=':
            case '!':
                if (pos + 1 < sourceCode.length() && sourceCode.charAt(pos + 1) == '=') {
                    sb.append('=');
                    pos++;
                }
                break;
            default:
        }
        pos++;
        String symbol = sb.toString();
        return new Token(symbol, Token.symbolToTokenType(symbol), line); // @@ [If a single '&' or '|' is encountered, it is treated as a valid symbol and passed to Token.symbolToTokenType(), which returns TokenType.INVALID. However, the lexer does not check for this INVALID type and does not report an error 'a' at this point. Instead, it silently adds an invalid token, and no error is ever recorded in ErrorHandler.]
    }
